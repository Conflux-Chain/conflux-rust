3#[unroll_for_loops]
4pub(super) const fn make_table_u32<const N: usize>(pow: u32) -> [u32; N] {
5    let mut ans = [0; N];
6    for i in 0..64 {
7        if i < N {
8            let val = (i + 2).pow(pow);
9            ans[i] = if val > u32::MAX as usize {
10                // zero represents infinity
11                0
12            } else {
13                val as u32
14            };
15        }
16    }
17    ans
18}

31#[inline]
32pub(super) fn search_table<I: Copy + Into<u64>, const N: usize>(
33    input: u64, table: &[I; N], pow: u32,
34) -> u64 {
35    if input < 1 << pow {
36        return if input >= 1 { 1 } else { 0 };
37    }
38
39    let mut index = N / 2;
40    let mut step = N / 2;
41    while step > 1 {
42        step /= 2;
43        let val = table[index].into();
44        if input < val || val == 0 {
45            index -= step;
46        } else {
47            index += step;
48        }
49    }
50
51    let val = table[index].into();
52    2 + if input < val || val == 0 {
53        index as u64 - 1
54    } else {
55        index as u64
56    }
57}
