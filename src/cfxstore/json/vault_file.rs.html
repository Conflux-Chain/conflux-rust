17use super::Crypto;
18use serde_derive::{Deserialize, Serialize};
19use serde_json;
20use std::io::{Read, Write};
21
22/// Vault meta file
23#[derive(Debug, PartialEq, Serialize, Deserialize)]
24pub struct VaultFile {
25    /// Vault password, encrypted with vault password
26    pub crypto: Crypto,
27    /// Vault metadata string
28    pub meta: Option<String>,
29}
30
31impl VaultFile {
32    pub fn load<R>(reader: R) -> Result<Self, serde_json::Error>
33    where R: Read {
34        serde_json::from_reader(reader)
35    }
36
37    pub fn write<W>(&self, writer: &mut W) -> Result<(), serde_json::Error>
38    where W: Write {
39        serde_json::to_writer(writer, self)
40    }
41}

43#[cfg(test)]
44mod test {
45    use crate::json::{Aes128Ctr, Cipher, Crypto, Kdf, Pbkdf2, Prf, VaultFile};
46    use serde_json;
47
48    #[test]
49    fn to_and_from_json() {
50        let file = VaultFile {
51			crypto: Crypto {
52				cipher: Cipher::Aes128Ctr(Aes128Ctr {
53					iv: "0155e3690be19fbfbecabcd440aa284b".into(),
54				}),
55				ciphertext: "4d6938a1f49b7782".into(),
56				kdf: Kdf::Pbkdf2(Pbkdf2 {
57					c: 1024,
58					dklen: 32,
59					prf: Prf::HmacSha256,
60					salt: "b6a9338a7ccd39288a86dba73bfecd9101b4f3db9c9830e7c76afdbd4f6872e5".into(),
61				}),
62				mac: "16381463ea11c6eb2239a9f339c2e780516d29d234ce30ac5f166f9080b5a262".into(),
63			},
64			meta: Some("{}".into()),
65		};
66
67        let serialized = serde_json::to_string(&file).unwrap();
68        let deserialized = serde_json::from_str(&serialized).unwrap();
69
70        assert_eq!(file, deserialized);
71    }
