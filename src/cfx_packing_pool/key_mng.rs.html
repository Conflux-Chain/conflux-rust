30impl<TX: PackingPoolTransaction> KeyMngTrait<PackingPoolMap<TX>>
31    for PackingPoolExtMap<TX::Sender>
32{
33    #[inline]
34    fn view_update(
35        &mut self, key: &TX::Sender, value: Option<&PackingBatch<TX>>,
36        old_value: Option<&PackingBatch<TX>>,
37    ) {
38        match (value, old_value) {
39            (Some(v), _) => {
40                self.insert(*key, v.first_gas_price());
41            }
42            (None, Some(_)) => {
43                self.remove(key);
44            }
45            (None, None) => {}
46        }
47    }
48
49    fn len(&self) -> usize { self.0.len() }
50
51    #[inline]
52    fn get_sort_key(&self, key: &TX::Sender) -> Option<U256> {
53        self.get(key).cloned()
54    }
55
56    #[inline]
57    fn make_sort_key(
58        &self, _key: &TX::Sender, value: &PackingBatch<TX>,
59    ) -> U256 {
60        value.first_gas_price()
61    }
