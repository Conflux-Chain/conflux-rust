18#[derive(Clone)]
19pub struct Throttle<S> {
20    service: S,
21    manager: TokenBucketManager,
22}
23
24impl<S> Throttle<S> {
25    pub fn new(file: Option<&str>, section: &str, s: S) -> Self {
26        let manager = match file {
27            Some(file) => TokenBucketManager::load(file, Some(section))
28                .expect("invalid throttling configuration file"),
29            None => TokenBucketManager::default(),
30        };
31
32        Throttle {
33            service: s,
34            manager,
35        }
36    }
37
38    pub fn before(&self, name: &str) -> RpcResult<()> {
39        let bucket = match self.manager.get(name) {
40            Some(bucket) => bucket,
41            None => return Ok(()),
42        };
43
44        let result = bucket.lock().throttle_default();
45
46        match result {
47            ThrottleResult::Success => Ok(()),
48            ThrottleResult::Throttled(wait_time) => {
49                debug!("RPC {} throttled in {:?}", name, wait_time);
50                let err = request_rejected_too_many_request_error(Some(
51                    format!("throttled in {:?}", wait_time),
52                ));
53                bail!(err)
54            }
55            ThrottleResult::AlreadyThrottled => {
56                debug!("RPC {} already throttled", name);
57                let err = request_rejected_too_many_request_error(Some(
58                    "already throttled, please try again later".into(),
59                ));
60                bail!(err)
61            }
62        }
63    }
64}
65
66impl<S> RpcServiceT for Throttle<S>
67where S: RpcServiceT<
68            MethodResponse = MethodResponse,
69            BatchResponse = MethodResponse,
70            NotificationResponse = MethodResponse,
71        > + Send
72        + Sync
73        + Clone
74        + 'static
75{
76    type BatchResponse = S::BatchResponse;
77    type MethodResponse = S::MethodResponse;
78    type NotificationResponse = S::NotificationResponse;
79
80    fn call<'a>(
81        &self, req: Request<'a>,
82    ) -> impl Future<Output = Self::MethodResponse> + Send + 'a {
83        let throttle_result = self.before(req.method_name());
84        match throttle_result {
85            Ok(_) => {
86                debug!("throttle interceptor: method `{}` success", req.method);
87                ResponseFuture::future(self.service.call(req))
88            }
89            Err(e) => {
90                debug!("throttle interceptor: method `{}` failed", req.method);
91                ResponseFuture::ready(MethodResponse::error(req.id, e))
92            }
93        }
94    }
95
96    fn batch<'a>(
97        &self, mut batch: Batch<'a>,
98    ) -> impl Future<Output = Self::BatchResponse> + Send + 'a {
99        for entry in batch.iter_mut() {
100            let (id, name) = match entry {
101                Ok(BatchEntry::Call(req)) => {
102                    (req.id.clone(), req.method_name())
103                }
104                Ok(BatchEntry::Notification(_)) => continue,
105                Err(_) => continue,
106            };
107
108            let throttle_result = self.before(name);
109            if let Err(e) = throttle_result {
110                // This will create a new error response for batch and replace
111                // the method call
112                *entry = Err(BatchEntryErr::new(id, e));
113            }
114        }
115
116        self.service.batch(batch)
117    }
118
119    fn notification<'a>(
120        &self, n: Notification<'a>,
121    ) -> impl Future<Output = Self::NotificationResponse> + Send + 'a {
122        let throttle_result = self.before(n.method_name());
123        match throttle_result {
124            Ok(_) => ResponseFuture::future(self.service.notification(n)),
125            // Notifications are not expected to return a response so just
126            // ignore if the rate limit is reached.
127            Err(_e) => ResponseFuture::ready(MethodResponse::notification()),
128        }
129    }
130}
