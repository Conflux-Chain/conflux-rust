89impl PosStateConfigTrait for OnceCell<PosStateConfig> {
90    fn round_per_term(&self, view: u64) -> Round {
91        let conf = self.get().unwrap();
92        if view < conf.cip136_transition_view {
93            conf.round_per_term
94        } else {
95            conf.cip136_round_per_term
96        }
97    }
98
99    /// A term `n` is open for election in the view range
100    /// `(n * ROUND_PER_TERM - ELECTION_TERM_START_ROUND, n * ROUND_PER_TERM -
101    /// ELECTION_TERM_END_ROUND]`
102    fn election_term_start_round(&self, view: u64) -> Round {
103        self.round_per_term(view) / 2 * 3
104    }
105
106    fn election_term_end_round(&self, view: u64) -> Round {
107        self.round_per_term(view) / 2
108    }
109
110    fn first_start_election_view(&self) -> u64 {
111        TERM_LIST_LEN as u64 * self.round_per_term(0)
112            - self.election_term_start_round(0)
113    }
114
115    fn first_end_election_view(&self) -> u64 {
116        TERM_LIST_LEN as u64 * self.round_per_term(0)
117            - self.election_term_end_round(0)
118    }
119
120    fn term_max_size(&self) -> usize { self.get().unwrap().term_max_size }
121
122    fn term_elected_size(&self) -> usize {
123        self.get().unwrap().term_elected_size
124    }
125
126    fn in_queue_locked_views(&self, view: u64) -> u64 {
127        let conf = self.get().unwrap();
128        if view >= conf.cip99_transition_view
129            && view < conf.cip136_transition_view
130        {
131            conf.cip99_in_queue_locked_views
132        } else if view >= conf.cip136_transition_view {
133            conf.cip136_in_queue_locked_views
134        } else {
135            conf.in_queue_locked_views
136        }
137    }
138
139    fn out_queue_locked_views(&self, view: u64) -> u64 {
140        let conf = self.get().unwrap();
141        if view >= conf.cip99_transition_view
142            && view < conf.cip136_transition_view
143        {
144            conf.cip99_out_queue_locked_views
145        } else if view >= conf.cip136_transition_view {
146            conf.cip136_out_queue_locked_views
147        } else {
148            conf.out_queue_locked_views
149        }
150    }
151
152    fn force_retired_locked_views(&self, view: u64) -> u64 {
153        self.out_queue_locked_views(view)
154    }
155
156    fn force_retire_check_epoch_count(&self, view: u64) -> u64 {
157        let conf = self.get().unwrap();
158        if view >= conf.cip99_transition_view {
159            // This is set according to the value of `TERM_LIST_LEN`.
160            // Since `TERM_LIST_LEN` is hardcoded, we do not parameterize this.
161            CIP99_FORCE_RETIRE_EPOCH_COUNT
162        } else {
163            1
164        }
165    }
166
167    fn max_nonce_per_account(&self, view: u64) -> u64 {
168        let conf = self.get().unwrap();
169        if view >= conf.nonce_limit_transition_view {
170            conf.max_nonce_per_account
171        } else {
172            u64::MAX
173        }
174    }
175
176    fn get_term_view(&self, view: u64) -> (u64, u64) {
177        let conf = self.get().unwrap();
178        if view < conf.cip136_transition_view {
179            (view / conf.round_per_term, view % conf.round_per_term)
180        } else {
181            let transition_term =
182                conf.cip136_transition_view / conf.round_per_term;
183            let view_after = view - conf.cip136_transition_view;
184            (
185                transition_term + view_after / conf.cip136_round_per_term,
186                view_after % conf.cip136_round_per_term,
187            )
188        }
189    }
190
191    fn get_starting_view_for_term(&self, term: u64) -> Option<u64> {
192        let conf = self.get().unwrap();
193        let transition_term = conf.cip136_transition_view / conf.round_per_term;
194        if term < transition_term {
195            Some(term * conf.round_per_term)
196        } else {
197            (term - transition_term)
198                .checked_mul(conf.cip136_round_per_term)
199                .map(|v| v + conf.cip136_transition_view)
200        }
201    }
202
203    fn dispute_locked_views(&self, view: u64) -> Option<u64> {
204        let conf = self.get().unwrap();
205        if view < conf.cip156_transition_view {
206            None
207        } else {
208            Some(conf.cip156_dispute_locked_views)
209        }
210    }
211}
