26/// A Diem `NetworkAddress` represented as a stack of protocols.
27///
28/// Most addresses look like: `/ip4/<addr>/tcp/<port>/ln-handshake/<version>`
29///
30/// The Noise IK protocol and x25519 key exchange (from Diem's P2P layer) have
31/// been removed — Conflux uses its own P2P networking.
32#[derive(Clone, Eq, Hash, PartialEq)]
33pub struct NetworkAddress(Vec<Protocol>);
34
35/// A single protocol in the [`NetworkAddress`] protocol stack.
36#[derive(Clone, Debug, Eq, Hash, PartialEq, Deserialize, Serialize)]
37#[cfg_attr(any(test, feature = "fuzzing"), derive(Arbitrary))]
38pub enum Protocol {
39    Ip4(Ipv4Addr),
40    Ip6(Ipv6Addr),
41    Dns(DnsName),
42    Dns4(DnsName),
43    Dns6(DnsName),
44    Tcp(u16),
45    Memory(u16),
46    Handshake(u8),
47}
48
49/// A minimally parsed DNS name.
50#[derive(Clone, Debug, Eq, Hash, PartialEq, Serialize)]
51pub struct DnsName(String);
52
53/// Possible errors when parsing a human-readable [`NetworkAddress`].
54#[derive(Error, Debug)]
55pub enum ParseError {
56    #[error("unknown protocol type: '{0}'")]
57    UnknownProtocolType(String),
58
59    #[error("unexpected end of protocol string")]
60    UnexpectedEnd,
61
62    #[error("error parsing ip4/ip6 address: {0}")]
63    ParseAddrError(#[from] net::AddrParseError),
64
65    #[error("error parsing integer: {0}")]
66    ParseIntError(#[from] num::ParseIntError),
67
68    #[error("network address cannot be empty")]
69    EmptyProtocolString,
70
71    #[error("protocol string must start with '/'")]
72    InvalidProtocolString,
73
74    #[error("dns name cannot be empty")]
75    EmptyDnsNameString,
76
77    #[error("dns name cannot contain '/' characters")]
78    InvalidDnsNameCharacter,
79
80    #[error("dns name is too long: len: {0} bytes, max len: 255 bytes")]
81    DnsNameTooLong(usize),
82}
83
84#[derive(Error, Debug)]
85#[error("network address cannot be empty")]
86pub struct EmptyError;

127impl FromStr for NetworkAddress {
128    type Err = ParseError;
129
130    fn from_str(s: &str) -> Result<Self, Self::Err> {
131        if s.is_empty() {
132            return Err(ParseError::EmptyProtocolString);
133        }
134
135        let mut protocols = Vec::new();
136        let mut parts_iter = s.split('/');
137
138        // the first character must be '/'
139        if parts_iter.next() != Some("") {
140            return Err(ParseError::InvalidProtocolString);
141        }
142
143        // parse all `Protocol`s
144        while let Some(protocol_type) = parts_iter.next() {
145            protocols.push(Protocol::parse(protocol_type, &mut parts_iter)?);
146        }
147
148        Ok(NetworkAddress::new(protocols))
149    }
150}

152impl ToSocketAddrs for NetworkAddress {
153    type Iter = std::vec::IntoIter<SocketAddr>;
154
155    fn to_socket_addrs(&self) -> Result<Self::Iter, std::io::Error> {
156        if let Some(((ipaddr, port), _)) = parse_ip_tcp(self.as_slice()) {
157            Ok(vec![SocketAddr::new(ipaddr, port)].into_iter())
158        } else if let Some(((ip_filter, dns_name, port), _)) =
159            parse_dns_tcp(self.as_slice())
160        {
161            format!("{}:{}", dns_name, port).to_socket_addrs().map(|v| {
162                v.filter(|addr| ip_filter.matches(addr.ip()))
163                    .collect::<Vec<_>>()
164                    .into_iter()
165            })
166        } else {
167            Ok(vec![].into_iter())
168        }
169    }
170}

266impl fmt::Display for Protocol {
267    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
268        use self::Protocol::*;
269        match self {
270            Ip4(addr) => write!(f, "/ip4/{}", addr),
271            Ip6(addr) => write!(f, "/ip6/{}", addr),
272            Dns(domain) => write!(f, "/dns/{}", domain),
273            Dns4(domain) => write!(f, "/dns4/{}", domain),
274            Dns6(domain) => write!(f, "/dns6/{}", domain),
275            Tcp(port) => write!(f, "/tcp/{}", port),
276            Memory(port) => write!(f, "/memory/{}", port),
277            Handshake(version) => write!(f, "/ln-handshake/{}", version),
278        }
279    }
280}

282fn parse_one<'a, T>(
283    args: &mut impl Iterator<Item = &'a str>,
284) -> Result<T, ParseError>
285where
286    T: FromStr,
287    T::Err: Into<ParseError>,
288{
289    let next_arg = args.next().ok_or(ParseError::UnexpectedEnd)?;
290    next_arg.parse().map_err(Into::into)
291}
292
293impl Protocol {
294    fn parse<'a>(
295        protocol_type: &str, args: &mut impl Iterator<Item = &'a str>,
296    ) -> Result<Protocol, ParseError> {
297        let protocol = match protocol_type {
298            "ip4" => Protocol::Ip4(parse_one(args)?),
299            "ip6" => Protocol::Ip6(parse_one(args)?),
300            "dns" => Protocol::Dns(parse_one(args)?),
301            "dns4" => Protocol::Dns4(parse_one(args)?),
302            "dns6" => Protocol::Dns6(parse_one(args)?),
303            "tcp" => Protocol::Tcp(parse_one(args)?),
304            "memory" => Protocol::Memory(parse_one(args)?),
305            "ln-handshake" => Protocol::Handshake(parse_one(args)?),
306            unknown => {
307                return Err(ParseError::UnknownProtocolType(
308                    unknown.to_string(),
309                ))
310            }
311        };
312        Ok(protocol)
313    }
314}

329impl DnsName {
330    fn validate(s: &str) -> Result<(), ParseError> {
331        if s.is_empty() {
332            Err(ParseError::EmptyDnsNameString)
333        } else if s.as_bytes().len() > MAX_DNS_NAME_SIZE {
334            Err(ParseError::DnsNameTooLong(s.as_bytes().len()))
335        } else if s.contains('/') {
336            Err(ParseError::InvalidDnsNameCharacter)
337        } else {
338            Ok(())
339        }
340    }
341}
342
343impl From<DnsName> for String {
344    fn from(dns_name: DnsName) -> String { dns_name.0 }
345}
346
347impl AsRef<str> for DnsName {
348    fn as_ref(&self) -> &str { self.0.as_ref() }
349}

351impl TryFrom<String> for DnsName {
352    type Error = ParseError;
353
354    fn try_from(s: String) -> Result<Self, Self::Error> {
355        DnsName::validate(s.as_str()).map(|_| DnsName(s))
356    }
357}

406/// parse the `&[Protocol]` into the `"/memory/<port>"` prefix and unparsed
407/// `&[Protocol]` suffix.
408pub fn parse_memory(protos: &[Protocol]) -> Option<(u16, &[Protocol])> {
409    protos
410        .split_first()
411        .and_then(|(first, suffix)| match first {
412            Protocol::Memory(port) => Some((*port, suffix)),
413            _ => None,
414        })
415}
416
417/// parse the `&[Protocol]` into the `"/ip4/<addr>/tcp/<port>"` or
418/// `"/ip6/<addr>/tcp/<port>"` prefix and unparsed `&[Protocol]` suffix.
419pub fn parse_ip_tcp(
420    protos: &[Protocol],
421) -> Option<((IpAddr, u16), &[Protocol])> {
422    use Protocol::*;
423
424    if protos.len() < 2 {
425        return None;
426    }
427
428    let (prefix, suffix) = protos.split_at(2);
429    match prefix {
430        [Ip4(ip), Tcp(port)] => Some(((IpAddr::V4(*ip), *port), suffix)),
431        [Ip6(ip), Tcp(port)] => Some(((IpAddr::V6(*ip), *port), suffix)),
432        _ => None,
433    }
434}
435
436#[derive(Copy, Clone, Debug, Eq, PartialEq)]
437pub enum IpFilter {
438    Any,
439    OnlyIp4,
440    OnlyIp6,
441}
442
443impl IpFilter {
444    pub fn matches(&self, ipaddr: IpAddr) -> bool {
445        match self {
446            IpFilter::Any => true,
447            IpFilter::OnlyIp4 => ipaddr.is_ipv4(),
448            IpFilter::OnlyIp6 => ipaddr.is_ipv6(),
449        }
450    }
451}
452
453/// parse the `&[Protocol]` into the `"/dns/<domain>/tcp/<port>"`,
454/// `"/dns4/<domain>/tcp/<port>"`, or `"/dns6/<domain>/tcp/<port>"` prefix and
455/// unparsed `&[Protocol]` suffix.
456pub fn parse_dns_tcp(
457    protos: &[Protocol],
458) -> Option<((IpFilter, &DnsName, u16), &[Protocol])> {
459    use Protocol::*;
460
461    if protos.len() < 2 {
462        return None;
463    }
464
465    let (prefix, suffix) = protos.split_at(2);
466    match prefix {
467        [Dns(name), Tcp(port)] => Some(((IpFilter::Any, name, *port), suffix)),
468        [Dns4(name), Tcp(port)] => {
469            Some(((IpFilter::OnlyIp4, name, *port), suffix))
470        }
471        [Dns6(name), Tcp(port)] => {
472            Some(((IpFilter::OnlyIp6, name, *port), suffix))
473        }
474        _ => None,
475    }
476}
477
478pub fn parse_tcp(protos: &[Protocol]) -> Option<((String, u16), &[Protocol])> {
479    use Protocol::*;
480
481    if protos.len() < 2 {
482        return None;
483    }
484
485    let (prefix, suffix) = protos.split_at(2);
486    match prefix {
487        [Ip4(ip), Tcp(port)] => Some(((ip.to_string(), *port), suffix)),
488        [Ip6(ip), Tcp(port)] => Some(((ip.to_string(), *port), suffix)),
489        [Dns(name), Tcp(port)] => Some(((name.to_string(), *port), suffix)),
490        [Dns4(name), Tcp(port)] => Some(((name.to_string(), *port), suffix)),
491        [Dns6(name), Tcp(port)] => Some(((name.to_string(), *port), suffix)),
492        _ => None,
493    }
494}
495
496/// parse the `&[Protocol]` into the `"/ln-handshake/<version>"` prefix and
497/// unparsed `&[Protocol]` suffix.
498pub fn parse_handshake(protos: &[Protocol]) -> Option<(u8, &[Protocol])> {
499    match protos.split_first() {
500        Some((Protocol::Handshake(version), suffix)) => {
501            Some((*version, suffix))
502        }
503        _ => None,
504    }
505}
