8use crate::{
9    access_path::AccessPath,
10    account_config::{
11        constants::{xus_tag, CORE_CODE_ADDRESS, DIEM_MODULE_NAME},
12        resources::PreburnResource,
13    },
14};
15use move_core_types::{
16    language_storage::{StructTag, TypeTag},
17    move_resource::MoveResource,
18};
19#[cfg(any(test, feature = "fuzzing"))]
20use proptest_derive::Arbitrary;
21use serde::{Deserialize, Serialize};
22
23/// The preburn along with metadata held in a preburn queue.
24#[derive(Debug, Serialize, Deserialize)]
25#[cfg_attr(any(test, feature = "fuzzing"), derive(Arbitrary))]
26pub struct PreburnWithMetadataResource {
27    preburn: PreburnResource,
28    metadata: Vec<u8>,
29}
30
31impl PreburnWithMetadataResource {
32    pub fn new(preburn: PreburnResource, metadata: Vec<u8>) -> Self {
33        Self { preburn, metadata }
34    }
35
36    pub fn preburn(&self) -> &PreburnResource { &self.preburn }
37
38    pub fn metadata(&self) -> &[u8] { &self.metadata }
39
40    // TODO/XXX: remove this once the MoveResource trait allows type arguments
41    // to `struct_tag`.
42    pub fn struct_tag_for_currency(currency_typetag: TypeTag) -> StructTag {
43        StructTag {
44            address: CORE_CODE_ADDRESS,
45            name: PreburnWithMetadataResource::struct_identifier(),
46            module: PreburnWithMetadataResource::module_identifier(),
47            type_params: vec![currency_typetag],
48        }
49    }
50
51    // TODO: remove this once the MoveResource trait allows type arguments to
52    // `resource_path`.
53    pub fn access_path_for(currency_typetag: TypeTag) -> Vec<u8> {
54        AccessPath::resource_access_vec(
55            PreburnWithMetadataResource::struct_tag_for_currency(
56                currency_typetag,
57            ),
58        )
59    }
60}
61
62impl MoveResource for PreburnWithMetadataResource {
63    const MODULE_NAME: &'static str = DIEM_MODULE_NAME;
64    const STRUCT_NAME: &'static str = "PreburnWithMetadata";
65
66    fn type_params() -> Vec<TypeTag> { vec![xus_tag()] }
67}
