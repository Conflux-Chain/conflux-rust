8use crate::{event::EventHandle, on_chain_config::OnChainConfig};
9use move_core_types::move_resource::MoveResource;
10use serde::{Deserialize, Serialize};
11
12#[derive(Debug, Serialize, Deserialize)]
13pub struct Credential {
14    human_name: String,
15    base_url: String,
16    compliance_public_key: Vec<u8>,
17    expiration_date: u64,
18    compliance_key_rotation_events: EventHandle,
19    base_url_rotation_events: EventHandle,
20}
21
22impl Credential {
23    pub fn human_name(&self) -> &str { &self.human_name }
24
25    pub fn base_url(&self) -> &str { &self.base_url }
26
27    pub fn expiration_date(&self) -> u64 { self.expiration_date }
28
29    pub fn compliance_public_key(&self) -> &[u8] { &self.compliance_public_key }
30
31    pub fn compliance_key_rotation_events(&self) -> &EventHandle {
32        &self.compliance_key_rotation_events
33    }
34
35    pub fn base_url_rotation_events(&self) -> &EventHandle {
36        &self.base_url_rotation_events
37    }
38}
39
40impl MoveResource for Credential {
41    const MODULE_NAME: &'static str = "DualAttestation";
42    const STRUCT_NAME: &'static str = "Credential";
43}
