8use crate::{
9    access_path::Path,
10    diem_timestamp::DiemTimestampResource,
11    on_chain_config::{ConfigurationResource, OnChainConfig, ValidatorSet},
12};
13use anyhow::Result;
14use move_core_types::move_resource::MoveResource;
15use serde::{de::DeserializeOwned, Deserialize, Serialize};
16use std::{collections::btree_map::BTreeMap, convert::TryFrom, fmt};
17
18#[derive(Default, Deserialize, PartialEq, Serialize)]
19pub struct AccountState(BTreeMap<Vec<u8>, Vec<u8>>);
20
21impl AccountState {
22    pub fn get_configuration_resource(
23        &self,
24    ) -> Result<Option<ConfigurationResource>> {
25        self.get_resource::<ConfigurationResource>()
26    }
27
28    pub fn get_diem_timestamp_resource(
29        &self,
30    ) -> Result<Option<DiemTimestampResource>> {
31        self.get_resource::<DiemTimestampResource>()
32    }
33
34    pub fn get_validator_set(&self) -> Result<Option<ValidatorSet>> {
35        self.get_config::<ValidatorSet>()
36    }
37
38    pub fn get(&self, key: &[u8]) -> Option<&Vec<u8>> { self.0.get(key) }
39
40    pub fn get_resource_impl<T: DeserializeOwned>(
41        &self, key: &[u8],
42    ) -> Result<Option<T>> {
43        self.0
44            .get(key)
45            .map(|bytes| bcs::from_bytes(bytes))
46            .transpose()
47            .map_err(Into::into)
48    }
49
50    pub fn insert(&mut self, key: Vec<u8>, value: Vec<u8>) -> Option<Vec<u8>> {
51        self.0.insert(key, value)
52    }
53
54    pub fn remove(&mut self, key: &[u8]) -> Option<Vec<u8>> {
55        self.0.remove(key)
56    }
57
58    pub fn iter(
59        &self,
60    ) -> impl std::iter::Iterator<Item = (&Vec<u8>, &Vec<u8>)> {
61        self.0.iter()
62    }
63
64    pub fn get_config<T: OnChainConfig>(&self) -> Result<Option<T>> {
65        self.get_resource_impl(&T::CONFIG_ID.access_path().path)
66    }
67
68    pub fn get_resource<T: MoveResource + DeserializeOwned>(
69        &self,
70    ) -> Result<Option<T>> {
71        self.get_resource_impl(&T::struct_tag().access_vector())
72    }
73
74    /// Return an iterator over the module values stored under this account
75    pub fn get_modules(&self) -> impl Iterator<Item = &Vec<u8>> {
76        self.0.iter().filter_map(|(k, v)| {
77            match Path::try_from(k).expect("Invalid access path") {
78                Path::Code(_) => Some(v),
79                Path::Resource(_) => None,
80            }
81        })
82    }
83}
84
85impl fmt::Debug for AccountState {
86    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
87        let diem_timestamp_str = self
88            .get_diem_timestamp_resource()
89            .map(|diem_timestamp_opt| format!("{:#?}", diem_timestamp_opt))
90            .unwrap_or_else(|e| format!("parse: {:#?}", e));
91
92        let validator_set_str = self
93            .get_validator_set()
94            .map(|validator_set_opt| format!("{:#?}", validator_set_opt))
95            .unwrap_or_else(|e| format!("parse error: {:#?}", e));
96
97        write!(
98            f,
99            "{{ \n \
100             DiemTimestamp {{ {} }} \n \
101             ValidatorSet {{ {} }} \n \
102             }}",
103            diem_timestamp_str, validator_set_str,
104        )
105    }
106}
