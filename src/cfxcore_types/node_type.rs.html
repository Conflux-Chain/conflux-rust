5use malloc_size_of_derive::MallocSizeOf as DeriveMallocSizeOf;
6use rlp::{Decodable, DecoderError, Encodable, Rlp, RlpStream};
7use std::str::FromStr;
8
9#[derive(Clone, Copy, Debug, PartialEq, DeriveMallocSizeOf)]
10#[repr(u8)]
11pub enum NodeType {
12    Archive,
13    Full,
14    Light,
15    Unknown,
16}
17
18impl Default for NodeType {
19    fn default() -> NodeType { NodeType::Unknown }
20}
21
22impl From<u8> for NodeType {
23    fn from(raw: u8) -> NodeType {
24        match raw {
25            0 => NodeType::Archive,
26            1 => NodeType::Full,
27            2 => NodeType::Light,
28            _ => NodeType::Unknown,
29        }
30    }
31}
32
33impl From<&NodeType> for u8 {
34    fn from(node_type: &NodeType) -> u8 {
35        match node_type {
36            NodeType::Archive => 0,
37            NodeType::Full => 1,
38            NodeType::Light => 2,
39            NodeType::Unknown => 0xff,
40        }
41    }
42}
43
44impl FromStr for NodeType {
45    type Err = String;
46
47    fn from_str(s: &str) -> Result<Self, Self::Err> {
48        let node_type = match s {
49            "archive" => Self::Archive,
50            "full" => Self::Full,
51            "light" => Self::Light,
52            _ => Self::Unknown,
53        };
54        Ok(node_type)
55    }
56}
57
58impl Encodable for NodeType {
59    fn rlp_append(&self, s: &mut RlpStream) {
60        let raw: u8 = self.into();
61        s.append_internal(&raw);
62    }
63}
64
65impl Decodable for NodeType {
66    fn decode(rlp: &Rlp) -> Result<Self, DecoderError> {
67        let raw: u8 = rlp.as_val()?;
68        Ok(NodeType::from(raw))
69    }
70}
