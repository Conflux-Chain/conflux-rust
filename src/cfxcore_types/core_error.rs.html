12#[derive(Debug, PartialEq, Clone, Eq)]
13/// Errors concerning block processing.
14pub enum BlockError {
15    /// Number field of header is invalid.
16    InvalidHeight(Mismatch<u64>),
17    /// Block has too much gas used.
18    TooMuchGasUsed(OutOfBounds<U256>),
19    /// State root header field is invalid.
20    InvalidStateRoot(Mismatch<H256>),
21    /// Gas used header field is invalid.
22    InvalidGasUsed(Mismatch<U256>),
23    /// Transactions root header field is invalid.
24    InvalidTransactionsRoot(Mismatch<H256>),
25    /// Difficulty is out of range; this can be used as an looser error prior
26    /// to getting a definitive value for difficulty. This error needs only
27    /// provide bounds of which it is out.
28    DifficultyOutOfBounds(OutOfBounds<U256>),
29    /// Difficulty header field is invalid; this is a strong error used after
30    /// getting a definitive value for difficulty (which is provided).
31    InvalidDifficulty(OutOfBounds<U256>),
32    /// Proof-of-work aspect of seal, which we assume is a 256-bit value, is
33    /// invalid.
34    InvalidProofOfWork(OutOfBounds<H256>),
35    /// Gas limit header field is invalid.
36    InvalidGasLimit(OutOfBounds<U256>),
37    /// Total gas limits of transactions in block is out of bound.
38    InvalidPackedGasLimit(OutOfBounds<U256>),
39    /// Total rlp sizes of transactions in block is out of bound.
40    InvalidBlockSize(OutOfBounds<u64>),
41    InvalidBasePrice(Mismatch<SpaceMap<U256>>),
42    /// Timestamp header field is invalid.
43    InvalidTimestamp(OutOfBounds<SystemTime>),
44    /// Timestamp header field is too far in future.
45    TemporarilyInvalid(OutOfBounds<SystemTime>),
46    /// Too many referees in a block
47    TooManyReferees(OutOfBounds<usize>),
48    /// Too long custom data in header
49    TooLongCustomInHeader(OutOfBounds<usize>),
50    /// Too many transactions from a particular address.
51    TooManyTransactions(Address),
52    /// Parent given is unknown.
53    UnknownParent(H256),
54    /// Duplicate parent or referee hashes exist.
55    DuplicateParentOrRefereeHashes(H256),
56    /// The value in `custom` does not match the specification.
57    InvalidCustom(Vec<Bytes>, Vec<Bytes>),
58    /// Should have a PoS reference but it's not set.
59    MissingPosReference,
60    /// Should not have a PoS reference but it's set.
61    UnexpectedPosReference,
62    /// Should have a base fee but it's not set.
63    MissingBaseFee,
64    /// Should not have a base fee but it's set.
65    UnexpectedBaseFee,
66    /// The PoS reference violates the validity rule (it should extend the PoS
67    /// reference of the parent and referees).
68    InvalidPosReference,
69}
70
71impl fmt::Display for BlockError {
72    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
73        use self::BlockError::*;
74
75        let msg = match *self {
76            InvalidHeight(ref mis) => format!("Invalid block height: {}", mis),
77            TooMuchGasUsed(ref oob) => {
78                format!("Block has too much gas used. {}", oob)
79            }
80            InvalidStateRoot(ref mis) => {
81                format!("Invalid state root in header: {}", mis)
82            }
83            InvalidGasUsed(ref mis) => {
84                format!("Invalid gas used in header: {}", mis)
85            }
86            InvalidTransactionsRoot(ref mis) => {
87                format!("Invalid transactions root in header: {}", mis)
88            }
89            DifficultyOutOfBounds(ref oob) => {
90                format!("Invalid block difficulty: {}", oob)
91            }
92            InvalidDifficulty(ref oob) => {
93                format!("Invalid block difficulty: {}", oob)
94            }
95            InvalidProofOfWork(ref oob) => {
96                format!("Block has invalid PoW: {}", oob)
97            }
98            InvalidGasLimit(ref oob) => format!("Invalid gas limit: {}", oob),
99            InvalidBasePrice(ref mis) => {
100                format!("Invalid base price: {:?}", mis)
101            }
102            InvalidPackedGasLimit(ref oob) => {
103                format!("Invalid packed gas limit: {}", oob)
104            }
105            InvalidBlockSize(ref oob) => format!("Invalid block size: {}", oob),
106            InvalidTimestamp(ref oob) => {
107                let oob =
108                    oob.map(|st| st.elapsed().unwrap_or_default().as_secs());
109                format!("Invalid timestamp in header: {}", oob)
110            }
111            TemporarilyInvalid(ref oob) => {
112                let oob =
113                    oob.map(|st| st.elapsed().unwrap_or_default().as_secs());
114                format!("Future timestamp in header: {}", oob)
115            }
116            UnknownParent(ref hash) => format!("Unknown parent: {}", hash),
117            TooManyReferees(ref num) => format!("Too many referees: {}", num),
118            TooLongCustomInHeader(ref num) => {
119                format!("Too long custom data in block header: {}", num)
120            }
121            TooManyTransactions(ref address) => {
122                format!("Too many transactions from: {}", address)
123            }
124            DuplicateParentOrRefereeHashes(ref hash) => {
125                format!("Duplicate parent or referee hashes: {}", hash)
126            }
127            InvalidCustom(ref header_custom, ref expected_custom_prefix) => {
128                format!(
129                    "Invalid custom in header: expect prefix {:?}, get {:?}",
130                    expected_custom_prefix, header_custom
131                )
132            }
133            MissingPosReference => "Missing PoS reference".into(),
134            UnexpectedPosReference => "Should not have PoS reference".into(),
135            MissingBaseFee => "Missing base fee".into(),
136            UnexpectedBaseFee => "Should not have base fee".into(),
137            InvalidPosReference => "The PoS reference is invalid".into(),
138        };
139
140        f.write_fmt(format_args!("Block error ({})", msg))
141    }
142}

144impl error::Error for BlockError {
145    fn description(&self) -> &str { "Block error" }
146}

148#[derive(Error, Debug)]
149pub enum CoreError {
150    #[error(transparent)]
151    Block(#[from] BlockError),
152    #[error(transparent)]
153    Transaction(#[from] TransactionError),
154    #[error(transparent)]
155    Filter(#[from] FilterError),
156    #[error("PoW hash is invalid or out of date.")]
157    PowHashInvalid,
158}
