50    /// Get all of the events given a transaction version.
51    /// We don't need a proof for this because it's only used to get all events
52    /// for a version which can be proved from the root hash of the event tree.
53    pub fn get_events_by_version(
54        &self, version: Version,
55    ) -> Result<Vec<ContractEvent>> {
56        let mut events = vec![];
57
58        let mut iter = self.db.iter::<EventSchema>(ReadOptions::default())?;
59        // Grab the first event and then iterate until we get all events for
60        // this version.
61        iter.seek(&version)?;
62        while let Some(((ver, index), event)) = iter.next().transpose()? {
63            if ver != version {
64                break;
65            }
66            events.push(event);
67        }
68
69        Ok(events)
70    }
71
72    pub fn get_events_by_version_iter(
73        &self, start_version: Version, num_versions: usize,
74    ) -> Result<EventsByVersionIter<'_>> {
75        let mut iter = self.db.iter::<EventSchema>(Default::default())?;
76        iter.seek(&start_version)?;
77
78        Ok(EventsByVersionIter {
79            inner: iter.peekable(),
80            expected_next_version: start_version,
81            end_version: start_version
82                .checked_add(num_versions as u64)
83                .ok_or_else(|| format_err!("Too many versions requested."))?,
84        })
85    }
86
87    fn get_event_by_version_and_index(
88        &self, version: Version, index: u64,
89    ) -> Result<ContractEvent> {
90        self.db
91            .get::<EventSchema>(&(version, index))?
92            .ok_or_else(|| {
93                DiemDbError::NotFound(format!(
94                    "Event {} of Txn {}",
95                    index, version
96                ))
97                .into()
98            })
99    }

101    /// Get the event raw data given transaction version and the index of the
102    /// event queried.
103    pub fn get_event_with_proof_by_version_and_index(
104        &self, version: Version, index: u64,
105    ) -> Result<(ContractEvent, EventAccumulatorProof)> {
106        // Get event content.
107        let event = self.get_event_by_version_and_index(version, index)?;
108
109        // Get the number of events in total for the transaction at `version`.
110        let mut iter = self.db.iter::<EventSchema>(ReadOptions::default())?;
111        iter.seek_for_prev(&(version + 1))?;
112        let num_events = match iter.next().transpose()? {
113            Some(((ver, index), _)) if ver == version => (index + 1),
114            _ => unreachable!(), /* since we've already got at least one
115                                  * event above */
116        };
117
118        // Get proof.
119        let proof = Accumulator::get_proof(
120            &EventHashReader::new(self, version),
121            num_events,
122            index,
123        )?;
124
125        Ok((event, proof))
126    }

187    /// Save contract events yielded by the transaction at `version` and return
188    /// root hash of the event accumulator formed by these events.
189    pub fn put_events(
190        &self, version: u64, events: &[ContractEvent], cs: &mut ChangeSet,
191    ) -> Result<HashValue> {
192        // Event table and indices updates
193        events.iter().enumerate().try_for_each::<_, Result<_>>(
194            |(idx, event)| {
195                cs.batch.put::<EventSchema>(&(version, idx as u64), event)?;
196                cs.batch.put::<EventByKeySchema>(
197                    event.key(),
198                    &(version, idx as u64),
199                )?;
200                cs.batch.put::<EventByVersionSchema>(
201                    &(*event.key(), version),
202                    &(idx as u64),
203                )?;
204                Ok(())
205            },
206        )?;
207
208        // EventAccumulatorSchema updates
209        let event_hashes: Vec<HashValue> =
210            events.iter().map(ContractEvent::hash).collect();
211        let (root_hash, writes) =
212            EmptyAccumulator::append(&EmptyReader, 0, &event_hashes)?;
213        writes.into_iter().try_for_each(|(pos, hash)| {
214            cs.batch
215                .put::<EventAccumulatorSchema>(&(version, pos), &hash)
216        })?;
217
218        Ok(root_hash)
219    }
220
221    pub(crate) fn put_events_multiple_versions(
222        &self, first_version: u64, event_vecs: &[Vec<ContractEvent>],
223        cs: &mut ChangeSet,
224    ) -> Result<Vec<HashValue>> {
225        event_vecs
226            .iter()
227            .enumerate()
228            .map(|(idx, events)| {
229                let version = first_version
230                    .checked_add(idx as Version)
231                    .ok_or_else(|| format_err!("version overflow"))?;
232                self.put_events(version, events, cs)
233            })
234            .collect::<Result<Vec<_>>>()
235    }

238type Accumulator<'a> =
239    MerkleAccumulator<EventHashReader<'a>, EventAccumulatorHasher>;
240
241struct EventHashReader<'a> {
242    store: &'a EventStore,
243    version: Version,
244}
245
246impl<'a> EventHashReader<'a> {
247    fn new(store: &'a EventStore, version: Version) -> Self {
248        Self { store, version }
249    }
250}
251
252impl<'a> HashReader for EventHashReader<'a> {
253    fn get(&self, position: Position) -> Result<HashValue> {
254        self.store
255            .db
256            .get::<EventAccumulatorSchema>(&(self.version, position))?
257            .ok_or_else(|| {
258                format_err!("Hash at position {:?} not found.", position)
259            })
260    }
261}

263type EmptyAccumulator = MerkleAccumulator<EmptyReader, EventAccumulatorHasher>;

265struct EmptyReader;

267// Asserts `get()` is never called.

272pub struct EventsByVersionIter<'a> {
273    inner: Peekable<SchemaIterator<'a, EventSchema>>,
274    expected_next_version: Version,
275    end_version: Version,
276}
277
278impl<'a> EventsByVersionIter<'a> {
279    fn next_impl(&mut self) -> Result<Option<Vec<ContractEvent>>> {
280        if self.expected_next_version >= self.end_version {
281            return Ok(None);
282        }
283
284        let mut ret = Vec::new();
285        while let Some(res) = self.inner.peek() {
286            let ((version, _index), _event) = res.as_ref().map_err(|e| {
287                format_err!("Hit error iterating events: {}", e)
288            })?;
289            if *version != self.expected_next_version {
290                break;
291            }
292            let ((_version, _index), event) =
293                self.inner.next().transpose()?.expect("Known to exist.");
294            ret.push(event);
295        }
296        self.expected_next_version = self
297            .expected_next_version
298            .checked_add(1)
299            .ok_or_else(|| format_err!("expected version overflowed."))?;
300        Ok(Some(ret))
301    }
302}
303
304impl<'a> Iterator for EventsByVersionIter<'a> {
305    type Item = Result<Vec<ContractEvent>>;
306
307    fn next(&mut self) -> Option<Self::Item> { self.next_impl().transpose() }
308}
