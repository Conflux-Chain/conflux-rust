8//! This file defines transaction store APIs that are related to committed
9//! signed transactions.
10
11use crate::{
12    change_set::ChangeSet,
13    errors::DiemDbError,
14    schema::{
15        transaction::TransactionSchema,
16        transaction_by_account::TransactionByAccountSchema,
17    },
18};
19use anyhow::Result;
20use diem_types::{
21    block_metadata::BlockMetadata,
22    transaction::{Transaction, Version},
23};
24use schemadb::DB;
25use std::sync::Arc;
26
27#[derive(Debug)]
28pub(crate) struct TransactionStore {
29    db: Arc<DB>,
30}
31
32impl TransactionStore {
33    pub fn new(db: Arc<DB>) -> Self { Self { db } }
34
35    /// Get signed transaction given `version`
36    pub fn get_transaction(&self, version: Version) -> Result<Transaction> {
37        self.db.get::<TransactionSchema>(&version)?.ok_or_else(|| {
38            DiemDbError::NotFound(format!("Txn {}", version)).into()
39        })
40    }
41
42    /// Returns the block metadata carried on the block metadata transaction at
43    /// or preceding `version`, together with the version of the block
44    /// metadata transaction. Returns None if there's no such transaction at
45    /// or preceding `version` (it's likely the genesis version 0).
46    pub fn get_block_metadata(
47        &self, version: Version,
48    ) -> Result<Option<(Version, BlockMetadata)>> {
49        // Maximum TPS from benchmark is around 1000.
50        const MAX_VERSIONS_TO_SEARCH: usize = 1000 * 3;
51
52        // Linear search via `DB::rev_iter()` here, NOT expecting performance
53        // hit, due to the fact that the iterator caches data block and
54        // that there are limited number of transactions in each block.
55        let mut iter =
56            self.db.rev_iter::<TransactionSchema>(Default::default())?;
57        iter.seek(&version)?;
58        for res in iter.take(MAX_VERSIONS_TO_SEARCH) {
59            let (v, txn) = res?;
60            if let Transaction::BlockMetadata(block_meta) = txn {
61                return Ok(Some((v, block_meta)));
62            } else if v == 0 {
63                return Ok(None);
64            }
65        }
66
67        Err(DiemDbError::NotFound(format!(
68            "BlockMetadata preceding version {}",
69            version
70        ))
71        .into())
72    }
73
74    /// Save signed transaction at `version`
75    pub fn put_transaction(
76        &self, version: Version, transaction: &Transaction, cs: &mut ChangeSet,
77    ) -> Result<()> {
78        if let Transaction::UserTransaction(txn) = transaction {
79            // TODO(lpl): Find a proper way to keep account-related info.
80            cs.batch.put::<TransactionByAccountSchema>(
81                &(txn.sender(), 0),
82                &version,
83            )?;
84        }
85        cs.batch.put::<TransactionSchema>(&version, &transaction)?;
86
87        Ok(())
88    }
89}
