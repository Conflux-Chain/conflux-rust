40/// The `Pruner` is meant to be part of a `DiemDB` instance and runs in the
41/// background to prune old data.
42///
43/// It creates a worker thread on construction and joins it on destruction. When
44/// destructed, it quits the worker thread eagerly without waiting for all
45/// pending work to be done.
46#[derive(Debug)]
47pub(crate) struct Pruner {
48    /// Other than the latest version, how many historical versions to keep
49    /// being readable. For example, this being 0 means keep only the
50    /// latest version.
51    historical_versions_to_keep: u64,
52    /// The worker thread handle, created upon Pruner instance construction and
53    /// joined upon its destruction. It only becomes `None` after joined in
54    /// `drop()`.
55    worker_thread: Option<JoinHandle<()>>,
56    /// The sender side of the channel talking to the worker thread.
57    command_sender: Mutex<Sender<Command>>,
58    /// (For tests) A way for the worker thread to inform the `Pruner` the
59    /// pruning progress. If it sets this atomic value to `V`, all versions
60    /// before `V` can no longer be accessed.
61    #[allow(dead_code)]
62    worker_progress: Arc<AtomicU64>,
63}
64
65impl Pruner {
66    /// Creates a worker thread that waits on a channel for pruning commands.
67    pub fn new(db: Arc<DB>, historical_versions_to_keep: u64) -> Self {
68        let (command_sender, command_receiver) = channel();
69
70        let worker_progress = Arc::new(AtomicU64::new(0));
71        let worker_progress_clone = Arc::clone(&worker_progress);
72
73        DIEM_STORAGE_PRUNE_WINDOW.set(historical_versions_to_keep as i64);
74        let worker_thread = std::thread::Builder::new()
75            .name("pos_ledger_db_pruner".into())
76            .spawn(move || {
77                Worker::new(db, command_receiver, worker_progress_clone).work()
78            })
79            .expect("Creating pruner thread should succeed.");
80
81        Self {
82            historical_versions_to_keep,
83            worker_thread: Some(worker_thread),
84            command_sender: Mutex::new(command_sender),
85            worker_progress,
86        }
87    }
88
89    /// Sends pruning command to the worker thread when necessary.
90    pub fn wake(&self, latest_version: Version) {
91        if latest_version > self.historical_versions_to_keep {
92            let least_readable_version =
93                latest_version - self.historical_versions_to_keep;
94            self.command_sender
95                .lock()
96                .send(Command::Prune {
97                    least_readable_version,
98                })
99                .expect("Receiver should not destruct prematurely.");
100        }
101    }
102
103    /// (For tests only.) Notifies the worker thread and waits for it to finish
104    /// its job by polling an internal counter.
105    #[cfg(test)]
106    pub fn wake_and_wait(&self, latest_version: Version) -> Result<()> {
107        self.wake(latest_version);
108
109        if latest_version > self.historical_versions_to_keep {
110            let least_readable_version =
111                latest_version - self.historical_versions_to_keep;
112            // Assuming no big pruning chunks will be issued by a test.
113            const TIMEOUT: Duration = Duration::from_secs(10);
114            let end = Instant::now() + TIMEOUT;
115
116            while Instant::now() < end {
117                if self.worker_progress.load(Ordering::Relaxed)
118                    >= least_readable_version
119                {
120                    return Ok(());
121                }
122                sleep(Duration::from_millis(1));
123            }
124            anyhow::bail!("Timeout waiting for pruner worker.");
125        }
126        Ok(())
127    }
128}
129
130impl Drop for Pruner {
131    fn drop(&mut self) {
132        self.command_sender
133            .lock()
134            .send(Command::Quit)
135            .expect("Receiver should not destruct.");
136        self.worker_thread
137            .take()
138            .expect("Worker thread must exist.")
139            .join()
140            .expect("Worker thread should join peacefully.");
141    }
142}
143
144enum Command {
145    Quit,
146    Prune { least_readable_version: Version },
147}
148
149struct Worker {
150    db: Arc<DB>,
151    command_receiver: Receiver<Command>,
152    target_least_readable_version: Version,
153    /// Keeps a record of the pruning progress. If this equals to version `V`,
154    /// we know versions smaller than `V` are no longer readable.
155    /// This being an atomic value is to communicate the info with the Pruner
156    /// thread (for tests).
157    least_readable_version: Arc<AtomicU64>,
158    /// Indicates if there's NOT any pending work to do currently, to hint
159    /// `Self::receive_commands()` to `recv()` blocking-ly.
160    blocking_recv: bool,
161    index_min_nonpurged_version: Version,
162    index_purged_at: Instant,
163}
164
165impl Worker {
166    const MAX_VERSIONS_TO_PRUNE_PER_BATCH: usize = 100;
167
168    fn new(
169        db: Arc<DB>, command_receiver: Receiver<Command>,
170        least_readable_version: Arc<AtomicU64>,
171    ) -> Self {
172        Self {
173            db,
174            command_receiver,
175            least_readable_version,
176            target_least_readable_version: 0,
177            blocking_recv: true,
178            index_min_nonpurged_version: 0,
179            index_purged_at: Instant::now(),
180        }
181    }
182
183    fn work(mut self) {
184        self.initialize();
185
186        while self.receive_commands() {
187            // Process a reasonably small batch of work before trying to receive
188            // commands again, in case `Command::Quit` is received
189            // (that's when we should quit.)
190            match prune_state(
191                Arc::clone(&self.db),
192                self.least_readable_version.load(Ordering::Relaxed),
193                self.target_least_readable_version,
194                Self::MAX_VERSIONS_TO_PRUNE_PER_BATCH,
195            ) {
196                Ok(least_readable_version) => {
197                    self.record_progress(least_readable_version);
198
199                    // Make next recv() blocking if all done.
200                    self.blocking_recv = least_readable_version
201                        == self.target_least_readable_version;
202
203                    // Try to purge the log.
204                    if let Err(e) = self.maybe_purge_index() {
205                        diem_warn!(
206                            error = ?e,
207                            "Failed purging state node index, ignored.",
208                        );
209                    }
210                }
211                Err(e) => {
212                    diem_error!(
213                        error = ?e,
214                        "Error pruning stale state nodes.",
215                    );
216                    // On error, stop retrying vigorously by making next recv()
217                    // blocking.
218                    self.blocking_recv = true;
219                }
220            }
221        }
222    }
223
224    /// Find out the first undeleted item in the stale node index.
225    ///
226    /// Seeking from the beginning (version 0) is potentially costly, we do it
227    /// once upon worker thread start, record the progress and seek from
228    /// that position afterwards.
229    fn initialize(&mut self) {
230        loop {
231            match self.get_least_readable_version() {
232                Ok(least_readable_version) => {
233                    diem_info!(
234                        least_readable_version = least_readable_version,
235                        "[state pruner worker] initialized."
236                    );
237                    self.target_least_readable_version = least_readable_version;
238                    self.record_progress(least_readable_version);
239                    return;
240                }
241                Err(e) => {
242                    diem_error!(
243                        error = ?e,
244                        "[state pruner worker] Error on first seek. Retrying in 1 second.",
245                    );
246                    sleep(Duration::from_secs(1));
247                }
248            }
249        }
250    }
251
252    fn get_least_readable_version(&self) -> Result<Version> {
253        let mut iter = self
254            .db
255            .iter::<StaleNodeIndexSchema>(ReadOptions::default())?;
256        iter.seek_to_first();
257        Ok(iter.next().transpose()?.map_or(0, |(index, _)| {
258            index
259                .stale_since_version
260                .checked_sub(1)
261                .expect("Nothing is stale since version 0.")
262        }))
263    }
264
265    /// Log the progress.
266    fn record_progress(&mut self, least_readable_version: Version) {
267        self.least_readable_version
268            .store(least_readable_version, Ordering::Relaxed);
269        DIEM_STORAGE_PRUNER_LEAST_READABLE_STATE_VERSION
270            .set(least_readable_version as i64);
271    }
272
273    /// Tries to receive all pending commands, blocking waits for the next
274    /// command if no work needs to be done, otherwise quits with `true` to
275    /// allow the outer loop to do some work before getting back here.
276    ///
277    /// Returns `false` if `Command::Quit` is received, to break the outer loop
278    /// and let `work_loop()` return.
279    fn receive_commands(&mut self) -> bool {
280        loop {
281            let command = if self.blocking_recv {
282                // Worker has nothing to do, blocking wait for the next command.
283                self.command_receiver
284                    .recv()
285                    .expect("Sender should not destruct prematurely.")
286            } else {
287                // Worker has pending work to do, non-blocking recv.
288                match self.command_receiver.try_recv() {
289                    Ok(command) => command,
290                    // Channel has drained, yield control to the outer loop.
291                    Err(_) => return true,
292                }
293            };
294
295            match command {
296                // On `Command::Quit` inform the outer loop to quit by returning
297                // `false`.
298                Command::Quit => return false,
299                Command::Prune {
300                    least_readable_version,
301                } => {
302                    if least_readable_version
303                        > self.target_least_readable_version
304                    {
305                        self.target_least_readable_version =
306                            least_readable_version;
307                        // Switch to non-blocking to allow some work to be done
308                        // after the channel has
309                        // drained.
310                        self.blocking_recv = false;
311                    }
312                }
313            }
314        }
315    }
316
317    /// Purge the stale node index so that after restart not too much already
318    /// pruned stuff is dealt with again (although no harm is done deleting
319    /// those then non-existent things.)
320    ///
321    /// We issue (range) deletes on the index only periodically instead of after
322    /// every pruning batch to avoid sending too many deletions to the DB,
323    /// which takes disk space and slows it down.
324    fn maybe_purge_index(&mut self) -> Result<()> {
325        const MIN_INTERVAL: Duration = Duration::from_secs(60);
326        const MIN_VERSIONS: u64 = 60000;
327
328        // A deletion is issued at most once in one minute and when the pruner
329        // has progressed by at least 60000 versions (assuming the
330        // pruner deletes as slow as 1000 versions per second,
331        // this imposes at most one minute of work in vain after restarting.)
332        let now = Instant::now();
333        if now - self.index_purged_at > MIN_INTERVAL {
334            let least_readable_version =
335                self.least_readable_version.load(Ordering::Relaxed);
336
337            if least_readable_version - self.index_min_nonpurged_version + 1
338                > MIN_VERSIONS
339            {
340                let new_min_non_purged_version = least_readable_version + 1;
341                self.db.range_delete::<StaleNodeIndexSchema, Version>(
342                    &self.index_min_nonpurged_version,
343                    &new_min_non_purged_version, // end is exclusive
344                )?;
345                self.index_min_nonpurged_version = new_min_non_purged_version;
346                self.index_purged_at = now;
347            }
348        }
349
350        Ok(())
351    }
352}

354struct StaleNodeIndicesByVersionIterator<'a> {
355    inner: Peekable<SchemaIterator<'a, StaleNodeIndexSchema>>,
356    target_least_readable_version: Version,
357}
358
359impl<'a> StaleNodeIndicesByVersionIterator<'a> {
360    fn new(
361        db: &'a DB, least_readable_version: Version,
362        target_least_readable_version: Version,
363    ) -> Result<Self> {
364        let mut iter =
365            db.iter::<StaleNodeIndexSchema>(ReadOptions::default())?;
366        iter.seek(&least_readable_version)?;
367
368        Ok(Self {
369            inner: iter.peekable(),
370            target_least_readable_version,
371        })
372    }
373
374    fn next_result(&mut self) -> Result<Option<Vec<StaleNodeIndex>>> {
375        match self.inner.next().transpose()? {
376            None => Ok(None),
377            Some((index, _)) => {
378                let version = index.stale_since_version;
379                if version > self.target_least_readable_version {
380                    return Ok(None);
381                }
382
383                let mut indices = vec![index];
384                while let Some(res) = self.inner.peek() {
385                    if let Ok((index_ref, _)) = res {
386                        if index_ref.stale_since_version != version {
387                            break;
388                        }
389                    }
390
391                    let (index, _) = self
392                        .inner
393                        .next()
394                        .transpose()?
395                        .expect("Should be Some.");
396                    indices.push(index);
397                }
398
399                Ok(Some(indices))
400            }
401        }
402    }
403}
404
405impl<'a> Iterator for StaleNodeIndicesByVersionIterator<'a> {
406    type Item = Result<Vec<StaleNodeIndex>>;
407
408    fn next(&mut self) -> Option<Self::Item> { self.next_result().transpose() }
409}
410
411pub fn prune_state(
412    db: Arc<DB>, least_readable_version: Version,
413    target_least_readable_version: Version, max_versions: usize,
414) -> Result<Version> {
415    let indices = StaleNodeIndicesByVersionIterator::new(
416        &db,
417        least_readable_version,
418        target_least_readable_version,
419    )?
420    .take(max_versions) // Iterator<Item = Result<Vec<StaleNodeIndex>>>
421    .collect::<Result<Vec<_>>>()? // now Vec<Vec<StaleNodeIndex>>
422    .into_iter()
423    .flatten()
424    .collect::<Vec<_>>();
425
426    if indices.is_empty() {
427        Ok(least_readable_version)
428    } else {
429        let _timer = DIEM_STORAGE_OTHER_TIMERS_SECONDS
430            .with_label_values(&["pruner_commit"])
431            .start_timer();
432        let new_least_readable_version =
433            indices.last().expect("Should exist.").stale_since_version;
434        let mut batch = SchemaBatch::new();
435        indices.into_iter().try_for_each(|index| {
436            batch.delete::<JellyfishMerkleNodeSchema>(&index.node_key)
437        })?;
438        db.write_schemas(batch, false)?;
439        Ok(new_least_readable_version)
440    }
