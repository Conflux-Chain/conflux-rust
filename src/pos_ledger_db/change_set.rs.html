8use crate::ledger_counters::LedgerCounterBumps;
9use diem_types::transaction::Version;
10use schemadb::SchemaBatch;
11use std::collections::HashMap;
12
13/// Structure that collects changes to be made to the DB in one transaction.
14///
15/// To be specific it carries a batch of db alternations and counter increases
16/// that'll be converted to DB alternations on "sealing". This is required to be
17/// converted to `SealedChangeSet` before committing to the DB.
18pub(crate) struct ChangeSet {
19    /// A batch of db alternations.
20    pub batch: SchemaBatch,
21    /// Counter bumps to be made on commit.
22    counter_bumps: HashMap<Version, LedgerCounterBumps>,
23}
24
25impl ChangeSet {
26    /// Constructor.
27    pub fn new() -> Self {
28        Self {
29            batch: SchemaBatch::new(),
30            counter_bumps: HashMap::new(),
31        }
32    }
33
34    pub fn counter_bumps(
35        &mut self, version: Version,
36    ) -> &mut LedgerCounterBumps {
37        self.counter_bumps
38            .entry(version)
39            .or_insert_with(LedgerCounterBumps::new)
40    }
41
42    #[cfg(test)]
43    pub fn new_with_bumps(
44        counter_bumps: HashMap<Version, LedgerCounterBumps>,
45    ) -> Self {
46        Self {
47            batch: SchemaBatch::new(),
48            counter_bumps,
49        }
50    }
51}
52
53/// ChangeSet that's ready to be committed to the DB.
54///
55/// This is a wrapper type just to make sure `ChangeSet` to be committed is
56/// sealed properly.
57pub(crate) struct SealedChangeSet {
58    /// A batch of db alternations.
59    pub batch: SchemaBatch,
60}
