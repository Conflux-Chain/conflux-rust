5use crate::RpcAddress;
6use cfx_addr::Network;
7use cfx_parameters::staking::DRIPS_PER_STORAGE_COLLATERAL_UNIT;
8use cfx_types::U256;
9use primitives::SponsorInfo as PrimitiveSponsorInfo;
10use serde::{Deserialize, Serialize};
11
12#[derive(Debug, Serialize, Deserialize, Clone)]
13#[serde(rename_all = "camelCase")]
14pub struct SponsorInfo {
15    /// This is the address of the sponsor for gas cost of the contract.
16    pub sponsor_for_gas: RpcAddress,
17    /// This is the address of the sponsor for collateral of the contract.
18    pub sponsor_for_collateral: RpcAddress,
19    /// This is the upper bound of sponsor gas cost per tx.
20    pub sponsor_gas_bound: U256,
21    /// This is the amount of tokens sponsor for gas cost to the contract.
22    pub sponsor_balance_for_gas: U256,
23    /// This is the amount of tokens sponsor for collateral to the contract.
24    pub sponsor_balance_for_collateral: U256,
25    /// This is the amount of unused storage points (in terms of bytes).
26    pub available_storage_points: U256,
27    /// This is the amount of used storage points (in terms of bytes).
28    pub used_storage_points: U256,
29}
30
31impl SponsorInfo {
32    pub fn default(network: Network) -> Result<Self, String> {
33        Ok(Self {
34            sponsor_for_gas: RpcAddress::null(network)?,
35            sponsor_for_collateral: RpcAddress::null(network)?,
36            sponsor_gas_bound: Default::default(),
37            sponsor_balance_for_gas: Default::default(),
38            sponsor_balance_for_collateral: Default::default(),
39            available_storage_points: Default::default(),
40            used_storage_points: Default::default(),
41        })
42    }
43
44    pub fn try_from(
45        sponsor_info: PrimitiveSponsorInfo, network: Network,
46    ) -> Result<Self, String> {
47        Ok(Self {
48            sponsor_for_gas: RpcAddress::try_from_h160(
49                sponsor_info.sponsor_for_gas,
50                network,
51            )?,
52            sponsor_for_collateral: RpcAddress::try_from_h160(
53                sponsor_info.sponsor_for_collateral,
54                network,
55            )?,
56            sponsor_gas_bound: sponsor_info.sponsor_gas_bound,
57            sponsor_balance_for_gas: sponsor_info.sponsor_balance_for_gas,
58            sponsor_balance_for_collateral: sponsor_info
59                .sponsor_balance_for_collateral,
60            available_storage_points: sponsor_info
61                .storage_points
62                .as_ref()
63                .map_or(U256::zero(), |x| {
64                    x.unused / *DRIPS_PER_STORAGE_COLLATERAL_UNIT
65                }),
66            used_storage_points: sponsor_info
67                .storage_points
68                .as_ref()
69                .map_or(U256::zero(), |x| {
70                    x.used / *DRIPS_PER_STORAGE_COLLATERAL_UNIT
71                }),
72        })
73    }
74}
