34pub trait MptValueKind: Debug {
35    fn value_eq(&self, maybe_value: Option<&[u8]>) -> bool;
36}
37
38impl MptValueKind for () {
39    fn value_eq(&self, maybe_value: Option<&[u8]>) -> bool {
40        maybe_value.is_none()
41    }
42}
43
44impl MptValueKind for Box<[u8]> {
45    fn value_eq(&self, maybe_value: Option<&[u8]>) -> bool {
46        maybe_value.map_or(false, |v| v.eq(&**self))
47    }
48}

50pub fn check_key_value_load<Value: MptValueKind>(
51    snapshot_db: &SnapshotDbSqlite,
52    mut kv_iter: impl FallibleIterator<Item = (Vec<u8>, Value), Error = Error>,
53    check_value: bool,
54) -> Result<u64> {
55    let mut checker_count = 0;
56    let mut mpt = snapshot_db.open_snapshot_mpt_shared()?;
57
58    let mut cursor = MptCursor::<
59        &mut dyn SnapshotMptTraitRead,
60        BasicPathNode<&mut dyn SnapshotMptTraitRead>,
61    >::new(&mut mpt);
62    cursor.load_root()?;
63    while let Some((access_key, expected_value)) = kv_iter.next()? {
64        let terminal =
65            cursor.open_path_for_key::<access_mode::Read>(&access_key)?;
66        if check_value {
67            let mpt_value = match terminal {
68                CursorOpenPathTerminal::Arrived => {
69                    cursor.current_node_mut().value_as_slice().into_option()
70                }
71                CursorOpenPathTerminal::ChildNotFound { .. } => None,
72                CursorOpenPathTerminal::PathDiverted(_) => None,
73            };
74            if !expected_value.value_eq(mpt_value) {
75                error!(
76                    "mpt value doesn't match snapshot kv. Expected {:?}, got {:?}",
77                    expected_value, mpt_value,
78                );
79            }
80        }
81        checker_count += 1;
82    }
83    cursor.finish()?;
84
85    Ok(checker_count)
86}
