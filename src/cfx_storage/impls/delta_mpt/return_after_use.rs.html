5use std::mem::swap;
6
7/// While mutable references can only be passed around as method parameters,
8/// ReturnAfterUse can pass mutable object in classes.
9pub struct ReturnAfterUse<'a, T: 'a> {
10    origin: Option<&'a mut Option<T>>,
11    current: Option<T>,
12}
13
14impl<'a, T: Default> Default for ReturnAfterUse<'a, T> {
15    fn default() -> Self {
16        Self {
17            origin: None,
18            current: Some(T::default()),
19        }
20    }
21}
22
23impl<'a, T> Drop for ReturnAfterUse<'a, T> {
24    fn drop(&mut self) {
25        match &mut self.origin {
26            Some(origin_mut) => swap(*origin_mut, &mut self.current),
27            None => {}
28        }
29    }
30}
31
32impl<'a, T> ReturnAfterUse<'a, T> {
33    pub fn new_from_value(val: T) -> Self {
34        Self {
35            origin: None,
36            current: Some(val),
37        }
38    }
39
40    pub fn new(option: &'a mut Option<T>) -> Self {
41        let mut ret = Self {
42            origin: None,
43            current: None,
44        };
45        swap(&mut ret.current, option);
46        ret.origin = Some(option);
47
48        ret
49    }
50
51    pub fn new_from_origin<'b: 'a>(
52        origin: &'a mut ReturnAfterUse<'b, T>,
53    ) -> ReturnAfterUse<'a, T> {
54        Self::new(&mut origin.current)
55    }
56
57    pub fn get_ref(&self) -> &T { return self.current.as_ref().unwrap(); }
58
59    pub fn get_mut(&mut self) -> &mut T {
60        return self.current.as_mut().unwrap();
61    }
62}
