11/// Key length should be multiple of 8.
12// TODO(yz): align key @8B with mask.
13pub type KeyPart<'a> = &'a [u8];
14
15pub enum WalkStop<'key, ChildIdType> {
16    // Path matching fails on the compressed path.
17    //
18    // if put, a new node should be created to replace the current node from
19    // parent children table; modify this node or create a new node to
20    // insert as children of new node, (update path) then the child that
21    // should be followed is nil at the new node.
22    //
23    // if get / delete (not found)
24    PathDiverted {
25        /// Key may terminate on the path.
26        key_child_index: Option<u8>,
27        key_remaining: CompressedPathRef<'key>,
28        matched_path: CompressedPathRaw,
29        unmatched_child_index: u8,
30        unmatched_path_remaining: CompressedPathRaw,
31    },
32
33    // If exactly at this node.
34    // if put, update this node
35    // if delete, may cause deletion / path compression (delete this node,
36    // parent update child, update path of original child node)
37    Arrived,
38
39    Descent {
40        key_remaining: KeyPart<'key>,
41        child_index: u8,
42        child_node: ChildIdType,
43    },
44
45    // To descent, however child doesn't exists:
46    // to modify this node or create a new node to replace this node (update
47    // child) Then create a new node for remaining key_part. if put single
48    // version, this node changes, parent update merkle.
49    //
50    // if get / delete (not found)
51    ChildNotFound {
52        key_remaining: CompressedPathRef<'key>,
53        child_index: u8,
54    },
55}
56
57impl<'key, ChildIdType> WalkStop<'key, ChildIdType> {
58    pub fn path_diverted_uninitialized() -> Self {
59        WalkStop::PathDiverted {
60            key_child_index: None,
61            key_remaining: Default::default(),
62            matched_path: Default::default(),
63            unmatched_child_index: 0,
64            unmatched_path_remaining: Default::default(),
65        }
66    }
67}
68
69pub trait GetChildTrait<'node> {
70    type ChildIdType: 'node;
71
72    fn get_child(&'node self, child_index: u8) -> Option<Self::ChildIdType>;
73}
74
75pub trait TrieNodeWalkTrait<'node>:
76    TrieNodeTrait + GetChildTrait<'node>
77{
78    fn walk<'key, AM: access_mode::AccessMode>(
79        &'node self, key: KeyPart<'key>,
80    ) -> WalkStop<'key, Self::ChildIdType> {
81        walk::<AM, _>(key, &self.compressed_path_ref(), self)
82    }
83}
84
85/// Traverse.
86///
87/// When a trie node start with the second nibble, the trie node has a
88/// compressed path of step 1. The nibble in the compressed path is
89/// the same as its child index.
90///
91/// The start of key is always aligned with compressed path of
92/// current node, e.g. if compressed path starts at the second-half, so
93/// should be key.
94pub(super) fn walk<
95    'key,
96    'node,
97    AM: access_mode::AccessMode,
98    Node: GetChildTrait<'node>,
99>(
100    key: KeyPart<'key>, path: &dyn CompressedPathTrait, node: &'node Node,
101) -> WalkStop<'key, Node::ChildIdType> {
102    let path_slice = path.path_slice();
103    let path_mask = path.path_mask();
104    let matched_path_begin_mask = CompressedPathRaw::second_nibble(path_mask);
105    let mut unmatched_path_mask =
106        CompressedPathRaw::clear_second_nibble(path_mask);
107
108    // Compare bytes till the last full byte. The first byte is always
109    // included because even if it's the second-half, it must be
110    // already matched before entering this TrieNode.
111    let memcmp_len = min(
112        path_slice.len()
113            - (CompressedPathRaw::no_second_nibble(path_mask) as usize),
114        key.len(),
115    );
116
117    for i in 0..memcmp_len {
118        if path_slice[i] != key[i] {
119            if AM::READ_ONLY {
120                return WalkStop::path_diverted_uninitialized();
121            } else {
122                let matched_path: CompressedPathRaw;
123                let key_child_index: u8;
124                let key_remaining;
125                let unmatched_child_index: u8;
126                let unmatched_path_remaining: &[u8];
127
128                if CompressedPathRaw::first_nibble(path_slice[i] ^ key[i]) == 0
129                {
130                    // "First half" matched
131                    matched_path = CompressedPathRaw::new_and_apply_mask(
132                        &path_slice[0..i + 1],
133                        matched_path_begin_mask
134                            | CompressedPathRaw::second_nibble_mask(),
135                    );
136
137                    key_child_index = CompressedPathRaw::second_nibble(key[i]);
138                    key_remaining = CompressedPathRef::new(
139                        &key[i + 1..],
140                        CompressedPathRaw::NO_MISSING_NIBBLE,
141                    );
142                    unmatched_child_index =
143                        CompressedPathRaw::second_nibble(path_slice[i]);
144                    unmatched_path_remaining = &path_slice[i + 1..];
145                } else {
146                    matched_path = CompressedPathRaw::new(
147                        &path_slice[0..i],
148                        matched_path_begin_mask,
149                    );
150                    key_child_index = CompressedPathRaw::first_nibble(key[i]);
151                    key_remaining = CompressedPathRef::new(
152                        &key[i..],
153                        CompressedPathRaw::first_nibble_mask(),
154                    );
155                    unmatched_path_mask |=
156                        CompressedPathRaw::first_nibble_mask();
157                    unmatched_child_index =
158                        CompressedPathRaw::first_nibble(path_slice[i]);
159                    unmatched_path_remaining = &path_slice[i..];
160                }
161                return WalkStop::PathDiverted {
162                    key_child_index: Some(key_child_index),
163                    key_remaining: key_remaining.into(),
164                    matched_path,
165                    unmatched_child_index,
166                    unmatched_path_remaining: CompressedPathRaw::new(
167                        unmatched_path_remaining,
168                        unmatched_path_mask,
169                    ),
170                };
171            }
172        }
173    }
174    // Key is fully consumed, get value attached.
175    if key.len() == memcmp_len {
176        // Compressed path isn't fully consumed.
177        if path_slice.len() > memcmp_len {
178            if AM::READ_ONLY {
179                return WalkStop::path_diverted_uninitialized();
180            } else {
181                return WalkStop::PathDiverted {
182                    // key_remaining is empty, and key_child_index doesn't
183                    // make sense, but we need to mark it.
184                    key_remaining: Default::default(),
185                    key_child_index: None,
186                    matched_path: CompressedPathRaw::new(
187                        &path_slice[0..memcmp_len],
188                        matched_path_begin_mask,
189                    ),
190                    unmatched_child_index: CompressedPathRaw::first_nibble(
191                        path_slice[memcmp_len],
192                    ),
193                    unmatched_path_remaining: CompressedPathRaw::new(
194                        &path_slice[memcmp_len..],
195                        unmatched_path_mask
196                            | CompressedPathRaw::first_nibble_mask(),
197                    ),
198                };
199            }
200        } else {
201            return WalkStop::Arrived;
202        }
203    } else {
204        // Key is not fully consumed.
205
206        // When path is fully consumed, check if child exists under child_index.
207        let child_index;
208        let key_remaining;
209
210        if path_slice.len() == memcmp_len {
211            // Compressed path is fully consumed. Descend into one child.
212            child_index = CompressedPathRaw::first_nibble(key[memcmp_len]);
213            key_remaining = CompressedPathRef::new(
214                &key[memcmp_len..],
215                CompressedPathRaw::first_nibble_mask(),
216            );
217        } else {
218            // One half byte remaining to match with path. Consume it in the
219            // key.
220            if CompressedPathRaw::first_nibble(
221                path_slice[memcmp_len] ^ key[memcmp_len],
222            ) != 0
223            {
224                // Mismatch.
225                if AM::READ_ONLY {
226                    return WalkStop::path_diverted_uninitialized();
227                } else {
228                    return WalkStop::PathDiverted {
229                        key_child_index: Some(CompressedPathRaw::first_nibble(
230                            key[memcmp_len],
231                        )),
232                        key_remaining: CompressedPathRef::new(
233                            &key[memcmp_len..],
234                            CompressedPathRaw::first_nibble_mask(),
235                        ),
236                        matched_path: CompressedPathRaw::new(
237                            &path_slice[0..memcmp_len],
238                            matched_path_begin_mask,
239                        ),
240                        unmatched_child_index: CompressedPathRaw::first_nibble(
241                            path_slice[memcmp_len],
242                        ),
243                        unmatched_path_remaining: CompressedPathRaw::new(
244                            &path_slice[memcmp_len..],
245                            unmatched_path_mask
246                                | CompressedPathRaw::first_nibble_mask(),
247                        ),
248                    };
249                }
250            } else {
251                child_index = CompressedPathRaw::second_nibble(key[memcmp_len]);
252                key_remaining = CompressedPathRef::new(
253                    &key[memcmp_len + 1..],
254                    CompressedPathRaw::NO_MISSING_NIBBLE,
255                );
256            }
257        }
258
259        match node.get_child(child_index) {
260            Option::None => {
261                return WalkStop::ChildNotFound {
262                    key_remaining,
263                    child_index,
264                };
265            }
266            Option::Some(child_node) => {
267                return WalkStop::Descent {
268                    key_remaining: key_remaining.path_slice,
269                    child_index,
270                    child_node,
271                };
272            }
273        }
274    }
275}
