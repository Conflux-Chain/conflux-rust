5/// Merge an MPT with sorted deletion and insertion stream.
6/// The merge can be in-place or writing to a new MPT (save-as mode).
7///
8/// In the merging process, the merger keeps a path from root to a node. When
9/// the next key to process from insertion or deletion stream lies outside the
10/// subtree of the last node in the path, the last node is closed. New nodes are
11/// open if the key goes further down extending the path. In in-place mode, node
12/// deletions are executed and node writes happen when a modified node is
13/// closed. In save-as mode, node deletion is no-op, and node writes happen
14/// when a node is closed, and when opening the new node. All modified
15/// node and skipped subtree are saved into the new MPT.
16///
17/// In a save-as copy, the base_mpt could be implemented as an iterator of the
18/// original MPT because the original MPT is exactly visited in string order
19/// by the path_db_key. (See definition of path_db_key below.)
20// TODO(yz): In future merge can be made into multiple threads easily by merging
21// different children in parallel then combine the root node.
22pub struct MptMerger<'a> {
23    rw_cursor: MptCursorRw<
24        MergeMptsInRequest<'a>,
25        ReadWritePathNode<MergeMptsInRequest<'a>>,
26    >,
27}
28
29impl<'a> MptMerger<'a> {
30    pub fn new(
31        maybe_readonly_mpt: Option<&'a mut dyn SnapshotMptTraitReadAndIterate>,
32        out_mpt: &'a mut dyn SnapshotMptTraitRw,
33    ) -> Self {
34        Self {
35            rw_cursor: MptCursorRw::new(MergeMptsInRequest {
36                maybe_readonly_mpt,
37                out_mpt,
38            }),
39        }
40    }
41
42    pub fn merge(
43        &mut self, inserter: &DumpedMptKvIterator,
44    ) -> Result<MerkleHash> {
45        self.rw_cursor.load_root(false)?;
46
47        struct Merger<'x, 'a: 'x> {
48            merger: &'x mut MptMerger<'a>,
49        }
50
51        impl<'x, 'a: 'x> Merger<'x, 'a> {
52            fn merger_mut(&mut self) -> &mut MptMerger<'a> { self.merger }
53        }
54
55        impl<'x, 'a: 'x> KVInserter<MptKeyValue> for Merger<'x, 'a> {
56            fn push(&mut self, v: MptKeyValue) -> Result<()> {
57                let (key, value) = v;
58                if value.len() > 0 {
59                    self.merger_mut().rw_cursor.insert(&key, value)?;
60                } else {
61                    self.merger_mut().rw_cursor.delete(&key)?;
62                }
63                Ok(())
64            }
65        }
66
67        inserter.iterate(&mut Merger { merger: self })?;
68
69        self.rw_cursor.finish()
70    }
71
72    /// The iterators operate on key, value store.
73    pub fn merge_insertion_deletion_separated<'k>(
74        &mut self,
75        mut delete_keys_iter: impl FallibleIterator<
76            Item = (Vec<u8>, ()),
77            Error = Error,
78        >,
79        mut set_keys_iter: impl FallibleIterator<Item = MptKeyValue, Error = Error>,
80        in_reconstruct_snapshot_state: bool,
81    ) -> Result<MerkleHash> {
82        self.rw_cursor.load_root(in_reconstruct_snapshot_state)?;
83
84        let mut key_to_delete = delete_keys_iter.next()?;
85        let mut key_value_to_set = set_keys_iter.next()?;
86
87        loop {
88            if key_to_delete.is_none() {
89                if key_value_to_set.is_some() {
90                    let (key, value) = key_value_to_set.unwrap();
91                    self.rw_cursor.insert(&key, value)?;
92                    while let Some((key, value)) = set_keys_iter.next()? {
93                        self.rw_cursor.insert(&key, value)?;
94                    }
95                }
96                break;
97            };
98
99            if key_value_to_set.is_none() {
100                if key_to_delete.is_some() {
101                    self.rw_cursor
102                        .delete(&key_to_delete.as_ref().unwrap().0)?;
103                    while let Some((key, _)) = delete_keys_iter.next()? {
104                        self.rw_cursor.delete(&key)?;
105                    }
106                }
107                break;
108            }
109
110            // In a diff, if there is a deletion of the same key of a insertion,
111            // delete only happens before the insertion because the inserted key
112            // value must present in the final merged result for it to be in the
113            // diff.
114            let key_delete = &key_to_delete.as_ref().unwrap().0;
115            if key_delete <= &key_value_to_set.as_ref().unwrap().0 {
116                self.rw_cursor.delete(key_delete)?;
117                key_to_delete = delete_keys_iter.next()?;
118            } else {
119                let (key_insert, value) = key_value_to_set.take().unwrap();
120                self.rw_cursor.insert(&key_insert, value)?;
121                key_value_to_set = set_keys_iter.next()?;
122            }
123        }
124
125        self.rw_cursor.finish()
126    }
127}
128
129struct MergeMptsInRequest<'a> {
130    maybe_readonly_mpt: Option<&'a mut dyn SnapshotMptTraitReadAndIterate>,
131    out_mpt: &'a mut dyn SnapshotMptTraitRw,
132}
133
134impl GetReadMpt for MergeMptsInRequest<'_> {
135    fn get_merkle_root(&self) -> MerkleHash {
136        if self.maybe_readonly_mpt.is_some() {
137            self.maybe_readonly_mpt.as_ref().unwrap().get_merkle_root()
138        } else {
139            self.out_mpt.get_merkle_root()
140        }
141    }
142
143    fn get_read_mpt(&mut self) -> &mut dyn SnapshotMptTraitRead {
144        if self.maybe_readonly_mpt.is_some() {
145            self.maybe_readonly_mpt.as_mut().unwrap().as_readonly()
146        } else {
147            self.out_mpt.as_readonly()
148        }
149    }
150}
151
152impl GetRwMpt for MergeMptsInRequest<'_> {
153    fn get_write_mpt(&mut self) -> &mut dyn SnapshotMptTraitRw { self.out_mpt }
154
155    fn get_write_and_read_mpt(
156        &mut self,
157    ) -> (
158        &mut dyn SnapshotMptTraitRw,
159        Option<&mut dyn SnapshotMptTraitReadAndIterate>,
160    ) {
161        (
162            self.out_mpt,
163            // Can't use map() here because it would be a compilation error.
164            match &mut self.maybe_readonly_mpt {
165                None => None,
166                Some(x) => Some(*x),
167            },
168        )
169    }
170
171    fn is_save_as_write(&self) -> bool { self.maybe_readonly_mpt.is_some() }
172
173    fn is_in_place_update(&self) -> bool { self.maybe_readonly_mpt.is_none() }
174}
