25fn encode_metrics(
26    encoder: impl Encoder, whitelist: &'static [&'static str],
27) -> Vec<u8> {
28    let mut metric_families = gather_metrics();
29    if !whitelist.is_empty() {
30        metric_families = whitelist_metrics(metric_families, whitelist);
31    }
32    let mut buffer = vec![];
33    encoder.encode(&metric_families, &mut buffer).unwrap();
34
35    NUM_METRICS
36        .with_label_values(&["total_bytes"])
37        .inc_by(buffer.len() as u64);
38    buffer
39}

41// filtering metrics from the prometheus collections
42// only return the whitelisted metrics
43fn whitelist_metrics(
44    metric_families: Vec<MetricFamily>, whitelist: &'static [&'static str],
45) -> Vec<MetricFamily> {
46    let mut whitelist_metrics = Vec::new();
47    for mf in metric_families {
48        let name = mf.get_name();
49        if whitelist.contains(&name) {
50            whitelist_metrics.push(mf.clone());
51        }
52    }
53    whitelist_metrics
54}
55
56// filtering metrics from the Json format metrics
57// only return the whitelisted metrics
58fn whitelist_json_metrics(
59    json_metrics: HashMap<String, String>, whitelist: &'static [&'static str],
60) -> HashMap<&'static str, String> {
61    let mut whitelist_metrics: HashMap<&'static str, String> = HashMap::new();
62    for key in whitelist {
63        if let Some(metric) = json_metrics.get(*key) {
64            whitelist_metrics.insert(key, metric.clone());
65        }
66    }
67    whitelist_metrics
68}

70async fn serve_metrics(
71    req: Request<Body>,
72) -> Result<Response<Body>, hyper::Error> {
73    let mut resp = Response::new(Body::empty());
74    match (req.method(), req.uri().path()) {
75        (&Method::GET, "/-/healthy") => {
76            *resp.body_mut() = Body::from("diem-node:ok");
77        }
78        (&Method::GET, "/metrics") => {
79            //Prometheus server expects metrics to be on host:port/metrics
80            let encoder = TextEncoder::new();
81            let buffer = encode_metrics(encoder, &[]);
82            *resp.body_mut() = Body::from(buffer);
83        }
84        // expose non-numeric metrics to host:port/json_metrics
85        (&Method::GET, "/json_metrics") => {
86            let json_metrics = get_json_metrics();
87            let encoded_metrics = serde_json::to_string(&json_metrics).unwrap();
88            *resp.body_mut() = Body::from(encoded_metrics);
89        }
90        (&Method::GET, "/counters") => {
91            // Json encoded diem_metrics;
92            let encoder = JsonEncoder;
93            let buffer = encode_metrics(encoder, &[]);
94            *resp.body_mut() = Body::from(buffer);
95        }
96        _ => {
97            *resp.status_mut() = StatusCode::NOT_FOUND;
98        }
99    };
100
101    Ok(resp)
102}

104async fn serve_public_metrics(
105    req: Request<Body>,
106) -> Result<Response<Body>, hyper::Error> {
107    let mut resp = Response::new(Body::empty());
108    match (req.method(), req.uri().path()) {
109        (&Method::GET, "/metrics") => {
110            let encoder = TextEncoder::new();
111            // encode public metrics defined in
112            // common/metrics/src/public_metrics.rs
113            let buffer = encode_metrics(encoder, PUBLIC_METRICS);
114            *resp.body_mut() = Body::from(buffer);
115        }
116        (&Method::GET, "/json_metrics") => {
117            let json_metrics = get_json_metrics();
118            let whitelist_json_metrics =
119                whitelist_json_metrics(json_metrics, PUBLIC_METRICS);
120            let encoded_metrics =
121                serde_json::to_string(&whitelist_json_metrics).unwrap();
122            *resp.body_mut() = Body::from(encoded_metrics);
123        }
124        _ => {
125            *resp.status_mut() = StatusCode::NOT_FOUND;
126        }
127    };
128
129    Ok(resp)
130}

132pub fn start_server(host: String, port: u16, public_metric: bool) {
133    // Only called from places that guarantee that host is parsable, but this
134    // must be assumed.
135    let addr: SocketAddr = (host.as_str(), port)
136        .to_socket_addrs()
137        .unwrap_or_else(|_| {
138            unreachable!("Failed to parse {}:{} as address", host, port)
139        })
140        .next()
141        .unwrap();
142
143    if public_metric {
144        thread::spawn(move || {
145            let make_service = make_service_fn(|_| {
146                future::ok::<_, hyper::Error>(service_fn(serve_public_metrics))
147            });
148
149            let rt = runtime::Builder::new_current_thread()
150                .enable_io()
151                .build()
152                .unwrap();
153            rt.block_on(async {
154                let server = Server::bind(&addr).serve(make_service);
155                server.await
156            })
157            .unwrap();
158        });
159    } else {
160        thread::spawn(move || {
161            let make_service = make_service_fn(|_| {
162                future::ok::<_, hyper::Error>(service_fn(serve_metrics))
163            });
164
165            let rt = runtime::Builder::new_current_thread()
166                .enable_io()
167                .build()
168                .unwrap();
169            rt.block_on(async {
170                let server = Server::bind(&addr).serve(make_service);
171                server.await
172            })
173            .unwrap();
174        });
175    }
176}
