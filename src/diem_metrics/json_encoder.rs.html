8use prometheus::{
9    proto::{LabelPair, Metric, MetricFamily, MetricType},
10    Encoder, Result,
11};
12use std::{collections::HashMap, io::Write};
13
14const JSON_FORMAT: &str = "application/json";
15
16/// An implementation of an [`Encoder`](::Encoder) that converts a
17/// `MetricFamily` proto message into `fbagent` json
18///
19/// This implementation converts metric{dimensions,...} -> value to a flat
20/// string with a value. e.g., "requests{method="GET", service="accounts"} -> 8
21/// into requests.GET.account -> 8
22/// For now, it ignores timestamps (if set on the metric)
23#[derive(Debug, Default)]
24pub struct JsonEncoder;
25
26impl Encoder for JsonEncoder {
27    fn encode<W: Write>(
28        &self, metric_familys: &[MetricFamily], writer: &mut W,
29    ) -> Result<()> {
30        let mut export_me: HashMap<String, f64> = HashMap::new();
31
32        for mf in metric_familys {
33            let name = mf.get_name();
34            let metric_type = mf.get_field_type();
35
36            for m in mf.get_metric() {
37                match metric_type {
38                    MetricType::COUNTER => {
39                        export_me.insert(
40                            flatten_metric_with_labels(name, m),
41                            m.get_counter().get_value(),
42                        );
43                    }
44                    MetricType::GAUGE => {
45                        export_me.insert(
46                            flatten_metric_with_labels(name, m),
47                            m.get_gauge().get_value(),
48                        );
49                    }
50                    MetricType::HISTOGRAM => {
51                        // write the sum and counts
52                        let h = m.get_histogram();
53                        export_me.insert(
54                            flatten_metric_with_labels(
55                                &format!("{}_count", name),
56                                m,
57                            ),
58                            h.get_sample_count() as f64,
59                        );
60                        export_me.insert(
61                            flatten_metric_with_labels(
62                                &format!("{}_sum", name),
63                                m,
64                            ),
65                            h.get_sample_sum(),
66                        );
67                    }
68                    _ => {
69                        // do nothing; unimplemented
70                    }
71                }
72            }
73        }
74
75        writer
76            .write_all(serde_json::to_string(&export_me).unwrap().as_bytes())?;
77        Ok(())
78    }
79
80    fn format_type(&self) -> &str { JSON_FORMAT }
81}

83/**
84This method takes Prometheus metrics with dimensions (represented as label:value tags)
85and converts it into a dot-separated string.
86
87Example:
88Prometheus metric: error_count{method: "get_account", error="connection_error"}
89Result: error_count.get_account.connection_error
90
91If the set of labels is empty, only the name is returned
92Example:
93Prometheus metric: errors
94Result: errors
95
96This is useful when exporting metric data to flat time series.
97*/
98fn flatten_metric_with_labels(name: &str, metric: &Metric) -> String {
99    let res = String::from(name);
100
101    if metric.get_label().is_empty() {
102        res
103    } else {
104        // string-list.join(".")
105        let values: Vec<&str> = metric
106            .get_label()
107            .iter()
108            .map(LabelPair::get_value)
109            .filter(|&x| !x.is_empty())
110            .collect();
111        let values = values.join(".");
112        if !values.is_empty() {
113            format!("{}.{}", res, values)
114        } else {
115            res
116        }
117    }
118}

120#[cfg(test)]
121mod tests {
122    use super::*;
123    use prometheus::{
124        core::{Collector, Metric},
125        IntCounter, IntCounterVec, Opts,
126    };
127    use serde_json::Value;
128
129    #[test]
130    fn test_flatten_labels() {
131        // generate counters for testing
132        let counter = IntCounter::new("counter_1", "Test counter 1").unwrap();
133        let res = flatten_metric_with_labels("counter_1", &counter.metric());
134        assert_eq!("counter_1", res.as_str());
135
136        let counter = IntCounterVec::new(
137            Opts::new("counter_2", "Example counter for testing"),
138            &["label_me"],
139        )
140        .unwrap();
141        let res = flatten_metric_with_labels(
142            "counter_2",
143            &counter.with_label_values(&[""]).metric(),
144        );
145        assert_eq!("counter_2", res.as_str());
146
147        let res = flatten_metric_with_labels(
148            "counter_2",
149            &counter.with_label_values(&["hello"]).metric(),
150        );
151        assert_eq!("counter_2.hello", res.as_str());
152
153        let counter = IntCounterVec::new(
154            Opts::new("counter_2", "Example counter for testing"),
155            &["label_me", "label_me_too"],
156        )
157        .unwrap();
158        let res = flatten_metric_with_labels(
159            "counter_3",
160            &counter.with_label_values(&["", ""]).metric(),
161        );
162        assert_eq!("counter_3", res.as_str());
163
164        let res = flatten_metric_with_labels(
165            "counter_3",
166            &counter.with_label_values(&["hello", "world"]).metric(),
167        );
168        assert_eq!("counter_3.hello.world", res.as_str());
169    }

171    #[test]
172    fn test_encoder() {
173        let counter = IntCounterVec::new(
174            Opts::new("testing_count", "Test Counter"),
175            &["method", "result"],
176        )
177        .unwrap();
178        // add some test data
179        counter.with_label_values(&["get", "302"]).inc();
180        counter.with_label_values(&["get", "302"]).inc();
181        counter.with_label_values(&["get", "404"]).inc();
182        counter.with_label_values(&["put", ""]).inc();
183
184        let metric_family = counter.collect();
185        let mut data_writer = Vec::<u8>::new();
186        let encoder = JsonEncoder;
187        let res = encoder.encode(&metric_family, &mut data_writer);
188        assert!(res.is_ok());
189
190        let expected: &str = r#"
191        {
192            "testing_count.get.302": 2.0,
193            "testing_count.get.404": 1.0,
194            "testing_count.put": 1.0
195        }"#;
196
197        let v: Value = serde_json::from_slice(&data_writer).unwrap();
198        let expected_v: Value = serde_json::from_str(expected).unwrap();
199
200        assert_eq!(v, expected_v);
201    }
