32pub fn invalid_params_check<T, E: std::fmt::Display>(
33    param: &str, r: std::result::Result<T, E>,
34) -> Result<T, ErrorObjectOwned> {
35    match r {
36        Ok(t) => Ok(t),
37        Err(e) => {
38            Err(invalid_params(param.into(), Some(format!("{}", e))).into())
39        }
40    }
41}
