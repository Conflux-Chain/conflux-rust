32pub fn to_hex(bytes: &[u8]) -> String {
33    let mut v = Vec::with_capacity(bytes.len() * 2);
34    for &byte in bytes.iter() {
35        v.push(CHARS[(byte >> 4) as usize]);
36        v.push(CHARS[(byte & 0xf) as usize]);
37    }
38
39    unsafe { String::from_utf8_unchecked(v) }
40}

42pub fn get_cache_size(block_height: u64) -> usize {
43    // TODO: Memoise
44    let mut sz: u64 =
45        CACHE_BYTES_INIT + CACHE_BYTES_GROWTH * stage(block_height);
46    sz = sz - NODE_BYTES as u64;

53pub fn get_data_size(block_height: u64) -> usize {

87// We use `inline(always)` because I was experiencing an 100% slowdown and
88// `perf` showed that these calls were taking up ~30% of the runtime. Adding
89// these annotations fixes the issue. Remove at your peril, if and only if you
90// have benchmarks to prove that this doesn't reintroduce the performance
91// regression. It's not caused by the `debug_assert_eq!` either, your guess is
92// as good as mine.
93impl Node {
94    #[inline(always)]
95    pub fn as_bytes(&self) -> &NodeBytes { unsafe { &self.bytes } }
