5use crate::{
6    block_data_manager::BlockDataManager, channel::Channel,
7    consensus::consensus_inner::ConsensusGraphInner, Notifications,
8};
9use cfx_parameters::{consensus::*, light::BLAME_CHECK_OFFSET};
10use primitives::BlockHeader;
11use std::{collections::VecDeque, sync::Arc};
12
13pub struct BlameVerifier {
14    /// Channel used to send verified headers to the light sync layer.
15    /// Each element is <height, maybe_witness>
16    blame_sender: Arc<Channel<(u64, Option<u64>)>>,
17
18    /// Block data manager.
19    data_man: Arc<BlockDataManager>,
20
21    /// Last epoch received from ConsensusNewBlockHandler.
22    last_epoch_received: u64,
23
24    /// Next epoch we plan to process.
25    next_epoch_to_process: u64,
26
27    /// Queue of epochs that need to be re-processed.
28    queue: VecDeque<u64>,
29}
30
31impl BlameVerifier {
32    pub fn new(
33        data_man: Arc<BlockDataManager>, notifications: Arc<Notifications>,
34    ) -> Self {
35        let blame_sender = notifications.blame_verification_results.clone();
36
37        let stable_hash = data_man.get_cur_consensus_era_stable_hash();
38
39        let stable_height = data_man
40            .block_header_by_hash(&stable_hash)
41            .expect("Current era stable header should exist")
42            .height();
43
44        let start_height = stable_height.saturating_sub(BLAME_CHECK_OFFSET);
45
46        debug!("Starting Blame Verifier from height {}", start_height);
47
48        let last_epoch_received = start_height;
49        let next_epoch_to_process = start_height + 1;
50        let queue = VecDeque::new();
51
52        Self {
53            blame_sender,
54            data_man,
55            last_epoch_received,
56            next_epoch_to_process,
57            queue,
58        }
59    }
60
61    fn header_from_height(
62        &self, inner: &ConsensusGraphInner, height: u64,
63    ) -> Option<Arc<BlockHeader>> {
64        let pivot_index = inner.height_to_pivot_index(height);
65        let pivot_arena_index = inner.pivot_chain[pivot_index];
66        let pivot_hash = inner.arena[pivot_arena_index].hash;
67        self.data_man.block_header_by_hash(&pivot_hash)
68    }
69
70    fn first_trusted_header_starting_from(
71        &self, inner: &ConsensusGraphInner, height: u64,
72    ) -> Option<u64> {
73        // check if `height` is available in memory
74        let pivot_index = match height {
75            h if h < inner.get_cur_era_genesis_height() => return None,
76            h => inner.height_to_pivot_index(h),
77        };
78
79        inner
80            .find_first_trusted_starting_from(
81                pivot_index,
82                Some(1000), /* blame_bound */
83                10,         /* min_vote_count */
84            )
85            .map(|index| inner.pivot_index_to_height(index))
86    }
87
88    /// Add `epoch` to the queue and start processing it.
89    pub fn process(&mut self, inner: &ConsensusGraphInner, epoch: u64) {
90        // we need to keep an offset so that we have
91        // enough headers to calculate the blame ratio
92        // TODO(thegaram): choose better value for `BLAME_CHECK_OFFSET`
93        let epoch = match epoch {
94            e if e < BLAME_CHECK_OFFSET => return,
95            e => e - BLAME_CHECK_OFFSET,
96        };
97
98        trace!("Blame verification received epoch {:?}", epoch);
99        self.queue.push_back(epoch);
100
101        loop {
102            // process while there are unprocessed epochs
103            let epoch = match self.queue.pop_front() {
104                Some(e) => e,
105                None => break,
106            };
107
108            // process until we encounter an epoch for which
109            // there is no blame information available
110            if !self.check(inner, epoch) {
111                break;
112            }
113        }
114    }
115
116    /// Check the blame corresponding to `epoch` and send the verification
117    /// results to the light node sync layer.
118    /// Returns false if the epoch cannot be processed, true otherwise.
119    #[rustfmt::skip]
120    pub fn check(&mut self, inner: &ConsensusGraphInner, epoch: u64) -> bool {
121        trace!(
122            "Blame verification is processing epoch {:?} (last_epoch_received = {}, next_epoch_to_process = {})",
123            epoch, self.last_epoch_received, self.next_epoch_to_process
124        );
125
126        match epoch {
127            // pivot chain reorg
128            //
129            //                                   ---        ---
130            //                               .- | D | <--- | E | <--- ...
131            //  ---        ---        ---    |   ---        ---
132            // | A | <--- | B | <--- | C | <-*
133            //  ---        ---        ---    |   ---
134            //                               .- | F | <--- ...
135            //                                   ---
136            //
137            // example (BLAME_CHECK_OFFSET = 2):
138            //    check is called with    C, D, E, C, F
139            //    we will process epochs  A, B, C, A, B
140            //
141            // TODO(thegaram): can a fork change the blame status of a header?
142
143            e if e <= self.last_epoch_received => {
144                // re-process from fork point
145                debug!("Chain reorg ({} --> {}), re-executing", self.last_epoch_received, e);
146                self.last_epoch_received = e;
147                self.next_epoch_to_process = e;
148            }
149
150            // sanity check: epochs are sent in order, one-by-one
151            e if e > self.last_epoch_received + 1 => {
152                error!(
153                    "Unexpected epoch number: e = {}, last_epoch_received = {}",
154                    e, self.last_epoch_received
155                );
156
157                assert!(false);
158            }
159
160            // epoch already handled through witness
161            //
162            //                 blames
163            //              ............
164            //              v          |
165            //  ---        ---        ---        ---
166            // | A | <--- | B | <--- | C | <--- | D | <--- ...
167            //  ---        ---        ---        ---
168            //
169            // we receive B and proceed to request all blamed headers (B, C);
170            // set last-epoch-received to B and next-epoch-to-process to D;
171            // we will skip C in the next iteration (it is covered already).
172
173            e if e < self.next_epoch_to_process => {
174                debug!("Epoch already covered, skipping (e = {}, next_epoch_to_process = {})", e, self.next_epoch_to_process);
175                self.last_epoch_received = e;
176                return true;
177            }
178
179            // sanity check: no epochs are skipped
180            e if e > self.next_epoch_to_process => {
181                error!("Unexpected epoch number: e = {}, next_epoch_to_process = {}", e, self.next_epoch_to_process);
182                assert!(false);
183            }
184
185            // in most cases, we will iterate over the pivot chain sequentially;
186            // at each step, the epoch we receive (e) will be the same as the
187            // next-epoch-to-process (nep)
188            //
189            //                         e
190            //  ---        ---        ---        ---
191            // |   | <--- |   | <--- |   | <--- |   | <--- ...
192            //  ---        ---        ---        ---
193            //             ler        nep
194
195            // e == last_epoch_received + 1
196            // e == next_epoch_to_process
197            e => {
198                self.last_epoch_received = e;
199            }
200        }
201
202        // convert epoch number into pivot height
203        let height = epoch + DEFERRED_STATE_EPOCH_COUNT;
204
205        // check blame
206        debug!(
207            "Finding witness for header at height {} (epoch {})...",
208            height, epoch
209        );
210
211        match self.first_trusted_header_starting_from(inner, height) {
212            // no witness found
213            None => {
214                warn!(
215                    "No witness found for epoch {} (height {});
216                    best_epoch_number = {}",
217                    epoch,
218                    height,
219                    inner.best_epoch_number(),
220                );
221
222                // this can happen in two cases:
223                //
224                // (1) we are lagging behind so much that `height`
225                //     is no longer maintained in memory.
226                //       --> consensus and blame verification are
227                //           in sync so this should not happen.
228                //
229                // (2) there are too many blamed blocks on the
230                //     `BLAME_CHECK_OFFSET` suffix of the pivot
231                //     chain so we cannot reliably determine the
232                //     witness.
233                //       --> we will retry during the next invocation of `check`
234                //
235                // example for (2):
236                //
237                //       blame     blame     blame
238                //       ......    ......    ......
239                //      /     |   /     |   /     |
240                //  ---       ---       ---       ---
241                // | A | <-- | B | <-- | C | <-- | D | <--- ...
242                //  ---       ---       ---       ---
243                //
244                // if we have such a section of the pivot chain of length
245                // `BLAME_CHECK_OFFSET` or more, then at the point of
246                // receiving A we might not be able to find the corresponding
247                // witness. in this case, we store this epoch for processing
248                // later. we assume here that the pivot chain will eventually
249                // normalize and we will be able to find the witness later.
250
251                // save for further processing and terminate
252                self.queue.push_front(epoch);
253                return false;
254            }
255
256            // header is not blamed (i.e. it is its own witness)
257            Some(w) if w == height => {
258                trace!("Epoch {} (height {}) is NOT blamed", epoch, height);
259
260                let header = self
261                    .header_from_height(inner, height)
262                    .expect("Pivot header exists");
263
264                // normal case: blaming blocks have been covered previously,
265                // so this block must be non-blaming
266                if header.blame() == 0 {
267                    // send non-blaming header
268                    self.blame_sender.send((height, None));
269                }
270
271                // special case
272                //
273                //      blame
274                //   ...........                                ---        ---        ---
275                //   v          |                           .- | E | <--- | F | <--- | G | <--- ...
276                //  ---        ---        ---        ---    |   ---        ---        ---
277                // | A | <--- | B | <--- | C | <--- | D | <-*
278                //  ---        ---        ---        ---    |   ---
279                //                                          .- | H | <--- ...
280                //                                              ---
281                //
282                // example (BLAME_CHECK_OFFSET = 2)
283                //    check is called with    C, D, E, F, G, D, H
284                //    we will process epochs  A, B, C, D, E, B, C
285                //
286                // after the chain reorg, we will start re-executing from B
287                // B was already covered in A's iteration but it is blaming
288                //   --> we do nothing, skip it
289
290                else {
291                    // EMPTY
292                }
293
294                // continue from the next header on the pivot chain
295                self.next_epoch_to_process = epoch + 1;
296            }
297
298            // header is blamed
299            Some(w) => {
300                debug!(
301                    "Epoch {} (height {}) is blamed, requesting witness {}",
302                    epoch, height, w
303                );
304
305                // this request covers all blamed headers:
306                // [height, height + 1, ..., w]
307                self.blame_sender.send((height, Some(w)));
308
309                // skip all subsequent headers requested
310                assert!(w > DEFERRED_STATE_EPOCH_COUNT);
311                let witness_epoch = w - DEFERRED_STATE_EPOCH_COUNT;
312                self.next_epoch_to_process = witness_epoch + 1;
313            }
314        }
315
316        true
317    }
318}
