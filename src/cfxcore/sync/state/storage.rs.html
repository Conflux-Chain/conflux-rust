21#[derive(
22    Clone, Hash, Ord, PartialOrd, PartialEq, Eq, Debug, DeriveMallocSizeOf,
23)]
24pub enum SnapshotSyncCandidate {
25    OneStepSync {
26        height: u64,
27        snapshot_epoch_id: EpochId,
28    },
29    FullSync {
30        height: u64,
31        snapshot_epoch_id: EpochId,
32    },
33    IncSync {
34        height: u64,
35        base_snapshot_epoch_id: EpochId,
36        snapshot_epoch_id: EpochId,
37    },
38}
39
40impl SnapshotSyncCandidate {
41    fn to_type_id(&self) -> u8 {
42        match self {
43            SnapshotSyncCandidate::OneStepSync { .. } => 0,
44            SnapshotSyncCandidate::FullSync { .. } => 1,
45            SnapshotSyncCandidate::IncSync { .. } => 2,
46        }
47    }
48
49    pub fn get_snapshot_epoch_id(&self) -> &EpochId {
50        match self {
51            SnapshotSyncCandidate::OneStepSync {
52                snapshot_epoch_id, ..
53            } => snapshot_epoch_id,
54            SnapshotSyncCandidate::FullSync {
55                snapshot_epoch_id, ..
56            } => snapshot_epoch_id,
57            SnapshotSyncCandidate::IncSync {
58                snapshot_epoch_id, ..
59            } => snapshot_epoch_id,
60        }
61    }
62}
63
64impl Encodable for SnapshotSyncCandidate {
65    fn rlp_append(&self, s: &mut RlpStream) {
66        match &self {
67            SnapshotSyncCandidate::OneStepSync {
68                height,
69                snapshot_epoch_id,
70            } => {
71                s.begin_list(3)
72                    .append(&self.to_type_id())
73                    .append(height)
74                    .append(snapshot_epoch_id);
75            }
76            SnapshotSyncCandidate::FullSync {
77                height,
78                snapshot_epoch_id,
79            } => {
80                s.begin_list(3)
81                    .append(&self.to_type_id())
82                    .append(height)
83                    .append(snapshot_epoch_id);
84            }
85            SnapshotSyncCandidate::IncSync {
86                height,
87                base_snapshot_epoch_id,
88                snapshot_epoch_id,
89            } => {
90                s.begin_list(4)
91                    .append(&self.to_type_id())
92                    .append(height)
93                    .append(base_snapshot_epoch_id)
94                    .append(snapshot_epoch_id);
95            }
96        }
97    }
98}

100impl Decodable for SnapshotSyncCandidate {
101    fn decode(rlp: &Rlp) -> Result<Self, DecoderError> {
102        let type_id: u8 = rlp.val_at(0)?;
103        let parsed = match type_id {
104            0 => SnapshotSyncCandidate::OneStepSync {
105                height: rlp.val_at(1)?,
106                snapshot_epoch_id: rlp.val_at(2)?,
107            },
108            1 => SnapshotSyncCandidate::FullSync {
109                height: rlp.val_at(1)?,
110                snapshot_epoch_id: rlp.val_at(2)?,
111            },
112            2 => SnapshotSyncCandidate::IncSync {
113                height: rlp.val_at(1)?,
114                base_snapshot_epoch_id: rlp.val_at(2)?,
115                snapshot_epoch_id: rlp.val_at(3)?,
116            },
117            _ => {
118                return Err(DecoderError::Custom(
119                    "Unknown SnapshotSyncCandidate type id",
120                ))
121            }
122        };
123        debug_assert_eq!(parsed.to_type_id(), type_id);
124        Ok(parsed)
125    }
126}

147pub struct RangedManifest {
148    pub chunk_boundaries: Vec<Vec<u8>>,
149    pub chunk_boundary_proofs: Vec<TrieProof>,
150    pub next: Option<Vec<u8>>,
151}
152
153impl Encodable for RangedManifest {
154    fn rlp_append(&self, s: &mut RlpStream) {
155        s.begin_list(3)
156            .append_list::<Vec<u8>, Vec<u8>>(&self.chunk_boundaries)
157            .append_list(&self.chunk_boundary_proofs)
158            .append(&self.next);
159    }
160}

162impl Decodable for RangedManifest {
163    fn decode(rlp: &Rlp) -> Result<Self, DecoderError> {
164        Ok(RangedManifest {
165            chunk_boundaries: rlp.list_at(0)?,
166            chunk_boundary_proofs: rlp.list_at(1)?,
167            next: rlp.val_at(2)?,
168        })
169    }
170}

172impl RangedManifest {
173    /// Validate the manifest with specified snapshot merkle root and the
174    /// requested start chunk key. Basically, the retrieved chunks should
175    /// not be empty, and the proofs of all chunk keys are valid.
176    pub fn validate(&self, snapshot_root: &MerkleHash) -> Result<(), Error> {
177        if self.chunk_boundaries.len() != self.chunk_boundary_proofs.len() {
178            bail!(Error::InvalidSnapshotManifest(
179                "chunk and proof number do not match".into(),
180            ));
181        }
182        if let Some(next) = &self.next {
183            if next != self.chunk_boundaries.last().unwrap() {
184                bail!(Error::InvalidSnapshotManifest(
185                    "next does not match last boundary".into(),
186                ));
187            }
188        }
189
190        // validate the trie proof for all chunks
191        for (chunk_index, proof) in
192            self.chunk_boundary_proofs.iter().enumerate()
193        {
194            if proof.get_merkle_root() != snapshot_root {
195                warn!(
196                    "Manifest merkle root should be {:?}, get {:?}",
197                    snapshot_root,
198                    proof.get_merkle_root()
199                );
200                bail!(Error::InvalidSnapshotManifest(
201                    "invalid proof merkle root".into(),
202                ));
203            }
204            if !proof.if_proves_key(&self.chunk_boundaries[chunk_index]).0 {
205                bail!(Error::InvalidSnapshotManifest("invalid proof".into(),));
206            }
207        }
208        Ok(())
209    }

211    pub fn convert_boundaries_to_chunks(
212        chunk_boundaries: Vec<Vec<u8>>,
213    ) -> Vec<ChunkKey> {
214        let mut chunks = Vec::with_capacity(chunk_boundaries.len());
215        let mut lower = None;
216        for key in chunk_boundaries {
217            chunks.push(ChunkKey {
218                lower_bound_incl: lower,
219                upper_bound_excl: Some(key.clone()),
220            });
221            lower = Some(key);
222        }
223        chunks.push(ChunkKey {
224            lower_bound_incl: lower,
225            upper_bound_excl: None,
226        });
227        chunks
228    }

230    pub fn load(
231        snapshot_to_sync: &SnapshotSyncCandidate, start_key: Option<Vec<u8>>,
232        storage_manager: &StorageManager, chunk_size: u64, max_chunks: usize,
233    ) -> Result<Option<(RangedManifest, MerkleHash)>, Error> {
234        let snapshot_epoch_id = match snapshot_to_sync {
235            SnapshotSyncCandidate::FullSync {
236                snapshot_epoch_id, ..
237            } => snapshot_epoch_id,
238            SnapshotSyncCandidate::IncSync { .. } => {
239                unimplemented!();
240            }
241            SnapshotSyncCandidate::OneStepSync { .. } => {
242                unimplemented!();
243            }
244        };
245        debug!(
246            "begin to load manifest, snapshot_epoch_id = {:?}, start_key = {:?}",
247            snapshot_epoch_id, start_key
248        );
249
250        let snapshot_db_manager =
251            storage_manager.get_storage_manager().get_snapshot_manager();
252
253        let snapshot_db = match snapshot_db_manager.get_snapshot_by_epoch_id(
254            snapshot_epoch_id,
255            /* try_open = */ true,
256            true,
257        )? {
258            Some(db) => db,
259            None => {
260                debug!(
261                    "failed to load manifest, cannot find snapshot {:?}",
262                    snapshot_epoch_id
263                );
264                return Ok(None);
265            }
266        };
267        let mut snapshot_mpt = snapshot_db.open_snapshot_mpt_shared()?;
268        let merkle_root = snapshot_mpt.merkle_root;
269        let mut slicer = match start_key {
270            Some(ref key) => MptSlicer::new_from_key(&mut snapshot_mpt, key)?,
271            None => MptSlicer::new(&mut snapshot_mpt)?,
272        };
273
274        let mut manifest = RangedManifest::default();
275        let mut has_next = true;
276
277        for i in 0..max_chunks {
278            trace!("cut chunks for manifest, loop = {}", i);
279            slicer.advance(chunk_size)?;
280            match slicer.get_range_end_key() {
281                None => {
282                    has_next = false;
283                    break;
284                }
285                Some(key) => {
286                    manifest.chunk_boundaries.push(key.to_vec());
287                    manifest.chunk_boundary_proofs.push(slicer.to_proof());
288                }
289            }
290        }
291
292        if has_next {
293            manifest.next = Some(
294                manifest
295                    .chunk_boundaries
296                    .last()
297                    .expect("boundaries not empty if has next")
298                    .clone(),
299            );
300        }
301
302        debug!(
303            "succeed to load manifest, chunks = {}, next_chunk_key = {:?}",
304            manifest.chunk_boundaries.len(),
305            manifest.next
306        );
307
308        Ok(Some((manifest, merkle_root)))
309    }
310}

312#[derive(Default)]
313pub struct Chunk {
314    pub keys: Vec<Vec<u8>>,
315    pub values: Vec<Vec<u8>>,
316}
317
318impl Encodable for Chunk {
319    fn rlp_append(&self, s: &mut RlpStream) {
320        s.begin_list(2)
321            .append_list::<Vec<u8>, Vec<u8>>(&self.keys)
322            .append_list::<Vec<u8>, Vec<u8>>(&self.values);
323    }
324}
325
326impl Decodable for Chunk {
327    fn decode(rlp: &Rlp) -> Result<Self, DecoderError> {
328        Ok(Chunk {
329            keys: rlp.list_at(0)?,
330            values: rlp.list_at(1)?,
331        })
332    }
333}
334
335impl Chunk {
336    /// Validate the chunk with specified key.
337    pub fn validate(&self, key: &ChunkKey) -> Result<(), Error> {
338        // chunk should not be empty
339        if self.keys.is_empty() {
340            // TODO: Now this may happen if the requested peer has opened
341            // maximal number of snapshots and cannot give a
342            // response temporarily, we should differentiate this
343            // from dishonest behaviors.
344            return Err(Error::EmptySnapshotChunk.into());
345        }
346        if self.keys.len() != self.values.len() {
347            return Err(Error::InvalidSnapshotChunk(
348                "keys and values do not match".into(),
349            )
350            .into());
351        }
352        // the key of first item in chunk should match with the requested key
353        if let Some(ref start_key) = key.lower_bound_incl {
354            if start_key != &self.keys[0] {
355                return Err(
356                    Error::InvalidSnapshotChunk("key mismatch".into()).into()
357                );
358            }
359        }
360
361        Ok(())
362    }
363
364    pub fn load(
365        snapshot_epoch_id: &H256, chunk_key: &ChunkKey,
366        storage_manager: &StorageManager, max_chunk_size: u64,
367    ) -> Result<Option<Chunk>, Error> {
368        debug!(
369            "begin to load chunk, snapshot_epoch_id = {:?}, key = {:?}",
370            snapshot_epoch_id, chunk_key
371        );
372
373        let snapshot_db_manager =
374            storage_manager.get_storage_manager().get_snapshot_manager();
375
376        let snapshot_db = match snapshot_db_manager.get_snapshot_by_epoch_id(
377            snapshot_epoch_id,
378            /* try_open = */ true,
379            false,
380        )? {
381            Some(db) => db,
382            None => {
383                debug!("failed to load chunk, cannot find snapshot by checkpoint {:?}",
384                       snapshot_epoch_id);
385                return Ok(None);
386            }
387        };
388
389        let mut kv_iterator = snapshot_db.snapshot_kv_iterator()?.take();
390        let lower_bound_incl =
391            chunk_key.lower_bound_incl.clone().unwrap_or_default();
392        let upper_bound_excl =
393            chunk_key.upper_bound_excl.as_ref().map(|k| k.as_slice());
394        let mut kvs = kv_iterator
395            .iter_range(lower_bound_incl.as_slice(), upper_bound_excl)?
396            .take();
397
398        let mut keys = Vec::new();
399        let mut values = Vec::new();
400        let mut chunk_size = 0;
401        while let Some((key, value)) = kvs.next()? {
402            chunk_size += rlp_key_value_len(key.len() as u16, value.len());
403            if chunk_size > max_chunk_size {
404                let msg =
405                    format!("Exceed max allowed chunk size {}", max_chunk_size);
406                error!("{}", msg);
407                return Err(Error::InvalidSnapshotChunk(msg).into());
408            }
409
410            keys.push(key);
411            values.push(value.into());
412        }
413
414        debug!(
415            "complete to load chunk, items = {}, chunk_key = {:?}",
416            keys.len(),
417            chunk_key
418        );
419
420        Ok(Some(Chunk { keys, values }))
421    }
