1use crate::sync::state::storage::SnapshotSyncCandidate;
2use network::node_table::NodeId;
3use primitives::EpochId;
4use std::{
5    collections::{BTreeMap, HashSet},
6    time::{Duration, Instant},
7};

9/// 1. Send CandidateRequest to all peers and set them to `pending_peers`. 1.1:
10///    All peers respond before timeout, we choose one candidate and start state
11///    sync. 1.2: Some peers timeout. 1.2.1: Have available candidates, we
12///    choose one and enter step 2. `pending_peers` is cleared before entering
13///    step 2. 1.2.2: No available candidates, reset the status and restart step
14///    1.
15/// 2. Send ManifestRequest to some random peer in `active_peers`. 2.1: Valid
16///    manifest received, move on to step 3. 2.2: Timeout/invalid/empty
17///    response. Remove from `active_peers`. 2.2.1: Have more available
18///    `active_peers`, restart step 2. 2.2.2: `active_peers` is empty, reset and
19///    restart step 1.
20/// 3. Send ChunkRequest to multiple peers in `active_peers`. 3.1: Valid chunks
21///    received. 3.1.1: All chunks received, set state_sync.status to Completed.
22///    3.1.2: More chunks to request. Request from `active_peers`. 3.2:
23///    Timeout/invalid/empty response. Remove from `active_peers`. 3.2.1: Have
24///    more available `active_peers`, push this chunk key back into the
25///    `pending_chunks` for later requesting. 3.2.2: `active_peers` is empty,
26///    reset and restart step 1.
27///
28/// All step start/restart are triggered by the periodic check of phase change.
29pub struct StateSyncCandidateManager {
30    /// The starting time of the ongoing candidate requesting
31    start_time: Instant,
32
33    pub current_era_genesis: EpochId,
34    candidates: Vec<SnapshotSyncCandidate>,
35
36    /// The map from state candidates to the set of peers that can support this
37    /// state
38    candidate_to_active_peers: BTreeMap<SnapshotSyncCandidate, HashSet<NodeId>>,
39    /// The peers who have been requested for candidate response but has not
40    /// replied
41    pending_peers: HashSet<NodeId>,
42
43    /// The chosen candidate that we are actually requesting state manifest and
44    /// chunks.
45    active_candidate: Option<usize>,
46}
47
48impl StateSyncCandidateManager {
49    fn new() -> Self {
50        Self {
51            start_time: Instant::now(),
52            current_era_genesis: Default::default(),
53            candidates: Default::default(),
54            candidate_to_active_peers: BTreeMap::new(),
55            pending_peers: Default::default(),
56            active_candidate: None,
57        }
58    }
59
60    pub fn reset(
61        &mut self, current_era_genesis: EpochId,
62        candidates: Vec<SnapshotSyncCandidate>, peers: Vec<NodeId>,
63    ) {
64        let mut candidates_map = BTreeMap::new();
65        for candidate in &candidates {
66            candidates_map.insert(candidate.clone(), HashSet::new());
67        }
68        self.start_time = Instant::now();
69        self.current_era_genesis = current_era_genesis;
70        self.candidates = candidates;
71        self.candidate_to_active_peers = candidates_map;
72        self.pending_peers = peers.into_iter().collect();
73        self.active_candidate = None;
74    }
75
76    /// Update the status about candidate choosing.
77    /// Return the peer and epoch to retrieve the state manifest
78    pub fn on_peer_response(
79        &mut self, peer: &NodeId,
80        supported_candidates: &Vec<SnapshotSyncCandidate>,
81        requested_candidates: &Vec<SnapshotSyncCandidate>,
82    ) {
83        if !self.pending_peers.remove(peer) {
84            debug!("Receive response from unexpected peer {:?}, possibly from old requests", peer);
85            return;
86        }
87        let mut requested_candidates_set: HashSet<&SnapshotSyncCandidate> =
88            requested_candidates.iter().collect();
89        for candidate in supported_candidates {
90            match self.candidate_to_active_peers.get_mut(candidate) {
91                Some(peer_set) => {
92                    peer_set.insert(*peer);
93                }
94                None => {
95                    if requested_candidates_set.contains(candidate) {
96                        debug!("requested candidate {:?} is stale", candidate);
97                    } else {
98                        debug!(
99                            "Receive unexpected candidate {:?} from peer {:?}",
100                            candidate, peer
101                        );
102                    }
103                }
104            }
105            requested_candidates_set.remove(&candidate);
106        }
107        for unsupported_candidate in requested_candidates_set {
108            match self
109                .candidate_to_active_peers
110                .get_mut(unsupported_candidate)
111            {
112                Some(peer_set) => {
113                    peer_set.remove(peer);
114                }
115                None => {
116                    debug!(
117                        "requested candidate {:?} is stale",
118                        unsupported_candidate,
119                    );
120                }
121            }
122        }
123    }
124
125    pub fn on_peer_disconnected(&mut self, peer: &NodeId) {
126        for peers in self.candidate_to_active_peers.values_mut() {
127            peers.remove(peer);
128        }
129    }
130
131    pub fn pending_peers(&self) -> &HashSet<NodeId> { &self.pending_peers }
132
133    pub fn get_active_candidate_and_peers(
134        &self,
135    ) -> Option<(SnapshotSyncCandidate, HashSet<NodeId>)> {
136        self.active_candidate.map(|i| {
137            let candidate = self.candidates[i].clone();
138            let active_peers = self
139                .candidate_to_active_peers
140                .get(&candidate)
141                .expect("Active candidate has a non-empty peer set")
142                .clone();
143            (candidate, active_peers)
144        })
145    }
146
147    pub fn set_active_candidate(&mut self) {
148        let mut candidate_index = self.active_candidate.map_or(0, |i| i + 1);
149        let max_candidate_index = self.candidates.len();
150        while candidate_index < max_candidate_index {
151            self.active_candidate = Some(candidate_index);
152            let candidate = &self.candidates[candidate_index];
153            let peer_set =
154                self.candidate_to_active_peers.get(candidate).unwrap();
155            if peer_set.is_empty() {
156                debug!(
157                    "StateSync: candidate {}={:?}, active_peers={:?}",
158                    candidate_index, candidate, peer_set
159                );
160            } else {
161                debug!(
162                    "StateSync: set active_candidate {}={:?}, active_peers={:?}",
163                    candidate_index, candidate, peer_set
164                );
165                return;
166            }
167            candidate_index += 1;
168        }
169        // All sync candidates failed.
170        if candidate_index == max_candidate_index {
171            self.active_candidate = None;
172        }
173    }
174
175    pub fn check_timeout(&mut self, candidate_timeout: &Duration) {
176        if !self.pending_peers.is_empty()
177            && self.start_time.elapsed() > *candidate_timeout
178        {
179            self.pending_peers.clear();
180        }
181    }
182
183    /// Return `true ` if we are not requesting either candidates or states
184    pub fn is_inactive(&self) -> bool {
185        self.pending_peers.is_empty() && self.active_candidate.is_none()
186    }
187}
188
189impl Default for StateSyncCandidateManager {
190    fn default() -> Self { Self::new() }
