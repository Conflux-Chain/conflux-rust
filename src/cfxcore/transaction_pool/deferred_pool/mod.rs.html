23/// The `DeferredPool` is designed to organize transactions for each address
24/// based on their nonce. It efficiently maintains and queries transactions even
25/// when received nonces are non-sequential. In addition, it calculates
26/// transactions that are ready to be packed for each address and stores them in
27/// the `packing_pool`. The transactions in the `packing_pool` should always be
28/// a subset of the transactions in the nonce pools for all addresses.
29#[derive(DeriveMallocSizeOf)]
30pub struct DeferredPool {
31    /// Store transactions organized in binary balanced trees keyed by nonce
32    /// for each address.
33    buckets: HashMap<AddressWithSpace, NoncePool>,
34    /// Store transactions that are ready to be packed for each address, and
35    /// implements random sampling logic.
36    packing_pool: SpaceMap<PackingPool<Arc<SignedTransaction>>>,
37}
38

71    pub fn estimate_packing_gas_limit(
72        &self, space: Space, gas_target: U256, parent_base_price: U256,
73        min_base_price: U256,
74    ) -> (U256, U256) {
75        let estimated_gas_limit = self
76            .packing_pool
77            .in_space(space)
78            .estimate_packing_gas_limit(
79                gas_target,
80                parent_base_price,
81                min_base_price,
82            );
83        let packing_gas_limit = U256::min(gas_target * 2, estimated_gas_limit);
84        let price_limit = compute_next_price(
85            gas_target,
86            packing_gas_limit,
87            parent_base_price,
88            min_base_price,
89        );
90        (packing_gas_limit, price_limit)
91    }

93    #[inline]
94    pub fn packing_sampler<
95        'a,
96        F: Fn(&SignedTransaction) -> PackingCheckResult,
97    >(
98        &'a mut self, space: Space, block_gas_limit: U256,
99        block_size_limit: usize, tx_num_limit: usize, tx_min_price: U256,
100        validity: F,
101    ) -> (Vec<Arc<SignedTransaction>>, U256, usize) {
102        if block_gas_limit.is_zero()
103            || block_size_limit == 0
104            || tx_num_limit == 0
105        {
106            return (vec![], 0.into(), 0);
107        }
108
109        debug!(
110            "txpool::packing_sampler start space={:?} block_gas_limit={} block_size_limit={} tx_limit={} min_price={}",
111            space,
112            block_gas_limit,
113            block_size_limit,
114            tx_num_limit,
115            tx_min_price
116        );
117
118        let mut to_pack_txs = Vec::new();
119        let mut to_drop_txs = Vec::new();
120
121        let mut minimum_unit_gas_limit = U256::from(21000);
122        let mut minimum_unit_tx_size = 80;
123
124        let mut rng = XorShiftRng::from_os_rng();
125
126        // When a sampled transaction exceeds the remaining capacity (gas limit
127        // or size) in a block, we skip it and look for the next transaction.
128        // However, if the remaining space is too small, we might sample a large
129        // number of transactions and still fail to find one that meets the
130        // criteria.
131
132        // Here, we maintain a threshold. When the remaining capacity is less
133        // than the threshold, the packing process stopped. The threshold
134        // increases by 1/16 for each fail due to insufficient capacity. This
135        // way, the packing process can always stop after a finite number of
136        // failures.
137
138        let mut rest_size_limit = block_size_limit;
139        let mut rest_gas_limit = block_gas_limit;
140
141        'all: for (sender, sender_txs, _) in self
142            .packing_pool
143            .in_space_mut(space)
144            .tx_sampler(&mut rng, block_gas_limit.into())
145        {
146            'sender: for tx in sender_txs.iter() {
147                if tx.gas_price() < &tx_min_price {
148                    trace!(
149                        "txpool::packing_sampler skip sender={:?} nonce={} reason=low_price tx_price={} min_price={}",
150                        sender,
151                        tx.nonce(),
152                        tx.gas_price(),
153                        tx_min_price
154                    );
155                    break 'sender;
156                }
157                match validity(&*tx) {
158                    PackingCheckResult::Pack => {}
159                    PackingCheckResult::Pending => {
160                        trace!(
161                            "txpool::packing_sampler stop sender={:?} nonce={} reason=pending",
162                            sender,
163                            tx.nonce()
164                        );
165                        break 'sender;
166                    }
167                    PackingCheckResult::Drop => {
168                        to_drop_txs.push(tx.clone());
169                        trace!(
170                            "txpool::packing_sampler drop sender={:?} nonce={} reason=invalid",
171                            sender,
172                            tx.nonce()
173                        );
174                        break 'sender;
175                    }
176                }
177
178                let gas_limit = *tx.gas_limit();
179                if gas_limit > rest_gas_limit {
180                    if gas_limit >= minimum_unit_gas_limit {
181                        minimum_unit_gas_limit += minimum_unit_gas_limit >> 4;
182                        trace!(
183                            "txpool::packing_sampler sender={:?} nonce={} gas_limit={} exceeds remaining_gas={} adjust_threshold={}",
184                            sender,
185                            tx.nonce(),
186                            gas_limit,
187                            rest_gas_limit,
188                            minimum_unit_gas_limit
189                        );
190                        break 'sender;
191                    } else {
192                        trace!(
193                            "txpool::packing_sampler stop all space={:?} reason=gas_exhausted remaining_gas={} next_gas_limit={}",
194                            space,
195                            rest_gas_limit,
196                            gas_limit
197                        );
198                        break 'all;
199                    }
200                }
201
202                let tx_size = tx.rlp_size();
203                if tx_size > rest_size_limit {
204                    if tx_size >= minimum_unit_tx_size {
205                        minimum_unit_tx_size += minimum_unit_tx_size >> 4;
206                        trace!(
207                            "txpool::packing_sampler sender={:?} nonce={} tx_size={} exceeds remaining_size={} adjust_threshold_size={}",
208                            sender,
209                            tx.nonce(),
210                            tx_size,
211                            rest_size_limit,
212                            minimum_unit_tx_size
213                        );
214                        break 'sender;
215                    } else {
216                        trace!(
217                            "txpool::packing_sampler stop all space={:?} reason=size_exhausted remaining_size={} next_size={}",
218                            space,
219                            rest_size_limit,
220                            tx_size
221                        );
222                        break 'all;
223                    }
224                }
225
226                rest_gas_limit -= gas_limit;
227                rest_size_limit -= tx_size;
228
229                to_pack_txs.push(tx.clone());
230                trace!(
231                    "txpool::packing_sampler select sender={:?} nonce={} remaining_gas={} remaining_size={} count={}",
232                    sender,
233                    tx.nonce(),
234                    rest_gas_limit,
235                    rest_size_limit,
236                    to_pack_txs.len()
237                );
238                if to_pack_txs.len() >= tx_num_limit {
239                    debug!(
240                        "txpool::packing_sampler reached tx limit {}",
241                        tx_num_limit
242                    );
243                    break 'all;
244                }
245            }
246        }
247
248        // Maybe we can remove to drop txs from deferred pool. But removing them
249        // directly may break gc logic. So we only update packing
250        // pool now.
251        for tx in to_drop_txs {
252            trace!(
253                "txpool::packing_sampler prune sender={:?} nonce={}",
254                tx.sender(),
255                tx.nonce()
256            );
257            self.packing_pool
258                .in_space_mut(space)
259                .split_off_suffix(tx.sender(), tx.nonce());
260        }
261
262        let gas_used = block_gas_limit - rest_gas_limit;
263        let size_used = block_size_limit - rest_size_limit;
264        debug!(
265            "txpool::packing_sampler finish space={:?}  packed={} gas_used={} size_used={}",
266            space,
267            to_pack_txs.len(),
268            gas_used,
269            size_used
270        );
271        if to_pack_txs.is_empty() && self.packing_pool.in_space(space).len() > 0
272        {
273            debug!(
274                "txpool::packing_sampler no_tx_selected space={:?} ready_entries={} block_gas_limit={} block_size_limit={} tx_limit={} remaining_gas={} remaining_size={}",
275                space,
276                self.packing_pool.in_space(space).len(),
277                block_gas_limit,
278                block_size_limit,
279                tx_num_limit,
280                rest_gas_limit,
281                rest_size_limit
282            );
283        }
284        (to_pack_txs, gas_used, size_used)
285    }

287    pub fn insert(&mut self, tx: TxWithReadyInfo, force: bool) -> InsertResult {
288        let bucket = self
289            .buckets
290            .entry(tx.sender())
291            .or_insert_with(|| NoncePool::new());
292
293        let res = bucket.insert(&tx, force);
294        if matches!(res, InsertResult::Updated(_)) {
295            // The transactions in the packing_pool must be consistent with the
296            // nonce pool. However, the replaced transactions have not undergone
297            // a readiness check, so we will temporarily remove them from the
298            // packing_pool.
299            self.packing_pool
300                .in_space_mut(tx.space())
301                .split_off_suffix(tx.sender(), tx.nonce());
302        }
303        res
304    }

338    pub fn remove_lowest_nonce(
339        &mut self, addr: &AddressWithSpace,
340    ) -> Option<TxWithReadyInfo> {
341        let bucket = self.buckets.get_mut(addr)?;
342        let ret = bucket.remove_lowest_nonce();
343        if bucket.is_empty() {
344            self.buckets.remove(addr);
345            self.packing_pool.in_space_mut(addr.space).remove(*addr);
346            return ret;
347        }
348
349        let tx = ret.as_ref()?;
350        let removed_tx = self
351            .packing_pool
352            .in_space_mut(addr.space)
353            .split_off_prefix(tx.sender(), &(tx.nonce() + 1));
354        if let Some(removed_tx) = removed_tx.first() {
355            if removed_tx.nonce() < tx.nonce() {
356                warn!("Internal Issue: Packing pool has inconsistent tranaction with nonce pool.");
357            } else if removed_tx.nonce() == tx.nonce() {
358                // TODO: remove the lowest nonce makes the rest nonce
359                info!("a ready tx is garbage-collected");
360                GC_READY_COUNTER.inc(1);
361            }
362        }
363
364        ret
365    }

378    pub fn recalculate_readiness_with_local_info(
379        &mut self, addr: &AddressWithSpace, nonce: U256, balance: U256,
380    ) -> Option<Arc<SignedTransaction>> {
381        let bucket = match self.buckets.get_mut(addr) {
382            Some(bucket) => bucket,
383            None => {
384                trace!(
385                    "txpool::packing readiness addr={:?} missing bucket",
386                    addr
387                );
388                return None;
389            }
390        };
391        let pack_info =
392            bucket.recalculate_readiness_with_local_info(nonce, balance);
393
394        let (first_tx, last_valid_nonce) = if let Some(info) = pack_info {
395            info
396        } else {
397            trace!(
398                "txpool::packing readiness addr={:?} no contiguous unpaid tx (nonce={:?}, balance={:?})",
399                addr, nonce, balance
400            );
401            // If cannot found such transaction, clear item in packing pool
402            let _ = self.packing_pool.in_space_mut(addr.space).remove(*addr);
403            return None;
404        };
405
406        trace!(
407            "txpool::packing readiness addr={:?} candidate window start_nonce={:?} last_valid_nonce={:?} first_tx_hash={:?}",
408            addr,
409            first_tx.nonce(),
410            last_valid_nonce,
411            first_tx.transaction.hash()
412        );
413
414        let first_valid_nonce = *first_tx.nonce();
415        let current_txs = if let Some(txs) = self
416            .packing_pool
417            .in_space(addr.space)
418            .get_transactions(addr)
419            .filter(|txs| txs.first().unwrap().nonce() <= &first_valid_nonce)
420        {
421            txs
422        } else {
423            // If one of the following condition happens, we organize a new
424            // batch
425            //  1. the packing batch is absent
426            //  2. the nonce of first valid transaction becomes smaller
427            // (unlikely happens unless execution revert)
428            let config = self.packing_pool.in_space(addr.space).config();
429            let batch =
430                bucket.make_packing_batch(first_tx, config, last_valid_nonce);
431            let _ = self.packing_pool.in_space_mut(addr.space).replace(batch);
432            return Some(first_tx.transaction.clone());
433        };
434
435        let current_first_nonce = *current_txs.first().unwrap().nonce();
436        let current_last_nonce = *current_txs.last().unwrap().nonce();
437        // There must be current_first_nonce <= first_valid_nonce
438        if current_first_nonce < first_valid_nonce {
439            let dropped = self
440                .packing_pool
441                .in_space_mut(addr.space)
442                .split_off_prefix(*addr, &first_valid_nonce);
443            if !dropped.is_empty() {
444                trace!(
445                    "txpool::packing readiness addr={:?} dropped {} txs with nonce < {:?}",
446                    addr,
447                    dropped.len(),
448                    first_valid_nonce
449                );
450            }
451        }
452
453        if current_last_nonce > last_valid_nonce {
454            let dropped = self
455                .packing_pool
456                .in_space_mut(addr.space)
457                .split_off_suffix(*addr, &(last_valid_nonce + 1));
458            if !dropped.is_empty() {
459                trace!(
460                    "txpool::packing readiness addr={:?} dropped {} txs with nonce > {:?}",
461                    addr,
462                    dropped.len(),
463                    last_valid_nonce
464                );
465            }
466        } else if current_last_nonce < last_valid_nonce {
467            for tx in bucket.iter_tx_by_nonce(&current_last_nonce) {
468                if tx.nonce() > &last_valid_nonce {
469                    break;
470                }
471                let (evicted, res) = self
472                    .packing_pool
473                    .in_space_mut(addr.space)
474                    .insert(tx.transaction.clone());
475                match &res {
476                    Ok(_) => {
477                        trace!(
478                            "txpool::packing readiness addr={:?} promoted tx hash={:?} nonce={:?} evicted={}",
479                            addr,
480                            tx.transaction.hash(),
481                            tx.nonce(),
482                            evicted.len()
483                        );
484                    }
485                    Err(e) => {
486                        trace!(
487                            "txpool::packing readiness addr={:?} failed to promote tx hash={:?} nonce={:?} err={:?}",
488                            addr,
489                            tx.transaction.hash(),
490                            tx.nonce(),
491                            e
492                        );
493                    }
494                }
495
496                if res.is_err() {
497                    break;
498                }
499            }
500        }
501
502        return Some(first_tx.transaction.clone());
503    }

515    pub fn get_pending_transactions<'a>(
516        &'a self, addr: &AddressWithSpace, start_nonce: &U256,
517        local_nonce: &U256, local_balance: &U256,
518    ) -> (Vec<&'a TxWithReadyInfo>, Option<PendingReason>) {
519        match self.buckets.get(addr) {
520            Some(bucket) => {
521                let pending_txs = bucket.get_pending_transactions(start_nonce);
522                let pending_reason = pending_txs.first().and_then(|tx| {
523                    bucket.check_pending_reason_with_local_info(
524                        *local_nonce,
525                        *local_balance,
526                        &tx.transaction.as_ref(),
527                    )
528                });
529                (pending_txs, pending_reason)
530            }
531            None => (Vec::new(), None),
532        }
533    }

535    pub fn eth_content<F>(
536        &self, space: Option<Space>, get_nonce_and_balance: F,
537    ) -> (
538        BTreeMap<AddressWithSpace, BTreeMap<U256, Arc<SignedTransaction>>>,
539        BTreeMap<AddressWithSpace, BTreeMap<U256, Arc<SignedTransaction>>>,
540    )
541    where F: Fn(&AddressWithSpace) -> (U256, U256) {
542        let mut total_pending = BTreeMap::new();
543        let mut total_queued = BTreeMap::new();
544        for (addr, pool) in self.buckets.iter() {
545            if let Some(addr_space) = space {
546                if addr_space != addr.space {
547                    continue;
548                }
549            }
550            let (nonce, balance) = get_nonce_and_balance(addr);
551            let (pending, queued) = pool.eth_content(nonce, balance);
552            total_pending.insert(*addr, pending);
553            total_queued.insert(*addr, queued);
554        }
555        (total_pending, total_queued)
556    }

584    pub fn last_succ_nonce(
585        &self, addr: AddressWithSpace, from_nonce: U256,
586    ) -> Option<U256> {
587        let bucket = self.buckets.get(&addr)?;
588        let mut next_nonce = from_nonce;
589        loop {
590            let nonce = bucket.succ_nonce(&next_nonce);
591            if nonce.is_none() {
592                break;
593            }
594            if nonce.unwrap() > next_nonce {
595                break;
596            }
597            next_nonce += 1.into();
598        }
599        Some(next_nonce)
600    }

641    pub fn pending_tx_number<F>(
642        &self, space: Option<Space>, get_nonce_and_balance: F,
643    ) -> u64
644    where F: Fn(&AddressWithSpace) -> (U256, U256) {
645        self.buckets
646            .iter()
647            .filter(|item| {
648                if let Some(space) = space {
649                    item.0.space == space
650                } else {
651                    true
652                }
653            })
654            .map(|(addr, nonce_pool)| {
655                let (nonce, balance) = get_nonce_and_balance(addr);
656                if let Some((tx, nonce)) = nonce_pool
657                    .recalculate_readiness_with_local_info(nonce, balance)
658                {
659                    (nonce - tx.nonce() + 1).as_u64()
660                } else {
661                    0u64
662                }
663            })
664            .sum()
665    }
