77pub fn start_pos_consensus(
78    config: &NodeConfig, network: Arc<NetworkService>,
79    protocol_config: ProtocolConfiguration,
80    own_pos_public_key: Option<(ConsensusPublicKey, ConsensusVRFPublicKey)>,
81    pos_genesis_state: GenesisPosState,
82    consensus_network_receiver: ConsensusNetworkReceivers,
83    mempool_network_receiver: MemPoolNetworkReceivers,
84    test_command_receiver: channel::Receiver<TestCommand>,
85    hsb_protocol: Arc<HotStuffSynchronizationProtocol>,
86) -> PosDropHandle {
87    crash_handler::setup_panic_handler();
88
89    let mut logger = diem_logger::Logger::new();
90    logger
91        .channel_size(config.logger.chan_size)
92        .is_async(config.logger.is_async)
93        .level(config.logger.level)
94        .read_env();
95    if let Some(log_file) = config.logger.file.clone() {
96        if let Some(parent) = log_file.parent() {
97            fs::create_dir_all(parent).expect(&format!(
98                "error creating PoS log file directory: parent={:?}",
99                parent
100            ));
101        }
102        let writer = match config.logger.rotation_count {
103            Some(count) => Box::new(RollingFileWriter::new(
104                log_file,
105                count,
106                config.logger.rotation_file_size_mb.unwrap_or(500),
107            ))
108                as Box<dyn Writer + Send + Sync + 'static>,
109            None => Box::new(FileWriter::new(log_file))
110                as Box<dyn Writer + Send + Sync + 'static>,
111        };
112        logger.printer(writer);
113    }
114    let _logger = Some(logger.build());
115
116    // Let's now log some important information, since the logger is set up
117    diem_info!(config = config, "Loaded Pos config");
118
119    if fail::has_failpoints() {
120        diem_warn!("Failpoints is enabled");
121        if let Some(failpoints) = &config.failpoints {
122            for (point, actions) in failpoints {
123                fail::cfg(point, actions)
124                    .expect("fail to set actions for failpoint");
125            }
126        }
127    } else if config.failpoints.is_some() {
128        diem_warn!("failpoints is set in config, but the binary doesn't compile with this feature");
129    }
130
131    setup_pos_environment(
132        &config,
133        network,
134        protocol_config,
135        own_pos_public_key,
136        pos_genesis_state,
137        consensus_network_receiver,
138        mempool_network_receiver,
139        test_command_receiver,
140        hsb_protocol,
141    )
142}

144pub fn setup_pos_environment(
145    node_config: &NodeConfig, network: Arc<NetworkService>,
146    protocol_config: ProtocolConfiguration,
147    own_pos_public_key: Option<(ConsensusPublicKey, ConsensusVRFPublicKey)>,
148    pos_genesis_state: GenesisPosState,
149    consensus_network_receiver: ConsensusNetworkReceivers,
150    mempool_network_receiver: MemPoolNetworkReceivers,
151    test_command_receiver: channel::Receiver<TestCommand>,
152    hsb_protocol: Arc<HotStuffSynchronizationProtocol>,
153) -> PosDropHandle {
154    let mut instant = Instant::now();
155    let (pos_ledger_db, db_rw) = DbReaderWriter::wrap(
156        PosLedgerDB::open(
157            &node_config.storage.dir(),
158            false, /* readonly */
159            node_config.storage.rocksdb_config,
160        )
161        .expect("DB should open."),
162    );
163
164    // If the DB hasn't been bootstrapped yet, commit genesis.
165    maybe_bootstrap(
166        &db_rw,
167        Some(PivotBlockDecision {
168            block_hash: protocol_config.pos_genesis_pivot_decision,
169            height: 0,
170        }),
171        pos_genesis_state.initial_seed.as_bytes().to_vec(),
172        pos_genesis_state
173            .initial_nodes
174            .into_iter()
175            .map(|node| {
176                (NodeID::new(node.bls_key, node.vrf_key), node.voting_power)
177            })
178            .collect(),
179        pos_genesis_state.initial_committee,
180    )
181    .expect("Db-bootstrapper should not fail.");
182
183    debug!(
184        "Storage service started in {} ms",
185        instant.elapsed().as_millis()
186    );
187
188    // Channel for consensus to notify mempool of committed transactions.
189    let (mempool_commit_sender, mempool_commit_receiver) =
190        channel(INTRA_NODE_CHANNEL_BUFFER_SIZE);
191
192    let (consensus_to_mempool_sender, consensus_requests) =
193        channel(INTRA_NODE_CHANNEL_BUFFER_SIZE);
194
195    let network_sender = NetworkSender {
196        network,
197        protocol_handler: hsb_protocol,
198    };
199
200    let (mp_client_sender, mp_client_events) =
201        channel(AC_SMP_CHANNEL_BUFFER_SIZE);
202
203    let db_with_cache = Arc::new(CachedPosLedgerDB::new(db_rw));
204
205    instant = Instant::now();
206    let mempool = diem_mempool::bootstrap(
207        node_config,
208        db_with_cache.clone(),
209        network_sender.clone(),
210        mempool_network_receiver,
211        mp_client_events,
212        consensus_requests,
213        mempool_commit_receiver,
214    );
215    debug!("Mempool started in {} ms", instant.elapsed().as_millis());
216
217    // Initialize and start consensus.
218    instant = Instant::now();
219    debug!("own_pos_public_key: {:?}", own_pos_public_key);
220    let (consensus_runtime, pow_handler, stopped, consensus_db) =
221        start_consensus(
222            node_config,
223            network_sender,
224            consensus_network_receiver,
225            consensus_to_mempool_sender,
226            mempool_commit_sender.clone(),
227            node_config.consensus.mempool_commit_timeout_ms,
228            pos_ledger_db.clone(),
229            db_with_cache.clone(),
230            own_pos_public_key.map_or_else(
231                || AccountAddress::random(),
232                |public_key| {
233                    from_consensus_public_key(&public_key.0, &public_key.1)
234                },
235            ),
236            mp_client_sender.clone(),
237            test_command_receiver,
238            protocol_config.pos_started_as_voter,
239        );
240    debug!("Consensus started in {} ms", instant.elapsed().as_millis());
241
242    PosDropHandle {
243        pow_handler,
244        _consensus_runtime: consensus_runtime,
245        stopped,
246        _mempool: mempool,
247        pos_ledger_db,
248        cached_db: db_with_cache,
249        consensus_db,
250        tx_sender: mp_client_sender,
251    }
252}
253
254impl Drop for PosDropHandle {
255    fn drop(&mut self) {
256        debug!("Drop PosDropHandle");
257        self.stopped.store(true, Ordering::SeqCst);
258        self.pow_handler.stop();
259    }
