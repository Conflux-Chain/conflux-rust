22pub struct TxnsLog {
23    txns: Vec<(
24        AccountAddress,
25        HashValue,
26        Option<String>,
27        Option<SystemTime>,
28    )>,
29}
30
31impl TxnsLog {
32    pub fn new() -> Self { Self { txns: vec![] } }
33
34    pub fn new_txn(account: AccountAddress, hash: HashValue) -> Self {
35        Self {
36            txns: vec![(account, hash, None, None)],
37        }
38    }
39
40    pub fn add(&mut self, account: AccountAddress, hash: HashValue) {
41        self.txns.push((account, hash, None, None));
42    }
43
44    pub fn add_with_status(
45        &mut self, account: AccountAddress, hash: HashValue, status: &str,
46    ) {
47        self.txns
48            .push((account, hash, Some(status.to_string()), None));
49    }
50
51    pub fn add_full_metadata(
52        &mut self, account: AccountAddress, hash: HashValue, status: &str,
53        timestamp: Option<SystemTime>,
54    ) {
55        self.txns
56            .push((account, hash, Some(status.to_string()), timestamp));
57    }
58}
59
60impl fmt::Display for TxnsLog {
61    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
62        let mut txns = "".to_string();
63
64        for (account, hash, status, timestamp) in self.txns.iter() {
65            let mut txn = format!("{}:{}", account, hash);
66            if let Some(status) = status {
67                txn += &format!(":{}", status)
68            }
69            if let Some(timestamp) = timestamp {
70                txn += &format!(":{:?}", timestamp)
71            }
72
73            txns += &format!("{} ", txn);
74        }
75
76        write!(f, "{}", txns)
77    }
78}

80#[derive(Schema)]
81pub struct LogSchema<'a> {
82    name: LogEntry,
83    event: Option<LogEvent>,
84    #[schema(debug)]
85    error: Option<&'a Error>,
86    #[schema(display)]
87    peer: Option<&'a PeerNetworkId>,
88    is_upstream_peer: Option<bool>,
89    #[schema(display)]
90    reconfig_update: Option<OnChainConfigPayload>,
91    #[schema(display)]
92    txns: Option<TxnsLog>,
93    account: Option<AccountAddress>,
94    #[schema(display)]
95    consensus_msg: Option<&'a ConsensusRequest>,
96    #[schema(display)]
97    state_sync_msg: Option<&'a CommitNotification>,
98    network_level: Option<usize>,
99    upstream_network: Option<&'a NetworkId>,
100    #[schema(debug)]
101    batch_id: Option<&'a BatchId>,
102    backpressure: Option<bool>,
103}
104
105impl<'a> LogSchema<'a> {
106    pub fn new(name: LogEntry) -> Self { Self::new_event(name, None) }
107
108    pub fn event_log(name: LogEntry, event: LogEvent) -> Self {
109        Self::new_event(name, Some(event))
110    }
111
112    pub fn new_event(name: LogEntry, event: Option<LogEvent>) -> Self {
113        Self {
114            name,
115            event,
116            error: None,
117            peer: None,
118            is_upstream_peer: None,
119            reconfig_update: None,
120            account: None,
121            txns: None,
122            consensus_msg: None,
123            state_sync_msg: None,
124            network_level: None,
125            upstream_network: None,
126            batch_id: None,
127            backpressure: None,
128        }
129    }
130}
