34pub struct Mempool {
35    // Stores the metadata of all transactions in mempool (of all states).
36    pub transactions: TransactionStore,
37
38    pub system_transaction_timeout: Duration,
39}
40
41impl Mempool {
42    pub fn new(config: &NodeConfig) -> Self {
43        Mempool {
44            transactions: TransactionStore::new(&config.mempool),
45            system_transaction_timeout: Duration::from_secs(
46                config.mempool.system_transaction_timeout_secs,
47            ),
48        }
49    }
50
51    /// This function will be called once the transaction has been stored.
52    pub(crate) fn remove_transaction(&mut self, hash: HashValue) {
53        self.transactions.commit_transaction(hash);
54    }
55
56    /// Used to add a transaction to the Mempool.
57    /// Performs basic validation: checks account's sequence number.
58    pub(crate) fn add_txn(
59        &mut self, txn: SignedTransaction, timeline_state: TimelineState,
60    ) -> MempoolStatus {
61        diem_trace!(LogSchema::new(LogEntry::AddTxn)
62            .txns(TxnsLog::new_txn(txn.sender(), txn.hash())),);
63
64        let expiration_time = std::time::SystemTime::now()
65            .duration_since(std::time::UNIX_EPOCH)
66            .expect("System time is before UNIX_EPOCH")
67            + self.system_transaction_timeout;
68
69        let txn_info =
70            MempoolTransaction::new(txn, expiration_time, timeline_state);
71
72        self.transactions.insert(txn_info)
73    }
74
75    /// Fetches next block of transactions for consensus.
76    /// `batch_size` - size of requested block.
77    /// `seen_txns` - transactions that were sent to Consensus but were not
78    /// committed yet,  mempool should filter out such transactions.
79    #[allow(clippy::explicit_counter_loop)]
80    pub(crate) fn get_block(
81        &mut self, _batch_size: u64, mut seen: HashSet<TxnPointer>,
82        pos_state: &PosState, validators: ValidatorVerifier,
83    ) -> Vec<SignedTransaction> {
84        let mut block = vec![];
85        let mut block_log = TxnsLog::new();
86        let seen_size = seen.len();
87        let mut txn_walked = 0usize;
88        for txn in self.transactions.iter() {
89            txn_walked += 1;
90            if seen.contains(&TxnPointer::from(txn)) {
91                continue;
92            }
93            let validate_result = match txn.txn.payload() {
94                TransactionPayload::Election(election_payload) => {
95                    pos_state.validate_election(election_payload)
96                }
97                TransactionPayload::PivotDecision(_) => {
98                    seen.insert((txn.get_sender(), txn.get_hash()));
99                    continue;
100                }
101                TransactionPayload::Dispute(dispute_payload) => {
102                    // TODO(lpl): Only dispute a node once.
103                    pos_state.validate_dispute(dispute_payload).and(
104                        verify_dispute(dispute_payload)
105                            .then_some(())
106                            .ok_or(anyhow::anyhow!("invalid dispute")),
107                    )
108                }
109                _ => {
110                    continue;
111                }
112            };
113            if validate_result.is_ok() {
114                block.push(txn.txn.clone());
115                block_log.add(txn.get_sender(), txn.get_hash());
116                seen.insert((txn.get_sender(), txn.get_hash()));
117            }
118        }
119        let mut max_pivot_height = 0;
120        let mut chosen_pivot_tx = None;
121        // iterate all pivot decision transaction
122        for pivot_decision_set in self.transactions.iter_pivot_decision() {
123            let mut pivot_decision_opt = None;
124            diem_debug!("get_block: 0 {:?}", pivot_decision_set.len());
125            for (account, hash) in pivot_decision_set.iter() {
126                if validators.get_public_key(account).is_some() {
127                    if pivot_decision_opt.is_none() {
128                        if let Some(txn) = self.transactions.get(hash) {
129                            pivot_decision_opt = Some(txn);
130                        }
131                    }
132                }
133            }
134            diem_debug!("get_block: 1 {:?}", pivot_decision_opt);
135            if validators
136                .check_voting_power(
137                    pivot_decision_set.iter().map(|(addr, _)| addr),
138                )
139                .is_ok()
140            {
141                let pivot_decision = pivot_decision_opt.unwrap();
142                let pivot_height = match pivot_decision.payload() {
143                    TransactionPayload::PivotDecision(decision) => {
144                        decision.height
145                    }
146                    _ => unreachable!(),
147                };
148                if pivot_height > max_pivot_height
149                    && pivot_height > pos_state.pivot_decision().height
150                {
151                    max_pivot_height = pivot_height;
152                    chosen_pivot_tx = Some(pivot_decision);
153                }
154            }
155            diem_debug!("get_block: 2 {:?}", chosen_pivot_tx);
156        }
157        if let Some(tx) = chosen_pivot_tx {
158            let pivot_decision_hash = match tx.payload() {
159                TransactionPayload::PivotDecision(decision) => decision.hash(),
160                _ => unreachable!(),
161            };
162            // aggregate signatures
163            let txn_hashes =
164                self.transactions.get_pivot_decisions(&pivot_decision_hash);
165            let senders: Vec<AccountAddress> =
166                validators.get_ordered_account_addresses_iter().collect();
167            let mut signatures = vec![];
168            for hash in &txn_hashes {
169                if let Some(txn) = self.transactions.get(hash) {
170                    match txn.authenticator() {
171                        TransactionAuthenticator::BLS { signature, .. } => {
172                            if let Ok(index) =
173                                senders.binary_search(&txn.sender())
174                            {
175                                signatures.push((signature, index));
176                            }
177                        }
178                        _ => unreachable!(),
179                    }
180                }
181            }
182            let new_tx =
183                SignedTransaction::new_multisig(tx.raw_txn(), signatures);
184            block_log.add(new_tx.sender(), new_tx.hash());
185            block.push(new_tx);
186        }
187
188        diem_debug!(
189            LogSchema::new(LogEntry::GetBlock).txns(block_log),
190            seen_consensus = seen_size,
191            walked = txn_walked,
192            seen_after = seen.len(),
193            result_size = block.len(),
194            block_size = block.len()
195        );
196        block
197    }

199    /// Periodic core mempool garbage collection.
200    /// Removes all expired transactions.
201    pub(crate) fn gc(&mut self) { self.transactions.gc_by_system_ttl(); }
202
203    /// Garbage collection based on client-specified expiration time.
204    pub(crate) fn gc_by_expiration_time(&mut self, block_time: Duration) {
205        self.transactions.gc_by_expiration_time(block_time);
206    }
207
208    /// Read `count` transactions from timeline since `timeline_id`.
209    /// Returns block of transactions and new last_timeline_id.
210    pub(crate) fn read_timeline(
211        &mut self, timeline_id: u64, count: usize,
212    ) -> (Vec<SignedTransaction>, u64) {
213        self.transactions.read_timeline(timeline_id, count)
214    }
215
216    /// Read transactions from timeline from `start_id` (exclusive) to `end_id`
217    /// (inclusive).
218    pub(crate) fn timeline_range(
219        &mut self, start_id: u64, end_id: u64,
220    ) -> Vec<SignedTransaction> {
221        self.transactions.timeline_range(start_id, end_id)
222    }
