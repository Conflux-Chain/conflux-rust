31/// TransactionStore is in-memory storage for all transactions in mempool.
32pub struct TransactionStore {
33    // normal transactions
34    transactions: AccountTransactions,
35    // pivot decision helper structure
36    pivot_decisions: HashMap<HashValue, HashSet<(AccountAddress, HashValue)>>,
37
38    // TTLIndex based on client-specified expiration time
39    expiration_time_index: TTLIndex,
40    // TTLIndex based on system expiration time
41    // we keep it separate from `expiration_time_index` so Mempool can't be
42    // clogged  by old transactions even if it hasn't received commit
43    // callbacks for a while
44    system_ttl_index: TTLIndex,
45    timeline_index: TimelineIndex,
46
47    // configuration
48    _capacity: usize,
49}
50
51pub type PivotDecisionIter<'a> =
52    Values<'a, HashValue, HashSet<(AccountAddress, HashValue)>>;
53
54impl TransactionStore {
55    pub(crate) fn new(config: &MempoolConfig) -> Self {
56        Self {
57            // main DS
58            transactions: AccountTransactions::new(),
59            pivot_decisions: HashMap::new(),
60
61            // various indexes
62            system_ttl_index: TTLIndex::new(Box::new(
63                |t: &MempoolTransaction| t.expiration_time,
64            )),
65            expiration_time_index: TTLIndex::new(Box::new(
66                |t: &MempoolTransaction| {
67                    Duration::from_secs(t.txn.expiration_timestamp_secs())
68                },
69            )),
70            timeline_index: TimelineIndex::new(),
71
72            // configuration
73            _capacity: config.capacity,
74        }
75    }
76
77    /// Fetch transaction by account address + hash.
78    pub(crate) fn get(&self, hash: &HashValue) -> Option<SignedTransaction> {
79        if let Some(txn) = self.transactions.get(hash) {
80            return Some(txn.txn.clone());
81        }
82        None
83    }
84
85    /// Fetch pivot decisions by pivot hash.
86    pub(crate) fn get_pivot_decisions(
87        &self, hash: &HashValue,
88    ) -> Vec<HashValue> {
89        if let Some(decisions) = self.pivot_decisions.get(hash) {
90            decisions
91                .iter()
92                .map(|(_, tx_hash)| tx_hash.clone())
93                .collect::<_>()
94        } else {
95            vec![]
96        }
97    }
98
99    /// Insert transaction into TransactionStore. Performs validation checks and
100    /// updates indexes.
101    pub(crate) fn insert(
102        &mut self, mut txn: MempoolTransaction,
103    ) -> MempoolStatus {
104        let address = txn.get_sender();
105        let hash = txn.get_hash();
106        let has_tx = self.get(&hash).is_some();
107
108        if has_tx {
109            return MempoolStatus::new(MempoolStatusCode::Accepted);
110        }
111
112        self.timeline_index.insert(&mut txn);
113
114        // TODO(linxi): evict transaction when mempool is full
115
116        // insert into storage and other indexes
117        self.system_ttl_index.insert(&txn);
118        self.expiration_time_index.insert(&txn);
119
120        let payload = txn.txn.clone().into_raw_transaction().into_payload();
121        if let TransactionPayload::PivotDecision(pivot_decision) = payload {
122            let pivot_decision_hash = pivot_decision.hash();
123            self.pivot_decisions
124                .entry(pivot_decision_hash)
125                .or_insert_with(HashSet::new);
126            if let Some(account_decision) =
127                self.pivot_decisions.get_mut(&pivot_decision_hash)
128            {
129                diem_debug!("txpool::insert pivot {:?}", hash);
130                account_decision.insert((address, hash));
131            }
132            self.transactions.insert(hash, txn, true);
133        } else {
134            self.transactions.insert(hash, txn, false);
135        }
136        diem_debug!(
137            LogSchema::new(LogEntry::AddTxn)
138                .txns(TxnsLog::new_txn(address, hash)),
139            hash = hash,
140            has_tx = has_tx
141        );
142
143        MempoolStatus::new(MempoolStatusCode::Accepted)
144    }

146    /// Handles transaction commit: deletes the transaction and cleans up
147    /// its entries in the timeline and TTL indexes.
148    pub(crate) fn commit_transaction(
149        &mut self, _account: &AccountAddress, hash: HashValue,
150    ) {
151        let mut txns_log = TxnsLog::new();
152        if let Some(transaction) = self.transactions.remove(&hash) {
153            txns_log.add(transaction.get_sender(), transaction.get_hash());
154            self.index_remove(&transaction);
155            // handle pivot decision
156            let payload = transaction.txn.into_raw_transaction().into_payload();
157            if let TransactionPayload::PivotDecision(pivot_decision) = payload {
158                let pivot_decision_hash = pivot_decision.hash();
159                if let Some(indices) =
160                    self.pivot_decisions.remove(&pivot_decision_hash)
161                {
162                    for (_, hash) in indices {
163                        if let Some(txn) = self.transactions.remove(&hash) {
164                            txns_log.add(txn.get_sender(), txn.get_hash());
165                            self.index_remove(&txn);
166                        }
167                    }
168                }
169            }
170        }
171        diem_debug!(LogSchema::new(LogEntry::CleanCommittedTxn).txns(txns_log));
172    }
173
174    pub(crate) fn reject_transaction(
175        &mut self, account: &AccountAddress, _hash: HashValue,
176    ) {
177        let mut txns_log = TxnsLog::new();
178        let mut hashes = Vec::new();
179        for txn in self.transactions.iter() {
180            if txn.get_sender() == *account {
181                txns_log.add(txn.get_sender(), txn.get_hash());
182                hashes.push(txn.get_hash());
183            }
184        }
185        for txn in self.transactions.iter_pivot_decision() {
186            if txn.get_sender() == *account {
187                txns_log.add(txn.get_sender(), txn.get_hash());
188                hashes.push(txn.get_hash());
189            }
190        }
191        for hash in hashes {
192            if let Some(txn) = self.transactions.remove(&hash) {
193                self.index_remove(&txn);
194            }
195        }
196        diem_debug!(LogSchema::new(LogEntry::CleanRejectedTxn).txns(txns_log));
197    }
198
199    /// Removes transaction from all indexes.
200    fn index_remove(&mut self, txn: &MempoolTransaction) {
201        self.system_ttl_index.remove(&txn);
202        self.expiration_time_index.remove(&txn);
203        self.timeline_index.remove(&txn);
204    }
205
206    /// Read `count` transactions from timeline since `timeline_id`.
207    /// Returns block of transactions and new last_timeline_id.
208    pub(crate) fn read_timeline(
209        &mut self, timeline_id: u64, count: usize,
210    ) -> (Vec<SignedTransaction>, u64) {
211        let mut batch = vec![];
212        let mut last_timeline_id = timeline_id;
213        for (_, hash) in self.timeline_index.read_timeline(timeline_id, count) {
214            if let Some(txn) = self.transactions.get(&hash) {
215                batch.push(txn.txn.clone());
216                if let TimelineState::Ready(timeline_id) = txn.timeline_state {
217                    last_timeline_id = timeline_id;
218                }
219            }
220        }
221        (batch, last_timeline_id)
222    }
223
224    pub(crate) fn timeline_range(
225        &mut self, start_id: u64, end_id: u64,
226    ) -> Vec<SignedTransaction> {
227        self.timeline_index
228            .timeline_range(start_id, end_id)
229            .iter()
230            .filter_map(|(_, hash)| {
231                self.transactions.get(hash).map(|txn| txn.txn.clone())
232            })
233            .collect()
234    }

236    /// Garbage collect old transactions.
237    pub(crate) fn gc_by_system_ttl(&mut self) {
238        let now = diem_infallible::duration_since_epoch();
239        self.gc(now, true);
240    }
241
242    /// Garbage collect old transactions based on client-specified expiration
243    /// time.
244    pub(crate) fn gc_by_expiration_time(&mut self, block_time: Duration) {
245        self.gc(block_time, false);
246    }
247
248    fn gc(&mut self, now: Duration, by_system_ttl: bool) {
249        let (index, log_event) = if by_system_ttl {
250            (&mut self.system_ttl_index, LogEvent::SystemTTLExpiration)
251        } else {
252            (&mut self.expiration_time_index, LogEvent::ClientExpiration)
253        };
254
255        let mut gc_txns = index.gc(now);
256        // sort the expired txns by order of sequence number per account
257        gc_txns.sort_by_key(|key| (key.address, key.hash));
258        let mut gc_iter = gc_txns.iter().peekable();
259
260        let mut gc_txns_log = TxnsLog::new();
261        while let Some(key) = gc_iter.next() {
262            if let Some(txn) = self.transactions.remove(&key.hash) {
263                gc_txns_log.add(txn.get_sender(), txn.get_hash());
264                self.index_remove(&txn);
265                if let TransactionPayload::PivotDecision(pivot_decision) =
266                    txn.txn.into_raw_transaction().into_payload()
267                {
268                    self.pivot_decisions.remove(&pivot_decision.hash());
269                }
270            }
271        }
272
273        diem_debug!(LogSchema::event_log(LogEntry::GCRemoveTxns, log_event)
274            .txns(gc_txns_log));
275    }
276
277    pub(crate) fn iter(&self) -> AccountTransactionIter<'_> {
278        self.transactions.iter()
279    }
280
281    pub(crate) fn iter_pivot_decision(&self) -> PivotDecisionIter<'_> {
282        self.pivot_decisions.values()
283    }
