28/// TransactionStore is in-memory storage for all transactions in mempool.
29pub struct TransactionStore {
30    // normal transactions
31    transactions: AccountTransactions,
32    // pivot decision helper structure
33    pivot_decisions: HashMap<HashValue, HashSet<(AccountAddress, HashValue)>>,
34
35    // TTL keyed on `MempoolTransaction.expiration_time = add_time +
36    // system_transaction_timeout`. Every txn is evicted after the timeout,
37    // regardless of its payload — keeps old txns from clogging the mempool
38    // when commit callbacks are delayed.
39    system_ttl_index: TTLIndex,
40    timeline_index: TimelineIndex,
41}
42
43pub type PivotDecisionIter<'a> =
44    Values<'a, HashValue, HashSet<(AccountAddress, HashValue)>>;
45
46impl TransactionStore {
47    pub(crate) fn new(_config: &MempoolConfig) -> Self {
48        Self {
49            // main DS
50            transactions: AccountTransactions::new(),
51            pivot_decisions: HashMap::new(),
52
53            // various indexes
54            system_ttl_index: TTLIndex::new(Box::new(
55                |t: &MempoolTransaction| t.expiration_time,
56            )),
57            timeline_index: TimelineIndex::new(),
58        }
59    }
60
61    /// Fetch transaction by account address + hash.
62    pub(crate) fn get(&self, hash: &HashValue) -> Option<SignedTransaction> {
63        if let Some(txn) = self.transactions.get(hash) {
64            return Some(txn.txn.clone());
65        }
66        None
67    }
68
69    /// Fetch pivot decisions by pivot hash.
70    pub(crate) fn get_pivot_decisions(
71        &self, hash: &HashValue,
72    ) -> Vec<HashValue> {
73        if let Some(decisions) = self.pivot_decisions.get(hash) {
74            decisions
75                .iter()
76                .map(|(_, tx_hash)| tx_hash.clone())
77                .collect::<_>()
78        } else {
79            vec![]
80        }
81    }
82
83    /// Insert transaction into TransactionStore. Performs validation checks and
84    /// updates indexes.
85    pub(crate) fn insert(
86        &mut self, mut txn: MempoolTransaction,
87    ) -> MempoolStatus {
88        let address = txn.get_sender();
89        let hash = txn.get_hash();
90        let has_tx = self.get(&hash).is_some();
91
92        if has_tx {
93            return MempoolStatus::new(MempoolStatusCode::Accepted);
94        }
95
96        self.timeline_index.insert(&mut txn);
97        self.system_ttl_index.insert(&txn);
98
99        let payload = txn.txn.clone().into_raw_transaction().into_payload();
100        if let TransactionPayload::PivotDecision(pivot_decision) = payload {
101            let pivot_decision_hash = pivot_decision.hash();
102            self.pivot_decisions
103                .entry(pivot_decision_hash)
104                .or_insert_with(HashSet::new);
105            if let Some(account_decision) =
106                self.pivot_decisions.get_mut(&pivot_decision_hash)
107            {
108                diem_debug!("txpool::insert pivot {:?}", hash);
109                account_decision.insert((address, hash));
110            }
111            self.transactions.insert(hash, txn, true);
112        } else {
113            self.transactions.insert(hash, txn, false);
114        }
115        diem_debug!(
116            LogSchema::new(LogEntry::AddTxn)
117                .txns(TxnsLog::new_txn(address, hash)),
118            hash = hash,
119            has_tx = has_tx
120        );
121
122        MempoolStatus::new(MempoolStatusCode::Accepted)
123    }
124
125    /// Handles transaction commit: deletes the transaction and cleans up
126    /// its entries in the timeline and TTL indexes.
127    pub(crate) fn commit_transaction(&mut self, hash: HashValue) {
128        let mut txns_log = TxnsLog::new();
129        if let Some(transaction) = self.transactions.remove(&hash) {
130            txns_log.add(transaction.get_sender(), transaction.get_hash());
131            self.index_remove(&transaction);
132            // handle pivot decision
133            let payload = transaction.txn.into_raw_transaction().into_payload();
134            if let TransactionPayload::PivotDecision(pivot_decision) = payload {
135                let pivot_decision_hash = pivot_decision.hash();
136                if let Some(indices) =
137                    self.pivot_decisions.remove(&pivot_decision_hash)
138                {
139                    for (_, hash) in indices {
140                        if let Some(txn) = self.transactions.remove(&hash) {
141                            txns_log.add(txn.get_sender(), txn.get_hash());
142                            self.index_remove(&txn);
143                        }
144                    }
145                }
146            }
147        }
148        diem_debug!(LogSchema::new(LogEntry::CleanCommittedTxn).txns(txns_log));
149    }
150
151    /// Removes transaction from all indexes.
152    fn index_remove(&mut self, txn: &MempoolTransaction) {
153        self.system_ttl_index.remove(&txn);
154        self.timeline_index.remove(&txn);
155    }
156
157    /// Read `count` transactions from timeline since `timeline_id`.
158    /// Returns block of transactions and new last_timeline_id.
159    pub(crate) fn read_timeline(
160        &mut self, timeline_id: u64, count: usize,
161    ) -> (Vec<SignedTransaction>, u64) {
162        let mut batch = vec![];
163        let mut last_timeline_id = timeline_id;
164        for (_, hash) in self.timeline_index.read_timeline(timeline_id, count) {
165            if let Some(txn) = self.transactions.get(&hash) {
166                batch.push(txn.txn.clone());
167                if let TimelineState::Ready(timeline_id) = txn.timeline_state {
168                    last_timeline_id = timeline_id;
169                }
170            }
171        }
172        (batch, last_timeline_id)
173    }
174
175    pub(crate) fn timeline_range(
176        &mut self, start_id: u64, end_id: u64,
177    ) -> Vec<SignedTransaction> {
178        self.timeline_index
179            .timeline_range(start_id, end_id)
180            .iter()
181            .filter_map(|(_, hash)| {
182                self.transactions.get(hash).map(|txn| txn.txn.clone())
183            })
184            .collect()
185    }
186
187    /// Garbage collect old transactions by system TTL.
188    pub(crate) fn gc_by_system_ttl(&mut self) {
189        let now = std::time::SystemTime::now()
190            .duration_since(std::time::UNIX_EPOCH)
191            .expect("System time is before UNIX_EPOCH");
192
193        let mut gc_txns = self.system_ttl_index.gc(now);
194        gc_txns.sort_by_key(|key| (key.address, key.hash));
195
196        let mut gc_txns_log = TxnsLog::new();
197        for key in gc_txns.iter() {
198            if let Some(txn) = self.transactions.remove(&key.hash) {
199                gc_txns_log.add(txn.get_sender(), txn.get_hash());
200                self.index_remove(&txn);
201                if let TransactionPayload::PivotDecision(pivot_decision) =
202                    txn.txn.into_raw_transaction().into_payload()
203                {
204                    self.pivot_decisions.remove(&pivot_decision.hash());
205                }
206            }
207        }
208
209        diem_debug!(LogSchema::event_log(
210            LogEntry::GCRemoveTxns,
211            LogEvent::SystemTTLExpiration
212        )
213        .txns(gc_txns_log));
214    }
215
216    pub(crate) fn iter(&self) -> AccountTransactionIter<'_> {
217        self.transactions.iter()
218    }
219
220    pub(crate) fn iter_pivot_decision(&self) -> PivotDecisionIter<'_> {
221        self.pivot_decisions.values()
222    }
223}
