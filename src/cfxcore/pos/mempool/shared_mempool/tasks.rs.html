47/// Attempts broadcast to `peer` and schedules the next broadcast.
48pub(crate) fn execute_broadcast(
49    peer: NodeId, backoff: bool, smp: &mut SharedMempool,
50    scheduled_broadcasts: &mut FuturesUnordered<ScheduledBroadcast>,
51    broadcasting_peers: &mut HashSet<NodeId>, executor: Handle,
52) {
53    diem_trace!("execute_broadcast starts: peer={}", peer);
54    let peer_manager = &smp.peer_manager.clone();
55    peer_manager.execute_broadcast(peer.clone(), backoff, smp);
56    let schedule_backoff = peer_manager.is_backoff_mode(&peer);
57
58    let interval_ms = if schedule_backoff {
59        smp.config.shared_mempool_backoff_interval_ms
60    } else {
61        smp.config.shared_mempool_tick_interval_ms
62    };
63
64    if peer_manager.contains_peer(&peer) {
65        // Make sure we only has one broadcast task for one peer id.
66        broadcasting_peers.insert(peer);
67        scheduled_broadcasts.push(ScheduledBroadcast::new(
68            Instant::now() + Duration::from_millis(interval_ms),
69            peer,
70            schedule_backoff,
71            executor,
72        ));
73    } else {
74        // The peer has been disconnected,
75        // so it can be added again after reconnection.
76        broadcasting_peers.remove(&peer);
77    }
78    diem_trace!("execute_broadcast end: peer={}", peer);
79}

85/// Processes transactions directly submitted by client.
86pub(crate) async fn process_client_transaction_submission(
87    smp: SharedMempool, transaction: SignedTransaction,
88    callback: oneshot::Sender<Result<SubmissionStatus>>, timer: HistogramTimer,
89) {
90    timer.stop_and_record();
91    let _timer = counters::process_txn_submit_latency_timer(
92        counters::CLIENT_LABEL,
93        counters::CLIENT_LABEL,
94    );
95    let statuses = process_incoming_transactions(
96        &smp,
97        vec![transaction],
98        TimelineState::NotReady,
99    )
100    .await;
101    log_txn_process_results(&statuses, None);
102
103    if let Some(status) = statuses.get(0) {
104        if callback.send(Ok(status.1.clone())).is_err() {
105            diem_error!(LogSchema::event_log(
106                LogEntry::JsonRpc,
107                LogEvent::CallbackFail
108            ));
109            counters::CLIENT_CALLBACK_FAIL.inc();
110        }
111    }
112}

114/// Processes transactions from other nodes.
115pub(crate) async fn process_transaction_broadcast(
116    smp: SharedMempool, transactions: Vec<SignedTransaction>,
117    request_id: Vec<u8>, timeline_state: TimelineState, peer: NodeId,
118    timer: HistogramTimer,
119) {
120    diem_trace!("process_transaction_broadcast starts: peer={}", peer);
121    timer.stop_and_record();
122    let results = process_incoming_transactions(
123        &smp,
124        transactions.clone(),
125        timeline_state,
126    )
127    .await;
128    log_txn_process_results(&results, Some(peer.clone()));
129
130    let ack_response = gen_ack_response(request_id, results, &peer);
131    if let Err(e) = smp
132        .network_sender
133        .send_message_with_peer_id(&peer, &ack_response)
134    {
135        counters::network_send_fail_inc(counters::ACK_TXNS);
136        diem_error!(LogSchema::event_log(
137            LogEntry::BroadcastACK,
138            LogEvent::NetworkSendFail
139        )
140        .error(&e.into()));
141        return;
142    }
143    notify_subscribers(SharedMempoolNotification::ACK, &smp.subscribers);
144    diem_trace!("process_transaction_broadcast ends: peer={}", peer);
145}
146
147fn gen_ack_response(
148    request_id: Vec<u8>, results: Vec<SubmissionStatusBundle>, peer: &NodeId,
149) -> MempoolSyncMsg {
150    let mut backoff = false;
151    let mut retry = false;
152    for r in results.into_iter() {
153        let submission_status = r.1;
154        if submission_status.0.code == MempoolStatusCode::MempoolIsFull {
155            backoff = true;
156        }
157        if is_txn_retryable(submission_status) {
158            retry = true;
159        }
160
161        if backoff && retry {
162            break;
163        }
164    }
165
166    diem_trace!(
167        "request[{:?}] from peer[{:?}] retry[{:?}]",
168        request_id,
169        peer,
170        retry
171    );
172
173    MempoolSyncMsg::BroadcastTransactionsResponse {
174        request_id,
175        retry,
176        backoff,
177    }
178}
179
180fn is_txn_retryable(result: SubmissionStatus) -> bool {
181    result.0.code == MempoolStatusCode::MempoolIsFull
182}

184/// Submits a list of SignedTransaction to the local mempool
185/// and returns a vector containing AdmissionControlStatus.
186pub(crate) async fn process_incoming_transactions(
187    smp: &SharedMempool, transactions: Vec<SignedTransaction>,
188    timeline_state: TimelineState,
189) -> Vec<SubmissionStatusBundle> {
190    let mut statuses = vec![];
191
192    let start_storage_read = Instant::now();
193    // Track latency for storage read fetching sequence number
194    let storage_read_latency = start_storage_read.elapsed();
195    counters::PROCESS_TXN_BREAKDOWN_LATENCY
196        .with_label_values(&[counters::FETCH_SEQ_NUM_LABEL])
197        .observe(
198            storage_read_latency.as_secs_f64() / transactions.len() as f64,
199        );
200
201    // Track latency: VM validation
202    let vm_validation_timer = counters::PROCESS_TXN_BREAKDOWN_LATENCY
203        .with_label_values(&[counters::VM_VALIDATION_LABEL])
204        .start_timer();
205    // Filter out already received transactions, so we do not need to process
206    // them.
207    let transactions: Vec<SignedTransaction> = {
208        let mempool = smp.mempool.lock();
209        transactions
210            .into_iter()
211            .filter(|tx| mempool.transactions.get(&tx.hash()).is_none())
212            .collect()
213    };
214    let validation_results = transactions
215        .par_iter()
216        .map(|t| {
217            smp.validator
218                .read()
219                .validate_transaction(&t, smp.commited_pos_state.clone())
220        })
221        .collect::<Vec<_>>();
222    vm_validation_timer.stop_and_record();
223
224    {
225        let mut mempool = smp.mempool.lock();
226        for (idx, transaction) in transactions.into_iter().enumerate() {
227            match validation_results[idx] {
228                None => {
229                    let mempool_status =
230                        mempool.add_txn(transaction.clone(), timeline_state);
231                    statuses.push((transaction, (mempool_status, None)));
232                }
233                Some(validation_status) => {
234                    statuses.push((
235                        transaction,
236                        (
237                            MempoolStatus::new(MempoolStatusCode::VmError),
238                            Some(validation_status),
239                        ),
240                    ));
241                }
242            }
243        }
244    }
245    notify_subscribers(
246        SharedMempoolNotification::NewTransactions,
247        &smp.subscribers,
248    );
249    statuses
250}
251
252fn log_txn_process_results(
253    results: &[SubmissionStatusBundle], sender: Option<NodeId>,
254) {
255    let sender = match sender {
256        Some(peer) => peer,
257        None => {
258            return;
259        }
260    };
261    for (txn, (_mempool_status, maybe_vm_status)) in results.iter() {
262        if let Some(vm_status) = maybe_vm_status {
263            diem_trace!(
264                SecurityEvent::InvalidTransactionMempool,
265                failed_transaction = txn,
266                vm_status = vm_status,
267                sender = sender,
268            );
269        }
270    }
271}

277pub(crate) async fn process_committed_transactions(
278    mempool: Arc<Mutex<CoreMempool>>, req: CommitNotification,
279) {
280    let start_time = Instant::now();
281    diem_debug!(LogSchema::event_log(
282        LogEntry::StateSyncCommit,
283        LogEvent::Received
284    )
285    .state_sync_msg(&req));
286    counters::mempool_service_transactions(
287        counters::COMMIT_STATE_SYNC_LABEL,
288        req.transactions.len(),
289    );
290    commit_txns(&mempool, req.transactions, req.block_timestamp_usecs, false)
291        .await;
292    let result = if req.callback.send(Ok(CommitResponse::success())).is_err() {
293        diem_error!(LogSchema::event_log(
294            LogEntry::StateSyncCommit,
295            LogEvent::CallbackFail
296        ));
297        counters::REQUEST_FAIL_LABEL
298    } else {
299        counters::REQUEST_SUCCESS_LABEL
300    };
301    let latency = start_time.elapsed();
302    counters::mempool_service_latency(
303        counters::COMMIT_STATE_SYNC_LABEL,
304        result,
305        latency,
306    );
307}

309pub(crate) async fn process_consensus_request(
310    db: Arc<CachedPosLedgerDB>, mempool: &Mutex<CoreMempool>,
311    req: ConsensusRequest,
312) {
313    // Start latency timer
314    let start_time = Instant::now();
315    diem_debug!(
316        LogSchema::event_log(LogEntry::Consensus, LogEvent::Received)
317            .consensus_msg(&req)
318    );
319
320    let (resp, callback, counter_label) = match req {
321        ConsensusRequest::GetBlockRequest(
322            max_block_size,
323            transactions,
324            parent_block_id,
325            validators,
326            callback,
327        ) => {
328            let exclude_transactions: HashSet<TxnPointer> = transactions
329                .iter()
330                .map(|txn| (txn.sender, txn.hash))
331                .collect();
332            let mut txns;
333            {
334                let mut mempool = mempool.lock();
335                // gc before pulling block as extra protection against txns that
336                // may expire in consensus Note: this gc
337                // operation relies on the fact that consensus uses the system
338                // time to determine block timestamp
339                let curr_time = diem_infallible::duration_since_epoch();
340                mempool.gc_by_expiration_time(curr_time);
341                let block_size = cmp::max(max_block_size, 1);
342                let pos_state = db
343                    .get_pos_state(&parent_block_id)
344                    .expect("pos_state should exist");
345                txns = mempool.get_block(
346                    block_size,
347                    exclude_transactions,
348                    &pos_state,
349                    validators,
350                );
351            }
352            counters::mempool_service_transactions(
353                counters::GET_BLOCK_LABEL,
354                txns.len(),
355            );
356            txns.len();
357            let pulled_block =
358                txns.drain(..).map(SignedTransaction::into).collect();
359
360            (
361                ConsensusResponse::GetBlockResponse(pulled_block),
362                callback,
363                counters::GET_BLOCK_LABEL,
364            )
365        }
366        ConsensusRequest::RejectNotification(transactions, callback) => {
367            counters::mempool_service_transactions(
368                counters::COMMIT_CONSENSUS_LABEL,
369                transactions.len(),
370            );
371            commit_txns(mempool, transactions, 0, true).await;
372            (
373                ConsensusResponse::CommitResponse(),
374                callback,
375                counters::COMMIT_CONSENSUS_LABEL,
376            )
377        }
378    };
379    // Send back to callback
380    let result = if callback.send(Ok(resp)).is_err() {
381        diem_error!(LogSchema::event_log(
382            LogEntry::Consensus,
383            LogEvent::CallbackFail
384        ));
385        counters::REQUEST_FAIL_LABEL
386    } else {
387        counters::REQUEST_SUCCESS_LABEL
388    };
389    let latency = start_time.elapsed();
390    counters::mempool_service_latency(counter_label, result, latency);
391}
392
393async fn commit_txns(
394    mempool: &Mutex<CoreMempool>, transactions: Vec<CommittedTransaction>,
395    block_timestamp_usecs: u64, is_rejected: bool,
396) {
397    let mut pool = mempool.lock();
398
399    for transaction in transactions {
400        pool.remove_transaction(
401            &transaction.sender,
402            transaction.hash,
403            is_rejected,
404        );
405    }
406
407    if block_timestamp_usecs > 0 {
408        pool.gc_by_expiration_time(Duration::from_micros(
409            block_timestamp_usecs,
410        ));
411    }
412}
