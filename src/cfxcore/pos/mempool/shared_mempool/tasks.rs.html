47/// Attempts broadcast to `peer` and schedules the next broadcast.
48pub(crate) fn execute_broadcast(
49    peer: NodeId, backoff: bool, smp: &mut SharedMempool,
50    scheduled_broadcasts: &mut FuturesUnordered<ScheduledBroadcast>,
51    broadcasting_peers: &mut HashSet<NodeId>, executor: Handle,
52) {
53    diem_trace!("execute_broadcast starts: peer={}", peer);
54    let peer_manager = &smp.peer_manager.clone();
55    peer_manager.execute_broadcast(peer.clone(), backoff, smp);
56    let schedule_backoff = peer_manager.is_backoff_mode(&peer);
57
58    let interval_ms = if schedule_backoff {
59        smp.config.shared_mempool_backoff_interval_ms
60    } else {
61        smp.config.shared_mempool_tick_interval_ms
62    };
63
64    if peer_manager.contains_peer(&peer) {
65        // Make sure we only has one broadcast task for one peer id.
66        broadcasting_peers.insert(peer);
67        scheduled_broadcasts.push(ScheduledBroadcast::new(
68            Instant::now() + Duration::from_millis(interval_ms),
69            peer,
70            schedule_backoff,
71            executor,
72        ));
73    } else {
74        // The peer has been disconnected,
75        // so it can be added again after reconnection.
76        broadcasting_peers.remove(&peer);
77    }
78    diem_trace!("execute_broadcast end: peer={}", peer);
79}

85/// Processes transactions directly submitted by client.
86pub(crate) async fn process_client_transaction_submission(
87    smp: SharedMempool, transaction: SignedTransaction,
88    callback: oneshot::Sender<Result<SubmissionStatus>>, timer: HistogramTimer,
89) {
90    timer.stop_and_record();
91    let _timer = counters::process_txn_submit_latency_timer(
92        counters::CLIENT_LABEL,
93        counters::CLIENT_LABEL,
94    );
95    let statuses = process_incoming_transactions(
96        &smp,
97        vec![transaction],
98        TimelineState::NotReady,
99    )
100    .await;
101    log_txn_process_results(&statuses, None);
102
103    if let Some(status) = statuses.get(0) {
104        if callback.send(Ok(status.1.clone())).is_err() {
105            diem_error!(LogSchema::event_log(
106                LogEntry::JsonRpc,
107                LogEvent::CallbackFail
108            ));
109            counters::CLIENT_CALLBACK_FAIL.inc();
110        }
111    }
112}

114/// Processes transactions from other nodes.
115pub(crate) async fn process_transaction_broadcast(
116    smp: SharedMempool, transactions: Vec<SignedTransaction>,
117    request_id: Vec<u8>, timeline_state: TimelineState, peer: NodeId,
118    timer: HistogramTimer,
119) {
120    diem_trace!("process_transaction_broadcast starts: peer={}", peer);
121    timer.stop_and_record();
122    /*let _timer = counters::process_txn_submit_latency_timer(
123        peer.raw_network_id().as_str(),
124        peer.peer_id().short_str().as_str(),
125    );*/
126    let results = process_incoming_transactions(
127        &smp,
128        transactions.clone(),
129        timeline_state,
130    )
131    .await;
132    log_txn_process_results(&results, Some(peer.clone()));
133
134    let ack_response = gen_ack_response(request_id, results, &peer);
135    if let Err(e) = smp
136        .network_sender
137        .send_message_with_peer_id(&peer, &ack_response)
138    {
139        counters::network_send_fail_inc(counters::ACK_TXNS);
140        diem_error!(LogSchema::event_log(
141            LogEntry::BroadcastACK,
142            LogEvent::NetworkSendFail
143        )
144        //.peer(&peer)
145        .error(&e.into()));
146        return;
147    }
148    notify_subscribers(SharedMempoolNotification::ACK, &smp.subscribers);
149    diem_trace!("process_transaction_broadcast ends: peer={}", peer);
150}

152fn gen_ack_response(
153    request_id: Vec<u8>, results: Vec<SubmissionStatusBundle>, peer: &NodeId,
154) -> MempoolSyncMsg {
155    let mut backoff = false;
156    let mut retry = false;
157    for r in results.into_iter() {
158        let submission_status = r.1;
159        if submission_status.0.code == MempoolStatusCode::MempoolIsFull {
160            backoff = true;
161        }
162        if is_txn_retryable(submission_status) {
163            retry = true;
164        }
165
166        if backoff && retry {
167            break;
168        }
169    }
170
171    diem_trace!(
172        "request[{:?}] from peer[{:?}] retry[{:?}]",
173        request_id,
174        peer,
175        retry
176    );
177
178    update_ack_counter(&peer, counters::SENT_LABEL, retry, backoff);
179    MempoolSyncMsg::BroadcastTransactionsResponse {
180        request_id,
181        retry,
182        backoff,
183    }
184}

210/// Submits a list of SignedTransaction to the local mempool
211/// and returns a vector containing AdmissionControlStatus.
212pub(crate) async fn process_incoming_transactions(
213    smp: &SharedMempool, transactions: Vec<SignedTransaction>,
214    timeline_state: TimelineState,
215) -> Vec<SubmissionStatusBundle> {
216    let mut statuses = vec![];
217
218    let start_storage_read = Instant::now();
219    // Track latency for storage read fetching sequence number
220    let storage_read_latency = start_storage_read.elapsed();
221    counters::PROCESS_TXN_BREAKDOWN_LATENCY
222        .with_label_values(&[counters::FETCH_SEQ_NUM_LABEL])
223        .observe(
224            storage_read_latency.as_secs_f64() / transactions.len() as f64,
225        );
226
227    // Track latency: VM validation
228    let vm_validation_timer = counters::PROCESS_TXN_BREAKDOWN_LATENCY
229        .with_label_values(&[counters::VM_VALIDATION_LABEL])
230        .start_timer();
231    // Filter out already received transactions, so we do not need to process
232    // them.
233    let transactions: Vec<SignedTransaction> = {
234        let mempool = smp.mempool.lock();
235        transactions
236            .into_iter()
237            .filter(|tx| mempool.transactions.get(&tx.hash()).is_none())
238            .collect()
239    };
240    let validation_results = transactions
241        .par_iter()
242        .map(|t| {
243            smp.validator
244                .read()
245                .validate_transaction(&t, smp.commited_pos_state.clone())
246        })
247        .collect::<Vec<_>>();
248    vm_validation_timer.stop_and_record();
249
250    {
251        let mut mempool = smp.mempool.lock();
252        for (idx, transaction) in transactions.into_iter().enumerate() {
253            if let Some(validation_result) = &validation_results[idx] {
254                match validation_result.status() {
255                    None => {
256                        let ranking_score = validation_result.score();
257                        let governance_role =
258                            validation_result.governance_role();
259                        let mempool_status = mempool.add_txn(
260                            transaction.clone(),
261                            ranking_score,
262                            timeline_state,
263                            governance_role,
264                        );
265                        statuses.push((transaction, (mempool_status, None)));
266                    }
267                    Some(validation_status) => {
268                        statuses.push((
269                            transaction.clone(),
270                            (
271                                MempoolStatus::new(MempoolStatusCode::VmError),
272                                Some(validation_status),
273                            ),
274                        ));
275                    }
276                }
277            }
278        }
279    }
280    notify_subscribers(
281        SharedMempoolNotification::NewTransactions,
282        &smp.subscribers,
283    );
284    statuses
285}

287fn log_txn_process_results(
288    results: &[SubmissionStatusBundle], sender: Option<NodeId>,
289) {
290    let sender = match sender {
291        Some(peer) => peer,
292        None => {
293            return;
294        }
295    };
296    for (txn, (_mempool_status, maybe_vm_status)) in results.iter() {
297        if let Some(vm_status) = maybe_vm_status {
298            diem_trace!(
299                SecurityEvent::InvalidTransactionMempool,
300                failed_transaction = txn,
301                vm_status = vm_status,
302                sender = sender,
303            );
304            /*counters::shared_mempool_transactions_processed_inc(
305                counters::VM_VALIDATION_LABEL,
306                &network,
307                &sender,
308            );*/
309            continue;
310        }
311        /*

313            MempoolStatusCode::Accepted => {
314                counters::shared_mempool_transactions_processed_inc(
315                    counters::SUCCESS_LABEL,
316                    &network,
317                    &sender,
318                )
319            }
320            _ => counters::shared_mempool_transactions_processed_inc(
321                &mempool_status.code.to_string(),
322                &network,

333pub(crate) async fn process_committed_transactions(
334    mempool: Arc<Mutex<CoreMempool>>, req: CommitNotification,
335) {
336    let start_time = Instant::now();
337    diem_debug!(LogSchema::event_log(
338        LogEntry::StateSyncCommit,
339        LogEvent::Received
340    )
341    .state_sync_msg(&req));
342    counters::mempool_service_transactions(
343        counters::COMMIT_STATE_SYNC_LABEL,
344        req.transactions.len(),
345    );
346    commit_txns(&mempool, req.transactions, req.block_timestamp_usecs, false)
347        .await;
348    let result = if req.callback.send(Ok(CommitResponse::success())).is_err() {
349        diem_error!(LogSchema::event_log(
350            LogEntry::StateSyncCommit,
351            LogEvent::CallbackFail
352        ));
353        counters::REQUEST_FAIL_LABEL
354    } else {
355        counters::REQUEST_SUCCESS_LABEL
356    };
357    let latency = start_time.elapsed();
358    counters::mempool_service_latency(
359        counters::COMMIT_STATE_SYNC_LABEL,
360        result,
361        latency,
362    );
363}

365pub(crate) async fn process_consensus_request(
366    db: Arc<CachedPosLedgerDB>, mempool: &Mutex<CoreMempool>,
367    req: ConsensusRequest,
368) {
369    // Start latency timer
370    let start_time = Instant::now();
371    diem_debug!(
372        LogSchema::event_log(LogEntry::Consensus, LogEvent::Received)
373            .consensus_msg(&req)
374    );
375
376    let (resp, callback, counter_label) = match req {
377        ConsensusRequest::GetBlockRequest(
378            max_block_size,
379            transactions,
380            parent_block_id,
381            validators,
382            callback,
383        ) => {
384            let exclude_transactions: HashSet<TxnPointer> = transactions
385                .iter()
386                .map(|txn| (txn.sender, txn.hash))
387                .collect();
388            let mut txns;
389            {
390                let mut mempool = mempool.lock();
391                // gc before pulling block as extra protection against txns that
392                // may expire in consensus Note: this gc
393                // operation relies on the fact that consensus uses the system
394                // time to determine block timestamp
395                let curr_time = diem_infallible::duration_since_epoch();
396                mempool.gc_by_expiration_time(curr_time);
397                let block_size = cmp::max(max_block_size, 1);
398                let pos_state = db
399                    .get_pos_state(&parent_block_id)
400                    .expect("pos_state should exist");
401                txns = mempool.get_block(
402                    block_size,
403                    exclude_transactions,
404                    &pos_state,
405                    validators,
406                );
407            }
408            counters::mempool_service_transactions(
409                counters::GET_BLOCK_LABEL,
410                txns.len(),
411            );
412            txns.len();
413            let pulled_block =
414                txns.drain(..).map(SignedTransaction::into).collect();
415
416            (
417                ConsensusResponse::GetBlockResponse(pulled_block),
418                callback,
419                counters::GET_BLOCK_LABEL,
420            )
421        }
422        ConsensusRequest::RejectNotification(transactions, callback) => {
423            counters::mempool_service_transactions(
424                counters::COMMIT_CONSENSUS_LABEL,
425                transactions.len(),
426            );
427            commit_txns(mempool, transactions, 0, true).await;
428            (
429                ConsensusResponse::CommitResponse(),
430                callback,
431                counters::COMMIT_CONSENSUS_LABEL,
432            )
433        }
434    };
435    // Send back to callback
436    let result = if callback.send(Ok(resp)).is_err() {
437        diem_error!(LogSchema::event_log(
438            LogEntry::Consensus,
439            LogEvent::CallbackFail
440        ));
441        counters::REQUEST_FAIL_LABEL
442    } else {
443        counters::REQUEST_SUCCESS_LABEL
444    };
445    let latency = start_time.elapsed();
446    counters::mempool_service_latency(counter_label, result, latency);
447}
448
449async fn commit_txns(
450    mempool: &Mutex<CoreMempool>, transactions: Vec<CommittedTransaction>,
451    block_timestamp_usecs: u64, is_rejected: bool,
452) {
453    let mut pool = mempool.lock();
454
455    for transaction in transactions {
456        pool.remove_transaction(
457            &transaction.sender,
458            transaction.hash,
459            is_rejected,
460        );
461    }
462
463    if block_timestamp_usecs > 0 {
464        pool.gc_by_expiration_time(Duration::from_micros(
465            block_timestamp_usecs,
466        ));
467    }
468}
