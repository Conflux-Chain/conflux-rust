43/// Coordinator that handles inbound network events and outbound txn broadcasts.
44pub(crate) async fn coordinator(
45    mut smp: SharedMempool, executor: Handle,
46    mut network_receivers: NetworkReceivers,
47    mut client_events: mpsc::Receiver<(
48        SignedTransaction,
49        oneshot::Sender<Result<SubmissionStatus>>,
50    )>,
51    mut consensus_requests: mpsc::Receiver<ConsensusRequest>,
52    mut commit_notifications: mpsc::Receiver<CommitNotification>,
53) {
54    diem_info!(LogSchema::event_log(
55        LogEntry::CoordinatorRuntime,
56        LogEvent::Start
57    ));
58    let mut scheduled_broadcasts = FuturesUnordered::new();
59    let mut broadcasting_peers = HashSet::new();
60
61    // Use a BoundedExecutor to restrict only `workers_available` concurrent
62    // worker tasks that can process incoming transactions.
63    let workers_available =
64        smp.config.shared_mempool_max_concurrent_inbound_syncs;
65    let bounded_executor =
66        BoundedExecutor::new(workers_available, executor.clone());
67
68    loop {
69        let _timer = counters::MAIN_LOOP.start_timer();
70        ::futures::select! {
71            (msg, callback) = client_events.select_next_some() => {
72                handle_client_event(&mut smp, &bounded_executor, msg, callback).await;
73            },
74            msg = consensus_requests.select_next_some() => {
75                tasks::process_consensus_request(smp.db_with_cache.clone(), &smp.mempool, msg).await;
76            }
77            msg = commit_notifications.select_next_some() => {
78                handle_commit_notification(&mut smp, msg);
79            }
80            (peer, backoff) = scheduled_broadcasts.select_next_some() => {
81                // diem_debug!("scheduled_broadcasts");
82                tasks::execute_broadcast(peer, backoff, &mut smp, &mut scheduled_broadcasts, &mut broadcasting_peers, executor.clone());
83            },
84            (peer, event) = network_receivers.network_events.select_next_some() => {
85                diem_debug!("network_events to scheduled_broadcasts");
86                match event {
87                        NetworkEvent::PeerConnected => {
88                        if smp.peer_manager.add_peer(peer) && !broadcasting_peers.contains(&peer) {
89                            // Only spawn tx broadcast for new peers.
90                            tasks::execute_broadcast(peer, true, &mut smp, &mut scheduled_broadcasts,&mut broadcasting_peers, executor.clone());
91                        }
92                    }
93                    NetworkEvent::PeerDisconnected => {
94                        smp.peer_manager.disable_peer(peer);
95                    }
96                }
97            },
98            (peer, msg) = network_receivers.mempool_sync_message.select_next_some() => {
99                diem_debug!("receive mempool_sync_message");
100                handle_mempool_sync_msg(&bounded_executor, &mut smp, peer, msg).await;
101            }
102            complete => break,
103        }
104    }
105    diem_error!(LogSchema::event_log(
106        LogEntry::CoordinatorRuntime,
107        LogEvent::Terminated
108    ));
109}

111async fn handle_client_event(
112    smp: &mut SharedMempool, bounded_executor: &BoundedExecutor,
113    msg: SignedTransaction,
114    callback: oneshot::Sender<
115        anyhow::Result<(MempoolStatus, Option<DiscardedVMStatus>)>,
116    >,
117) {
118    diem_debug!("handle_client_event");
119    // This timer measures how long it took for the bounded executor to
120    // *schedule* the task.
121    let _timer = counters::task_spawn_latency_timer(
122        counters::CLIENT_EVENT_LABEL,
123        counters::SPAWN_LABEL,
124    );
125    // This timer measures how long it took for the task to go from scheduled to
126    // started.
127    let task_start_timer = counters::task_spawn_latency_timer(
128        counters::CLIENT_EVENT_LABEL,
129        counters::START_LABEL,
130    );
131    bounded_executor
132        .spawn(tasks::process_client_transaction_submission(
133            smp.clone(),
134            msg,
135            callback,
136            task_start_timer,
137        ))
138        .await;
139}
140
141fn handle_commit_notification(
142    smp: &mut SharedMempool, msg: CommitNotification,
143) {
144    let _timer = counters::task_spawn_latency_timer(
145        counters::STATE_SYNC_EVENT_LABEL,
146        counters::SPAWN_LABEL,
147    );
148    smp.update_pos_state();
149    tokio::spawn(tasks::process_committed_transactions(
150        smp.mempool.clone(),
151        msg,
152    ));
153}

155async fn handle_mempool_sync_msg(
156    bounded_executor: &BoundedExecutor, smp: &mut SharedMempool, peer: NodeId,
157    msg: MempoolSyncMsg,
158) {
159    counters::shared_mempool_event_inc("message");
160    match msg {
161        MempoolSyncMsg::BroadcastTransactionsRequest {
162            request_id,
163            transactions,
164        } => {
165            let smp_clone = smp.clone();
166            let timeline_state = TimelineState::NonQualified;
167            /*
168            match smp.peer_manager.is_upstream_peer(&peer, None) {
169                true => TimelineState::NonQualified,
170                false => TimelineState::NotReady,
171            };*/
172            // This timer measures how long it took for the bounded
173            // executor to *schedule* the task.
174            let _timer = counters::task_spawn_latency_timer(
175                counters::PEER_BROADCAST_EVENT_LABEL,
176                counters::SPAWN_LABEL,
177            );
178            // This timer measures how long it took for the task to go
179            // from scheduled to started.
180            let task_start_timer = counters::task_spawn_latency_timer(
181                counters::PEER_BROADCAST_EVENT_LABEL,
182                counters::START_LABEL,
183            );
184            bounded_executor
185                .spawn(tasks::process_transaction_broadcast(
186                    smp_clone,
187                    transactions,
188                    request_id,
189                    timeline_state,
190                    peer,
191                    task_start_timer,
192                ))
193                .await;
194        }
195        MempoolSyncMsg::BroadcastTransactionsResponse {
196            request_id,
197            retry,
198            backoff,
199        } => {
200            let ack_timestamp = SystemTime::now();
201            smp.peer_manager.process_broadcast_ack(
202                peer,
203                request_id,
204                retry,
205                backoff,
206                ack_timestamp,
207            );
208        }
209    }
210}

212/// Garbage collect all expired transactions by SystemTTL.
213pub(crate) async fn gc_coordinator(
214    mempool: Arc<Mutex<CoreMempool>>, gc_interval_ms: u64,
215) {
216    diem_info!(LogSchema::event_log(LogEntry::GCRuntime, LogEvent::Start));
217    let mut interval =
218        IntervalStream::new(interval(Duration::from_millis(gc_interval_ms)));
219    while let Some(_interval) = interval.next().await {
220        diem_sample!(
221            SampleRate::Duration(Duration::from_secs(60)),
222            diem_info!(LogSchema::event_log(
223                LogEntry::GCRuntime,
224                LogEvent::Live
225            ))
226        );
227        mempool.lock().gc();
228    }
229
230    diem_error!(LogSchema::event_log(
231        LogEntry::GCRuntime,
232        LogEvent::Terminated
233    ));
