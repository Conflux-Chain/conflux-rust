44/// Coordinator that handles inbound network events and outbound txn broadcasts.
45pub(crate) async fn coordinator(
46    mut smp: SharedMempool, executor: Handle,
47    mut network_receivers: NetworkReceivers,
48    mut client_events: mpsc::Receiver<(
49        SignedTransaction,
50        oneshot::Sender<Result<SubmissionStatus>>,
51    )>,
52    mut consensus_requests: mpsc::Receiver<ConsensusRequest>,
53    mut state_sync_requests: mpsc::Receiver<CommitNotification>,
54    mut mempool_reconfig_events: diem_channel::Receiver<
55        (),
56        OnChainConfigPayload,
57    >,
58) {
59    diem_info!(LogSchema::event_log(
60        LogEntry::CoordinatorRuntime,
61        LogEvent::Start
62    ));
63    let mut scheduled_broadcasts = FuturesUnordered::new();
64    let mut broadcasting_peers = HashSet::new();
65
66    // Use a BoundedExecutor to restrict only `workers_available` concurrent
67    // worker tasks that can process incoming transactions.
68    let workers_available =
69        smp.config.shared_mempool_max_concurrent_inbound_syncs;
70    let bounded_executor =
71        BoundedExecutor::new(workers_available, executor.clone());
72
73    loop {
74        let _timer = counters::MAIN_LOOP.start_timer();
75        ::futures::select! {
76            (msg, callback) = client_events.select_next_some() => {
77                handle_client_event(&mut smp, &bounded_executor, msg, callback).await;
78            },
79            msg = consensus_requests.select_next_some() => {
80                tasks::process_consensus_request(smp.db_with_cache.clone(), &smp.mempool, msg).await;
81            }
82            msg = state_sync_requests.select_next_some() => {
83                handle_state_sync_request(&mut smp, msg);
84            }
85            config_update = mempool_reconfig_events.select_next_some() => {
86                handle_mempool_reconfig_event(&mut smp, &bounded_executor, config_update).await;
87            },
88            (peer, backoff) = scheduled_broadcasts.select_next_some() => {
89                // diem_debug!("scheduled_broadcasts");
90                tasks::execute_broadcast(peer, backoff, &mut smp, &mut scheduled_broadcasts, &mut broadcasting_peers, executor.clone());
91            },
92            (peer, event) = network_receivers.network_events.select_next_some() => {
93                diem_debug!("network_events to scheduled_broadcasts");
94                match event {
95                        NetworkEvent::PeerConnected => {
96                        if smp.peer_manager.add_peer(peer) && !broadcasting_peers.contains(&peer) {
97                            // Only spawn tx broadcast for new peers.
98                            tasks::execute_broadcast(peer, true, &mut smp, &mut scheduled_broadcasts,&mut broadcasting_peers, executor.clone());
99                        }
100                    }
101                    NetworkEvent::PeerDisconnected => {
102                        smp.peer_manager.disable_peer(peer);
103                    }
104                }
105            },
106            (peer, msg) = network_receivers.mempool_sync_message.select_next_some() => {
107                diem_debug!("receive mempool_sync_message");
108                handle_mempool_sync_msg(&bounded_executor, &mut smp, peer, msg).await;
109            }
110            complete => break,
111        }
112    }
113    diem_error!(LogSchema::event_log(
114        LogEntry::CoordinatorRuntime,
115        LogEvent::Terminated
116    ));
117}

119async fn handle_client_event(
120    smp: &mut SharedMempool, bounded_executor: &BoundedExecutor,
121    msg: SignedTransaction,
122    callback: oneshot::Sender<
123        anyhow::Result<(MempoolStatus, Option<DiscardedVMStatus>)>,
124    >,
125) {
126    diem_debug!("handle_client_event");
127    // This timer measures how long it took for the bounded executor to
128    // *schedule* the task.
129    let _timer = counters::task_spawn_latency_timer(
130        counters::CLIENT_EVENT_LABEL,
131        counters::SPAWN_LABEL,
132    );
133    // This timer measures how long it took for the task to go from scheduled to
134    // started.
135    let task_start_timer = counters::task_spawn_latency_timer(
136        counters::CLIENT_EVENT_LABEL,
137        counters::START_LABEL,
138    );
139    bounded_executor
140        .spawn(tasks::process_client_transaction_submission(
141            smp.clone(),
142            msg,
143            callback,
144            task_start_timer,
145        ))
146        .await;
147}
148
149fn handle_state_sync_request(smp: &mut SharedMempool, msg: CommitNotification) {
150    let _timer = counters::task_spawn_latency_timer(
151        counters::STATE_SYNC_EVENT_LABEL,
152        counters::SPAWN_LABEL,
153    );
154    smp.update_pos_state();
155    tokio::spawn(tasks::process_state_sync_request(smp.mempool.clone(), msg));
156}

158async fn handle_mempool_reconfig_event(
159    smp: &mut SharedMempool, bounded_executor: &BoundedExecutor,
160    config_update: OnChainConfigPayload,
161) {
162    diem_info!(LogSchema::event_log(
163        LogEntry::ReconfigUpdate,
164        LogEvent::Received
165    ));
166    let _timer = counters::task_spawn_latency_timer(
167        counters::RECONFIG_EVENT_LABEL,
168        counters::SPAWN_LABEL,
169    );
170
171    bounded_executor
172        .spawn(tasks::process_config_update(
173            config_update,
174            smp.validator.clone(),
175        ))
176        .await;
177}

179async fn handle_mempool_sync_msg(
180    bounded_executor: &BoundedExecutor, smp: &mut SharedMempool, peer: NodeId,
181    msg: MempoolSyncMsg,
182) {
183    counters::shared_mempool_event_inc("message");
184    match msg {
185        MempoolSyncMsg::BroadcastTransactionsRequest {
186            request_id,
187            transactions,
188        } => {
189            let smp_clone = smp.clone();
190            let timeline_state = TimelineState::NonQualified;
191            /*
192            match smp.peer_manager.is_upstream_peer(&peer, None) {
193                true => TimelineState::NonQualified,
194                false => TimelineState::NotReady,
195            };*/
196            // This timer measures how long it took for the bounded
197            // executor to *schedule* the task.
198            let _timer = counters::task_spawn_latency_timer(
199                counters::PEER_BROADCAST_EVENT_LABEL,
200                counters::SPAWN_LABEL,
201            );
202            // This timer measures how long it took for the task to go
203            // from scheduled to started.
204            let task_start_timer = counters::task_spawn_latency_timer(
205                counters::PEER_BROADCAST_EVENT_LABEL,
206                counters::START_LABEL,
207            );
208            bounded_executor
209                .spawn(tasks::process_transaction_broadcast(
210                    smp_clone,
211                    transactions,
212                    request_id,
213                    timeline_state,
214                    peer,
215                    task_start_timer,
216                ))
217                .await;
218        }
219        MempoolSyncMsg::BroadcastTransactionsResponse {
220            request_id,
221            retry,
222            backoff,
223        } => {
224            let ack_timestamp = SystemTime::now();
225            smp.peer_manager.process_broadcast_ack(
226                peer,
227                request_id,
228                retry,
229                backoff,
230                ack_timestamp,
231            );
232        }
233    }
234}

236/// Garbage collect all expired transactions by SystemTTL.
237pub(crate) async fn gc_coordinator(
238    mempool: Arc<Mutex<CoreMempool>>, gc_interval_ms: u64,
239) {
240    diem_info!(LogSchema::event_log(LogEntry::GCRuntime, LogEvent::Start));
241    let mut interval =
242        IntervalStream::new(interval(Duration::from_millis(gc_interval_ms)));
243    while let Some(_interval) = interval.next().await {
244        diem_sample!(
245            SampleRate::Duration(Duration::from_secs(60)),
246            diem_info!(LogSchema::event_log(
247                LogEntry::GCRuntime,
248                LogEvent::Live
249            ))
250        );
251        mempool.lock().gc();
252    }
253
254    diem_error!(LogSchema::event_log(
255        LogEntry::GCRuntime,
256        LogEvent::Terminated
257    ));
