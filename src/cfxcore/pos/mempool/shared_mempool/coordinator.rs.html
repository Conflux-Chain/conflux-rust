38/// Coordinator that handles inbound network events and outbound txn broadcasts.
39pub(crate) async fn coordinator(
40    mut smp: SharedMempool, executor: Handle,
41    mut network_receivers: NetworkReceivers,
42    mut client_events: mpsc::Receiver<(
43        SignedTransaction,
44        oneshot::Sender<Result<SubmissionStatus>>,
45    )>,
46    mut consensus_requests: mpsc::Receiver<ConsensusRequest>,
47    mut commit_notifications: mpsc::Receiver<CommitNotification>,
48) {
49    diem_info!(LogSchema::event_log(
50        LogEntry::CoordinatorRuntime,
51        LogEvent::Start
52    ));
53    let mut scheduled_broadcasts = FuturesUnordered::new();
54    let mut broadcasting_peers = HashSet::new();
55
56    // Use a BoundedExecutor to restrict only `workers_available` concurrent
57    // worker tasks that can process incoming transactions.
58    let workers_available =
59        smp.config.shared_mempool_max_concurrent_inbound_syncs;
60    let bounded_executor =
61        BoundedExecutor::new(workers_available, executor.clone());
62
63    loop {
64        ::futures::select! {
65            (msg, callback) = client_events.select_next_some() => {
66                handle_client_event(&mut smp, &bounded_executor, msg, callback).await;
67            },
68            msg = consensus_requests.select_next_some() => {
69                tasks::process_consensus_request(smp.db_with_cache.clone(), &smp.mempool, msg).await;
70            }
71            msg = commit_notifications.select_next_some() => {
72                handle_commit_notification(&mut smp, msg);
73            }
74            (peer, backoff) = scheduled_broadcasts.select_next_some() => {
75                // diem_debug!("scheduled_broadcasts");
76                tasks::execute_broadcast(peer, backoff, &mut smp, &mut scheduled_broadcasts, &mut broadcasting_peers, executor.clone());
77            },
78            (peer, event) = network_receivers.network_events.select_next_some() => {
79                diem_debug!("network_events to scheduled_broadcasts");
80                match event {
81                        NetworkEvent::PeerConnected => {
82                        if smp.peer_manager.add_peer(peer) && !broadcasting_peers.contains(&peer) {
83                            // Only spawn tx broadcast for new peers.
84                            tasks::execute_broadcast(peer, true, &mut smp, &mut scheduled_broadcasts,&mut broadcasting_peers, executor.clone());
85                        }
86                    }
87                    NetworkEvent::PeerDisconnected => {
88                        smp.peer_manager.disable_peer(peer);
89                    }
90                }
91            },
92            (peer, msg) = network_receivers.mempool_sync_message.select_next_some() => {
93                diem_debug!("receive mempool_sync_message");
94                handle_mempool_sync_msg(&bounded_executor, &mut smp, peer, msg).await;
95            }
96            complete => break,
97        }
98    }
99    diem_error!(LogSchema::event_log(
100        LogEntry::CoordinatorRuntime,
101        LogEvent::Terminated
102    ));
103}
104
105async fn handle_client_event(
106    smp: &mut SharedMempool, bounded_executor: &BoundedExecutor,
107    msg: SignedTransaction,
108    callback: oneshot::Sender<
109        anyhow::Result<(MempoolStatus, Option<DiscardedVMStatus>)>,
110    >,
111) {
112    diem_debug!("handle_client_event");
113    bounded_executor
114        .spawn(tasks::process_client_transaction_submission(
115            smp.clone(),
116            msg,
117            callback,
118        ))
119        .await;
120}
121
122fn handle_commit_notification(
123    smp: &mut SharedMempool, msg: CommitNotification,
124) {
125    smp.update_pos_state();
126    tokio::spawn(tasks::process_committed_transactions(
127        smp.mempool.clone(),
128        msg,
129    ));
130}
131
132async fn handle_mempool_sync_msg(
133    bounded_executor: &BoundedExecutor, smp: &mut SharedMempool, peer: NodeId,
134    msg: MempoolSyncMsg,
135) {
136    match msg {
137        MempoolSyncMsg::BroadcastTransactionsRequest {
138            request_id,
139            transactions,
140        } => {
141            let smp_clone = smp.clone();
142            let timeline_state = TimelineState::NonQualified;
143            bounded_executor
144                .spawn(tasks::process_transaction_broadcast(
145                    smp_clone,
146                    transactions,
147                    request_id,
148                    timeline_state,
149                    peer,
150                ))
151                .await;
152        }
153        MempoolSyncMsg::BroadcastTransactionsResponse {
154            request_id,
155            retry,
156            backoff,
157        } => {
158            smp.peer_manager
159                .process_broadcast_ack(peer, request_id, retry, backoff);
160        }
161    }
162}

164/// Garbage collect all expired transactions by SystemTTL.
165pub(crate) async fn gc_coordinator(
166    mempool: Arc<Mutex<CoreMempool>>, gc_interval_ms: u64,
167) {
168    diem_info!(LogSchema::event_log(LogEntry::GCRuntime, LogEvent::Start));
169    let mut interval =
170        IntervalStream::new(interval(Duration::from_millis(gc_interval_ms)));
171    while let Some(_interval) = interval.next().await {
172        diem_sample!(
173            SampleRate::Duration(Duration::from_secs(60)),
174            diem_info!(LogSchema::event_log(
175                LogEntry::GCRuntime,
176                LogEvent::Live
177            ))
178        );
179        mempool.lock().gc();
180    }
181
182    diem_error!(LogSchema::event_log(
183        LogEntry::GCRuntime,
184        LogEvent::Terminated
185    ));
186}
