27pub fn set_timestamp(timestamp_type: TimestampType, time_as_usecs: u64) {
28    TIMESTAMP
29        .with_label_values(&[timestamp_type.as_str()])
30        .set((time_as_usecs / 1000) as i64)
31}
32
33pub enum TimestampType {
34    /// Current ledger committed timestamp
35    Committed,
36    /// Current computers clock
37    Real,
38    /// Current ledger synced timestamp
39    Synced,
40}
41
42impl TimestampType {
43    pub fn as_str(&self) -> &'static str {
44        match self {
45            TimestampType::Committed => "committed",
46            TimestampType::Real => "real",
47            TimestampType::Synced => "synced",
48        }
49    }
50}
51
52pub fn set_version(version_type: VersionType, version: u64) {
53    VERSION
54        .with_label_values(&[version_type.as_str()])
55        .set(version as i64)
56}
57
58pub fn get_version(version_type: VersionType) -> u64 {
59    VERSION.with_label_values(&[version_type.as_str()]).get() as u64
60}
61
62pub enum VersionType {
63    /// Version of latest ledger info committed.
64    Committed,
65    /// Highest known version or version proceeding it
66    Highest,
67    /// Version of most recent txn that was synced (even if it is not backed by
68    /// an LI)
69    Synced,
70    /// Current version a node is trying to catch up to usually within the
71    /// current epoch
72    Target,
73}
74
75impl VersionType {
76    pub fn as_str(&self) -> &'static str {
77        match self {
78            VersionType::Committed => "committed",
79            VersionType::Highest => "highest",
80            VersionType::Synced => "synced",
81            VersionType::Target => "target",
82        }
83    }
84}
