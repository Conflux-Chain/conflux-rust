8use diem_logger::Schema;
9use diem_types::{
10    contract_event::ContractEvent, ledger_info::LedgerInfoWithSignatures,
11};
12use serde::Serialize;
13
14#[derive(Clone, Schema)]
15pub struct LogSchema {
16    name: LogEntry,
17    event: Option<LogEvent>,
18    subscription_name: Option<String>,
19    count: Option<usize>,
20    reconfig_events: Option<Vec<ContractEvent>>,
21    version: Option<u64>,
22    local_li_version: Option<u64>,
23    local_synced_version: Option<u64>,
24    local_epoch: Option<u64>,
25    #[schema(display)]
26    ledger_info: Option<LedgerInfoWithSignatures>,
27    old_epoch: Option<u64>,
28    new_epoch: Option<u64>,
29    target_version: Option<u64>,
30}
31
32impl LogSchema {
33    pub fn new(name: LogEntry) -> Self { Self::new_event(name, None) }
34
35    pub fn event_log(name: LogEntry, event: LogEvent) -> Self {
36        Self::new_event(name, Some(event))
37    }
38
39    fn new_event(name: LogEntry, event: Option<LogEvent>) -> Self {
40        Self {
41            name,
42            event,
43            version: None,
44            subscription_name: None,
45            reconfig_events: None,
46            count: None,
47            local_li_version: None,
48            local_synced_version: None,
49            local_epoch: None,
50            ledger_info: None,
51            new_epoch: None,
52            old_epoch: None,
53            target_version: None,
54        }
55    }
56}
57
58#[derive(Clone, Copy, Serialize)]
59#[serde(rename_all = "snake_case")]
60pub enum LogEntry {
61    Reconfig,
62    RuntimeStart,
63    ConsensusCommit,
64    SyncRequest,
65    EpochChange,
66    CommitFlow,
67    ProgressCheck,
68}
69
70#[derive(Clone, Copy, Serialize)]
71#[serde(rename_all = "snake_case")]
72pub enum LogEvent {
73    CallbackFail,
74    Complete,
75    Fail,
76    Timeout,
77    PublishError,
78    Success,
79    PostCommitFail,
80}
