21#[derive(Clone, Schema)]
22pub struct LogSchema<'a> {
23    name: LogEntry,
24    event: Option<LogEvent>,
25    #[schema(debug)]
26    error: Option<&'a Error>,
27    #[schema(display)]
28    peer: Option<&'a PeerNetworkId>,
29    is_valid_peer: Option<bool>,
30    #[schema(display)]
31    chunk_request: Option<GetChunkRequest>,
32    version: Option<u64>,
33    #[schema(display)]
34    chunk_response: Option<GetChunkResponse>,
35    #[schema(display)]
36    waypoint: Option<Waypoint>,
37    subscription_name: Option<String>,
38    count: Option<usize>,
39    reconfig_events: Option<Vec<ContractEvent>>,
40    local_li_version: Option<u64>,
41    local_synced_version: Option<u64>,
42    local_epoch: Option<u64>,
43    #[schema(display)]
44    ledger_info: Option<LedgerInfoWithSignatures>,
45    old_epoch: Option<u64>,
46    new_epoch: Option<u64>,
47    request_version: Option<u64>,
48    target_version: Option<u64>,
49    old_multicast_level: Option<NetworkId>,
50    new_multicast_level: Option<NetworkId>,
51    //chunk_req_info: Option<&'a ChunkRequestInfo>,
52}
53
54impl<'a> LogSchema<'a> {
55    pub fn new(name: LogEntry) -> Self { Self::new_event(name, None) }
56
57    pub fn event_log(name: LogEntry, event: LogEvent) -> Self {
58        Self::new_event(name, Some(event))
59    }
60
61    fn new_event(name: LogEntry, event: Option<LogEvent>) -> Self {
62        Self {
63            name,
64            event,
65            peer: None,
66            is_valid_peer: None,
67            error: None,
68            chunk_request: None,
69            chunk_response: None,
70            version: None,
71            waypoint: None,
72            subscription_name: None,
73            reconfig_events: None,
74            count: None,
75            local_li_version: None,
76            local_synced_version: None,
77            local_epoch: None,
78            ledger_info: None,
79            new_epoch: None,
80            old_epoch: None,
81            request_version: None,
82            target_version: None,
83            old_multicast_level: None,
84            new_multicast_level: None,
85            //chunk_req_info: None,
86        }
87    }
88}
