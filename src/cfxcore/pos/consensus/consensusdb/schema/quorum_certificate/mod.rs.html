8//! This module defines physical storage schema for consensus quorum certificate
9//! (of a block).
10//!
11//! Serialized quorum certificate bytes identified by block_hash.
12//! ```text
13//! |<---key---->|<----value--->|
14//! | block_hash |  QuorumCert  |
15//! ```
16
17use super::QC_CF_NAME;
18use anyhow::Result;
19use consensus_types::quorum_cert::{QuorumCert, QuorumCertUnchecked};
20use diem_crypto::HashValue;
21use schemadb::{
22    define_schema,
23    schema::{KeyCodec, ValueCodec},
24};
25
26define_schema!(QCSchema, HashValue, QuorumCert, QC_CF_NAME);
27
28impl KeyCodec<QCSchema> for HashValue {
29    fn encode_key(&self) -> Result<Vec<u8>> { Ok(self.to_vec()) }
30
31    fn decode_key(data: &[u8]) -> Result<Self> {
32        Ok(HashValue::from_slice(data)?)
33    }
34}
35
36impl ValueCodec<QCSchema> for QuorumCert {
37    fn encode_value(&self) -> Result<Vec<u8>> { Ok(bcs::to_bytes(self)?) }
38
39    fn decode_value(data: &[u8]) -> Result<Self> {
40        Ok(bcs::from_bytes::<QuorumCertUnchecked>(data).map(Into::into)?)
41    }
42}
