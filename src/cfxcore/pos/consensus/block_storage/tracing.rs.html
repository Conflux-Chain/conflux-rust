8use crate::pos::consensus::counters;
9use diem_infallible::duration_since_epoch;
10use std::time::Duration;
11
12pub struct BlockStage;
13
14impl BlockStage {
15    pub const COMMITTED: &'static str = "committed";
16    pub const EXECUTED: &'static str = "executed";
17    pub const QC_ADDED: &'static str = "qc_added";
18    pub const QC_AGGREGATED: &'static str = "qc_aggregated";
19    pub const RECEIVED: &'static str = "received";
20    pub const SIGNED: &'static str = "signed";
21    pub const SYNCED: &'static str = "synced";
22    pub const VOTED: &'static str = "voted";
23}
24
25/// Record the time during each stage of a block.
26pub fn observe_block(timestamp: u64, stage: &'static str) {
27    if let Some(t) =
28        duration_since_epoch().checked_sub(Duration::from_micros(timestamp))
29    {
30        counters::BLOCK_TRACING
31            .with_label_values(&[stage])
32            .observe(t.as_secs_f64());
33    }
34}
