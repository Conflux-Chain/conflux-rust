29pub const BLOCK_FETCH_BATCH_MAX_SIZE: u64 = 60;

31#[derive(Debug, PartialEq)]
32/// Whether we need to do block retrieval if we want to insert a Quorum Cert.
33pub enum NeedFetchResult {
34    QCRoundBeforeRoot,
35    QCAlreadyExist,
36    QCBlockExist,
37    NeedFetch,
38}
39
40impl BlockStore {
41    /// Checks if quorum certificate can be inserted in block store without RPC
42    /// Returns the enum to indicate the detailed status.
43    pub fn need_fetch_for_quorum_cert(
44        &self, qc: &QuorumCert,
45    ) -> NeedFetchResult {
46        if qc.certified_block().round() < self.root().round() {
47            return NeedFetchResult::QCRoundBeforeRoot;
48        }
49        if self
50            .get_quorum_cert_for_block(qc.certified_block().id())
51            .is_some()
52        {
53            return NeedFetchResult::QCAlreadyExist;
54        }
55        if self.block_exists(qc.certified_block().id()) {
56            return NeedFetchResult::QCBlockExist;
57        }
58        NeedFetchResult::NeedFetch
59    }
60
61    pub async fn insert_quorum_cert(
62        &self, qc: &QuorumCert, retriever: &mut BlockRetriever,
63    ) -> anyhow::Result<()> {
64        match self.need_fetch_for_quorum_cert(&qc) {
65            NeedFetchResult::NeedFetch => {
66                self.fetch_quorum_cert(qc.clone(), retriever).await?
67            }
68            NeedFetchResult::QCBlockExist => {
69                self.insert_single_quorum_cert(qc.clone())?
70            }
71            _ => (),
72        }
73        if self.root().round() < qc.commit_info().round() {
74            let finality_proof = qc.ledger_info();
75            self.commit(finality_proof.clone()).await?;
76            if qc.ends_epoch() {
77                retriever
78                    .network
79                    .broadcast(
80                        ConsensusMsg::EpochChangeProof(Box::new(
81                            EpochChangeProof::new(
82                                vec![finality_proof.clone()],
83                                /* more = */ false,
84                            ),
85                        )),
86                        vec![],
87                    )
88                    .await;
89            }
90            // Wait for PoW to process fetched PoS references.
91            // We wait after commit `qc`, so PoW can process its pivot decision
92            // correctly.
93            // This is not needed for correctness/liveness, but we still add the
94            // waiting here to make the syncing more predictable and
95            // avoid unnecessarily message processing.
96            self.pow_handler
97                .wait_for_initialization(
98                    finality_proof
99                        .ledger_info()
100                        .pivot_decision()
101                        .clone()
102                        .unwrap()
103                        .block_hash,
104                )
105                .await;
106        }
107        Ok(())
108    }
109
110    /// Insert the quorum certificate separately from the block, used to split
111    /// the processing of updating the consensus state(with qc) and deciding
112    /// whether to vote(with block) The missing ancestors are going to be
113    /// retrieved from the given peer. If a given peer fails to provide the
114    /// missing ancestors, the qc is not going to be added.
115    async fn fetch_quorum_cert(
116        &self, qc: QuorumCert, retriever: &mut BlockRetriever,
117    ) -> anyhow::Result<()> {
118        debug!("fetch_quorum_cert: qc={:?}", qc);
119        let mut pending = vec![];
120        let mut retrieve_qc = qc.clone();
121        loop {
122            // round 0 blocks is the genesis of every epoch.
123            if self.block_exists(retrieve_qc.certified_block().id())
124                || retrieve_qc.certified_block().round() <= self.root().round()
125            {
126                break;
127            }
128            // This will not underflow because of the check above.
129            let round_gap =
130                retrieve_qc.certified_block().round() - self.root().round();
131            let mut blocks = retriever
132                .retrieve_block_for_qc(
133                    &retrieve_qc,
134                    round_gap.min(BLOCK_FETCH_BATCH_MAX_SIZE),
135                )
136                .await?;
137            // retriever ensures that the blocks are chained.
138            retrieve_qc = blocks
139                .last()
140                .expect("checked by retriever")
141                .quorum_cert()
142                .clone();
143            pending.append(&mut blocks);
144        }
145
146        if !pending.is_empty() {
147            // Execute the blocks in catch_up mode.
148            while let Some(block) = pending.pop() {
149                // We may receive more blocks than needed in a batch, so check
150                // again here.
151                if self.block_exists(block.id())
152                    || block.round() <= self.root().round()
153                {
154                    continue;
155                }
156
157                let block_qc = block.quorum_cert().clone();
158                self.insert_single_quorum_cert(block_qc.clone())?;
159                self.execute_and_insert_block(
160                    block, true, /* catch_up_mode */
161                    true, /* force_recompute */
162                )?;
163                if block_qc.commit_info().round() > self.root().round() {
164                    if let Err(e) =
165                        self.commit(block_qc.ledger_info().clone()).await
166                    {
167                        // TODO(lpl): Blocks not committed before crash
168                        // should be committed
169                        // here? Make sure
170                        // they are recovered to
171                        // BlockStore during start.
172                        diem_warn!("fetch_quorum_cert: commit error={:?}", e);
173                    }
174                } else {
175                    diem_debug!(
176                        "skip commit, qc_round={} root_root={}",
177                        block_qc.commit_info().round(),
178                        self.root().round()
179                    );
180                }
181            }
182        }
183
184        self.insert_single_quorum_cert(qc)
185    }
186}

188/// BlockRetriever is used internally to retrieve blocks
189pub struct BlockRetriever {
190    network: ConsensusNetworkSender,
191    preferred_peer: Author,
192}
193
194impl BlockRetriever {
195    pub fn new(
196        network: ConsensusNetworkSender, preferred_peer: Author,
197    ) -> Self {
198        Self {
199            network,
200            preferred_peer,
201        }
202    }
203
204    /// Retrieve chain of n blocks for given QC
205    ///
206    /// Returns Result with Vec that has a size of `[1, num_blocks]`.
207    /// This guarantee is based on BlockRetrievalResponse::verify that ensures
208    /// that number of blocks in response is within the range.
209    ///
210    /// The first attempt of block retrieval will always be sent to
211    /// preferred_peer to allow the leader to drive quorum certificate
212    /// creation The other peers from the quorum certificate
213    /// will be randomly tried next.  If all members of the quorum certificate
214    /// are exhausted, an error is returned
215    pub async fn retrieve_block_for_qc<'a>(
216        &'a mut self, qc: &'a QuorumCert, num_blocks: u64,
217    ) -> anyhow::Result<Vec<Block>> {
218        let block_id = qc.certified_block().id();
219        self.request_block(num_blocks, block_id).await
220    }
221
222    pub async fn retrieve_block_for_ledger_info(
223        &mut self, ledger_info: &LedgerInfoWithSignatures,
224    ) -> anyhow::Result<Block> {
225        let block_id = ledger_info.ledger_info().consensus_block_id();
226        let mut blocks = self.request_block(1, block_id).await?;
227        if blocks.len() == 1 {
228            Ok(blocks.remove(0))
229        } else {
230            bail!("retrieve_block_for_ledger_info returns incorrect block number: {}", blocks.len())
231        }
232    }
233
234    async fn request_block(
235        &mut self, num_blocks: u64, block_id: HashValue,
236    ) -> anyhow::Result<Vec<Block>> {
237        let mut peers: Vec<AccountAddress> = self
238            .network
239            .network_sender()
240            .protocol_handler
241            .pos_peer_mapping
242            .read()
243            .keys()
244            .map(Clone::clone)
245            .collect();
246        let mut attempt = 0_u32;
247        loop {
248            if peers.is_empty() {
249                bail!(
250                    "Failed to fetch block {} in {} attempts: no more peers available",
251                    block_id,
252                    attempt
253                );
254            }
255            let peer = self.pick_peer(attempt, &mut peers);
256            attempt += 1;
257
258            diem_debug!(
259                LogSchema::new(LogEvent::RetrieveBlock).remote_peer(peer),
260                block_id = block_id,
261                "Fetching block, attempt {}",
262                attempt
263            );
264            let response = self
265                .network
266                .request_block(
267                    BlockRetrievalRequest::new(block_id, num_blocks),
268                    peer,
269                    retrieval_timeout(attempt),
270                )
271                .await;
272            match response.and_then(|result| {
273                if result.status() == BlockRetrievalStatus::Succeeded {
274                    Ok(result.blocks().clone())
275                } else {
276                    Err(format_err!("{:?}", result.status()))
277                }
278            }) {
279                result @ Ok(_) => return result,
280                Err(e) => diem_warn!(
281                    remote_peer = peer,
282                    block_id = block_id,
283                    error = ?e, "Failed to fetch block, trying another peer",
284                ),
285            }
286        }
287    }
288
289    fn pick_peer(
290        &self, attempt: u32, peers: &mut Vec<AccountAddress>,
291    ) -> AccountAddress {
292        assert!(!peers.is_empty(), "pick_peer on empty peer list");
293
294        if attempt == 0 {
295            // remove preferred_peer if its in list of peers
296            // (strictly speaking it is not required to be there)
297            for i in 0..peers.len() {
298                if peers[i] == self.preferred_peer {
299                    peers.remove(i);
300                    break;
301                }
302            }
303            return self.preferred_peer;
304        }
305
306        let peer_idx = rng().random_range(0..peers.len());
307        peers.remove(peer_idx)
308    }
309}

311// Max timeout is 16s=RETRIEVAL_INITIAL_TIMEOUT*(2^RETRIEVAL_MAX_EXP)
312const RETRIEVAL_INITIAL_TIMEOUT: Duration = Duration::from_millis(1000);
313const RETRIEVAL_MAX_EXP: u32 = 4;
314
315/// Returns exponentially increasing timeout with
316/// limit of RETRIEVAL_INITIAL_TIMEOUT*(2^RETRIEVAL_MAX_EXP)
317#[allow(clippy::trivially_copy_pass_by_ref)]
318fn retrieval_timeout(attempt: u32) -> Duration {
319    assert!(attempt > 0, "retrieval_timeout attempt can't be 0");
320    let exp = RETRIEVAL_MAX_EXP.min(attempt - 1); // [0..RETRIEVAL_MAX_EXP]
321    RETRIEVAL_INITIAL_TIMEOUT * 2_u32.pow(exp)
322}
