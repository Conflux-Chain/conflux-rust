75/// RecoveryManager is used to process events in order to sync up with peer if
76/// we can't recover from local consensusdb RoundManager is used for normal
77/// event handling. We suppress clippy warning here because we expect most of
78/// the time we will have RoundManager
79#[allow(clippy::large_enum_variant)]
80pub enum RoundProcessor {
81    Recovery(RecoveryManager),
82    Normal(RoundManager),
83}

174    fn epoch_state(&self) -> &EpochState {
175        match self
176            .processor
177            .as_ref()
178            .expect("EpochManager not started yet")
179        {
180            RoundProcessor::Normal(p) => p.epoch_state(),
181            RoundProcessor::Recovery(p) => p.epoch_state(),
182        }
183    }

185    fn epoch(&self) -> u64 { self.epoch_state().epoch }

187    fn create_round_state(
188        &self, time_service: Arc<dyn TimeService>,
189        timeout_sender: channel::Sender<(u64, Round)>,
190        proposal_timeout_sender: channel::Sender<(u64, Round)>,
191        new_round_timeout_sender: channel::Sender<(u64, Round)>, epoch: u64,
192    ) -> RoundState {
193        // 1.5^6 ~= 11
194        // Timeout goes from initial_timeout to initial_timeout*11 in 6 steps
195        let base_interval = Duration::from_millis(
196            if epoch < self.config.cip113_transition_epoch {
197                self.config.round_initial_timeout_ms
198            } else {
199                self.config.cip113_round_initial_timeout_ms
200            },
201        );
202        let time_interval =
203            Box::new(ExponentialTimeInterval::new(base_interval, 1.2, 6));
204        RoundState::new(
205            time_interval,
206            time_service,
207            timeout_sender,
208            proposal_timeout_sender,
209            new_round_timeout_sender,
210        )
211    }

213    /// Create a proposer election handler based on proposers
214    fn create_proposer_election(
215        &self, epoch_state: &EpochState,
216    ) -> Box<dyn ProposerElection + Send + Sync> {
217        let proposers = epoch_state
218            .verifier()
219            .get_ordered_account_addresses_iter()
220            .collect::<Vec<_>>();
221        match &self.config.proposer_type {
222            ConsensusProposerType::RotatingProposer => Box::new(
223                RotatingProposer::new(proposers, self.config.contiguous_rounds),
224            ),
225            // We don't really have a fixed proposer!
226            ConsensusProposerType::FixedProposer => {
227                let proposer = choose_leader(proposers);
228                Box::new(RotatingProposer::new(
229                    vec![proposer],
230                    self.config.contiguous_rounds,
231                ))
232            }
233            ConsensusProposerType::RoundProposer(round_proposers) => {
234                // Hardcoded to the first proposer
235                let default_proposer = proposers.get(0).unwrap();
236                Box::new(RoundProposer::new(
237                    round_proposers.clone(),
238                    *default_proposer,
239                ))
240            }
241            ConsensusProposerType::VrfProposer => Box::new(VrfProposer::new(
242                self.author,
243                self.config
244                    .safety_rules
245                    .vrf_private_key
246                    .as_ref()
247                    .expect(
248                        "VRF private key mush be set for VRF leader election",
249                    )
250                    .private_key(),
251                self.config.safety_rules.vrf_proposal_threshold,
252                epoch_state.clone(),
253            )),
254        }
255    }

257    async fn process_epoch_retrieval(
258        &mut self, request: EpochRetrievalRequest, peer_id: AccountAddress,
259    ) -> anyhow::Result<()> {
260        diem_debug!(
261            LogSchema::new(LogEvent::ReceiveEpochRetrieval)
262                .remote_peer(peer_id)
263                .epoch(self.epoch()),
264            "[EpochManager] receive {}",
265            request,
266        );
267        let proof = self
268            .storage
269            .pos_ledger_db()
270            .get_epoch_ending_ledger_infos(
271                request.start_epoch,
272                request.end_epoch,
273            )
274            .map_err(DbError::from)
275            .context("[EpochManager] Failed to get epoch proof")?;
276        let msg = ConsensusMsg::EpochChangeProof(Box::new(proof));
277        self.network_sender.send_to(peer_id, &msg).context(format!(
278            "[EpochManager] Failed to send epoch proof to {}",
279            peer_id
280        ))
281    }

283    async fn process_different_epoch(
284        &mut self, different_epoch: u64, peer_id: AccountAddress,
285    ) -> anyhow::Result<()> {
286        diem_debug!(
287            LogSchema::new(LogEvent::ReceiveMessageFromDifferentEpoch)
288                .remote_peer(peer_id)
289                .epoch(self.epoch()),
290            remote_epoch = different_epoch,
291        );
292        match different_epoch.cmp(&self.epoch()) {
293            // We try to help nodes that have lower epoch than us
294            Ordering::Less => {
295                self.process_epoch_retrieval(
296                    EpochRetrievalRequest {
297                        start_epoch: different_epoch,
298                        end_epoch: self.epoch(),
299                    },
300                    peer_id,
301                )
302                .await
303            }
304            // We request proof to join higher epoch
305            Ordering::Greater => {
306                let request = EpochRetrievalRequest {
307                    start_epoch: self.epoch(),
308                    end_epoch: different_epoch,
309                };
310                let msg =
311                    ConsensusMsg::EpochRetrievalRequest(Box::new(request));
312                self.network_sender.send_to(peer_id, &msg).context(format!(
313                    "[EpochManager] Failed to send epoch retrieval to {}",
314                    peer_id
315                ))
316            }
317            Ordering::Equal => {
318                bail!("[EpochManager] Same epoch should not come to process_different_epoch");
319            }
320        }
321    }

323    async fn start_new_epoch(
324        &mut self, proof: EpochChangeProof, peer_id: AccountAddress,
325    ) -> anyhow::Result<()> {
326        let ledger_info = proof
327            .verify(self.epoch_state())
328            .context("[EpochManager] Invalid EpochChangeProof")?;
329        diem_debug!(
330            LogSchema::new(LogEvent::NewEpoch)
331                .epoch(ledger_info.ledger_info().next_block_epoch()),
332            "Received verified epoch change",
333        );
334
335        // make sure storage is on this ledger_info too, it should be no-op if
336        // it's already committed
337        // self.state_computer
338        //     .sync_to(ledger_info.clone())
339        //     .await
340        //     .context(format!(
341        //         "[EpochManager] State sync to new epoch {}",
342        //         ledger_info
343        //     ))?;
344        for ledger_info in proof.get_all_ledger_infos() {
345            let mut new_epoch = false;
346            match self.processor_mut() {
347                RoundProcessor::Recovery(_) => {
348                    bail!("start_new_epoch for Recovery processor");
349                }
350                RoundProcessor::Normal(p) => {
351                    if ledger_info.ledger_info().epoch()
352                        == p.epoch_state().epoch
353                    {
354                        p.sync_to_ledger_info(&ledger_info, peer_id).await?;
355                        new_epoch = true;
356                    } else {
357                        diem_error!(
358                            "Unexpected epoch change: me={} get={}",
359                            p.epoch_state().epoch,
360                            ledger_info.ledger_info().epoch()
361                        );
362                    }
363                }
364            }
365            if new_epoch {
366                monitor!("reconfig", self.expect_new_epoch().await);
367            }
368        }
369
370        Ok(())
371    }

373    async fn start_round_manager(
374        &mut self, recovery_data: RecoveryData, epoch_state: EpochState,
375    ) {
376        // Release the previous RoundManager, especially the SafetyRule client
377        self.processor = None;
378        let epoch = epoch_state.epoch;
379        counters::EPOCH.set(epoch_state.epoch as i64);
380        counters::CURRENT_EPOCH_VALIDATORS
381            .set(epoch_state.verifier().len() as i64);
382        diem_info!(
383            epoch = epoch_state.epoch,
384            validators = epoch_state.verifier().to_string(),
385            root_block = recovery_data.root_block(),
386            "Starting new epoch",
387        );
388        let last_vote = recovery_data.last_vote();
389
390        diem_info!(epoch = epoch, "Create BlockStore");
391        let block_store = Arc::new(BlockStore::new(
392            Arc::clone(&self.storage),
393            recovery_data,
394            Arc::clone(&self.state_computer),
395            self.config.max_pruned_blocks_in_mem,
396            Arc::clone(&self.time_service),
397            self.pow_handler.clone(),
398        ));
399
400        diem_info!(epoch = epoch, "Update SafetyRules");
401
402        let mut safety_rules = MetricsSafetyRules::new(
403            self.safety_rules_manager.client(),
404            self.storage.clone(),
405        );
406        if let Err(error) = safety_rules.perform_initialize() {
407            diem_error!(
408                epoch = epoch,
409                error = error,
410                "Unable to initialize safety rules.",
411            );
412        }
413
414        diem_info!(epoch = epoch, "Create ProposalGenerator");
415        // TODO(lpl): Decide key management.
416        // txn manager is required both by proposal generator (to pull the
417        // proposers) and by event processor (to update their status).
418        let proposal_generator =
419            match epoch_state.verifier().get_public_key(&self.author) {
420                Some(_public_key) => {
421                    let private_key = self
422                        .config
423                        .safety_rules
424                        .test
425                        .as_ref()
426                        .expect("test config set")
427                        .consensus_key
428                        .as_ref()
429                        .expect("private key set in pos")
430                        .private_key();
431                    Some(ProposalGenerator::new(
432                        self.author,
433                        block_store.clone(),
434                        self.txn_manager.clone(),
435                        self.time_service.clone(),
436                        self.config.max_block_size,
437                        self.pow_handler.clone(),
438                        private_key,
439                    ))
440                }
441                None => None,
442            };
443
444        diem_info!(epoch = epoch, "Create RoundState");
445        let round_state = self.create_round_state(
446            self.time_service.clone(),
447            self.timeout_sender.clone(),
448            self.proposal_timeout_sender.clone(),
449            self.new_round_timeout_sender.clone(),
450            epoch,
451        );
452
453        diem_info!(epoch = epoch, "Create ProposerElection");
454        let proposer_election = self.create_proposer_election(&epoch_state);
455        let network_sender = ConsensusNetworkSender::new(
456            self.author,
457            self.network_sender.clone(),
458            //self.self_sender.clone(),
459            epoch_state.verifier().clone(),
460        );
461
462        let mut processor = RoundManager::new(
463            epoch_state,
464            block_store,
465            round_state,
466            proposer_election,
467            proposal_generator,
468            safety_rules,
469            network_sender,
470            self.txn_manager.clone(),
471            self.storage.clone(),
472            self.config.sync_only,
473            self.tx_sender.clone(),
474            self.config.chain_id,
475            self.is_voting,
476            self.election_control.clone(),
477            self.config
478                .safety_rules
479                .test
480                .as_ref()
481                .and_then(|config| config.consensus_key.clone()),
482            self.config.safety_rules.vrf_private_key.clone(),
483        );
484        processor.start(last_vote).await;
485        self.processor = Some(RoundProcessor::Normal(processor));
486        diem_info!(epoch = epoch, "RoundManager started");
487    }

514    async fn start_processor(&mut self, payload: OnChainConfigPayload) {
515        let epoch_state: EpochState = payload.get().unwrap_or_else(|_| {
516            let validator_set: ValidatorSet = payload.get().unwrap();
517            EpochState::new(
518                payload.epoch(),
519                (&validator_set).into(),
520                // genesis pivot decision
521                self.storage
522                    .pos_ledger_db()
523                    .get_latest_ledger_info()
524                    .expect("non-empty ledger info")
525                    .ledger_info()
526                    .pivot_decision()
527                    .unwrap()
528                    .block_hash
529                    .as_bytes()
530                    .to_vec(),
531            )
532        });
533        diem_debug!("start_processor: epoch_state={:?}", epoch_state);
534
535        match self.storage.start() {
536            LivenessStorageData::RecoveryData(initial_data) => {
537                self.start_round_manager(initial_data, epoch_state).await
538            }
539            LivenessStorageData::LedgerRecoveryData(ledger_recovery_data) => {
540                self.start_recovery_manager(ledger_recovery_data, epoch_state)
541                    .await
542            }
543        }
544    }

546    async fn process_message(
547        &mut self, peer_id: AccountAddress, consensus_msg: ConsensusMsg,
548    ) -> anyhow::Result<()> {
549        // we can't verify signatures from a different epoch
550        let maybe_unverified_event =
551            self.process_epoch(peer_id, consensus_msg).await?;
552
553        // This msg is duplicate or unuseful, so we do not need to verify it.
554        if !self.filter_unverified_event(&maybe_unverified_event) {
555            return Ok(());
556        }
557
558        if let Some(unverified_event) = maybe_unverified_event {
559            // same epoch -> run well-formedness + signature check
560            let verified_event = unverified_event
561                .clone()
562                .verify(
563                    &self.epoch_state().verifier(),
564                    self.epoch_state().vrf_seed.as_slice(),
565                )
566                .context("[EpochManager] Verify event")
567                .map_err(|err| {
568                    diem_error!(
569                        SecurityEvent::ConsensusInvalidMessage,
570                        remote_peer = peer_id,
571                        error = ?err,
572                        unverified_event = unverified_event
573                    );
574                    err
575                })?;
576
577            // process the verified event
578            self.process_event(peer_id, verified_event).await?;
579        }
580        Ok(())
581    }

583    async fn process_epoch(
584        &mut self, peer_id: AccountAddress, msg: ConsensusMsg,
585    ) -> anyhow::Result<Option<UnverifiedEvent>> {
586        match msg {
587            ConsensusMsg::ProposalMsg(_)
588            | ConsensusMsg::SyncInfo(_)
589            | ConsensusMsg::VoteMsg(_) => {
590                let event: UnverifiedEvent = msg.into();
591                if event.epoch() == self.epoch() {
592                    return Ok(Some(event));
593                } else {
594                    monitor!(
595                        "process_different_epoch_consensus_msg",
596                        self.process_different_epoch(event.epoch(), peer_id)
597                            .await?
598                    );
599                }
600            }
601            ConsensusMsg::EpochChangeProof(proof) => {
602                let msg_epoch = proof.epoch()?;
603                diem_debug!(
604                    LogSchema::new(LogEvent::ReceiveEpochChangeProof)
605                        .remote_peer(peer_id)
606                        .epoch(self.epoch()),
607                    "Proof from epoch {}",
608                    msg_epoch,
609                );
610                if msg_epoch == self.epoch() {
611                    monitor!(
612                        "process_epoch_proof",
613                        self.start_new_epoch(*proof, peer_id).await?
614                    );
615                } else {
616                    debug!(
617                        "[EpochManager] Unexpected epoch proof from epoch {}, local epoch {}",
618                        msg_epoch,
619                        self.epoch()
620                    );
621                }
622            }
623            ConsensusMsg::EpochRetrievalRequest(request) => {
624                ensure!(
625                    request.end_epoch <= self.epoch(),
626                    "[EpochManager] Received EpochRetrievalRequest beyond what we have locally"
627                );
628                monitor!(
629                    "process_epoch_retrieval",
630                    self.process_epoch_retrieval(*request, peer_id).await?
631                );
632            }
633            _ => {
634                bail!("[EpochManager] Unexpected messages: {:?}", msg);
635            }
636        }
637        Ok(None)
638    }

640    async fn process_event(
641        &mut self, peer_id: AccountAddress, event: VerifiedEvent,
642    ) -> anyhow::Result<()> {
643        match self.processor_mut() {
644            RoundProcessor::Recovery(p) => {
645                let recovery_data = match event {
646                    VerifiedEvent::ProposalMsg(proposal) => {
647                        p.process_proposal_msg(*proposal).await
648                    }
649                    VerifiedEvent::VoteMsg(vote) => {
650                        p.process_vote_msg(*vote).await
651                    }
652                    VerifiedEvent::SyncInfo(sync_info) => {
653                        p.sync_up(&sync_info, peer_id).await
654                    }
655                }?;
656                let epoch_state = p.epoch_state().clone();
657                diem_info!("Recovered from SyncProcessor");
658                self.start_round_manager(recovery_data, epoch_state).await;
659                Ok(())
660            }
661            RoundProcessor::Normal(p) => match event {
662                VerifiedEvent::ProposalMsg(proposal) => monitor!(
663                    "process_proposal",
664                    p.process_proposal_msg(*proposal).await
665                ),
666                VerifiedEvent::VoteMsg(vote) => {
667                    monitor!("process_vote", p.process_vote_msg(*vote).await)
668                }
669                VerifiedEvent::SyncInfo(sync_info) => monitor!(
670                    "process_sync_info",
671                    p.process_sync_info_msg(*sync_info, peer_id).await
672                ),
673            },
674        }
675    }

677    /// Return true for events that we need to process.
678    fn filter_unverified_event(
679        &mut self, maybe_event: &Option<UnverifiedEvent>,
680    ) -> bool {
681        let event = match maybe_event {
682            Some(event) => event,
683            None => return false,
684        };
685        let processor = match self.processor_mut() {
686            RoundProcessor::Recovery(_) => return true,
687            RoundProcessor::Normal(p) => p,
688        };
689        match event {
690            UnverifiedEvent::ProposalMsg(p) => {
691                processor.filter_proposal(p.as_ref())
692            }
693            UnverifiedEvent::VoteMsg(v) => processor.filter_vote(v.as_ref()),
694            _ => true,
695        }
696    }

759    pub async fn start(
760        mut self, mut round_timeout_sender_rx: channel::Receiver<(u64, Round)>,
761        mut proposal_timeout_sender_rx: channel::Receiver<(u64, Round)>,
762        mut new_round_timeout_sender_rx: channel::Receiver<(u64, Round)>,
763        mut network_receivers: NetworkReceivers,
764        mut test_command_receiver: channel::Receiver<TestCommand>,
765        stopped: Arc<AtomicBool>,
766    ) {
767        // initial start of the processor
768        self.expect_new_epoch().await;
769        diem_debug!("EpochManager main_loop starts");
770        loop {
771            if stopped.load(AtomicOrdering::SeqCst) {
772                break;
773            }
774            let result = monitor!(
775                "main_loop",
776                select_biased! {
777                    command = test_command_receiver.select_next_some() => {
778                        self.process_test_command(command).await
779                    }
780                    round = round_timeout_sender_rx.select_next_some() => {
781                        monitor!("process_local_timeout", self.process_local_timeout(round).await)
782                    }
783                    round = proposal_timeout_sender_rx.select_next_some() => {
784                        monitor!("process_proposal_timeout", self.process_proposal_timeout(round).await)
785                    }
786                    round = new_round_timeout_sender_rx.select_next_some() => {
787                        monitor!("process_new_round_timeout", self.process_new_round_timeout(round).await)
788                    }
789                    msg = network_receivers.consensus_messages.select_next_some() => {
790                        let (peer, msg) = (msg.0, msg.1);
791                        monitor!("process_message", self.process_message(peer, msg).await.with_context(|| format!("from peer: {}", peer)))
792                    }
793                    block_retrieval = network_receivers.block_retrieval.select_next_some() => {
794                        monitor!("process_block_retrieval", self.process_block_retrieval(block_retrieval).await)
795                    }
796                }
797            );
798            let round_state =
799                if let RoundProcessor::Normal(p) = self.processor_mut() {
800                    Some(p.round_state())
801                } else {
802                    None
803                };
804            match result {
805                Ok(_) => diem_trace!(RoundStateLogSchema::new(round_state)),
806                Err(e) => {
807                    counters::ERROR_COUNT.inc();
808                    diem_error!(error = ?e, kind = error_kind(&e), RoundStateLogSchema::new(round_state));
809                }
810            }
811
812            // Continually capture the time of consensus process to ensure that
813            // clock skew between validators is reasonable and to
814            // find any unusual (possibly byzantine) clock behavior.
815            counters::OP_COUNTERS
816                .gauge("time_since_epoch_ms")
817                .set(duration_since_epoch().as_millis() as i64);
818        }
819    }

845                self.process_proposal_timeout(round).await
846            }

851                    }

853                };

856            }

861                    }
