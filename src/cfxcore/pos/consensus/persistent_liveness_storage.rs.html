30/// PersistentLivenessStorage is essential for maintaining liveness when a node
31/// crashes.  Specifically, upon a restart, a correct node will recover.  Even
32/// if all nodes crash, liveness is guaranteed.
33/// Blocks persisted are proposed but not yet committed.  The committed state is
34/// persisted via StateComputer.
35pub trait PersistentLivenessStorage: Send + Sync {
36    /// Persist the blocks and quorum certs into storage atomically.
37    fn save_tree(
38        &self, blocks: Vec<Block>, quorum_certs: Vec<QuorumCert>,
39    ) -> Result<()>;
40
41    /// Delete the corresponding blocks and quorum certs atomically.
42    fn prune_tree(&self, block_ids: Vec<HashValue>) -> Result<()>;
43
44    /// Persist consensus' state
45    fn save_vote(&self, vote: &Vote) -> Result<()>;
46
47    /// Construct data that can be recovered from ledger
48    #[allow(dead_code)]
49    fn recover_from_ledger(&self) -> LedgerRecoveryData;
50
51    /// Construct necessary data to start consensus.
52    fn start(&self) -> LivenessStorageData;
53
54    /// Persist the highest timeout certificate for improved liveness - proof
55    /// for other replicas to jump to this round
56    fn save_highest_timeout_cert(
57        &self, highest_timeout_cert: TimeoutCertificate,
58    ) -> Result<()>;
59
60    /// Retrieve a epoch change proof for SafetyRules so it can instantiate its
61    /// ValidatorVerifier.
62    fn retrieve_epoch_change_proof(
63        &self, version: u64,
64    ) -> Result<EpochChangeProof>;
65
66    fn save_ledger_blocks(&self, _blocks: Vec<Block>) -> Result<()> {
67        unimplemented!()
68    }
69
70    fn get_ledger_block(&self, _block_id: &HashValue) -> Result<Option<Block>> {
71        unimplemented!()
72    }
73
74    #[allow(dead_code)]
75    fn prune_staking_events(
76        &self, _committed_pivot_decision: &PivotBlockDecision,
77    ) -> Result<()> {
78        unimplemented!()
79    }
80
81    /// Returns a handle of the pos-ledger-db.
82    fn pos_ledger_db(&self) -> Arc<dyn DbReader>;
83}

88/// LedgerRecoveryData is a subset of RecoveryData that we can get solely from
89/// ledger info.
90#[derive(Clone)]
91pub struct LedgerRecoveryData {
92    storage_ledger: LedgerInfo,
93}
94
95impl LedgerRecoveryData {
96    pub fn new(storage_ledger: LedgerInfo) -> Self {
97        LedgerRecoveryData { storage_ledger }
98    }
99
100    pub fn commit_round(&self) -> Round { self.storage_ledger.round() }
101
102    /// Finds the root (last committed block) and returns the root block, the QC
103    /// to the root block and the ledger info for the root block, return an
104    /// error if it can not be found.
105    ///
106    /// We guarantee that the block corresponding to the storage's latest ledger
107    /// info always exists.
108    fn find_root(
109        &self, blocks: &mut Vec<Block>, quorum_certs: &mut Vec<QuorumCert>,
110    ) -> Result<RootInfo> {
111        diem_info!(
112            "The last committed block id as recorded in storage: {}",
113            self.storage_ledger
114        );
115
116        // We start from the block that storage's latest ledger info, if storage
117        // has end-epoch LedgerInfo, we generate the virtual genesis
118        // block
119        let root_id = if self.storage_ledger.ends_epoch() {
120            let genesis = Block::make_genesis_block_from_ledger_info(
121                &self.storage_ledger,
122            );
123            let genesis_qc =
124                QuorumCert::certificate_for_genesis_from_ledger_info(
125                    &self.storage_ledger,
126                    genesis.id(),
127                );
128            let genesis_id = genesis.id();
129            blocks.push(genesis);
130            quorum_certs.push(genesis_qc);
131            genesis_id
132        } else {
133            self.storage_ledger.consensus_block_id()
134        };
135
136        // sort by (epoch, round) to guarantee the topological order of parent
137        // <- child
138        blocks.sort_by_key(|b| (b.epoch(), b.round()));
139
140        let root_idx = blocks
141            .iter()
142            .position(|block| block.id() == root_id)
143            .ok_or_else(|| format_err!("unable to find root: {}", root_id))?;
144        let root_block = blocks.remove(root_idx);
145        let root_quorum_cert = quorum_certs
146            .iter()
147            .find(|qc| qc.certified_block().id() == root_block.id())
148            .ok_or_else(|| format_err!("No QC found for root: {}", root_id))?
149            .clone();
150        let root_ledger_info = quorum_certs
151            .iter()
152            .find(|qc| qc.commit_info().id() == root_block.id())
153            .ok_or_else(|| format_err!("No LI found for root: {}", root_id))?
154            .clone();
155
156        diem_info!("Consensus root block is {}", root_block);
157
158        Ok(RootInfo(root_block, root_quorum_cert, root_ledger_info))
159    }
160}

162pub struct RootMetadata {
163    pub accu_hash: HashValue,
164    pub frozen_root_hashes: Vec<HashValue>,
165    pub num_leaves: Version,
166    pub pivot_decision: Option<PivotBlockDecision>,
167}
168
169impl RootMetadata {
170    pub fn new(
171        num_leaves: u64, accu_hash: HashValue,
172        frozen_root_hashes: Vec<HashValue>,
173        pivot_decision: Option<PivotBlockDecision>,
174    ) -> Self {
175        Self {
176            num_leaves,
177            accu_hash,
178            frozen_root_hashes,
179            pivot_decision,
180        }
181    }
182
183    pub fn version(&self) -> Version { max(self.num_leaves, 1) - 1 }
184
185    #[cfg(any(test, feature = "fuzzing"))]
186    pub fn new_empty() -> Self {
187        Self::new(
188            0,
189            *diem_crypto::hash::ACCUMULATOR_PLACEHOLDER_HASH,
190            vec![],
191            None,
192        )
193    }

196/// The recovery data constructed from raw consensusdb data, it'll find the root
197/// value and blocks that need cleanup or return error if the input data is
198/// inconsistent.
199pub struct RecoveryData {
200    // The last vote message sent by this validator.
201    last_vote: Option<Vote>,
202    root: RootInfo,
203    root_metadata: RootMetadata,
204    // 1. the blocks guarantee the topological ordering - parent <- child.
205    // 2. all blocks are children of the root.
206    blocks: Vec<Block>,
207    quorum_certs: Vec<QuorumCert>,
208    blocks_to_prune: Option<Vec<HashValue>>,
209
210    // Liveness data
211    highest_timeout_certificate: Option<TimeoutCertificate>,
212}
213
214impl RecoveryData {
215    pub fn new(
216        last_vote: Option<Vote>, ledger_recovery_data: LedgerRecoveryData,
217        mut blocks: Vec<Block>, root_metadata: RootMetadata,
218        mut quorum_certs: Vec<QuorumCert>,
219        highest_timeout_certificate: Option<TimeoutCertificate>,
220    ) -> Result<Self> {
221        let root = ledger_recovery_data
222            .find_root(&mut blocks, &mut quorum_certs)
223            .with_context(|| {
224                // for better readability
225                quorum_certs.sort_by_key(|qc| qc.certified_block().round());
226                format!(
227                    "\nRoot id: {}\nBlocks in db: {}\nQuorum Certs in db: {}\n",
228                    ledger_recovery_data.storage_ledger.consensus_block_id(),
229                    blocks
230                        .iter()
231                        .map(|b| format!("\n\t{}", b))
232                        .collect::<Vec<String>>()
233                        .concat(),
234                    quorum_certs
235                        .iter()
236                        .map(|qc| format!("\n\t{}", qc))
237                        .collect::<Vec<String>>()
238                        .concat(),
239                )
240            })?;
241
242        let blocks_to_prune = Some(Self::find_blocks_to_prune(
243            root.0.id(),
244            &mut blocks,
245            &mut quorum_certs,
246        ));
247        let epoch = root.0.epoch();
248        Ok(RecoveryData {
249            last_vote: match last_vote {
250                Some(v) if v.epoch() == epoch => Some(v),
251                _ => None,
252            },
253            root,
254            root_metadata,
255            blocks,
256            quorum_certs,
257            blocks_to_prune,
258            highest_timeout_certificate: match highest_timeout_certificate {
259                Some(tc) if tc.epoch() == epoch => Some(tc),
260                _ => None,
261            },
262        })
263    }
264
265    pub fn root_block(&self) -> &Block { &self.root.0 }
266
267    pub fn last_vote(&self) -> Option<Vote> { self.last_vote.clone() }
268
269    pub fn take(self) -> (RootInfo, RootMetadata, Vec<Block>, Vec<QuorumCert>) {
270        (
271            self.root,
272            self.root_metadata,
273            self.blocks,
274            self.quorum_certs,
275        )
276    }
277
278    pub fn take_blocks_to_prune(&mut self) -> Vec<HashValue> {
279        self.blocks_to_prune
280            .take()
281            .expect("blocks_to_prune already taken")
282    }
283
284    pub fn highest_timeout_certificate(&self) -> Option<TimeoutCertificate> {
285        self.highest_timeout_certificate.clone()
286    }
287
288    fn find_blocks_to_prune(
289        root_id: HashValue, blocks: &mut Vec<Block>,
290        quorum_certs: &mut Vec<QuorumCert>,
291    ) -> Vec<HashValue> {
292        // prune all the blocks that don't have root as ancestor
293        let mut tree = HashSet::new();
294        let mut to_remove = vec![];
295        tree.insert(root_id);
296        // assume blocks are sorted by round already
297        blocks.retain(|block| {
298            if tree.contains(&block.parent_id()) {
299                tree.insert(block.id());
300                true
301            } else {
302                to_remove.push(block.id());
303                false
304            }
305        });
306        quorum_certs.retain(|qc| tree.contains(&qc.certified_block().id()));
307        to_remove
308    }
309}

311/// The proxy we use to persist data in diem db storage service via grpc.
312pub struct StorageWriteProxy {
313    db: Arc<ConsensusDB>,
314    pos_ledger_db: Arc<dyn DbReader>,
315}
316
317impl StorageWriteProxy {
318    pub fn new(config: &NodeConfig, pos_ledger_db: Arc<dyn DbReader>) -> Self {
319        let db = Arc::new(ConsensusDB::new(config.storage.dir()));
320        StorageWriteProxy { db, pos_ledger_db }
321    }
322
323    pub fn consensus_db(&self) -> Arc<ConsensusDB> { self.db.clone() }
324}
325
326impl PersistentLivenessStorage for StorageWriteProxy {
327    fn save_tree(
328        &self, blocks: Vec<Block>, quorum_certs: Vec<QuorumCert>,
329    ) -> Result<()> {
330        Ok(self
331            .db
332            .save_blocks_and_quorum_certificates(blocks, quorum_certs)?)
333    }
334
335    fn prune_tree(&self, block_ids: Vec<HashValue>) -> Result<()> {
336        if !block_ids.is_empty() {
337            // quorum certs that certified the block_ids will get removed
338            self.db.delete_blocks_and_quorum_certificates(block_ids)?;
339        }
340        Ok(())
341    }
342
343    fn save_vote(&self, vote: &Vote) -> Result<()> {
344        Ok(self.db.save_vote(bcs::to_bytes(vote)?)?)
345    }
346
347    fn recover_from_ledger(&self) -> LedgerRecoveryData {
348        let startup_info = self
349            .pos_ledger_db
350            .get_startup_info(true)
351            .expect("unable to read ledger info from storage")
352            .expect("startup info is None");
353
354        LedgerRecoveryData::new(
355            startup_info.latest_ledger_info.ledger_info().clone(),
356        )
357    }
358
359    fn start(&self) -> LivenessStorageData {
360        diem_info!("Start consensus recovery.");
361        let raw_data = self
362            .db
363            .get_data()
364            .expect("unable to recover consensus data");
365
366        let last_vote = raw_data.0.map(|vote_data| {
367            bcs::from_bytes(&vote_data[..])
368                .expect("unable to deserialize last vote msg")
369        });
370
371        let highest_timeout_certificate = raw_data.1.map(|ts| {
372            bcs::from_bytes(&ts[..])
373                .expect("unable to deserialize highest timeout certificate")
374        });
375        let blocks = raw_data.2;
376        let quorum_certs: Vec<_> = raw_data.3;
377        let blocks_repr: Vec<String> =
378            blocks.iter().map(|b| format!("\n\t{}", b)).collect();
379        diem_info!(
380            "The following blocks were restored from ConsensusDB : {}",
381            blocks_repr.concat()
382        );
383        let qc_repr: Vec<String> = quorum_certs
384            .iter()
385            .map(|qc| format!("\n\t{}", qc))
386            .collect();
387        diem_info!(
388            "The following quorum certs were restored from ConsensusDB: {}",
389            qc_repr.concat()
390        );
391
392        // find the block corresponding to storage latest ledger info
393        let startup_info = self
394            .pos_ledger_db
395            .get_startup_info(true)
396            .expect("unable to read ledger info from storage")
397            .expect("startup info is None");
398        diem_debug!("startup_info={:?}", startup_info);
399        let ledger_recovery_data = LedgerRecoveryData::new(
400            startup_info.latest_ledger_info.ledger_info().clone(),
401        );
402        let frozen_root_hashes = startup_info
403            .committed_tree_state
404            .ledger_frozen_subtree_hashes
405            .clone();
406        let root_executed_trees = ExecutedTrees::new_with_pos_state(
407            startup_info.committed_tree_state,
408            startup_info.committed_pos_state,
409        );
410        match RecoveryData::new(
411            last_vote,
412            ledger_recovery_data.clone(),
413            blocks,
414            RootMetadata::new(
415                root_executed_trees.txn_accumulator().num_leaves(),
416                root_executed_trees.state_id(),
417                frozen_root_hashes,
418                startup_info
419                    .latest_ledger_info
420                    .ledger_info()
421                    .pivot_decision()
422                    .cloned(),
423            ),
424            quorum_certs,
425            highest_timeout_certificate,
426        ) {
427            Ok(mut initial_data) => {
428                (self as &dyn PersistentLivenessStorage)
429                    .prune_tree(initial_data.take_blocks_to_prune())
430                    .expect("unable to prune dangling blocks during restart");
431                if initial_data.last_vote.is_none() {
432                    self.db
433                        .delete_last_vote_msg()
434                        .expect("unable to cleanup last vote");
435                }
436                if initial_data.highest_timeout_certificate.is_none() {
437                    self.db
438                        .delete_highest_timeout_certificate()
439                        .expect("unable to cleanup highest timeout cert");
440                }
441                diem_info!(
442                    "Starting up the consensus state machine with recovery data - [last_vote {}], [highest timeout certificate: {}]",
443                    initial_data.last_vote.as_ref().map_or("None".to_string(), |v| v.to_string()),
444                    initial_data.highest_timeout_certificate.as_ref().map_or("None".to_string(), |v| v.to_string()),
445                );
446
447                LivenessStorageData::RecoveryData(initial_data)
448            }
449            Err(e) => {
450                diem_error!(error = ?e, "Failed to construct recovery data");
451                LivenessStorageData::LedgerRecoveryData(ledger_recovery_data)
452            }
453        }
454    }
455
456    fn save_highest_timeout_cert(
457        &self, highest_timeout_cert: TimeoutCertificate,
458    ) -> Result<()> {
459        Ok(self.db.save_highest_timeout_certificate(bcs::to_bytes(
460            &highest_timeout_cert,
461        )?)?)
462    }
463
464    fn retrieve_epoch_change_proof(
465        &self, version: u64,
466    ) -> Result<EpochChangeProof> {
467        let (_, proofs, _) = self
468            .pos_ledger_db
469            .get_state_proof(version)
470            .map_err(DbError::from)?;
471        Ok(proofs)
472    }
473
474    fn pos_ledger_db(&self) -> Arc<dyn DbReader> { self.pos_ledger_db.clone() }
475
476    fn save_ledger_blocks(&self, blocks: Vec<Block>) -> Result<()> {
477        Ok(self.db.save_ledger_blocks(blocks)?)
478    }
479
480    fn get_ledger_block(&self, block_id: &HashValue) -> Result<Option<Block>> {
481        Ok(self.db.get_ledger_block(block_id)?)
482    }
483
484    fn prune_staking_events(
485        &self, committed_pivot_decision: &PivotBlockDecision,
486    ) -> Result<()> {
487        Ok(self
488            .db
489            .delete_staking_events_before(committed_pivot_decision.height)?)
490    }
