26/// PersistentLivenessStorage is essential for maintaining liveness when a node
27/// crashes.  Specifically, upon a restart, a correct node will recover.  Even
28/// if all nodes crash, liveness is guaranteed.
29/// Blocks persisted are proposed but not yet committed.  The committed state is
30/// persisted via StateComputer.
31pub trait PersistentLivenessStorage: Send + Sync {
32    /// Persist the blocks and quorum certs into storage atomically.
33    fn save_tree(
34        &self, blocks: Vec<Block>, quorum_certs: Vec<QuorumCert>,
35    ) -> Result<()>;
36
37    /// Delete the corresponding blocks and quorum certs atomically.
38    fn prune_tree(&self, block_ids: Vec<HashValue>) -> Result<()>;
39
40    /// Persist consensus' state
41    fn save_vote(&self, vote: &Vote) -> Result<()>;
42
43    /// Construct data that can be recovered from ledger
44    #[allow(dead_code)]
45    fn recover_from_ledger(&self) -> LedgerRecoveryData;
46
47    /// Construct necessary data to start consensus.
48    fn start(&self) -> LivenessStorageData;
49
50    /// Persist the highest timeout certificate for improved liveness - proof
51    /// for other replicas to jump to this round
52    fn save_highest_timeout_cert(
53        &self, highest_timeout_cert: TimeoutCertificate,
54    ) -> Result<()>;
55
56    fn save_ledger_blocks(&self, _blocks: Vec<Block>) -> Result<()> {
57        unimplemented!()
58    }
59
60    fn get_ledger_block(&self, _block_id: &HashValue) -> Result<Option<Block>> {
61        unimplemented!()
62    }
63
64    #[allow(dead_code)]
65    fn prune_staking_events(
66        &self, _committed_pivot_decision: &PivotBlockDecision,
67    ) -> Result<()> {
68        unimplemented!()
69    }
70
71    /// Returns a handle of the pos-ledger-db.
72    fn pos_ledger_db(&self) -> Arc<dyn DbReader>;
73}

78/// LedgerRecoveryData is a subset of RecoveryData that we can get solely from
79/// ledger info.
80#[derive(Clone)]
81pub struct LedgerRecoveryData {
82    storage_ledger: LedgerInfo,
83}
84
85impl LedgerRecoveryData {
86    pub fn new(storage_ledger: LedgerInfo) -> Self {
87        LedgerRecoveryData { storage_ledger }
88    }
89
90    pub fn commit_round(&self) -> Round { self.storage_ledger.round() }
91
92    /// Finds the root (last committed block) and returns the root block, the QC
93    /// to the root block and the ledger info for the root block, return an
94    /// error if it can not be found.
95    ///
96    /// We guarantee that the block corresponding to the storage's latest ledger
97    /// info always exists.
98    fn find_root(
99        &self, blocks: &mut Vec<Block>, quorum_certs: &mut Vec<QuorumCert>,
100    ) -> Result<RootInfo> {
101        diem_info!(
102            "The last committed block id as recorded in storage: {}",
103            self.storage_ledger
104        );
105
106        // We start from the block that storage's latest ledger info, if storage
107        // has end-epoch LedgerInfo, we generate the virtual genesis
108        // block
109        let root_id = if self.storage_ledger.ends_epoch() {
110            let genesis = Block::make_genesis_block_from_ledger_info(
111                &self.storage_ledger,
112            );
113            let genesis_qc =
114                QuorumCert::certificate_for_genesis_from_ledger_info(
115                    &self.storage_ledger,
116                    genesis.id(),
117                );
118            let genesis_id = genesis.id();
119            blocks.push(genesis);
120            quorum_certs.push(genesis_qc);
121            genesis_id
122        } else {
123            self.storage_ledger.consensus_block_id()
124        };
125
126        // sort by (epoch, round) to guarantee the topological order of parent
127        // <- child
128        blocks.sort_by_key(|b| (b.epoch(), b.round()));
129
130        let root_idx = blocks
131            .iter()
132            .position(|block| block.id() == root_id)
133            .ok_or_else(|| format_err!("unable to find root: {}", root_id))?;
134        let root_block = blocks.remove(root_idx);
135        let root_quorum_cert = quorum_certs
136            .iter()
137            .find(|qc| qc.certified_block().id() == root_block.id())
138            .ok_or_else(|| format_err!("No QC found for root: {}", root_id))?
139            .clone();
140        let root_ledger_info = quorum_certs
141            .iter()
142            .find(|qc| qc.commit_info().id() == root_block.id())
143            .ok_or_else(|| format_err!("No LI found for root: {}", root_id))?
144            .clone();
145
146        diem_info!("Consensus root block is {}", root_block);
147
148        Ok(RootInfo(root_block, root_quorum_cert, root_ledger_info))
149    }
150}

152pub struct RootMetadata {
153    pub accu_hash: HashValue,
154    pub frozen_root_hashes: Vec<HashValue>,
155    pub num_leaves: Version,
156    pub pivot_decision: Option<PivotBlockDecision>,
157}
158
159impl RootMetadata {
160    pub fn new(
161        num_leaves: u64, accu_hash: HashValue,
162        frozen_root_hashes: Vec<HashValue>,
163        pivot_decision: Option<PivotBlockDecision>,
164    ) -> Self {
165        Self {
166            num_leaves,
167            accu_hash,
168            frozen_root_hashes,
169            pivot_decision,
170        }
171    }
172
173    pub fn version(&self) -> Version { max(self.num_leaves, 1) - 1 }
174
175    #[cfg(any(test, feature = "fuzzing"))]
176    pub fn new_empty() -> Self {
177        Self::new(
178            0,
179            *diem_crypto::hash::ACCUMULATOR_PLACEHOLDER_HASH,
180            vec![],
181            None,
182        )
183    }

186/// The recovery data constructed from raw consensusdb data, it'll find the root
187/// value and blocks that need cleanup or return error if the input data is
188/// inconsistent.
189pub struct RecoveryData {
190    // The last vote message sent by this validator.
191    last_vote: Option<Vote>,
192    root: RootInfo,
193    root_metadata: RootMetadata,
194    // 1. the blocks guarantee the topological ordering - parent <- child.
195    // 2. all blocks are children of the root.
196    blocks: Vec<Block>,
197    quorum_certs: Vec<QuorumCert>,
198    blocks_to_prune: Option<Vec<HashValue>>,
199
200    // Liveness data
201    highest_timeout_certificate: Option<TimeoutCertificate>,
202}
203
204impl RecoveryData {
205    pub fn new(
206        last_vote: Option<Vote>, ledger_recovery_data: LedgerRecoveryData,
207        mut blocks: Vec<Block>, root_metadata: RootMetadata,
208        mut quorum_certs: Vec<QuorumCert>,
209        highest_timeout_certificate: Option<TimeoutCertificate>,
210    ) -> Result<Self> {
211        let root = ledger_recovery_data
212            .find_root(&mut blocks, &mut quorum_certs)
213            .with_context(|| {
214                // for better readability
215                quorum_certs.sort_by_key(|qc| qc.certified_block().round());
216                format!(
217                    "\nRoot id: {}\nBlocks in db: {}\nQuorum Certs in db: {}\n",
218                    ledger_recovery_data.storage_ledger.consensus_block_id(),
219                    blocks
220                        .iter()
221                        .map(|b| format!("\n\t{}", b))
222                        .collect::<Vec<String>>()
223                        .concat(),
224                    quorum_certs
225                        .iter()
226                        .map(|qc| format!("\n\t{}", qc))
227                        .collect::<Vec<String>>()
228                        .concat(),
229                )
230            })?;
231
232        let blocks_to_prune = Some(Self::find_blocks_to_prune(
233            root.0.id(),
234            &mut blocks,
235            &mut quorum_certs,
236        ));
237        let epoch = root.0.epoch();
238        Ok(RecoveryData {
239            last_vote: match last_vote {
240                Some(v) if v.epoch() == epoch => Some(v),
241                _ => None,
242            },
243            root,
244            root_metadata,
245            blocks,
246            quorum_certs,
247            blocks_to_prune,
248            highest_timeout_certificate: match highest_timeout_certificate {
249                Some(tc) if tc.epoch() == epoch => Some(tc),
250                _ => None,
251            },
252        })
253    }
254
255    pub fn root_block(&self) -> &Block { &self.root.0 }
256
257    pub fn last_vote(&self) -> Option<Vote> { self.last_vote.clone() }
258
259    pub fn take(self) -> (RootInfo, RootMetadata, Vec<Block>, Vec<QuorumCert>) {
260        (
261            self.root,
262            self.root_metadata,
263            self.blocks,
264            self.quorum_certs,
265        )
266    }
267
268    pub fn take_blocks_to_prune(&mut self) -> Vec<HashValue> {
269        self.blocks_to_prune
270            .take()
271            .expect("blocks_to_prune already taken")
272    }
273
274    pub fn highest_timeout_certificate(&self) -> Option<TimeoutCertificate> {
275        self.highest_timeout_certificate.clone()
276    }
277
278    fn find_blocks_to_prune(
279        root_id: HashValue, blocks: &mut Vec<Block>,
280        quorum_certs: &mut Vec<QuorumCert>,
281    ) -> Vec<HashValue> {
282        // prune all the blocks that don't have root as ancestor
283        let mut tree = HashSet::new();
284        let mut to_remove = vec![];
285        tree.insert(root_id);
286        // assume blocks are sorted by round already
287        blocks.retain(|block| {
288            if tree.contains(&block.parent_id()) {
289                tree.insert(block.id());
290                true
291            } else {
292                to_remove.push(block.id());
293                false
294            }
295        });
296        quorum_certs.retain(|qc| tree.contains(&qc.certified_block().id()));
297        to_remove
298    }
299}

301/// The proxy we use to persist data in diem db storage service via grpc.
302pub struct StorageWriteProxy {
303    db: Arc<ConsensusDB>,
304    pos_ledger_db: Arc<dyn DbReader>,
305}
306
307impl StorageWriteProxy {
308    pub fn new(config: &NodeConfig, pos_ledger_db: Arc<dyn DbReader>) -> Self {
309        let db = Arc::new(ConsensusDB::new(config.storage.dir()));
310        StorageWriteProxy { db, pos_ledger_db }
311    }
312
313    pub fn consensus_db(&self) -> Arc<ConsensusDB> { self.db.clone() }
314}
315
316impl PersistentLivenessStorage for StorageWriteProxy {
317    fn save_tree(
318        &self, blocks: Vec<Block>, quorum_certs: Vec<QuorumCert>,
319    ) -> Result<()> {
320        Ok(self
321            .db
322            .save_blocks_and_quorum_certificates(blocks, quorum_certs)?)
323    }
324
325    fn prune_tree(&self, block_ids: Vec<HashValue>) -> Result<()> {
326        if !block_ids.is_empty() {
327            // quorum certs that certified the block_ids will get removed
328            self.db.delete_blocks_and_quorum_certificates(block_ids)?;
329        }
330        Ok(())
331    }
332
333    fn save_vote(&self, vote: &Vote) -> Result<()> {
334        Ok(self.db.save_vote(bcs::to_bytes(vote)?)?)
335    }
336
337    fn recover_from_ledger(&self) -> LedgerRecoveryData {
338        let startup_info = self
339            .pos_ledger_db
340            .get_startup_info(true)
341            .expect("unable to read ledger info from storage")
342            .expect("startup info is None");
343
344        LedgerRecoveryData::new(
345            startup_info.latest_ledger_info.ledger_info().clone(),
346        )
347    }
348
349    fn start(&self) -> LivenessStorageData {
350        diem_info!("Start consensus recovery.");
351        let raw_data = self
352            .db
353            .get_data()
354            .expect("unable to recover consensus data");
355
356        let last_vote = raw_data.0.map(|vote_data| {
357            bcs::from_bytes(&vote_data[..])
358                .expect("unable to deserialize last vote msg")
359        });
360
361        let highest_timeout_certificate = raw_data.1.map(|ts| {
362            bcs::from_bytes(&ts[..])
363                .expect("unable to deserialize highest timeout certificate")
364        });
365        let blocks = raw_data.2;
366        let quorum_certs: Vec<_> = raw_data.3;
367        let blocks_repr: Vec<String> =
368            blocks.iter().map(|b| format!("\n\t{}", b)).collect();
369        diem_info!(
370            "The following blocks were restored from ConsensusDB : {}",
371            blocks_repr.concat()
372        );
373        let qc_repr: Vec<String> = quorum_certs
374            .iter()
375            .map(|qc| format!("\n\t{}", qc))
376            .collect();
377        diem_info!(
378            "The following quorum certs were restored from ConsensusDB: {}",
379            qc_repr.concat()
380        );
381
382        // find the block corresponding to storage latest ledger info
383        let startup_info = self
384            .pos_ledger_db
385            .get_startup_info(true)
386            .expect("unable to read ledger info from storage")
387            .expect("startup info is None");
388        diem_debug!("startup_info={:?}", startup_info);
389        let ledger_recovery_data = LedgerRecoveryData::new(
390            startup_info.latest_ledger_info.ledger_info().clone(),
391        );
392        let frozen_root_hashes = startup_info
393            .committed_tree_state
394            .ledger_frozen_subtree_hashes
395            .clone();
396        let root_executed_trees = ExecutedTrees::new_with_pos_state(
397            startup_info.committed_tree_state,
398            startup_info.committed_pos_state,
399        );
400        match RecoveryData::new(
401            last_vote,
402            ledger_recovery_data.clone(),
403            blocks,
404            RootMetadata::new(
405                root_executed_trees.txn_accumulator().num_leaves(),
406                root_executed_trees.state_id(),
407                frozen_root_hashes,
408                startup_info
409                    .latest_ledger_info
410                    .ledger_info()
411                    .pivot_decision()
412                    .cloned(),
413            ),
414            quorum_certs,
415            highest_timeout_certificate,
416        ) {
417            Ok(mut initial_data) => {
418                (self as &dyn PersistentLivenessStorage)
419                    .prune_tree(initial_data.take_blocks_to_prune())
420                    .expect("unable to prune dangling blocks during restart");
421                if initial_data.last_vote.is_none() {
422                    self.db
423                        .delete_last_vote_msg()
424                        .expect("unable to cleanup last vote");
425                }
426                if initial_data.highest_timeout_certificate.is_none() {
427                    self.db
428                        .delete_highest_timeout_certificate()
429                        .expect("unable to cleanup highest timeout cert");
430                }
431                diem_info!(
432                    "Starting up the consensus state machine with recovery data - [last_vote {}], [highest timeout certificate: {}]",
433                    initial_data.last_vote.as_ref().map_or("None".to_string(), |v| v.to_string()),
434                    initial_data.highest_timeout_certificate.as_ref().map_or("None".to_string(), |v| v.to_string()),
435                );
436
437                LivenessStorageData::RecoveryData(initial_data)
438            }
439            Err(e) => {
440                diem_error!(error = ?e, "Failed to construct recovery data");
441                LivenessStorageData::LedgerRecoveryData(ledger_recovery_data)
442            }
443        }
444    }
445
446    fn save_highest_timeout_cert(
447        &self, highest_timeout_cert: TimeoutCertificate,
448    ) -> Result<()> {
449        Ok(self.db.save_highest_timeout_certificate(bcs::to_bytes(
450            &highest_timeout_cert,
451        )?)?)
452    }
453
454    fn pos_ledger_db(&self) -> Arc<dyn DbReader> { self.pos_ledger_db.clone() }
455
456    fn save_ledger_blocks(&self, blocks: Vec<Block>) -> Result<()> {
457        Ok(self.db.save_ledger_blocks(blocks)?)
458    }
459
460    fn get_ledger_block(&self, block_id: &HashValue) -> Result<Option<Block>> {
461        Ok(self.db.get_ledger_block(block_id)?)
462    }
463
464    fn prune_staking_events(
465        &self, committed_pivot_decision: &PivotBlockDecision,
466    ) -> Result<()> {
467        Ok(self
468            .db
469            .delete_staking_events_before(committed_pivot_decision.height)?)
470    }
