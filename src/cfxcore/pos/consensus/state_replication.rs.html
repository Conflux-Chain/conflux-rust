8use super::error::{MempoolError, StateSyncError};

18/// Retrieves and updates the status of transactions on demand (e.g., via
19/// talking with Mempool)
20#[async_trait::async_trait]
21pub trait TxnManager: Send + Sync {
22    /// Brings new transactions to be applied.
23    /// The `exclude_txns` list includes the transactions that are already
24    /// pending in the branch of blocks consensus is trying to extend.
25    async fn pull_txns(
26        &self, max_size: u64, exclude: Vec<&Payload>, hash: HashValue,
27        validators: ValidatorVerifier,
28    ) -> Result<Payload, MempoolError>;
29
30    /// Notifies TxnManager about the executed result of the block,
31    /// which includes the specifics of what transactions succeeded and failed.
32    async fn notify(
33        &self, block: &Block, compute_result: &StateComputeResult,
34    ) -> Result<(), MempoolError>;
35}
36
37/// While Consensus is managing proposed blocks, `StateComputer` is managing the
38/// results of the (speculative) execution of their payload.
39/// StateComputer is using proposed block ids for identifying the transactions.
40#[async_trait::async_trait]
41pub trait StateComputer: Send + Sync {
42    /// How to execute a sequence of transactions and obtain the next state.
43    /// While some of the transactions succeed, some of them can fail.
44    /// In case all the transactions are failed, new_state_id is equal to the
45    /// previous state id.
46    fn compute(
47        &self,
48        // The block that will be computed.
49        block: &Block,
50        // The parent block root hash.
51        parent_block_id: HashValue,
52        catch_up_mode: bool,
53    ) -> Result<StateComputeResult, ExecutionError>;
54
55    /// Send a successful commit. A future is fulfilled when the state is
56    /// finalized.
57    async fn commit(
58        &self, block_ids: Vec<HashValue>,
59        finality_proof: LedgerInfoWithSignatures,
60    ) -> Result<(), ExecutionError>;
61
62    /// Best effort state synchronization to the given target LedgerInfo.
63    /// In case of success (`Result::Ok`) the LI of storage is at the given
64    /// target. In case of failure (`Result::Error`) the LI of storage
65    /// remains unchanged, and the validator can assume there were no
66    /// modifications to the storage made.
67    async fn sync_to(
68        &self, target: LedgerInfoWithSignatures,
69    ) -> Result<(), StateSyncError>;
70}
