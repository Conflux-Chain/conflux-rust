8use crate::pos::consensus::liveness::proposer_election::ProposerElection;
9use consensus_types::common::{Author, Round};
10
11/// The rotating proposer maps a round to an author according to a round-robin
12/// rotation. A fixed proposer strategy loses liveness when the fixed proposer
13/// is down. Rotating proposers won't gather quorum certificates to machine
14/// loss/byzantine behavior on f/n rounds.
15pub struct RotatingProposer {
16    // Ordering of proposers to rotate through (all honest replicas must agree
17    // on this)
18    proposers: Vec<Author>,
19    // Number of contiguous rounds (i.e. round numbers increase by 1) a
20    // proposer is active in a row
21    contiguous_rounds: u32,
22}
23
24/// Choose a proposer that is going to be the single leader (relevant for a mock
25/// fixed proposer election only).
26pub fn choose_leader(peers: Vec<Author>) -> Author {
27    // As it is just a tmp hack function, pick the min PeerId to be a proposer.
28    peers.into_iter().min().expect("No trusted peers found!")
29}
30
31impl RotatingProposer {
32    /// With only one proposer in the vector, it behaves the same as a fixed
33    /// proposer strategy.
34    pub fn new(proposers: Vec<Author>, contiguous_rounds: u32) -> Self {
35        Self {
36            proposers,
37            contiguous_rounds,
38        }
39    }
40}
41
42impl ProposerElection for RotatingProposer {
43    fn get_valid_proposer(&self, round: Round) -> Author {
44        self.proposers[((round / u64::from(self.contiguous_rounds))
45            % self.proposers.len() as u64) as usize]
46    }
47}
