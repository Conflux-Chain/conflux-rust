22/// A reason for starting a new round: introduced for monitoring / debug
23/// purposes.
24#[derive(Serialize, Eq, Debug, PartialEq)]
25pub enum NewRoundReason {
26    QCReady,
27    Timeout,
28}
29
30impl fmt::Display for NewRoundReason {
31    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
32        match self {
33            NewRoundReason::QCReady => write!(f, "QCReady"),
34            NewRoundReason::Timeout => write!(f, "TCReady"),
35        }
36    }
37}

39/// NewRoundEvents produced by RoundState are guaranteed to be monotonically
40/// increasing. NewRoundEvents are consumed by the rest of the system: they can
41/// cause sending new proposals or voting for some proposals that wouldn't have
42/// been voted otherwise. The duration is populated for debugging and testing
43#[derive(Debug, PartialEq, Eq)]
44pub struct NewRoundEvent {
45    pub round: Round,
46    pub reason: NewRoundReason,
47    pub timeout: Duration,
48}
49
50impl fmt::Display for NewRoundEvent {
51    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
52        write!(
53            f,
54            "NewRoundEvent: [round: {}, reason: {}, timeout: {:?}]",
55            self.round, self.reason, self.timeout
56        )
57    }
58}

60/// Determines the maximum round duration based on the round difference between
61/// the current round and the committed round
62pub trait RoundTimeInterval: Send + Sync + 'static {
63    /// Use the index of the round after the highest quorum certificate to
64    /// commit a block and return the duration for this round
65    ///
66    /// Round indices start at 0 (round index = 0 is the first round after the
67    /// round that led to the highest committed round).  Given that round r
68    /// is the highest round to commit a block, then round index 0 is round
69    /// r+1.  Note that for genesis does not follow the 3-chain rule for
70    /// commits, so round 1 has round index 0.  For example, if one wants to
71    /// calculate the round duration of round 6 and the highest committed round
72    /// is 3 (meaning the highest round to commit a block is round 5, then
73    /// the round index is 0.
74    fn get_round_duration(
75        &self, round_index_after_committed_qc: usize,
76    ) -> Duration;
77}

79/// Round durations increase exponentially
80/// Basically time interval is base * mul^power
81/// Where power=max(rounds_since_qc, max_exponent)
82#[derive(Clone)]
83pub struct ExponentialTimeInterval {
84    // Initial time interval duration after a successful quorum commit.
85    base_ms: u64,
86    // By how much we increase interval every time
87    exponent_base: f64,
88    // Maximum time interval won't exceed base * mul^max_pow.
89    // Theoretically, setting it means
90    // that we rely on synchrony assumptions when the known max messaging delay
91    // is max_interval.  Alternatively, we can consider using max_interval
92    // to meet partial synchrony assumptions where while delta is unknown,
93    // it is <= max_interval.
94    max_exponent: usize,
95}
96
97impl ExponentialTimeInterval {
98    #[cfg(any(test, feature = "fuzzing"))]
99    #[allow(unused)]
100    pub fn fixed(duration: Duration) -> Self { Self::new(duration, 1.0, 0) }
101
102    pub fn new(
103        base: Duration, exponent_base: f64, max_exponent: usize,
104    ) -> Self {
105        assert!(
106            max_exponent < 32,
107            "max_exponent for RoundStateTimeInterval should be <32"
108        );
109        assert!(
110            exponent_base.powf(max_exponent as f64).ceil()
111                < f64::from(std::u32::MAX),
112            "Maximum interval multiplier should be less then u32::Max"
113        );
114        ExponentialTimeInterval {
115            base_ms: base.as_millis() as u64, /* any reasonable ms timeout
116                                               * fits u64 perfectly */
117            exponent_base,
118            max_exponent,
119        }
120    }
121}
122
123impl RoundTimeInterval for ExponentialTimeInterval {
124    fn get_round_duration(
125        &self, round_index_after_committed_qc: usize,
126    ) -> Duration {
127        let pow = round_index_after_committed_qc.min(self.max_exponent) as u32;
128        let base_multiplier = self.exponent_base.powf(f64::from(pow));
129        let duration_ms =
130            ((self.base_ms as f64) * base_multiplier).ceil() as u64;
131        Duration::from_millis(duration_ms)
132    }
133}

135/// `RoundState` contains information about a specific round and moves forward
136/// when receives new certificates.
137///
138/// A round `r` starts in the following cases:
139/// * there is a QuorumCert for round `r-1`,
140/// * there is a TimeoutCertificate for round `r-1`.
141///
142/// Round interval calculation is the responsibility of the
143/// RoundStateTimeoutInterval trait. It depends on the delta between the current
144/// round and the highest committed round (the intuition is that we want to
145/// exponentially grow the interval the further the current round is from the
146/// last committed round).
147///
148/// Whenever a new round starts a local timeout is set following the round
149/// interval. This local timeout is going to send the timeout events once in
150/// interval until the new round starts.
151pub struct RoundState {
152    // Determines the time interval for a round given the number of
153    // non-committed rounds since last commit.
154    time_interval: Box<dyn RoundTimeInterval>,
155    // Highest known committed round as reported by the caller. The caller
156    // might choose not to inform the RoundState about certain committed
157    // rounds (e.g., NIL blocks): in this case the committed round in
158    // RoundState might lag behind the committed round of a block tree.
159    highest_committed_round: Round,
160    // Current round is max{highest_qc, highest_tc} + 1.
161    current_round: Round,
162    // The deadline for the next local timeout event. It is reset every time a
163    // new round start, or a previous deadline expires.
164    // Represents as Duration since UNIX_EPOCH.
165    current_round_deadline: Duration,
166    // Service for timer
167    time_service: Arc<dyn TimeService>,
168    // To send local timeout events to the subscriber (e.g., SMR)
169    timeout_sender: channel::Sender<(u64, Round)>,
170    // To send timeout events for proposal selection to the subscriber (e.g.,
171    // SMR)
172    proposal_timeout_sender: channel::Sender<(u64, Round)>,
173    new_round_timeout_sender: channel::Sender<(u64, Round)>,
174    new_round_sent: bool,
175    // Votes received for the current round.
176    pending_votes: PendingVotes,
177    // Vote sent locally for the current round.
178    vote_sent: Option<Vote>,
179}

181#[derive(Default, Schema)]
182pub struct RoundStateLogSchema<'a> {
183    round: Option<Round>,
184    committed_round: Option<Round>,
185    #[schema(display)]
186    pending_votes: Option<&'a PendingVotes>,
187    #[schema(display)]
188    self_vote: Option<&'a Vote>,
189}
190
191impl<'a> RoundStateLogSchema<'a> {
192    pub fn new(round_state: Option<&'a RoundState>) -> Self {
193        match round_state {
194            Some(state) => Self {
195                round: Some(state.current_round),
196                committed_round: Some(state.highest_committed_round),
197                pending_votes: Some(&state.pending_votes),
198                self_vote: state.vote_sent.as_ref(),
199            },
200            None => Self::default(),
201        }
202    }
203}

205impl RoundState {
206    pub fn new(
207        time_interval: Box<dyn RoundTimeInterval>,
208        time_service: Arc<dyn TimeService>,
209        timeout_sender: channel::Sender<(u64, Round)>,
210        proposal_timeout_sender: channel::Sender<(u64, Round)>,
211        new_round_timeout_sender: channel::Sender<(u64, Round)>,
212    ) -> Self {
213        // Our counters are initialized lazily, so they're not going to appear
214        // in Prometheus if some conditions never happen. Invoking get()
215        // function enforces creation.
216        counters::QC_ROUNDS_COUNT.get();
217        counters::TIMEOUT_ROUNDS_COUNT.get();
218        counters::TIMEOUT_COUNT.get();
219
220        Self {
221            time_interval,
222            highest_committed_round: 0,
223            current_round: 0,
224            current_round_deadline: time_service.get_current_timestamp(),
225            time_service,
226            timeout_sender,
227            proposal_timeout_sender,
228            new_round_timeout_sender,
229            new_round_sent: false,
230            pending_votes: PendingVotes::new(),
231            vote_sent: None,
232        }
233    }
234
235    /// Return the current round.
236    pub fn current_round(&self) -> Round { self.current_round }
237
238    /// Returns deadline for current round
239    pub fn current_round_deadline(&self) -> Duration {
240        self.current_round_deadline
241    }
242
243    /// In case the local timeout corresponds to the current round, reset the
244    /// timeout and return true. Otherwise ignore and return false.
245    pub fn process_local_timeout(&mut self, epoch_round: (u64, Round)) -> bool {
246        diem_info!(round = epoch_round.1, "Local timeout");
247        counters::TIMEOUT_COUNT.inc();
248        self.setup_timeout(epoch_round.0);
249        true
250    }
251
252    /// Notify the RoundState about the potentially new QC, TC, and highest
253    /// committed round. Note that some of these values might not be
254    /// available by the caller.
255    pub fn process_certificates(
256        &mut self, sync_info: SyncInfo,
257    ) -> Option<NewRoundEvent> {
258        if sync_info.highest_commit_round() > self.highest_committed_round {
259            self.highest_committed_round = sync_info.highest_commit_round();
260        }
261        let new_round = sync_info.highest_round() + 1;
262        if new_round > self.current_round {
263            // Start a new round.
264            self.current_round = new_round;
265            self.new_round_sent = false;
266            self.pending_votes = PendingVotes::new();
267            self.vote_sent = None;
268            let timeout = self.setup_timeout(sync_info.epoch());
269            // The new round reason is QCReady in case both QC and TC are equal
270            let new_round_reason =
271                if sync_info.highest_timeout_certificate().is_none() {
272                    NewRoundReason::QCReady
273                } else {
274                    NewRoundReason::Timeout
275                };
276            let new_round_event = NewRoundEvent {
277                round: self.current_round,
278                reason: new_round_reason,
279                timeout,
280            };
281            diem_debug!(
282                round = new_round,
283                "Starting new round: {}",
284                new_round_event
285            );
286            return Some(new_round_event);
287        }
288        None
289    }
290
291    pub fn insert_vote(
292        &mut self, vote: &Vote, verifier: &ValidatorVerifier,
293    ) -> VoteReceptionResult {
294        if vote.vote_data().proposed().round() == self.current_round {
295            self.pending_votes.insert_vote(vote, verifier)
296        } else {
297            VoteReceptionResult::UnexpectedRound(
298                vote.vote_data().proposed().round(),
299                self.current_round,
300            )
301        }
302    }
303
304    pub fn record_vote(&mut self, vote: Vote) {
305        if vote.vote_data().proposed().round() == self.current_round {
306            self.vote_sent = Some(vote);
307        }
308    }
309
310    pub fn vote_sent(&self) -> Option<Vote> { self.vote_sent.clone() }
311
312    pub fn get_round_certificate(
313        &self, verifier: &ValidatorVerifier,
314    ) -> VoteReceptionResult {
315        self.pending_votes.get_certificate(verifier)
316    }

318    /// Setup the timeout task and return the duration of the current timeout
319    fn setup_timeout(&mut self, epoch: u64) -> Duration {
320        let timeout_sender = self.timeout_sender.clone();
321        let timeout = self.setup_deadline();
322        diem_trace!(
323            "Scheduling timeout of {} ms for round {}",
324            timeout.as_millis(),
325            self.current_round
326        );
327        self.time_service.run_after(
328            timeout,
329            SendTask::make(timeout_sender, (epoch, self.current_round)),
330        );
331        timeout
332    }
333
334    /// Setup the current round deadline and return the duration of the current
335    /// round
336    fn setup_deadline(&mut self) -> Duration {
337        let round_index_after_committed_round =
338            self.get_round_index_after_committed_round();
339        let timeout = self
340            .time_interval
341            .get_round_duration(round_index_after_committed_round);
342        let now = self.time_service.get_current_timestamp();
343        diem_debug!(
344            round = self.current_round,
345            "{:?} passed since the previous deadline.",
346            now.checked_sub(self.current_round_deadline)
347                .map_or("0 ms".to_string(), |v| format!("{:?}", v))
348        );
349        diem_debug!(
350            round = self.current_round,
351            "Set round deadline to {:?} from now",
352            timeout
353        );
354        self.current_round_deadline = now + timeout;
355        timeout
356    }
357
358    /// Setup the timeout task and return the duration of the current timeout
359    pub fn setup_proposal_timeout(&self, epoch: u64) -> Duration {
360        let proposal_timeout_sender = self.proposal_timeout_sender.clone();
361        let timeout = self.setup_proposal_deadline();
362        diem_trace!(
363            "Scheduling proposal selection timeout of {} ms for round {}",
364            timeout.as_millis(),
365            self.current_round
366        );
367        self.time_service.run_after(
368            timeout,
369            SendTask::make(
370                proposal_timeout_sender,
371                (epoch, self.current_round),
372            ),
373        );
374        timeout
375    }
376
377    /// Setup the timeout task and return the duration of the current timeout
378    pub fn setup_new_round_timeout(&mut self, epoch: u64) -> Option<Duration> {
379        if self.new_round_sent {
380            return None;
381        }
382        let new_round_timeout_sender = self.new_round_timeout_sender.clone();
383        let timeout = self.setup_new_round_deadline();
384        diem_trace!(
385            "Scheduling new round timeout of {} ms for round {}",
386            timeout.as_millis(),
387            self.current_round
388        );
389        self.time_service.run_after(
390            timeout,
391            SendTask::make(
392                new_round_timeout_sender,
393                (epoch, self.current_round),
394            ),
395        );
396        self.new_round_sent = true;
397        Some(timeout)
398    }

417    fn setup_new_round_deadline(&self) -> Duration {
418        let round_index_after_committed_round =
419            self.get_round_index_after_committed_round();
420        let timeout = self
421            .time_interval
422            .get_round_duration(round_index_after_committed_round)
423            / 2;
424        diem_debug!(
425            round = self.current_round,
426            "Set proposal selection deadline to {:?} from now",
427            timeout
428        );
429        timeout
430    }
431
432    fn get_round_index_after_committed_round(&self) -> usize {
433        let round_index_after_committed_round = {
434            if self.highest_committed_round == 0 {
435                // Genesis doesn't require the 3-chain rule for commit, hence
436                // start the index at the round after genesis.
437                self.current_round - 1
438            } else if self.current_round < self.highest_committed_round + 3 {
439                0
440            } else {
441                self.current_round - self.highest_committed_round - 3
442            }
443        } as usize;
444        round_index_after_committed_round
445    }

447    pub fn vote_received(&self, vote: &Vote) -> bool {
448        self.pending_votes.vote_received(vote)
449    }
