15pub fn max_of_collection<I, T: Ord>(collection: I) -> Option<T>
16where I: Iterator<Item = T> {
17    collection.fold(None, |max_so_far, x| match max_so_far {
18        None => Some(x),
19        Some(max_so_far) => Some(cmp::max(max_so_far, x)),
20    })
21}

23// TODO: we validate for Conflux space currently.
24pub fn validate_chain_id(
25    ours: &ChainIdParamsOneChainInner, theirs: ChainIdParamsOneChainInner,
26    peer_height: u64,
27) -> Result<(), Error> {
28    if !ours.matches(&theirs, peer_height) {
29        let error_kind = Error::ChainIdMismatch {
30            ours: ours.clone(),
31            theirs,
32        };
33        debug!("{:?}", error_kind);
34        bail!(error_kind);
35    } else {
36        Ok(())
37    }
38}

40// TODO(thegaram): consider distinguishing between expected and unexpected
41// errors, e.g. some errors suggest the peer requested a non-existent item
42// (normal) while others suggest a local db inconsistency (exception).
43pub fn partition_results<I, E>(
44    it: impl Iterator<Item = Result<I, E>>,
45) -> (Vec<I>, Vec<E>)
46where
47    I: Debug,
48    E: Debug,
49{
50    let (success, failure): (Vec<_>, Vec<_>) = it.partition(Result::is_ok);
51    let success = success.into_iter().map(Result::unwrap).collect();
52    let failure = failure.into_iter().map(Result::unwrap_err).collect();
53    (success, failure)
54}
