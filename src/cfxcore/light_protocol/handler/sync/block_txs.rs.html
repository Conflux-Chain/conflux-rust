32#[derive(Debug)]
33#[allow(dead_code)]
34struct Statistics {
35    cached: usize,
36    in_flight: usize,
37    waiting: usize,
38}
39
40// prioritize earlier requests
41type MissingBlockTxs = TimeOrdered<H256>;
42
43type PendingBlockTxs = PendingItem<Vec<SignedTransaction>, ClonableError>;
44
45pub struct BlockTxs {
46    // helper API for retrieving ledger information
47    ledger: LedgerInfo,
48
49    // series of unique request ids
50    request_id_allocator: Arc<UniqueId>,
51
52    // sync and request manager
53    sync_manager: SyncManager<H256, MissingBlockTxs>,
54
55    // tx sync manager
56    txs: Arc<Txs>,
57
58    // block txs received from full node
59    verified: Arc<RwLock<LruCache<H256, PendingBlockTxs>>>,
60}
61
62impl BlockTxs {
63    pub fn new(
64        consensus: SharedConsensusGraph, peers: Arc<Peers<FullPeerState>>,
65        request_id_allocator: Arc<UniqueId>, txs: Arc<Txs>,
66    ) -> Self {
67        let ledger = LedgerInfo::new(consensus.clone());
68        let sync_manager =
69            SyncManager::new(peers.clone(), msgid::GET_BLOCK_TXS);
70
71        let cache = LruCache::with_expiry_duration(*CACHE_TIMEOUT);
72        let verified = Arc::new(RwLock::new(cache));
73
74        BlockTxs {
75            ledger,
76            request_id_allocator,
77            sync_manager,
78            txs,
79            verified,
80        }
81    }
82
83    #[inline]
84    pub fn print_stats(&self) {
85        debug!(
86            "block tx sync statistics: {:?}",
87            Statistics {
88                cached: self.verified.read().len(),
89                in_flight: self.sync_manager.num_in_flight(),
90                waiting: self.sync_manager.num_waiting(),
91            }
92        );
93    }
94
95    #[inline]
96    pub fn request(
97        &self, hash: H256,
98    ) -> impl Future<Output = Result<Vec<SignedTransaction>>> {
99        let mut verified = self.verified.write();
100
101        if !verified.contains_key(&hash) {
102            let missing = MissingBlockTxs::new(hash);
103            self.sync_manager.insert_waiting(std::iter::once(missing));
104        }
105
106        verified
107            .entry(hash)
108            .or_insert(PendingItem::pending())
109            .clear_error();
110
111        FutureItem::new(hash, self.verified.clone())
112            .map(|res| res.map_err(|e| e.into()))
113    }
114
115    #[inline]
116    pub fn receive(
117        &self, peer: &NodeId, id: RequestId,
118        block_txs: impl Iterator<Item = BlockTxsWithHash>,
119    ) -> Result<()> {
120        for BlockTxsWithHash { hash, block_txs } in block_txs {
121            trace!("Validating block_txs {:?} with hash {}", block_txs, hash);
122
123            match self.sync_manager.check_if_requested(peer, id, &hash)? {
124                None => continue,
125                Some(_) => self.validate_and_store(hash, block_txs)?,
126            };
127        }
128
129        Ok(())
130    }
131
132    #[inline]
133    pub fn validate_and_store(
134        &self, hash: H256, block_txs: Vec<SignedTransaction>,
135    ) -> Result<()> {
136        // validate block txs
137        if let Err(e) = self.validate_block_txs(hash, &block_txs) {
138            // forward error to both rpc caller(s) and sync handler
139            // so we need to make it clonable
140            let e = ClonableError::from(e);
141
142            self.verified
143                .write()
144                .entry(hash)
145                .or_insert(PendingItem::pending())
146                .set_error(e.clone());
147
148            bail!(e);
149        }
150
151        // store block bodies by block hash
152        self.verified
153            .write()
154            .entry(hash)
155            .or_insert(PendingItem::pending())
156            .set(block_txs);
157
158        self.sync_manager.remove_in_flight(&hash);
159        Ok(())
160    }
161
162    #[inline]
163    pub fn clean_up(&self) {
164        // remove timeout in-flight requests
165        let timeout = *BLOCK_TX_REQUEST_TIMEOUT;
166        let block_txs = self.sync_manager.remove_timeout_requests(timeout);
167        trace!("Timeout block-txs ({}): {:?}", block_txs.len(), block_txs);
168        self.sync_manager.insert_waiting(block_txs.into_iter());
169
170        // trigger cache cleanup
171        self.verified.write().get(&Default::default());
172    }
173
174    #[inline]
175    fn send_request(
176        &self, io: &dyn NetworkContext, peer: &NodeId, hashes: Vec<H256>,
177    ) -> Result<Option<RequestId>> {
178        if hashes.is_empty() {
179            return Ok(None);
180        }
181
182        let request_id = self.request_id_allocator.next();
183
184        trace!(
185            "send_request GetBlockTxs peer={:?} id={:?} hashes={:?}",
186            peer,
187            request_id,
188            hashes
189        );
190
191        let msg: Box<dyn Message> =
192            Box::new(GetBlockTxs { request_id, hashes });
193
194        msg.send(io, peer)?;
195        Ok(Some(request_id))
196    }
197
198    #[inline]
199    pub fn sync(&self, io: &dyn NetworkContext) {
200        self.sync_manager.sync(
201            MAX_BLOCK_TXS_IN_FLIGHT,
202            BLOCK_TX_REQUEST_BATCH_SIZE,
203            |peer, block_hashes| self.send_request(io, peer, block_hashes),
204        );
205    }
206
207    #[inline]
208    pub fn validate_block_txs(
209        &self, hash: H256, txs: &Vec<SignedTransaction>,
210    ) -> Result<()> {
211        // validate each transaction first
212        for tx in txs {
213            self.txs.validate_tx(&tx)?;
214        }
215
216        let expected = *self.ledger.header(hash)?.transactions_root();
217        let txs: Vec<_> = txs.iter().map(|tx| Arc::new(tx.clone())).collect();
218        let received = compute_transaction_root(&txs);
219
220        if received != expected {
221            bail!(Error::InvalidTxRoot {
222                hash,
223                expected,
224                received,
225            });
226        }
227
228        Ok(())
229    }
