18use cfx_types::H256;

26#[derive(Debug)]
27#[allow(dead_code)]
28struct Statistics {
29    cached: usize,
30    in_flight: usize,
31    waiting: usize,
32}
33
34// prioritize earlier requests
35type MissingTx = TimeOrdered<H256>;
36
37type PendingTx = PendingItem<SignedTransaction, ClonableError>;
38
39pub struct Txs {
40    // series of unique request ids
41    request_id_allocator: Arc<UniqueId>,
42
43    // sync and request manager
44    sync_manager: SyncManager<H256, MissingTx>,
45
46    // txs received from full node
47    verified: Arc<RwLock<LruCache<H256, PendingTx>>>,
48}
49
50impl Txs {
51    pub fn new(
52        peers: Arc<Peers<FullPeerState>>, request_id_allocator: Arc<UniqueId>,
53    ) -> Self {
54        let sync_manager = SyncManager::new(peers.clone(), msgid::GET_TXS);
55
56        let cache = LruCache::with_expiry_duration(*CACHE_TIMEOUT);
57        let verified = Arc::new(RwLock::new(cache));
58
59        Txs {
60            request_id_allocator,
61            sync_manager,
62            verified,
63        }
64    }
65
66    #[inline]
67    pub fn print_stats(&self) {
68        debug!(
69            "tx sync statistics: {:?}",
70            Statistics {
71                cached: self.verified.read().len(),
72                in_flight: self.sync_manager.num_in_flight(),
73                waiting: self.sync_manager.num_waiting(),
74            }
75        );
76    }
77
78    #[inline]
79    pub fn request_now(
80        &self, io: &dyn NetworkContext, hash: H256,
81    ) -> impl Future<Output = Result<SignedTransaction>> {
82        let mut verified = self.verified.write();
83
84        if !verified.contains_key(&hash) {
85            let missing = std::iter::once(MissingTx::new(hash));
86
87            self.sync_manager.request_now(missing, |peer, hashes| {
88                self.send_request(io, peer, hashes)
89            });
90        }
91
92        verified
93            .entry(hash)
94            .or_insert(PendingItem::pending())
95            .clear_error();
96
97        FutureItem::new(hash, self.verified.clone())
98            .map(|res| res.map_err(|e| e.into()))
99    }
100
101    #[inline]
102    pub fn receive(
103        &self, peer: &NodeId, id: RequestId,
104        txs: impl Iterator<Item = SignedTransaction>,
105    ) -> Result<()> {
106        for tx in txs {
107            let hash = tx.hash();
108            trace!("Validating tx {:?}", hash);
109
110            match self.sync_manager.check_if_requested(peer, id, &hash)? {
111                None => continue,
112                Some(_) => self.validate_and_store(tx)?,
113            };
114        }
115
116        Ok(())
117    }
118
119    #[inline]
120    fn validate_and_store(&self, tx: SignedTransaction) -> Result<()> {
121        let hash = tx.hash();
122
123        // validate tx
124        if let Err(e) = self.validate_tx(&tx) {
125            // forward error to both rpc caller(s) and sync handler
126            // so we need to make it clonable
127            let e = ClonableError::from(e);
128
129            self.verified
130                .write()
131                .entry(hash)
132                .or_insert(PendingItem::pending())
133                .set_error(e.clone());
134
135            bail!(e);
136        }
137
138        self.verified
139            .write()
140            .entry(hash)
141            .or_insert(PendingItem::pending())
142            .set(tx);
143
144        self.sync_manager.remove_in_flight(&hash);
145        Ok(())
146    }
147
148    #[inline]
149    pub fn clean_up(&self) {
150        // remove timeout in-flight requests
151        let timeout = *TX_REQUEST_TIMEOUT;
152        let txs = self.sync_manager.remove_timeout_requests(timeout);
153        trace!("Timeout txs ({}): {:?}", txs.len(), txs);
154        self.sync_manager.insert_waiting(txs.into_iter());
155
156        // trigger cache cleanup
157        self.verified.write().get(&Default::default());
158    }
159
160    #[inline]
161    fn send_request(
162        &self, io: &dyn NetworkContext, peer: &NodeId, hashes: Vec<H256>,
163    ) -> Result<Option<RequestId>> {
164        if hashes.is_empty() {
165            return Ok(None);
166        }
167
168        let request_id = self.request_id_allocator.next();
169
170        trace!(
171            "send_request GetTxs peer={:?} id={:?} hashes={:?}",
172            peer,
173            request_id,
174            hashes
175        );
176
177        let msg: Box<dyn Message> = Box::new(GetTxs { request_id, hashes });
178
179        msg.send(io, peer)?;
180        Ok(Some(request_id))
181    }
182
183    #[inline]
184    pub fn sync(&self, io: &dyn NetworkContext) {
185        self.sync_manager.sync(
186            MAX_TXS_IN_FLIGHT,
187            TX_REQUEST_BATCH_SIZE,
188            |peer, tx_hashes| self.send_request(io, peer, tx_hashes),
189        );
190    }
191
192    #[inline]
193    pub fn validate_tx(&self, tx: &SignedTransaction) -> Result<()> {
194        match tx.verify_public(false /* skip */) {
195            Ok(true) => {}
196            _ => {
197                warn!("Tx signature verification failed for {:?}", tx);
198                bail!(Error::InvalidTxSignature { hash: tx.hash() });
199            }
200        }
201
202        Ok(())
203    }
204}
