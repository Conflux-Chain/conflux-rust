30use super::types::{CallTrace, CallTraceNode, LogCallOrder};
31
32/// An arena of recorded traces.
33///
34/// This type will be populated via the
35/// [TracingInspector](super::TracingInspector).
36#[derive(Clone, Debug, PartialEq, Eq)]
37#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
38pub struct CallTraceArena {
39    /// The arena of recorded trace nodes
40    pub(crate) arena: Vec<CallTraceNode>,
41}
42
43impl CallTraceArena {
44    /// Pushes a new trace into the arena, returning the trace ID
45    ///
46    /// This appends a new trace to the arena, and also inserts a new entry in
47    /// the node's parent node children set if `attach_to_parent` is `true`.
48    /// E.g. if calls to precompiles should not be included in the call
49    /// graph this should be called with [PushTraceKind::PushOnly].
50    pub(crate) fn push_trace(
51        &mut self, mut entry: usize, kind: PushTraceKind, new_trace: CallTrace,
52    ) -> usize {
53        loop {
54            match new_trace.depth {
55                // The entry node, just update it
56                0 => {
57                    self.arena[0].trace = new_trace;
58                    return 0;
59                }
60                // We found the parent node, add the new trace as a child
61                _ if self.arena[entry].trace.depth == new_trace.depth - 1 => {
62                    let id = self.arena.len();
63                    let node = CallTraceNode {
64                        parent: Some(entry),
65                        trace: new_trace,
66                        idx: id,
67                        ..Default::default()
68                    };
69                    self.arena.push(node);
70
71                    // also track the child in the parent node
72                    if kind.is_attach_to_parent() {
73                        let parent = &mut self.arena[entry];
74                        let trace_location = parent.children.len();
75                        parent
76                            .ordering
77                            .push(LogCallOrder::Call(trace_location));
78                        parent.children.push(id);
79                    }
80
81                    return id;
82                }
83                _ => {
84                    // We haven't found the parent node, go deeper
85                    entry = *self.arena[entry]
86                        .children
87                        .last()
88                        .expect("Disconnected trace");
89                }
90            }
91        }
92    }
93
94    /// Returns the nodes in the arena
95    pub fn nodes(&self) -> &[CallTraceNode] { &self.arena }
96
97    /// Consumes the arena and returns the nodes
98    pub fn into_nodes(self) -> Vec<CallTraceNode> { self.arena }
99
100    /// Clears the arena
101    ///
102    /// Note that this method has no effect on the allocated capacity of the
103    /// arena.
104    #[inline]
105    pub fn clear(&mut self) { self.arena.clear(); }
106}
107
108/// How to push a trace into the arena
109pub(crate) enum PushTraceKind {
110    /// This will _only_ push the trace into the arena.
111    PushOnly,
112    /// This will push the trace into the arena, and also insert a new entry in
113    /// the node's parent node children set.
114    PushAndAttachToParent,
115}
116
117impl PushTraceKind {
118    #[inline]
119    const fn is_attach_to_parent(&self) -> bool {
120        matches!(self, Self::PushAndAttachToParent)
121    }
122}
123
124impl Default for CallTraceArena {
125    fn default() -> Self {
126        // The first node is the root node
127        Self {
128            arena: vec![Default::default()],
129        }
130    }
131}
