46/// A type for creating geth style traces
47#[derive(Clone, Debug)]
48pub struct GethTraceBuilder {

50    nodes: Vec<CallTraceNode>,

59    ) -> Self {
60        Self { nodes, _config }
61    }

63    /// Fill in the geth trace with all steps of the trace and its children
64    /// traces in the order they appear in the transaction.
65    fn fill_geth_trace(
66        &self, main_trace_node: &CallTraceNode,
67        opts: &GethDefaultTracingOptions,
68        storage: &mut HashMap<Address, BTreeMap<B256, B256>>,
69        struct_logs: &mut Vec<StructLog>,
70    ) {
71        // A stack with all the steps of the trace and all its children's steps.
72        // This is used to process the steps in the order they appear in the
73        // transactions. Steps are grouped by their Call Trace Node, in
74        // order to process them all in the order they appear in the
75        // transaction, we need to process steps of call nodes when they appear.
76        // When we find a call step, we push all the steps of the child trace on
77        // the stack, so they are processed next. The very next step is
78        // the last item on the stack
79        let mut step_stack =
80            VecDeque::with_capacity(main_trace_node.trace.steps.len());
81
82        main_trace_node.push_steps_on_stack(&mut step_stack);
83
84        // Iterate over the steps inside the given trace
85        while let Some(CallTraceStepStackItem {
86            trace_node,
87            step,
88            call_child_id,
89        }) = step_stack.pop_back()
90        {
91            let mut log = step.convert_to_geth_struct_log(opts);
92
93            // Fill in storage depending on the options
94            if opts.is_storage_enabled() {
95                let contract_storage =
96                    storage.entry(step.contract).or_default();
97                if let Some(change) = step.storage_change {
98                    contract_storage
99                        .insert(change.key.into(), change.value.into());
100                    log.storage = Some(contract_storage.clone());
101                }
102            }
103
104            if opts.is_return_data_enabled() {
105                log.return_data = Some(trace_node.trace.output.clone());
106            }
107
108            // Add step to geth trace
109            struct_logs.push(log);
110
111            // If the step is a call, we first push all the steps of the child
112            // trace on the stack, so they are processed next
113            if let Some(call_child_id) = call_child_id {
114                let child_trace = &self.nodes[call_child_id];
115                child_trace.push_steps_on_stack(&mut step_stack);
116            }
117
118            // if we reached the limit, we stop
119            if let Some(limit) = opts.limit {
120                if limit > 0 && struct_logs.len() >= limit as usize {
121                    break;
122                }
123            }
124        }
125    }

127    /// Generate a geth-style trace e.g. for `debug_traceTransaction`
128    ///
129    /// This expects the gas used and return value for the
130    /// [ExecutionResult](revm_primitives::ExecutionResult) of the executed
131    /// transaction.
132    pub fn geth_traces(
133        &self, receipt_gas_used: u64, return_value: Bytes,
134        opts: GethDefaultTracingOptions,
135    ) -> DefaultFrame {
136        if self.nodes.is_empty() {
137            return Default::default();
138        }
139        // Fetch top-level trace
140        let main_trace_node = &self.nodes[0];
141        let main_trace = &main_trace_node.trace;
142
143        let mut struct_logs = Vec::new();
144        let mut storage = HashMap::new();
145        self.fill_geth_trace(
146            main_trace_node,
147            &opts,
148            &mut storage,
149            &mut struct_logs,
150        );
151
152        DefaultFrame {
153            // If the top-level trace succeeded, then it was a success
154            failed: !main_trace.success,
155            gas: receipt_gas_used,
156            return_value,
157            struct_logs,
158        }
159    }

161    /// Generate a geth-style traces for the call tracer.
162    ///
163    /// This decodes all call frames from the recorded traces.
164    ///
165    /// This expects the gas used for the
166    /// [ExecutionResult](revm_primitives::ExecutionResult) of the executed
167    /// transaction.
168    pub fn geth_call_traces(
169        &self, opts: CallConfig, gas_used: u64,
170    ) -> CallFrame {
171        if self.nodes.is_empty() {
172            return Default::default();
173        }
174
175        let include_logs = opts.with_log.unwrap_or_default();
176        // first fill up the root
177        let main_trace_node = &self.nodes[0];
178        let mut root_call_frame =
179            main_trace_node.geth_empty_call_frame(include_logs);
180        root_call_frame.gas_used = U256::from(gas_used);
181
182        // selfdestructs are not recorded as individual call traces but are
183        // derived from the call trace and are added as additional
184        // `CallFrame` objects to the parent call
185        if let Some(selfdestruct) =
186            main_trace_node.geth_selfdestruct_call_trace()
187        {
188            root_call_frame.calls.push(selfdestruct);
189        }
190
191        if opts.only_top_call.unwrap_or_default() {
192            return root_call_frame;
193        }
194
195        // fill all the call frames in the root call frame with the recorded
196        // traces. traces are identified by their index in the arena
197        // so we can populate the call frame tree by walking up the call tree
198        let mut call_frames = Vec::with_capacity(self.nodes.len());
199        call_frames.push((0, root_call_frame));
200
201        for (idx, trace) in self.nodes.iter().enumerate().skip(1) {
202            // selfdestructs are not recorded as individual call traces but are
203            // derived from the call trace and are added as
204            // additional `CallFrame` objects to the parent call
205            if let Some(selfdestruct) = trace.geth_selfdestruct_call_trace() {
206                call_frames
207                    .last_mut()
208                    .expect("not empty")
209                    .1
210                    .calls
211                    .push(selfdestruct);
212            }
213
214            // include logs only if call and all its parents were successful
215            let include_logs =
216                include_logs && !self.call_or_parent_failed(trace);
217            call_frames.push((idx, trace.geth_empty_call_frame(include_logs)));
218        }
219
220        // pop the _children_ calls frame and move it to the parent
221        // this will roll up the child frames to their parent; this works
222        // because `child idx > parent idx`
223        loop {
224            let (idx, call) = call_frames.pop().expect("call frames not empty");
225            let node = &self.nodes[idx];
226            if let Some(parent) = node.parent {
227                let parent_frame = &mut call_frames[parent];
228                // we need to ensure that calls are in order they are called:
229                // the last child node is the last call, but
230                // since we walk up the tree, we need to always
231                // insert at position 0
232                parent_frame.1.calls.insert(0, call);
233            } else {
234                debug_assert!(
235                    call_frames.is_empty(),
236                    "only one root node has no parent"
237                );
238                return call;
239            }
240        }
241    }

243    /// Returns true if the given trace or any of its parents failed.
244    fn call_or_parent_failed(&self, node: &CallTraceNode) -> bool {
245        if node.trace.is_error() {
246            return true;
247        }
248
249        let mut parent_idx = node.parent;
250        while let Some(idx) = parent_idx {
251            let next = &self.nodes[idx];
252            if next.trace.is_error() {
253                return true;
254            }
255
256            parent_idx = next.parent;
257        }
258        false
259    }

261    ///  Returns the accounts necessary for transaction execution.
262    ///
263    /// The prestate mode returns the accounts necessary to execute a given
264    /// transaction. diff_mode returns the differences between the
265    /// transaction's pre and post-state.
266    ///
267    /// * `state` - The state post-transaction execution.
268    /// * `diff_mode` - if prestate is in diff or prestate mode.
269    /// * `db` - The database to fetch state pre-transaction execution.
270    pub fn geth_prestate_traces<DB: DatabaseRef>(
271        &self, state: State, prestate_config: PreStateConfig, db: DB,
272    ) -> Result<PreStateFrame, DB::Error> {
273        let account_diffs = state.iter().map(|(addr, acc)| (*addr, acc));
274
275        if prestate_config.is_default_mode() {
276            let mut prestate = PreStateMode::default();
277            // we only want changed accounts for things like balance changes etc
278            for (addr, changed_acc) in account_diffs {
279                let db_acc = db.basic_ref(addr)?.unwrap_or_default();
280                let code = load_account_code(&db, &db_acc);
281                let mut acc_state = AccountState::from_account_info(
282                    db_acc.nonce,
283                    db_acc.balance,
284                    code,
285                );
286
287                // insert the original value of all modified storage slots
288                for (key, slot) in changed_acc.storage.iter() {
289                    acc_state
290                        .storage
291                        .insert((*key).into(), slot.original_value.into());
292                }
293
294                prestate.0.insert(addr, acc_state);
295            }
296
297            Ok(PreStateFrame::Default(prestate))
298        } else {
299            let mut state_diff = DiffMode::default();
300            let mut account_change_kinds =
301                HashMap::with_capacity(account_diffs.len());
302            for (addr, changed_acc) in account_diffs {
303                let db_acc = db.basic_ref(addr)?.unwrap_or_default();
304
305                let pre_code = load_account_code(&db, &db_acc);
306
307                let mut pre_state = AccountState::from_account_info(
308                    db_acc.nonce,
309                    db_acc.balance,
310                    pre_code,
311                );
312
313                let mut post_state = AccountState::from_account_info(
314                    changed_acc.info.nonce,
315                    changed_acc.info.balance,
316                    changed_acc
317                        .info
318                        .code
319                        .as_ref()
320                        .map(|code| code.original_bytes()),
321                );
322
323                // handle storage changes
324                for (key, slot) in changed_acc
325                    .storage
326                    .iter()
327                    .filter(|(_, slot)| slot.is_changed())
328                {
329                    pre_state
330                        .storage
331                        .insert((*key).into(), slot.original_value.into());
332                    post_state
333                        .storage
334                        .insert((*key).into(), slot.present_value.into());
335                }
336
337                state_diff.pre.insert(addr, pre_state);
338                state_diff.post.insert(addr, post_state);
339
340                // determine the change type
341                let pre_change = if changed_acc.is_created() {
342                    AccountChangeKind::Create
343                } else {
344                    AccountChangeKind::Modify
345                };
346                let post_change = if changed_acc.is_selfdestructed() {
347                    AccountChangeKind::SelfDestruct
348                } else {
349                    AccountChangeKind::Modify
350                };
351
352                account_change_kinds.insert(addr, (pre_change, post_change));
353            }
354
355            // ensure we're only keeping changed entries
356            state_diff.retain_changed().remove_zero_storage_values();
357
358            self.diff_traces(
359                &mut state_diff.pre,
360                &mut state_diff.post,
361                account_change_kinds,
362            );
363            Ok(PreStateFrame::Diff(state_diff))
364        }
365    }
366
367    /// Returns the difference between the pre and post state of the transaction
368    /// depending on the kind of changes of that account (pre,post)
369    fn diff_traces(
370        &self, pre: &mut BTreeMap<Address, AccountState>,
371        post: &mut BTreeMap<Address, AccountState>,
372        change_type: HashMap<Address, (AccountChangeKind, AccountChangeKind)>,
373    ) {
374        post.retain(|addr, post_state| {
375            // Don't keep destroyed accounts in the post state
376            if change_type
377                .get(addr)
378                .map(|ty| ty.1.is_selfdestruct())
379                .unwrap_or(false)
380            {
381                return false;
382            }
383            if let Some(pre_state) = pre.get(addr) {
384                // remove any unchanged account info
385                post_state.remove_matching_account_info(pre_state);
386            }
387
388            true
389        });
390
391        // Don't keep created accounts the pre state
392        pre.retain(|addr, _pre_state| {
393            // only keep accounts that are not created
394            change_type
395                .get(addr)
396                .map(|ty| !ty.0.is_created())
397                .unwrap_or(true)
398        });
399    }

402/// Loads the code for the given account from the account itself or the database
403///
404/// Returns None if the code hash is the KECCAK_EMPTY hash
405#[inline]
406pub(crate) fn load_account_code<DB: DatabaseRef>(
407    db: DB, db_acc: &AccountInfo,
408) -> Option<Bytes> {
409    db_acc
410        .code
411        .as_ref()
412        .map(|code| code.original_bytes())
413        .or_else(|| {
414            if db_acc.code_hash == KECCAK_EMPTY {
415                None
416            } else {
417                db.code_by_hash_ref(db_acc.code_hash)
418                    .ok()
419                    .map(|code| code.original_bytes())
420            }
421        })
422        .map(Into::into)
423}
