21//! Return data structures
22
23use cfx_types::U256;
24
25use super::Spec;
26
27/// Return data buffer. Holds memory from a previous call and a slice into that
28/// memory.
29#[derive(Debug)]
30pub struct ReturnData {
31    mem: Vec<u8>,
32    offset: usize,
33    size: usize,
34}
35
36impl ::std::ops::Deref for ReturnData {
37    type Target = [u8];
38
39    fn deref(&self) -> &[u8] { &self.mem[self.offset..self.offset + self.size] }
40}
41
42impl From<Vec<u8>> for ReturnData {
43    fn from(value: Vec<u8>) -> Self {
44        ReturnData {
45            offset: 0,
46            size: value.len(),
47            mem: value,
48        }
49    }
50}
51
52impl ReturnData {
53    /// Create empty `ReturnData`.
54    pub fn empty() -> Self {
55        ReturnData {
56            mem: Vec::new(),
57            offset: 0,
58            size: 0,
59        }
60    }
61
62    /// Create `ReturnData` from give buffer and slice.
63    pub fn new(mem: Vec<u8>, offset: usize, size: usize) -> Self {
64        ReturnData { mem, offset, size }
65    }
66}
67
68/// Gas Left: either it is a known value, or it needs to be computed by
69/// processing a return instruction.
70#[derive(Debug)]
71pub enum GasLeft {
72    /// Known gas left
73    Known(U256),
74    /// Return or Revert instruction must be processed.
75    NeedsReturn {
76        /// Amount of gas left.
77        gas_left: U256,
78        /// Return data buffer.
79        data: ReturnData,
80        /// Apply or revert state changes on revert.
81        apply_state: bool,
82    },
83}
84
85impl GasLeft {
86    pub fn charge_return_data_gas(
87        mut self, spec: &Spec,
88    ) -> super::Result<GasLeft> {
89        match &mut self {
90            GasLeft::NeedsReturn { gas_left, data, .. } => {
91                let length = data.len();
92                let return_cost =
93                    U256::from((length + 31) / 32 * spec.memory_gas);
94
95                if *gas_left < return_cost {
96                    Err(super::Error::OutOfGas)
97                } else {
98                    *gas_left -= return_cost;
99                    Ok(self)
100                }
101            }
102            _ => Ok(self),
103        }
104    }
105}
