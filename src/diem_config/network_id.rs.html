13/// Represents the Role that a peer plays in the network ecosystem rather than
14/// the type of node.
15#[derive(
16    Clone,
17    Copy,
18    Debug,
19    Deserialize,
20    Eq,
21    Hash,
22    Ord,
23    PartialEq,
24    PartialOrd,
25    Serialize,
26)]
27pub enum PeerRole {
28    Validator = 0,
29    PreferredUpstream,
30    Upstream,
31    ValidatorFullNode,
32    Downstream,
33    Known,
34    Unknown,
35}
36
37impl Default for PeerRole {
38    /// Default to least trusted
39    fn default() -> Self { PeerRole::Unknown }

108/// A representation of the network being used in communication.
109/// There should only be one of each NetworkId used for a single node (except
110/// for NetworkId::Public), and handshakes should verify that the NetworkId
111/// being used is the same during a handshake, to effectively ensure
112/// communication is restricted to a network.  Network should be checked that it
113/// is not the `DEFAULT_NETWORK`
114#[derive(Clone, Deserialize, Eq, Hash, PartialEq, Serialize)]
115#[serde(rename = "NetworkId", rename_all = "snake_case")]
116pub enum NetworkId {
117    Validator,
118    Public,
119    Private(String),
120}
121
122impl PartialOrd for NetworkId {
123    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
124        Some(self.cmp(other))
125    }
126}
127
128impl Ord for NetworkId {
129    /// Generalized ordering for determining which network is the most
130    /// important. The lower the ordering, the higher the importance (i.e.,
131    /// the validator network is less than all other networks because it has
132    /// the highest priority).
133    fn cmp(&self, other: &Self) -> Ordering {
134        // To simplify logic below, if it's the same it's equal
135        if self == other {
136            Ordering::Equal
137        } else {
138            // Everywhere below assumes that equal has already been covered
139            match self {
140                NetworkId::Validator => Ordering::Less,
141                NetworkId::Public => Ordering::Greater,
142                NetworkId::Private(_) => match other {
143                    NetworkId::Validator => Ordering::Greater,
144                    NetworkId::Public => Ordering::Less,
145                    NetworkId::Private(_) => {
146                        if self.is_vfn_network() {
147                            Ordering::Less
148                        } else {
149                            Ordering::Greater
150                        }
151                    }
152                },
153            }
154        }
155    }
156}

158/// An intra-node identifier for a network of a node unique for a network
159/// This extra layer on top of `NetworkId` mainly exists for the
160/// application-layer (e.g. mempool, state sync) to differentiate between
161/// multiple public networks that a node may belong to
162#[derive(Clone, Deserialize, Eq, Hash, PartialEq, Serialize)]
163pub struct NodeNetworkId(NetworkId, usize);
164
165impl NodeNetworkId {
166    pub fn new(network_id: NetworkId, num_id: usize) -> Self {
167        Self(network_id, num_id)
168    }
169
170    pub fn network_id(&self) -> NetworkId { self.0.clone() }
171}
172
173impl fmt::Debug for NodeNetworkId {
174    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
175        write!(f, "{}", self)
176    }
177}
178
179impl fmt::Display for NodeNetworkId {

181        write!(f, "{}:{}", self.0, self.1)
182    }
183}

202const VFN_NETWORK: &str = "vfn";
203
204impl NetworkId {
205    /// Convenience function to specify the VFN network
206    pub fn vfn_network() -> NetworkId {
207        NetworkId::Private(VFN_NETWORK.to_string())
208    }
209
210    pub fn is_vfn_network(&self) -> bool {
211        matches!(self, NetworkId::Private(network) if network == VFN_NETWORK)
212    }
213
214    pub fn is_validator_network(&self) -> bool {
215        matches!(self, NetworkId::Validator)
216    }
217
218    /// Roles for a prioritization of relative upstreams
219    pub fn upstream_roles(&self, role: &RoleType) -> &'static [PeerRole] {
220        match self {
221            NetworkId::Validator => &[PeerRole::Validator],
222            NetworkId::Public => &[
223                PeerRole::PreferredUpstream,
224                PeerRole::Upstream,
225                PeerRole::ValidatorFullNode,
226            ],
227            NetworkId::Private(_) => {
228                if self.is_vfn_network() {
229                    match role {
230                        RoleType::Validator => &[],
231                        RoleType::FullNode => &[PeerRole::Validator],
232                    }
233                } else {
234                    &[PeerRole::PreferredUpstream, PeerRole::Upstream]
235                }
236            }
237        }
238    }
239
240    /// Roles for a prioritization of relative downstreams
241    pub fn downstream_roles(&self, role: &RoleType) -> &'static [PeerRole] {
242        match self {
243            NetworkId::Validator => &[PeerRole::Validator],
244            // In order to allow fallbacks, we must allow for nodes to accept
245            // ValidatorFullNodes
246            NetworkId::Public => &[
247                PeerRole::ValidatorFullNode,
248                PeerRole::Downstream,
249                PeerRole::Known,
250                PeerRole::Unknown,
251            ],
252            NetworkId::Private(_) => {
253                if self.is_vfn_network() {
254                    match role {
255                        RoleType::Validator => &[PeerRole::ValidatorFullNode],
256                        RoleType::FullNode => &[],
257                    }
258                } else {
259                    // It's a private network, disallow unknown peers
260                    &[PeerRole::Downstream, PeerRole::Known]
261                }
262            }
263        }
264    }
265
266    pub fn as_str(&self) -> &str {
267        match self {
268            NetworkId::Validator => "Validator",
269            NetworkId::Public => "Public",
270            // We used to return "Private({info})" here; however, it's important
271            // to get the network id str without temp allocs, as this is in the
272            // metrics/logging hot path. In theory, someone could set their
273            // network id as `Private("Validator")`, which would result in
274            // confusing metrics/logging output for them, but would not affect
275            // correctness.
276            NetworkId::Private(info) => info.as_ref(),
277        }
278    }
279
280    fn serialize_str<S>(
281        &self, serializer: S,
282    ) -> std::result::Result<S::Ok, S::Error>
283    where S: Serializer {
284        self.as_str().serialize(serializer)
285    }
286}

288/// Identifier of a node, represented as (network_id, peer_id)
289#[derive(Clone, Deserialize, Eq, Hash, PartialEq, Serialize)]
290pub struct PeerNetworkId(pub NodeNetworkId, pub PeerId);
291
292impl PeerNetworkId {
293    pub fn network_id(&self) -> NodeNetworkId { self.0.clone() }
294
295    pub fn raw_network_id(&self) -> NetworkId { self.0.network_id() }
296
297    pub fn peer_id(&self) -> PeerId { self.1 }
298
299    #[cfg(any(test, feature = "fuzzing"))]
300    pub fn random() -> Self {
301        Self(
302            NodeNetworkId::new(NetworkId::default(), 0),
303            PeerId::random(),
304        )
305    }
306
307    #[cfg(any(test, feature = "fuzzing"))]
308    pub fn random_validator() -> Self {
309        Self(
310            NodeNetworkId::new(NetworkId::Validator, 0),
311            PeerId::random(),
312        )
313    }
314}
