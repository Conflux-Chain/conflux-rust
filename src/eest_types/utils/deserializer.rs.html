1use cfx_types::{address_util::hex_to_address, Address};
2use serde::{de, Deserialize};

4/// Deserializes a [string][String] as a [u64].
5pub fn deserialize_str_as_u64<'de, D>(
6    deserializer: D,
7) -> Result<u64, D::Error>
8where D: de::Deserializer<'de> {
9    let string = String::deserialize(deserializer)?;
10
11    if let Some(stripped) = string.strip_prefix("0x") {
12        u64::from_str_radix(stripped, 16)
13    } else {
14        string.parse()
15    }
16    .map_err(serde::de::Error::custom)
17}

19/// Deserializes a [string][String] as an optional [Address].
20pub fn deserialize_maybe_empty<'de, D>(
21    deserializer: D,
22) -> Result<Option<Address>, D::Error>
23where D: de::Deserializer<'de> {
24    let string = String::deserialize(deserializer)?;
25    if string.is_empty() {
26        Ok(None)
27    } else {
28        hex_to_address(&string).map_err(de::Error::custom).map(Some)
29    }
30}
