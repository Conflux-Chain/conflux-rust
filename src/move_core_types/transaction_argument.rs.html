8use crate::{account_address::AccountAddress, value::MoveValue};
9use serde::{Deserialize, Serialize};
10use std::fmt;
11
12#[derive(Clone, Hash, Eq, PartialEq, Serialize, Deserialize)]
13pub enum TransactionArgument {
14    U8(u8),
15    U64(u64),
16    U128(u128),
17    Address(AccountAddress),
18    U8Vector(#[serde(with = "serde_bytes")] Vec<u8>),
19    Bool(bool),
20}
21
22impl fmt::Debug for TransactionArgument {
23    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
24        match self {
25            TransactionArgument::U8(value) => write!(f, "{{U8: {}}}", value),
26            TransactionArgument::U64(value) => write!(f, "{{U64: {}}}", value),
27            TransactionArgument::U128(value) => {
28                write!(f, "{{U128: {}}}", value)
29            }
30            TransactionArgument::Bool(boolean) => {
31                write!(f, "{{BOOL: {}}}", boolean)
32            }
33            TransactionArgument::Address(address) => {
34                write!(f, "{{ADDRESS: {:?}}}", address)
35            }
36            TransactionArgument::U8Vector(vector) => {
37                write!(f, "{{U8Vector: 0x{}}}", hex::encode(vector))
38            }
39        }
40    }
41}
42
43/// Convert the transaction arguments into Move values.
44pub fn convert_txn_args(args: &[TransactionArgument]) -> Vec<Vec<u8>> {
45    args.iter()
46        .map(|arg| {
47            let mv = match arg {
48                TransactionArgument::U8(i) => MoveValue::U8(*i),
49                TransactionArgument::U64(i) => MoveValue::U64(*i),
50                TransactionArgument::U128(i) => MoveValue::U128(*i),
51                TransactionArgument::Address(a) => MoveValue::Address(*a),
52                TransactionArgument::Bool(b) => MoveValue::Bool(*b),
53                TransactionArgument::U8Vector(v) => {
54                    MoveValue::vector_u8(v.clone())
55                }
56            };
57            mv.simple_serialize()
58                .expect("transaction arguments must serialize")
59        })
60        .collect()
61}
