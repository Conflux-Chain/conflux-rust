8//! This module lays out the basic abstract costing schedule for bytecode
9//! instructions.
10//!
11//! It is important to note that the cost schedule defined in this file does not
12//! track hashing operations or other native operations; the cost of each native
13//! operation will be returned by the native function itself.
14use mirai_annotations::*;
15use serde::{Deserialize, Serialize};
16use std::{
17    ops::{Add, Div, Mul, Sub},
18    u64,
19};
20
21/// The underlying carrier for gas-related units and costs. Data with this type
22/// should not be manipulated directly, but instead be manipulated using the
23/// newtype wrappers defined around them and the functions defined in the
24/// `GasAlgebra` trait.
25pub type GasCarrier = u64;
26
27/// A trait encoding the operations permitted on the underlying carrier for the
28/// gas unit, and how other gas-related units can interact with other units --
29/// operations can only be performed across units with the same underlying
30/// carrier (i.e. as long as the underlying data is the same).
31pub trait GasAlgebra<GasCarrier>: Sized
32where GasCarrier: Add<Output = GasCarrier>
33        + Sub<Output = GasCarrier>
34        + Div<Output = GasCarrier>
35        + Mul<Output = GasCarrier>
36        + Copy
37{
38    /// Project a value into the gas algebra.
39    fn new(carrier: GasCarrier) -> Self;
40
41    /// Get the carrier.
42    fn get(&self) -> GasCarrier;
43
44    /// Map a function `f` of one argument over the underlying data.
45    fn map<F: Fn(GasCarrier) -> GasCarrier>(self, f: F) -> Self {
46        Self::new(f(self.get()))
47    }
48
49    /// Map a function `f` of two arguments over the underlying carrier. Note
50    /// that this function can take two different implementations of the
51    /// trait -- one for `self` the other for the second argument. But, we
52    /// enforce that they have the same underlying carrier.
53    fn map2<F: Fn(GasCarrier, GasCarrier) -> GasCarrier>(
54        self, other: impl GasAlgebra<GasCarrier>, f: F,
55    ) -> Self {
56        Self::new(f(self.get(), other.get()))
57    }
58
59    /// Apply a function `f` of two arguments to the carrier. Since `f` is not
60    /// an endomorphism, we return the resulting value, as opposed to the
61    /// result wrapped up in ourselves.
62    fn app<T, F: Fn(GasCarrier, GasCarrier) -> T>(
63        &self, other: &impl GasAlgebra<GasCarrier>, f: F,
64    ) -> T {
65        f(self.get(), other.get())
66    }
67
68    /// We allow casting between GasAlgebras as long as they have the same
69    /// underlying carrier -- i.e. they use the same type to store the
70    /// underlying value.
71    fn unitary_cast<T: GasAlgebra<GasCarrier>>(self) -> T { T::new(self.get()) }
72
73    /// Add the two `GasAlgebra`s together.
74    fn add(self, right: impl GasAlgebra<GasCarrier>) -> Self {
75        self.map2(right, Add::add)
76    }
77
78    /// Subtract one `GasAlgebra` from the other.
79    fn sub(self, right: impl GasAlgebra<GasCarrier>) -> Self {
80        self.map2(right, Sub::sub)
81    }
82
83    /// Multiply two `GasAlgebra`s together.
84    fn mul(self, right: impl GasAlgebra<GasCarrier>) -> Self {
85        self.map2(right, Mul::mul)
86    }
87
88    /// Divide one `GasAlgebra` by the other.
89    fn div(self, right: impl GasAlgebra<GasCarrier>) -> Self {
90        self.map2(right, Div::div)
91    }
92}

94// We would really like to be able to implement the standard arithmetic traits
95// over the GasAlgebra trait, but that isn't possible.
96macro_rules! define_gas_unit {
97    {
98        name: $name: ident,
99        carrier: $carrier: ty,
100        doc: $comment: literal
101    } => {
102        #[derive(Debug, Hash, Eq, PartialEq, Copy, Clone, Serialize, Deserialize)]
103        #[doc=$comment]
104        pub struct $name<GasCarrier>(GasCarrier);
105        impl GasAlgebra<$carrier> for $name<$carrier> {
106            fn new(c: GasCarrier) -> Self {
107                Self(c)
108            }
109            fn get(&self) -> GasCarrier {
110                self.0
111            }
112        }
113    }
114}

173#[derive(Clone, Debug, Serialize, PartialEq, Deserialize)]
174pub struct GasConstants {
175    /// The cost per-byte read from global storage.
176    pub global_memory_per_byte_cost: InternalGasUnits<GasCarrier>,
177
178    /// The cost per-byte written to storage.
179    pub global_memory_per_byte_write_cost: InternalGasUnits<GasCarrier>,
180
181    /// The flat minimum amount of gas required for any transaction.
182    /// Charged at the start of execution.
183    pub min_transaction_gas_units: InternalGasUnits<GasCarrier>,
184
185    /// Any transaction over this size will be charged an additional amount per
186    /// byte.
187    pub large_transaction_cutoff: AbstractMemorySize<GasCarrier>,
188
189    /// The units of gas that to be charged per byte over the
190    /// `large_transaction_cutoff` in addition to
191    /// `min_transaction_gas_units` for transactions whose size exceeds
192    /// `large_transaction_cutoff`.
193    pub intrinsic_gas_per_byte: InternalGasUnits<GasCarrier>,
194
195    /// ~5 microseconds should equal one unit of computational gas. We bound
196    /// the maximum computational time of any given transaction at roughly
197    /// 20 seconds. We want this number and `MAX_PRICE_PER_GAS_UNIT` to
198    /// always satisfy the inequality that `MAXIMUM_NUMBER_OF_GAS_UNITS *
199    /// MAX_PRICE_PER_GAS_UNIT < min(u64::MAX, GasUnits<GasCarrier>::MAX)`
200    /// NB: The bound is set quite high since custom scripts aren't allowed
201    /// except from predefined and vetted senders.
202    pub maximum_number_of_gas_units: GasUnits<GasCarrier>,
203
204    /// The minimum gas price that a transaction can be submitted with.
205    pub min_price_per_gas_unit: GasPrice<GasCarrier>,
206
207    /// The maximum gas unit price that a transaction can be submitted with.
208    pub max_price_per_gas_unit: GasPrice<GasCarrier>,
209
210    pub max_transaction_size_in_bytes: GasCarrier,
211
212    pub gas_unit_scaling_factor: GasCarrier,
213    pub default_account_size: AbstractMemorySize<GasCarrier>,
214}
215
216impl GasConstants {
217    pub fn to_internal_units(
218        &self, units: GasUnits<GasCarrier>,
219    ) -> InternalGasUnits<GasCarrier> {
220        InternalGasUnits::new(units.get() * self.gas_unit_scaling_factor)
221    }
222
223    pub fn to_external_units(
224        &self, units: InternalGasUnits<GasCarrier>,
225    ) -> GasUnits<GasCarrier> {
226        GasUnits::new(units.get() / self.gas_unit_scaling_factor)
227    }
228}

247/// The cost tables, keyed by the serialized form of the bytecode instruction.
248/// We use the serialized form as opposed to the instruction enum itself as the
249/// key since this will be the on-chain representation of bytecode instructions
250/// in the future.
251#[derive(Clone, Debug, Serialize, PartialEq, Deserialize)]
252pub struct CostTable {
253    pub instruction_table: Vec<GasCost>,
254    pub native_table: Vec<GasCost>,
255    pub gas_constants: GasConstants,
256}
257
258impl CostTable {
259    #[inline]
260    pub fn instruction_cost(&self, instr_index: u8) -> &GasCost {
261        precondition!(
262            instr_index > 0
263                && instr_index <= (self.instruction_table.len() as u8)
264        );
265        &self.instruction_table[(instr_index - 1) as usize]
266    }
267
268    #[inline]
269    pub fn native_cost(&self, native_index: u8) -> &GasCost {
270        precondition!(native_index < (self.native_table.len() as u8));
271        &self.native_table[native_index as usize]
272    }
273}

275/// The  `GasCost` tracks:
276/// - instruction cost: how much time/computational power is needed to perform
277///   the instruction
278/// - memory cost: how much memory is required for the instruction, and storage
279///   overhead
280#[derive(Clone, Debug, Serialize, Deserialize, PartialEq)]
281pub struct GasCost {
282    pub instruction_gas: InternalGasUnits<GasCarrier>,
283    pub memory_gas: InternalGasUnits<GasCarrier>,
284}
285
286impl GasCost {
287    pub fn new(instr_gas: GasCarrier, mem_gas: GasCarrier) -> Self {
288        Self {
289            instruction_gas: InternalGasUnits::new(instr_gas),
290            memory_gas: InternalGasUnits::new(mem_gas),
291        }
292    }
293
294    /// Convert a GasCost to a total gas charge in `InternalGasUnits`.
295    #[inline]
296    pub fn total(&self) -> InternalGasUnits<GasCarrier> {
297        self.instruction_gas.add(self.memory_gas)
298    }
