49/// A `VMStatus` is represented as either
50/// - `Executed` indicating successful execution
51/// - `Error` indicating an error from the VM itself
52/// - `MoveAbort` indicating an `abort` ocurred inside of a Move program
53#[derive(
54    Clone, PartialEq, Eq, PartialOrd, Ord, Hash, Serialize, Deserialize,
55)]
56#[cfg_attr(any(test, feature = "fuzzing"), derive(Arbitrary))]
57#[cfg_attr(any(test, feature = "fuzzing"), proptest(no_params))]
58pub enum VMStatus {
59    /// The VM status corresponding to an EXECUTED status code
60    Executed,
61
62    /// Indicates an error from the VM, e.g. OUT_OF_GAS, INVALID_AUTH_KEY,
63    /// RET_TYPE_MISMATCH_ERROR etc.
64    /// The code will neither EXECUTED nor ABORTED
65    Error(StatusCode),
66
67    /// Indicates an `abort` from inside Move code. Contains the location of
68    /// the abort and the code
69    MoveAbort(AbortLocation, /* code */ u64),
70
71    /// Indicates an failure from inside Move code, where the VM could not
72    /// continue exection, e.g. dividing by zero or a missing resource
73    ExecutionFailure {
74        status_code: StatusCode,
75        location: AbortLocation,
76        function: u16,
77        code_offset: u16,
78    },
79}

114/// A status type is one of 5 different variants, along with a fallback variant
115/// in the case that we don't recognize the status code.
116#[derive(Clone, PartialEq, Eq, Debug, Hash)]
117pub enum StatusType {
118    Validation,
119    Verification,
120    InvariantViolation,
121    Deserialization,
122    Execution,
123    Unknown,
124}

126impl VMStatus {
127    /// Return the status code for the `VMStatus`
128    pub fn status_code(&self) -> StatusCode {
129        match self {
130            Self::Executed => StatusCode::EXECUTED,
131            Self::MoveAbort(_, _) => StatusCode::ABORTED,
132            Self::ExecutionFailure { status_code, .. } => *status_code,
133            Self::Error(code) => {
134                let code = *code;
135                debug_assert!(code != StatusCode::EXECUTED);
136                debug_assert!(code != StatusCode::ABORTED);
137                debug_assert!(code.status_type() != StatusType::Execution);
138                code
139            }
140        }
141    }
142
143    /// Returns the Move abort code if the status is `MoveAbort`, and `None`
144    /// otherwise
145    pub fn move_abort_code(&self) -> Option<u64> {
146        match self {
147            Self::MoveAbort(_, code) => Some(*code),
148            Self::Error(_) | Self::ExecutionFailure { .. } | Self::Executed => {
149                None
150            }
151        }
152    }
153
154    /// Return the status type for this `VMStatus`. This is solely determined by
155    /// the `status_code`
156    pub fn status_type(&self) -> StatusType { self.status_code().status_type() }
157
158    /// Returns `Ok` with a recorded status if it should be kept, `Err` of the
159    /// error code if it should be discarded
160    pub fn keep_or_discard(self) -> Result<KeptVMStatus, DiscardedVMStatus> {
161        match self {
162            VMStatus::Executed => Ok(KeptVMStatus::Executed),
163            VMStatus::MoveAbort(location, code) => {
164                Ok(KeptVMStatus::MoveAbort(location, code))
165            }
166            VMStatus::ExecutionFailure {
167                status_code: StatusCode::OUT_OF_GAS,
168                ..
169            }
170            | VMStatus::Error(StatusCode::OUT_OF_GAS) => {
171                Ok(KeptVMStatus::OutOfGas)
172            }
173            VMStatus::ExecutionFailure {
174                status_code: _status_code,
175                location,
176                function,
177                code_offset,
178            } => Ok(KeptVMStatus::ExecutionFailure {
179                location,
180                function,
181                code_offset,
182            }),
183            VMStatus::Error(code) => {
184                match code.status_type() {
185                    // Any unknown error should be discarded
186                    StatusType::Unknown => Err(code),
187                    // Any error that is a validation status (i.e. an error
188                    // arising from the prologue) causes the
189                    // transaction to not be included.
190                    StatusType::Validation => Err(code),
191                    // If the VM encountered an invalid internal state, we
192                    // should discard the transaction.
193                    StatusType::InvariantViolation => Err(code),
194                    // A transaction that publishes code that cannot be verified
195                    // will be charged.
196                    StatusType::Verification => {
197                        Ok(KeptVMStatus::MiscellaneousError)
198                    }
199                    // If we are able to decode the`SignedTransaction`, but
200                    // failed to decode `SingedTransaction.
201                    // raw_transaction.payload` (i.e., the transaction script),
202                    // there should be a charge made to that user's account for
203                    // the gas fees related to decoding,
204                    // running the prologue etc.
205                    StatusType::Deserialization => {
206                        Ok(KeptVMStatus::MiscellaneousError)
207                    }
208                    // Any error encountered during the execution of the
209                    // transaction will charge gas.
210                    StatusType::Execution => {
211                        Ok(KeptVMStatus::ExecutionFailure {
212                            location: AbortLocation::Script,
213                            function: 0,
214                            code_offset: 0,
215                        })
216                    }
217                }
218            }
219        }
220    }
221}

223impl fmt::Display for StatusType {
224    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
225        let string = match self {
226            StatusType::Validation => "Validation",
227            StatusType::Verification => "Verification",
228            StatusType::InvariantViolation => "Invariant violation",
229            StatusType::Deserialization => "Deserialization",
230            StatusType::Execution => "Execution",
231            StatusType::Unknown => "Unknown",
232        };
233        write!(f, "{}", string)
234    }
235}

237impl fmt::Display for VMStatus {
238    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
239        let status_type = self.status_type();
240        let mut status =
241            format!("status {:#?} of type {}", self.status_code(), status_type);
242
243        if let VMStatus::MoveAbort(_, code) = self {
244            status = format!("{} with sub status {}", status, code);
245        }
246
247        write!(f, "{}", status)
248    }
249}

251impl fmt::Display for KeptVMStatus {
252    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
253        write!(f, "status ")?;
254        match self {
255            KeptVMStatus::Executed => write!(f, "EXECUTED"),
256            KeptVMStatus::OutOfGas => write!(f, "OUT_OF_GAS"),
257            KeptVMStatus::MiscellaneousError => write!(f, "MISCELLANEOUS_ERROR"),
258            KeptVMStatus::MoveAbort(location, code) => {
259                write!(f, "ABORTED with code {} in {}", code, location)
260            }
261            KeptVMStatus::ExecutionFailure {
262                location,
263                function,
264                code_offset,
265            } => write!(
266                f,
267                "EXECUTION_FAILURE at bytecode offset {} in function index {} in {}",
268                code_offset, function, location
269            ),
270        }
271    }
272}

274impl fmt::Debug for VMStatus {
275    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
276        match self {
277            VMStatus::Executed => write!(f, "EXECUTED"),
278            VMStatus::Error(code) => {
279                f.debug_struct("ERROR").field("status_code", code).finish()
280            }
281            VMStatus::MoveAbort(location, code) => f
282                .debug_struct("ABORTED")
283                .field("code", code)
284                .field("location", location)
285                .finish(),
286            VMStatus::ExecutionFailure {
287                status_code,
288                location,
289                function,
290                code_offset,
291            } => f
292                .debug_struct("EXECUTION_FAILURE")
293                .field("status_code", status_code)
294                .field("location", location)
295                .field("function_definition", function)
296                .field("code_offset", code_offset)
297                .finish(),
298        }
299    }
300}

693impl StatusCode {
694    /// Return the status type for this status code
695    pub fn status_type(self) -> StatusType {
696        let major_status_number: u64 = self.into();
697        if major_status_number >= VALIDATION_STATUS_MIN_CODE
698            && major_status_number <= VALIDATION_STATUS_MAX_CODE
699        {
700            return StatusType::Validation;
701        }
702
703        if major_status_number >= VERIFICATION_STATUS_MIN_CODE
704            && major_status_number <= VERIFICATION_STATUS_MAX_CODE
705        {
706            return StatusType::Verification;
707        }
708
709        if major_status_number >= INVARIANT_VIOLATION_STATUS_MIN_CODE
710            && major_status_number <= INVARIANT_VIOLATION_STATUS_MAX_CODE
711        {
712            return StatusType::InvariantViolation;
713        }
714
715        if major_status_number >= DESERIALIZATION_STATUS_MIN_CODE
716            && major_status_number <= DESERIALIZATION_STATUS_MAX_CODE
717        {
718            return StatusType::Deserialization;
719        }
720
721        if major_status_number >= EXECUTION_STATUS_MIN_CODE
722            && major_status_number <= EXECUTION_STATUS_MAX_CODE
723        {
724            return StatusType::Execution;
725        }
726
727        StatusType::Unknown
728    }
729}

796#[test]
797fn test_status_codes() {
798    use std::collections::HashSet;
799    // Make sure that within the 0-EXECUTION_STATUS_MAX_CODE that all of the
800    // status codes succeed when they should, and fail when they should.
801    for possible_major_status_code in 0..=EXECUTION_STATUS_MAX_CODE {
802        if STATUS_CODE_VALUES.contains(&possible_major_status_code) {
803            let status = StatusCode::try_from(possible_major_status_code);
804            assert!(status.is_ok());
805            let to_major_status_code = u64::from(status.unwrap());
806            assert_eq!(possible_major_status_code, to_major_status_code);
807        } else {
808            assert!(StatusCode::try_from(possible_major_status_code).is_err())
809        }
810    }
811
812    let mut seen_statuses = HashSet::new();
813    let mut seen_codes = HashSet::new();
814    // Now make sure that all of the error codes (including any that may be
815    // out-of-range) succeed. Make sure there aren't any duplicate mappings
816    for major_status_code in STATUS_CODE_VALUES.iter() {
817        assert!(
818            !seen_codes.contains(major_status_code),
819            "Duplicate major_status_code found"
820        );
821        seen_codes.insert(*major_status_code);
822        let status = StatusCode::try_from(*major_status_code);
823        assert!(status.is_ok());
824        let unwrapped_status = status.unwrap();
825        assert!(
826            !seen_statuses.contains(&unwrapped_status),
827            "Found duplicate u64 -> Status mapping"
828        );
829        seen_statuses.insert(unwrapped_status);
830        let to_major_status_code = u64::from(unwrapped_status);
831        assert_eq!(*major_status_code, to_major_status_code);
832    }
833}
