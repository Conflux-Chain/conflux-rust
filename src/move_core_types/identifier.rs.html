8//! An identifier is the name of an entity (module, resource, function, etc) in
9//! Move.
10//!
11//! A valid identifier consists of an ASCII string which satisfies any of the
12//! conditions:
13//!
14//! * The first character is a letter and the remaining characters are letters,
15//!   digits or underscores.
16//! * The first character is an underscore, and there is at least one further
17//!   letter, digit or underscore.
18//!
19//! The spec for allowed identifiers is similar to Rust's spec
20//! ([as of version 1.38](https://doc.rust-lang.org/1.38.0/reference/identifiers.html)).
21//!
22//! Allowed identifiers are currently restricted to ASCII due to unresolved
23//! issues with Unicode normalization. See [Rust issue #55467](https://github.com/rust-lang/rust/issues/55467) and the
24//! associated RFC for some discussion. Unicode identifiers may eventually be
25//! supported once these issues are worked out.
26//!
27//! This module only determines allowed identifiers at the bytecode level. Move
28//! source code will likely be more restrictive than even this, with a "raw
29//! identifier" escape hatch similar to Rust's `r#` identifiers.
30//!
31//! Among other things, identifiers are used to:
32//! * specify keys for lookups in storage
33//! * do cross-module lookups while executing transactions
34
35use anyhow::{bail, Result};
36#[cfg(any(test, feature = "fuzzing"))]
37use proptest::prelude::*;
38use ref_cast::RefCast;
39use serde::{Deserialize, Serialize};
40use std::{borrow::Borrow, fmt, ops::Deref};

42/// Return true if this character can appear in a Move identifier.
43///
44/// Note: there are stricter restrictions on whether a character can begin a
45/// Move identifier--only alphabetic characters are allowed here.
46pub fn is_valid_identifier_char(c: char) -> bool {
47    matches!(c, '_' | 'a'..='z' | 'A'..='Z' | '0'..='9')
48}
49
50/// Describes what identifiers are allowed.
51///
52/// For now this is deliberately restrictive -- we would like to evolve this in
53/// the future.
54// TODO: "<SELF>" is coded as an exception. It should be removed once
55// CompiledScript goes away.
56fn is_valid(s: &str) -> bool {
57    if s == "<SELF>" {
58        return true;
59    }
60    let len = s.len();
61    let mut chars = s.chars();
62    match chars.next() {
63        Some('a'..='z') | Some('A'..='Z') => {
64            chars.all(is_valid_identifier_char)
65        }
66        Some('_') if len > 1 => chars.all(is_valid_identifier_char),
67        _ => false,
68    }
69}

71/// A regex describing what identifiers are allowed. Used for proptests.
72// TODO: "<SELF>" is coded as an exception. It should be removed once
73// CompiledScript goes away.
74#[cfg(any(test, feature = "fuzzing"))]
75#[allow(dead_code)]
76pub(crate) static ALLOWED_IDENTIFIERS: &str =
77    r"(?:[a-zA-Z][a-zA-Z0-9_]*)|(?:_[a-zA-Z0-9_]+)|(?:<SELF>)";
78#[cfg(any(test, feature = "fuzzing"))]
79pub(crate) static ALLOWED_NO_SELF_IDENTIFIERS: &str =
80    r"(?:[a-zA-Z][a-zA-Z0-9_]*)|(?:_[a-zA-Z0-9_]+)";

82/// An owned identifier.
83///
84/// For more details, see the module level documentation.
85#[derive(
86    Clone, Debug, Eq, Hash, Ord, PartialEq, PartialOrd, Serialize, Deserialize,
87)]
88pub struct Identifier(Box<str>);
89// An identifier cannot be mutated so use Box<str> instead of String -- it is 1
90// word smaller.
91
92impl Identifier {
93    /// Creates a new `Identifier` instance.
94    pub fn new(s: impl Into<Box<str>>) -> Result<Self> {
95        let s = s.into();
96        if Self::is_valid(&s) {
97            Ok(Self(s))
98        } else {
99            bail!("Invalid identifier '{}'", s);
100        }
101    }
102
103    /// Returns true if this string is a valid identifier.
104    pub fn is_valid(s: impl AsRef<str>) -> bool { is_valid(s.as_ref()) }
105
106    /// Returns if this identifier is `"<SELF>"`.
107    /// TODO: remove once we fully separate CompiledScript & CompiledModule.
108    pub fn is_self(&self) -> bool { &*self.0 == "<SELF>" }
109
110    /// Converts a vector of bytes to an `Identifier`.
111    pub fn from_utf8(vec: Vec<u8>) -> Result<Self> {
112        let s = String::from_utf8(vec)?;
113        Self::new(s)
114    }
115
116    /// Creates a borrowed version of `self`.
117    pub fn as_ident_str(&self) -> &IdentStr { self }
118
119    /// Converts this `Identifier` into a `String`.
120    ///
121    /// This is not implemented as a `From` trait to discourage automatic
122    /// conversions -- these conversions should not typically happen.
123    pub fn into_string(self) -> String { self.0.into() }
124
125    /// Converts this `Identifier` into a UTF-8-encoded byte sequence.
126    pub fn into_bytes(self) -> Vec<u8> { self.into_string().into_bytes() }
127}
128
129impl<'a> From<&'a IdentStr> for Identifier {
130    fn from(ident_str: &'a IdentStr) -> Self { ident_str.to_owned() }
131}
132
133impl AsRef<IdentStr> for Identifier {
134    fn as_ref(&self) -> &IdentStr { self }
135}
136
137impl Deref for Identifier {
138    type Target = IdentStr;
139
140    fn deref(&self) -> &IdentStr {
141        // Identifier and IdentStr maintain the same invariants, so it is safe
142        // to convert.
143        IdentStr::ref_cast(&self.0)
144    }
145}
146
147impl fmt::Display for Identifier {
148    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
149        write!(f, "{}", &self.0)
150    }
151}

153/// A borrowed identifier.
154///
155/// For more details, see the module level documentation.
156#[derive(Debug, Eq, Hash, Ord, PartialEq, PartialOrd, RefCast)]
157#[repr(transparent)]
158pub struct IdentStr(str);
159
160impl IdentStr {
161    pub fn new(s: &str) -> Result<&IdentStr> {
162        if Self::is_valid(s) {
163            Ok(IdentStr::ref_cast(s))
164        } else {
165            bail!("Invalid identifier '{}'", s);
166        }
167    }
168
169    /// Returns true if this string is a valid identifier.
170    pub fn is_valid(s: impl AsRef<str>) -> bool { is_valid(s.as_ref()) }
171
172    /// Returns the length of `self` in bytes.
173    pub fn len(&self) -> usize { self.0.len() }
174
175    /// Returns `true` if `self` has a length of zero bytes.
176    pub fn is_empty(&self) -> bool { self.0.is_empty() }
177
178    /// Converts `self` to a `&str`.
179    ///
180    /// This is not implemented as a `From` trait to discourage automatic
181    /// conversions -- these conversions should not typically happen.
182    pub fn as_str(&self) -> &str { &self.0 }
183
184    /// Converts `self` to a byte slice.
185    pub fn as_bytes(&self) -> &[u8] { self.0.as_bytes() }
186}
187
188impl Borrow<IdentStr> for Identifier {
189    fn borrow(&self) -> &IdentStr { self }
190}
191
192impl ToOwned for IdentStr {
193    type Owned = Identifier;
194
195    fn to_owned(&self) -> Identifier { Identifier(self.0.into()) }
196}
197
198impl fmt::Display for IdentStr {
199    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
200        write!(f, "{}", &self.0)
201    }
202}

204#[cfg(any(test, feature = "fuzzing"))]
205impl Arbitrary for Identifier {
206    type Parameters = ();
207    type Strategy = BoxedStrategy<Self>;
208
209    fn arbitrary_with((): ()) -> Self::Strategy {
210        ALLOWED_NO_SELF_IDENTIFIERS
211            .prop_map(|s| {
212                // Identifier::new will verify that generated identifiers are
213                // correct.
214                Identifier::new(s).unwrap()
215            })
216            .boxed()
217    }
218}
