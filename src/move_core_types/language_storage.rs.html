26#[derive(
27    Serialize, Deserialize, Debug, PartialEq, Hash, Eq, Clone, PartialOrd, Ord,
28)]
29pub enum TypeTag {
30    Bool,
31    U8,
32    U64,
33    U128,
34    Address,
35    Signer,
36    Vector(Box<TypeTag>),
37    Struct(StructTag),
38}
39
40#[derive(
41    Serialize, Deserialize, Debug, PartialEq, Hash, Eq, Clone, PartialOrd, Ord,
42)]
43pub struct StructTag {
44    pub address: AccountAddress,
45    pub module: Identifier,
46    pub name: Identifier,
47    // TODO: rename to "type_args" (or better "ty_args"?)
48    pub type_params: Vec<TypeTag>,
49}
50
51impl StructTag {
52    pub fn access_vector(&self) -> Vec<u8> {
53        let mut key = vec![RESOURCE_TAG];
54        key.append(&mut bcs::to_bytes(self).unwrap());
55        key
56    }
57
58    pub fn module_id(&self) -> ModuleId {
59        ModuleId::new(self.address, self.module.to_owned())
60    }
61}

76    pub fn type_(&self) -> &StructTag { &self.type_ }

79impl ResourceKey {

85/// Represents the initial key into global storage where we first index by the
86/// address, and then the struct tag
87#[derive(
88    Serialize, Deserialize, Debug, PartialEq, Hash, Eq, Clone, PartialOrd, Ord,
89)]
90#[cfg_attr(any(test, feature = "fuzzing"), derive(Arbitrary))]
91#[cfg_attr(any(test, feature = "fuzzing"), proptest(no_params))]
92pub struct ModuleId {
93    address: AccountAddress,
94    name: Identifier,
95}
96
97impl From<ModuleId> for (AccountAddress, Identifier) {
98    fn from(module_id: ModuleId) -> Self { (module_id.address, module_id.name) }
99}
100
101impl ModuleId {
102    pub fn new(address: AccountAddress, name: Identifier) -> Self {
103        ModuleId { address, name }
104    }
105
106    pub fn name(&self) -> &IdentStr { &self.name }
107
108    pub fn address(&self) -> &AccountAddress { &self.address }
109
110    pub fn access_vector(&self) -> Vec<u8> {
111        let mut key = vec![CODE_TAG];
112        key.append(&mut bcs::to_bytes(self).unwrap());
113        key
114    }
115}
116
117impl Display for ModuleId {
118    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
119        write!(f, "{}::{}", self.address, self.name)
120    }
121}
122
123impl Display for StructTag {
124    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
125        write!(
126            f,
127            "0x{}::{}::{}",
128            self.address.short_str_lossless(),
129            self.module,
130            self.name
131        )?;
132        if let Some(first_ty) = self.type_params.first() {
133            write!(f, "<")?;
134            write!(f, "{}", first_ty)?;
135            for ty in self.type_params.iter().skip(1) {
136                write!(f, ", {}", ty)?;
137            }
138            write!(f, ">")?;
139        }
140        Ok(())
141    }
142}
143
144impl Display for TypeTag {
145    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
146        match self {
147            TypeTag::Struct(s) => write!(f, "{}", s),
148            TypeTag::Vector(ty) => write!(f, "Vector<{}>", ty),
149            TypeTag::U8 => write!(f, "U8"),
150            TypeTag::U64 => write!(f, "U64"),
151            TypeTag::U128 => write!(f, "U128"),
152            TypeTag::Address => write!(f, "Address"),
153            TypeTag::Signer => write!(f, "Signer"),
154            TypeTag::Bool => write!(f, "Bool"),
155        }
156    }
157}
