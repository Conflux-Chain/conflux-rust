19/// Vote is the struct that is ultimately sent by the voter in response for
20/// receiving a proposal.
21/// Vote carries the `LedgerInfo` of a block that is going to be committed in
22/// case this vote is gathers QuorumCertificate (see the detailed explanation in
23/// the comments of `LedgerInfo`).
24#[derive(Deserialize, Serialize, Clone, PartialEq, Eq)]
25pub struct Vote {
26    /// The data of the vote
27    vote_data: VoteData,
28    /// The identity of the voter.
29    author: Author,
30    /// LedgerInfo of a block that is going to be committed in case this vote
31    /// gathers QC.
32    ledger_info: LedgerInfo,
33    /// Signature of the LedgerInfo
34    signature: ConsensusSignature,
35    /// The round signatures can be aggregated into a timeout certificate if
36    /// present.
37    timeout_signature: Option<ConsensusSignature>,
38}
39
40// this is required by structured log
41impl Debug for Vote {
42    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
43        write!(f, "{}", self)
44    }
45}
46
47impl Display for Vote {
48    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
49        write!(
50            f,
51            "Vote: [vote data: {}, author: {}, is_timeout: {}, {}]",
52            self.vote_data,
53            self.author.short_str(),
54            self.is_timeout(),
55            self.ledger_info
56        )
57    }
58}
59
60impl Vote {
61    /// Generates a new Vote corresponding to the "fast-vote" path without the
62    /// round signatures that can be aggregated into a timeout certificate
63    pub fn new(
64        vote_data: VoteData, author: Author,
65        mut ledger_info_placeholder: LedgerInfo,
66        validator_signer: &ValidatorSigner,
67    ) -> Self {
68        ledger_info_placeholder.set_consensus_data_hash(vote_data.hash());
69        let signature = validator_signer.sign(&ledger_info_placeholder);
70        Self::new_with_signature(
71            vote_data,
72            author,
73            ledger_info_placeholder,
74            signature,
75        )
76    }
77
78    /// Generates a new Vote using a signature over the specified ledger_info
79    pub fn new_with_signature(
80        vote_data: VoteData, author: Author, ledger_info: LedgerInfo,
81        signature: ConsensusSignature,
82    ) -> Self {
83        Self {
84            vote_data,
85            author,
86            ledger_info,
87            signature,
88            timeout_signature: None,
89        }
90    }
91
92    /// Generates a round signature, which can then be used for aggregating a
93    /// timeout certificate. Typically called for generating vote messages
94    /// that are sent upon timeouts.
95    pub fn add_timeout_signature(&mut self, signature: ConsensusSignature) {
96        if self.timeout_signature.is_some() {
97            return; // round signature is already set
98        }
99
100        self.timeout_signature.replace(signature);
101    }
102
103    pub fn vote_data(&self) -> &VoteData { &self.vote_data }
104
105    /// Return the author of the vote
106    pub fn author(&self) -> Author { self.author }
107
108    /// Return the LedgerInfo associated with this vote
109    pub fn ledger_info(&self) -> &LedgerInfo { &self.ledger_info }
110
111    /// Return the signature of the vote
112    pub fn signature(&self) -> &ConsensusSignature { &self.signature }
113
114    /// Returns the hash of the data represent by a timeout proposal
115    pub fn timeout(&self) -> Timeout {
116        Timeout::new(
117            self.vote_data().proposed().epoch(),
118            self.vote_data().proposed().round(),
119        )
120    }
121
122    /// Return the epoch of the vote
123    pub fn epoch(&self) -> u64 { self.vote_data.proposed().epoch() }
124
125    /// Returns the signature for the vote_data().proposed().round() that can be
126    /// aggregated for TimeoutCertificate.
127    pub fn timeout_signature(&self) -> Option<&ConsensusSignature> {
128        self.timeout_signature.as_ref()
129    }
130
131    /// The vote message is considered a timeout vote message if it carries a
132    /// signature on the round, which can then be used for aggregating it to
133    /// the TimeoutCertificate.
134    pub fn is_timeout(&self) -> bool { self.timeout_signature.is_some() }
135
136    /// Verifies that the consensus data hash of LedgerInfo corresponds to the
137    /// vote info, and then verifies the signature.
138    pub fn verify(&self, validator: &ValidatorVerifier) -> anyhow::Result<()> {
139        ensure!(
140            self.ledger_info.consensus_data_hash() == self.vote_data.hash(),
141            "Vote's hash mismatch with LedgerInfo"
142        );
143        validator
144            .verify(self.author(), &self.ledger_info, &self.signature)
145            .context("Failed to verify Vote")?;
146        if let Some(timeout_signature) = &self.timeout_signature {
147            validator
148                .verify(self.author(), &self.timeout(), timeout_signature)
149                .context("Failed to verify Timeout Vote")?;
150        }
151        // Let us verify the vote data as well
152        self.vote_data().verify()?;
153        Ok(())
154    }
155}
