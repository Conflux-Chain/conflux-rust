19pub fn build_block(
20    b: &PrimitiveBlock, network: Network, consensus: &ConsensusGraph,
21    consensus_inner: &ConsensusGraphInner, data_man: &Arc<BlockDataManager>,
22    include_txs: bool, tx_space_filter: Option<Space>,
23) -> Result<Block, String> {
24    let block_hash = b.block_header.hash();
25
26    let epoch_number = consensus_inner
27        .get_block_epoch_number(&block_hash)
28        .or_else(|| data_man.block_epoch_number(&block_hash))
29        .map(Into::into);
30
31    let block_number = consensus.get_block_number(&block_hash)?.map(Into::into);
32
33    // get the block.gas_used
34    let tx_len = b.transactions.len();
35
36    let (gas_used, transactions) = {
37        let maybe_results = consensus_inner.block_execution_results_by_hash(
38            &b.hash(),
39            false, /* update_cache */
40        );
41
42        // calculate block gasUsed according block.execution_result and
43        // tx_space_filter
44        let gas_used_sum = match maybe_results {
45            Some(DataVersionTuple(_, ref execution_result)) => {
46                match tx_space_filter {
47                    Some(space_filter) => {
48                        let mut total_gas_used = U256::zero();
49                        let mut prev_acc_gas_used = U256::zero();
50                        for (idx, tx) in b.transactions.iter().enumerate() {
51                            let ref receipt =
52                                execution_result.block_receipts.receipts[idx];
53                            if tx.space() == space_filter {
54                                total_gas_used += receipt.accumulated_gas_used
55                                    - prev_acc_gas_used;
56                            }
57                            prev_acc_gas_used = receipt.accumulated_gas_used;
58                        }
59                        Some(total_gas_used)
60                    }
61                    None => Some(
62                        execution_result.block_receipts.receipts[tx_len - 1]
63                            .accumulated_gas_used,
64                    ),
65                }
66            }
67            None => None,
68        };
69
70        // prepare the transaction array according include_txs,
71        // execution_result, tx_space_filter
72        let transactions = match include_txs {
73            false => BlockTransactions::Hashes(
74                b.transaction_hashes(Some(Space::Native)),
75            ),
76            true => {
77                let tx_vec = match maybe_results {
78                    Some(DataVersionTuple(_, ref execution_result)) => {
79                        let maybe_state_root =
80                            data_man.get_executed_state_root(&b.hash());
81
82                        b.transactions
83                            .iter()
84                            .enumerate()
85                            .filter(|(_idx, tx)| {
86                                tx_space_filter.is_none()
87                                    || tx.space() == tx_space_filter.unwrap()
88                            })
89                            .enumerate()
90                            .map(|(new_index, (original_index, tx))| {
91                                let receipt = execution_result
92                                    .block_receipts
93                                    .receipts
94                                    .get(original_index)
95                                    .unwrap();
96                                let prior_gas_used = if original_index == 0 {
97                                    U256::zero()
98                                } else {
99                                    execution_result.block_receipts.receipts
100                                        [original_index - 1]
101                                        .accumulated_gas_used
102                                };
103                                match receipt.outcome_status {
104                                    TransactionStatus::Success
105                                    | TransactionStatus::Failure => {
106                                        let tx_index = TransactionIndex {
107                                            block_hash: b.hash(),
108                                            real_index: original_index,
109                                            is_phantom: false,
110                                            rpc_index: Some(new_index),
111                                        };
112                                        let tx_exec_error_msg =
113                                            &execution_result
114                                                .block_receipts
115                                                .tx_execution_error_messages
116                                                [original_index];
117                                        Transaction::from_signed(
118                                            tx,
119                                            Some(PackedOrExecuted::Executed(
120                                                Receipt::new(
121                                                    (**tx).clone(),
122                                                    receipt.clone(),
123                                                    tx_index,
124                                                    prior_gas_used,
125                                                    epoch_number,
126                                                    execution_result
127                                                        .block_receipts
128                                                        .block_number,
129                                                    b.block_header.base_price(),
130                                                    maybe_state_root,
131                                                    if tx_exec_error_msg
132                                                        .is_empty()
133                                                    {
134                                                        None
135                                                    } else {
136                                                        Some(
137                                                            tx_exec_error_msg
138                                                                .clone(),
139                                                        )
140                                                    },
141                                                    network,
142                                                    false,
143                                                    false,
144                                                )?,
145                                            )),
146                                            network,
147                                        )
148                                    }
149                                    TransactionStatus::Skipped => {
150                                        Transaction::from_signed(
151                                            tx, None, network,
152                                        )
153                                    }
154                                }
155                            })
156                            .collect::<Result<_, _>>()?
157                    }
158                    None => b
159                        .transactions
160                        .iter()
161                        .filter(|tx| {
162                            tx_space_filter.is_none()
163                                || tx.space() == tx_space_filter.unwrap()
164                        })
165                        .map(|x| Transaction::from_signed(x, None, network))
166                        .collect::<Result<_, _>>()?,
167                };
168                BlockTransactions::Full(tx_vec)
169            }
170        };
171
172        (gas_used_sum, transactions)
173    };
174
175    let base_fee_per_gas: Option<U256> =
176        b.block_header.base_price().map(|x| x[Space::Native]).into();
177
178    // if a block is 1559 block(has base_fee_per_gas) then it's
179    // block.gas_limit is 90% of the actual block.gas_limit
180    let gas_limit: U256 = b.block_header.core_space_gas_limit();
181
182    Ok(Block {
183        hash: H256::from(block_hash),
184        parent_hash: H256::from(b.block_header.parent_hash().clone()),
185        height: b.block_header.height().into(),
186        miner: RpcAddress::try_from_h160(*b.block_header.author(), network)?,
187        deferred_state_root: H256::from(
188            b.block_header.deferred_state_root().clone(),
189        ),
190        deferred_receipts_root: H256::from(
191            b.block_header.deferred_receipts_root().clone(),
192        ),
193        deferred_logs_bloom_hash: H256::from(
194            b.block_header.deferred_logs_bloom_hash().clone(),
195        ),
196        blame: U64::from(b.block_header.blame()),
197        transactions_root: H256::from(
198            b.block_header.transactions_root().clone(),
199        ),
200        // PrimitiveBlock does not contain this information
201        epoch_number: epoch_number.map(|e| U256::from(e)),
202        block_number,
203        // fee system
204        gas_used,
205        gas_limit,
206        base_fee_per_gas,
207        timestamp: b.block_header.timestamp().into(),
208        difficulty: b.block_header.difficulty().clone().into(),
209        pow_quality: b
210            .block_header
211            .pow_hash
212            .map(|h| pow::pow_hash_to_quality(&h, &b.block_header.nonce())),
213        adaptive: b.block_header.adaptive(),
214        referee_hashes: b
215            .block_header
216            .referee_hashes()
217            .iter()
218            .map(|x| H256::from(*x))
219            .collect(),
220        nonce: b.block_header.nonce().into(),
221        transactions,
222        custom: b
223            .block_header
224            .custom()
225            .clone()
226            .into_iter()
227            .map(Into::into)
228            .collect(),
229        size: Some(b.size().into()),
230        pos_reference: b.block_header.pos_reference().clone(),
231    })
232}

234pub fn build_header(
235    h: &PrimitiveBlockHeader, network: Network, consensus: SharedConsensusGraph,
236) -> Result<Header, String> {
237    let hash = h.hash();
238
239    let epoch_number = consensus
240        .get_block_epoch_number(&hash)
241        .or_else(|| consensus.data_manager().block_epoch_number(&hash))
242        .map(Into::into);
243
244    let block_number = consensus.get_block_number(&hash)?.map(Into::into);
245
246    let base_fee_per_gas: Option<U256> =
247        h.base_price().map(|x| x[Space::Native]).into();
248
249    let referee_hashes =
250        h.referee_hashes().iter().map(|x| H256::from(*x)).collect();
251
252    Ok(Header {
253        hash: H256::from(hash),
254        parent_hash: H256::from(*h.parent_hash()),
255        height: h.height().into(),
256        miner: RpcAddress::try_from_h160(*h.author(), network)?,
257        deferred_state_root: H256::from(*h.deferred_state_root()),
258        deferred_receipts_root: H256::from(*h.deferred_receipts_root()),
259        deferred_logs_bloom_hash: H256::from(*h.deferred_logs_bloom_hash()),
260        blame: U64::from(h.blame()),
261        transactions_root: H256::from(*h.transactions_root()),
262        epoch_number,
263        block_number,
264        gas_limit: h.gas_limit().into(),
265        base_fee_per_gas,
266        timestamp: h.timestamp().into(),
267        difficulty: h.difficulty().into(),
268        adaptive: h.adaptive(),
269        referee_hashes,
270        nonce: h.nonce().into(),
271        pow_quality: h
272            .pow_hash
273            .map(|pow_hash| pow::pow_hash_to_quality(&pow_hash, &h.nonce())),
274        pos_reference: *h.pos_reference(),
275        custom: h.custom().clone().into_iter().map(Into::into).collect(),
276    })
277}
