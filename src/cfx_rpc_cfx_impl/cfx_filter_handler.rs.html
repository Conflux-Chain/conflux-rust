39pub struct UnfinalizedEpochs {
40    pub epochs_queue: VecDeque<(u64, Vec<H256>)>,
41    pub epochs_map: HashMap<u64, Vec<Vec<H256>>>,
42}
43
44impl Default for UnfinalizedEpochs {
45    fn default() -> Self {
46        UnfinalizedEpochs {
47            epochs_queue: Default::default(),
48            epochs_map: Default::default(),
49        }
50    }

53pub struct CfxFilterHandler {
54    consensus: SharedConsensusGraph,
55    tx_pool: SharedTransactionPool,
56    polls: Mutex<PollManager<SyncPollFilter<RpcLog>>>,
57    unfinalized_epochs: Arc<RwLock<UnfinalizedEpochs>>,
58    logs_filter_max_limit: Option<usize>,
59    network: Network,
60}

62impl CfxFilterHandler {
63    pub fn new(
64        consensus: SharedConsensusGraph, tx_pool: SharedTransactionPool,
65        epochs_ordered: Arc<Channel<(u64, Vec<H256>)>>, executor: Arc<Runtime>,
66        poll_lifetime: u32, logs_filter_max_limit: Option<usize>,
67        network: Network,
68    ) -> Self {
69        let handler = CfxFilterHandler {
70            consensus,
71            tx_pool,
72            polls: Mutex::new(PollManager::new(poll_lifetime)),
73            unfinalized_epochs: Default::default(),
74            logs_filter_max_limit,
75            network,
76        };
77        handler.start_epochs_loop(epochs_ordered, executor);
78        handler
79    }
80
81    fn start_epochs_loop(
82        &self, epochs_ordered: Arc<Channel<(u64, Vec<H256>)>>,
83        executor: Arc<Runtime>,
84    ) {
85        let mut receiver = epochs_ordered.subscribe();
86        let consensus = self.consensus.clone();
87        let epochs = self.unfinalized_epochs.clone();
88
89        let fut = async move {
90            while let Some(epoch) = receiver.recv().await {
91                let mut epochs = epochs.write();
92                epochs.epochs_queue.push_back(epoch.clone());
93                epochs
94                    .epochs_map
95                    .entry(epoch.0)
96                    .or_insert(vec![])
97                    .push(epoch.1.clone());
98
99                let latest_finalized_epoch_number =
100                    consensus.latest_finalized_epoch_number();
101                debug!(
102                    "latest finalized epoch number: {}, received epochs: {:?}",
103                    latest_finalized_epoch_number, epoch
104                );
105
106                while let Some(e) = epochs.epochs_queue.front() {
107                    if e.0 < latest_finalized_epoch_number {
108                        let (k, _) = epochs.epochs_queue.pop_front().unwrap();
109                        if let Some(target) = epochs.epochs_map.get_mut(&k) {
110                            if target.len() == 1 {
111                                epochs.epochs_map.remove(&k);
112                            } else {
113                                target.remove(0);
114                            }
115                        }
116                    } else {
117                        break;
118                    }
119                }
120            }
121        };
122
123        executor.spawn(fut);
124    }

126    fn consensus_graph(&self) -> &ConsensusGraph { &self.consensus }

128    fn best_executed_epoch_number(&self) -> u64 {

132    fn block_hashes(&self, epoch_num: EpochNumber) -> Option<Vec<H256>> {
133        let _inner = self.consensus_graph().inner.read();

144    fn logs(&self, filter: LogFilter) -> RpcResult<Vec<RpcLog>> {
145        let logs = self
146            .consensus_graph()
147            .logs(filter)
148            .map_err(|e| ErrorObjectOwned::from(CoreError::from(e)))?;
149
150        Ok(logs
151            .iter()
152            .cloned()
153            .map(|l| RpcLog::try_from_localized(l, self.network))
154            .collect::<Result<_, _>>()
155            .map_err(|_| {
156                invalid_params_rpc_err("filter", Some("retrieve logs error"))
157            })?)
158    }
159
160    fn logs_for_epoch(
161        &self, filter: &LogFilter, epoch: (u64, Vec<H256>),
162        data_man: &Arc<BlockDataManager>,
163    ) -> RpcResult<Vec<RpcLog>> {
164        let logs = match retrieve_epoch_logs(data_man, epoch) {
165            Some(logs) => logs,
166            None => {
167                return Err(rpc_error_with_code(
168                    codes::UNSUPPORTED as i32,
169                    "Unable to retrieve logs for epoch",
170                ))
171            }
172        };
173
174        let logs: Vec<RpcLog> = logs
175            .iter()
176            .filter(|l| filter.matches(&l.entry))
177            .cloned()
178            .map(|l| RpcLog::try_from_localized(l, self.network))
179            .collect::<Result<_, _>>()
180            .map_err(|_| {
181                invalid_params_rpc_err(
182                    "filter",
183                    Some("retrieve logs for epoch error"),
184                )
185            })?;
186
187        Ok(logs)
188    }

190    fn epochs_since_last_request(
191        &self, last_epoch_number: u64,
192        recent_reported_epochs: &VecDeque<(u64, Vec<H256>)>,
193    ) -> RpcResult<(u64, Vec<(u64, Vec<H256>)>)> {
194        let last_block = if let Some((num, hash)) =
195            recent_reported_epochs.front().cloned()
196        {
197            if last_epoch_number != num {
198                return Err(rpc_error_with_code(
199                    codes::UNSUPPORTED as i32,
200                    "Last block number does not match",
201                ));
202            }
203            Some(hash)
204        } else {
205            None
206        };
207
208        let current_epoch_number = self.best_executed_epoch_number();
209        debug!("current epoch number {}", current_epoch_number);
210        let latest_epochs = self.unfinalized_epochs.read();
211
212        let mut idx = latest_epochs.epochs_queue.len() as i32 - 1;
213        while idx >= 0
214            && latest_epochs.epochs_queue[idx as usize].0
215                != current_epoch_number
216        {
217            if latest_epochs.epochs_queue[idx as usize].0 == last_epoch_number
218                && last_block
219                    == Some(latest_epochs.epochs_queue[idx as usize].1.clone())
220            {
221                return Ok((0, vec![]));
222            }
223            idx -= 1;
224        }
225
226        let mut end_epoch_number = current_epoch_number + 1;
227        let mut new_epochs = vec![];
228        let mut hs = HashSet::new();
229        while idx >= 0 {
230            let (num, blocks) =
231                latest_epochs.epochs_queue[idx as usize].clone();
232            if num == last_epoch_number
233                && (last_block.is_none() || last_block == Some(blocks.clone()))
234            {
235                break;
236            }
237
238            if num < end_epoch_number && !hs.contains(&num) {
239                hs.insert(num);
240                new_epochs.push((num, blocks));
241                end_epoch_number = num;
242            }
243
244            idx -= 1;
245        }
246        new_epochs.reverse();
247
248        let mut reorg_epochs = vec![];
249        let mut reorg_len = 0;
250        for i in 0..recent_reported_epochs.len() {
251            let (num, hash) = recent_reported_epochs[i].clone();
252
253            if num < end_epoch_number {
254                let pivot_hash =
255                    if let Some(v) = latest_epochs.epochs_map.get(&num) {
256                        v.last().unwrap().clone()
257                    } else {
258                        self.block_hashes(EpochNumber::Number(num))
259                            .expect("Epoch should exist")
260                    };
261
262                if pivot_hash == hash {
263                    break;
264                }
265
266                debug!("reorg for {}, pivot hash {:?}", num, pivot_hash);
267                reorg_epochs.push((num, pivot_hash));
268            }
269            reorg_len += 1;
270        }
271        reorg_epochs.reverse();
272
273        debug!(
274            "stable epochs from {} to {}",
275            last_epoch_number + 1,
276            end_epoch_number
277        );
278        for epoch_num in (last_epoch_number + 1)..end_epoch_number {
279            let hash = self
280                .block_hashes(EpochNumber::Number(epoch_num))
281                .expect("Epoch should exist");
282            reorg_epochs.push((epoch_num, hash));
283        }
284        reorg_epochs.append(&mut new_epochs);
285
286        info!(
287            "Chain reorg len: {}, new epochs len: {}",
288            reorg_len,
289            reorg_epochs.len()
290        );
291        Ok((reorg_len, reorg_epochs))
292    }

295impl CfxFilterRpcServer for CfxFilterHandler {
296    fn new_filter(&self, filter: CfxRpcLogFilter) -> RpcResult<H128> {
297        debug!("create filter: {:?}", filter);
298        let mut polls = self.polls.lock();
299        let epoch_number = self.best_executed_epoch_number();
300
301        let filter: LogFilter = filter.into_primitive()?;
302
303        let id = polls.create_poll(SyncPollFilter::new(PollFilter::Logs {
304            last_epoch_number: if epoch_number == 0 {
305                0
306            } else {
307                epoch_number - 1
308            },
309            filter,
310            include_pending: false,
311            previous_logs: VecDeque::with_capacity(MAX_BLOCK_HISTORY_SIZE),
312            recent_reported_epochs: VecDeque::with_capacity(
313                MAX_BLOCK_HISTORY_SIZE,
314            ),
315        }));
316
317        Ok(id.into())
318    }
319
320    fn new_block_filter(&self) -> RpcResult<H128> {
321        debug!("create block filter");
322        let mut polls = self.polls.lock();
323        let id = polls.create_poll(SyncPollFilter::new(PollFilter::Block {
324            last_epoch_number: self.best_executed_epoch_number(),
325            recent_reported_epochs: VecDeque::with_capacity(
326                MAX_BLOCK_HISTORY_SIZE,
327            ),
328        }));
329        Ok(id.into())
330    }
331
332    fn new_pending_transaction_filter(&self) -> RpcResult<H128> {
333        debug!("create pending transaction filter");
334        let mut polls = self.polls.lock();
335        let pending_transactions = self.pending_transaction_hashes();
336        let id = polls.create_poll(SyncPollFilter::new(
337            PollFilter::PendingTransaction(pending_transactions),
338        ));
339        Ok(id.into())
340    }
341
342    fn filter_changes(&self, filter_id: H128) -> RpcResult<CfxFilterChanges> {
343        info!("filter_changes id: {}", filter_id);
344        let filter = match self.polls.lock().poll_mut(&filter_id) {
345            Some(filter) => filter.clone(),
346            None => return Err(invalid_request_msg("Filter not found")),
347        };
348
349        filter.modify(|filter| match *filter {
350            PollFilter::Block {
351                ref mut last_epoch_number,
352                ref mut recent_reported_epochs,
353            } => {
354                let (reorg_len, epochs) = self.epochs_since_last_request(
355                    *last_epoch_number,
356                    recent_reported_epochs,
357                )?;
358
359                for _ in 0..reorg_len {
360                    recent_reported_epochs.pop_front();
361                }
362
363                let mut hashes = Vec::new();
364                for (num, blocks) in epochs.into_iter() {
365                    *last_epoch_number = num;
366                    hashes.append(&mut blocks.clone());
367
368                    if recent_reported_epochs.len() >= MAX_BLOCK_HISTORY_SIZE {
369                        recent_reported_epochs.pop_back();
370                    }
371                    recent_reported_epochs.push_front((num, blocks));
372                }
373
374                Ok(CfxFilterChanges::Hashes(hashes))
375            }
376            PollFilter::PendingTransaction(ref mut previous_hashes) => {
377                let current_hashes = self.pending_transaction_hashes();
378                let new_hashes = {
379                    current_hashes
380                        .difference(previous_hashes)
381                        .cloned()
382                        .map(Into::into)
383                        .collect()
384                };
385                *previous_hashes = current_hashes;
386                Ok(CfxFilterChanges::Hashes(new_hashes))
387            }
388            PollFilter::Logs {
389                ref mut last_epoch_number,
390                ref mut recent_reported_epochs,
391                ref mut previous_logs,
392                ref filter,
393                include_pending: _,
394            } => {
395                let (reorg_len, epochs) = self.epochs_since_last_request(
396                    *last_epoch_number,
397                    recent_reported_epochs,
398                )?;
399
400                let mut logs = vec![];
401
402                for _ in 0..reorg_len {
403                    recent_reported_epochs.pop_front().unwrap();
404                }
405
406                if reorg_len > 0 {
407                    logs.push(CfxFilterLog::ChainReorg(RevertTo {
408                        revert_to: epochs.first().unwrap().0.into(),
409                    }));
410                }
411                let data_man = self.consensus_graph().data_manager().clone();
412
413                for (num, blocks) in epochs.into_iter() {
414                    let log = match self.logs_for_epoch(
415                        &filter,
416                        (num, blocks.clone()),
417                        &data_man,
418                    ) {
419                        Ok(l) => l,
420                        _ => break,
421                    };
422
423                    log.iter()
424                        .for_each(|l| logs.push(CfxFilterLog::Log(l.clone())));
425
426                    *last_epoch_number = num;
427
428                    if recent_reported_epochs.len() >= MAX_BLOCK_HISTORY_SIZE {
429                        recent_reported_epochs.pop_back();
430                        previous_logs.pop_back();
431                    }
432                    recent_reported_epochs.push_front((num, blocks));
433                    previous_logs.push_front(log);
434                }
435
436                Ok(CfxFilterChanges::Logs(limit_logs(
437                    logs,
438                    self.logs_filter_max_limit,
439                )))
440            }
441        })
442    }
443
444    fn filter_logs(&self, filter_id: H128) -> RpcResult<Vec<RpcLog>> {
445        let (filter, _) = {
446            let mut polls = self.polls.lock();
447            match polls.poll(&filter_id).and_then(|f| {
448                f.modify(|filter| match *filter {
449                    PollFilter::Logs {
450                        ref filter,
451                        include_pending,
452                        ..
453                    } => Some((filter.clone(), include_pending)),
454                    _ => None,
455                })
456            }) {
457                Some((filter, include_pending)) => (filter, include_pending),
458                None => return Err(invalid_request_msg("Filter not found")),
459            }
460        };
461
462        Ok(limit_logs(self.logs(filter)?, self.logs_filter_max_limit))
463    }
464
465    fn uninstall_filter(&self, filter_id: H128) -> RpcResult<bool> {
466        Ok(self.polls.lock().remove_poll(&filter_id))
467    }

470fn retrieve_epoch_logs(
471    data_man: &Arc<BlockDataManager>, epoch: (u64, Vec<H256>),
472) -> Option<Vec<LocalizedLogEntry>> {
473    debug!("retrieve_epoch_logs {:?}", epoch);
474    let (epoch_number, hashes) = epoch;
475    let pivot = hashes.last().cloned().expect("epoch should not be empty");
476
477    // retrieve epoch receipts
478    let fut = hashes
479        .iter()
480        .map(|h| retrieve_block_receipts(&data_man, h, &pivot));
481
482    let receipts = fut.into_iter().collect::<Option<Vec<_>>>()?;
483
484    let mut logs = vec![];
485    let mut log_index = 0;
486
487    for (block_hash, block_receipts) in zip(hashes, receipts) {
488        // retrieve block transactions
489        let block = match data_man
490            .block_by_hash(&block_hash, true /* update_cache */)
491        {
492            Some(b) => b,
493            None => {
494                warn!("Unable to retrieve block {:?}", block_hash);
495                return None;
496            }
497        };
498
499        let txs = &block.transactions;
500        assert_eq!(block_receipts.receipts.len(), txs.len());
501
502        // construct logs
503        for (txid, (receipt, tx)) in
504            zip(&block_receipts.receipts, txs).enumerate()
505        {
506            let native_logs: Vec<_> = receipt
507                .logs
508                .iter()
509                .cloned()
510                .filter(|l| l.space == Space::Native)
511                .collect();
512
513            for (logid, entry) in native_logs.into_iter().enumerate() {
514                logs.push(LocalizedLogEntry {
515                    entry,
516                    block_hash,
517                    epoch_number,
518                    block_timestamp: Some(block.block_header.timestamp()),
519                    transaction_hash: tx.hash,
520                    transaction_index: txid,
521                    log_index,
522                    transaction_log_index: logid,
523                });
524
525                log_index += 1;
526            }
527        }
528    }
529
530    Some(logs)
531}

534    data_man: &Arc<BlockDataManager>, hash: &H256, pivot: &H256,
