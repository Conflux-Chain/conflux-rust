92    fn decode_raw_txs(
93        &self, raw_txs: Bytes, tx_data_len: usize,
94    ) -> RpcResult<Vec<Arc<SignedTransaction>>> {
95        let txs: Vec<TransactionWithSignature> =
96            Rlp::new(&raw_txs.into_vec()).as_list().map_err(|err| {
97                ErrorObjectOwned::owned(
98                    INVALID_PARAMS_CODE,
99                    format!("raw_txs decode error: {:?}", err),
100                    None::<()>,
101                )
102            })?;
103
104        let mut transactions = Vec::new();
105
106        for tx in txs {
107            let public = match tx.recover_public() {
108                Ok(public) => public,
109                Err(e) => {
110                    return Err(ErrorObjectOwned::owned(
111                        INVALID_PARAMS_CODE,
112                        format!("Recover public error: {:?}", e),
113                        None::<()>,
114                    ));
115                }
116            };
117
118            let mut signed_tx = SignedTransaction::new(public, tx);
119
120            if tx_data_len > 0 {
121                *signed_tx.transaction.transaction.unsigned.data_mut() =
122                    vec![0; tx_data_len];
123            }
124
125            transactions.push(Arc::new(signed_tx));
126        }
127
128        Ok(transactions)
129    }

131    fn build_rpc_block(
132        &self, b: &PrimitiveBlock, consensus_inner: &ConsensusGraphInner,
133        include_txs: bool,
134    ) -> Result<RpcBlock, String> {
135        let network = *self.network.get_network_type();
136        let block_hash = b.block_header.hash();
137
138        let epoch_number = consensus_inner
139            .get_block_epoch_number(&block_hash)
140            .or_else(|| self.data_man.block_epoch_number(&block_hash))
141            .map(Into::into);
142
143        let block_number = self
144            .consensus
145            .get_block_number(&block_hash)?
146            .map(Into::into);
147
148        let _tx_len = b.transactions.len();
149
150        let (gas_used, transactions) = {
151            let maybe_results = consensus_inner
152                .block_execution_results_by_hash(&b.hash(), false);
153
154            let gas_used_sum = match maybe_results {
155                Some(DataVersionTuple(_, ref execution_result)) => {
156                    let mut total_gas_used = U256::zero();
157                    let mut prev_acc_gas_used = U256::zero();
158                    for (idx, tx) in b.transactions.iter().enumerate() {
159                        let receipt =
160                            &execution_result.block_receipts.receipts[idx];
161                        if tx.space() == Space::Native {
162                            total_gas_used += receipt.accumulated_gas_used
163                                - prev_acc_gas_used;
164                        }
165                        prev_acc_gas_used = receipt.accumulated_gas_used;
166                    }
167                    Some(total_gas_used)
168                }
169                None => None,
170            };
171
172            let transactions = match include_txs {
173                false => BlockTransactions::Hashes(
174                    b.transaction_hashes(Some(Space::Native)),
175                ),
176                true => {
177                    let maybe_results = consensus_inner
178                        .block_execution_results_by_hash(&b.hash(), false);
179                    let tx_vec = match maybe_results {
180                        Some(DataVersionTuple(_, ref execution_result)) => {
181                            let maybe_state_root = self
182                                .data_man
183                                .get_executed_state_root(&b.hash());
184
185                            b.transactions
186                                .iter()
187                                .enumerate()
188                                .filter(|(_idx, tx)| {
189                                    tx.space() == Space::Native
190                                })
191                                .enumerate()
192                                .map(|(new_index, (original_index, tx))| {
193                                    let receipt = execution_result
194                                        .block_receipts
195                                        .receipts
196                                        .get(original_index)
197                                        .unwrap();
198                                    let prior_gas_used =
199                                        if original_index == 0 {
200                                            U256::zero()
201                                        } else {
202                                            execution_result
203                                                .block_receipts
204                                                .receipts[original_index - 1]
205                                                .accumulated_gas_used
206                                        };
207                                    match receipt.outcome_status {
208                                        TransactionStatus::Success
209                                        | TransactionStatus::Failure => {
210                                            let tx_index = TransactionIndex {
211                                                block_hash: b.hash(),
212                                                real_index: original_index,
213                                                is_phantom: false,
214                                                rpc_index: Some(new_index),
215                                            };
216                                            let tx_exec_error_msg =
217                                                &execution_result
218                                                    .block_receipts
219                                                    .tx_execution_error_messages
220                                                    [original_index];
221                                            RpcTransaction::from_signed(
222                                                tx,
223                                                Some(
224                                                    PackedOrExecuted::Executed(
225                                                        Receipt::new(
226                                                            (**tx).clone(),
227                                                            receipt.clone(),
228                                                            tx_index,
229                                                            prior_gas_used,
230                                                            epoch_number,
231                                                            execution_result
232                                                                .block_receipts
233                                                                .block_number,
234                                                            b.block_header
235                                                                .base_price(),
236                                                            maybe_state_root,
237                                                            if tx_exec_error_msg
238                                                                .is_empty()
239                                                            {
240                                                                None
241                                                            } else {
242                                                                Some(
243                                                                    tx_exec_error_msg
244                                                                        .clone(),
245                                                                )
246                                                            },
247                                                            network,
248                                                            false,
249                                                            false,
250                                                        )?,
251                                                    ),
252                                                ),
253                                                network,
254                                            )
255                                        }
256                                        TransactionStatus::Skipped => {
257                                            RpcTransaction::from_signed(
258                                                tx, None, network,
259                                            )
260                                        }
261                                    }
262                                })
263                                .collect::<Result<_, _>>()?
264                        }
265                        None => b
266                            .transactions
267                            .iter()
268                            .filter(|tx| tx.space() == Space::Native)
269                            .map(|x| {
270                                RpcTransaction::from_signed(x, None, network)
271                            })
272                            .collect::<Result<_, _>>()?,
273                    };
274                    BlockTransactions::Full(tx_vec)
275                }
276            };
277
278            (gas_used_sum, transactions)
279        };
280
281        let base_fee_per_gas: Option<U256> =
282            b.block_header.base_price().map(|x| x[Space::Native]).into();
283        let gas_limit: U256 = b.block_header.core_space_gas_limit();
284
285        Ok(RpcBlock {
286            hash: H256::from(block_hash),
287            parent_hash: H256::from(b.block_header.parent_hash().clone()),
288            height: b.block_header.height().into(),
289            miner: cfx_rpc_cfx_types::address::RpcAddress::try_from_h160(
290                *b.block_header.author(),
291                network,
292            )?,
293            deferred_state_root: H256::from(
294                b.block_header.deferred_state_root().clone(),
295            ),
296            deferred_receipts_root: H256::from(
297                b.block_header.deferred_receipts_root().clone(),
298            ),
299            deferred_logs_bloom_hash: H256::from(
300                b.block_header.deferred_logs_bloom_hash().clone(),
301            ),
302            blame: U64::from(b.block_header.blame()),
303            transactions_root: H256::from(
304                b.block_header.transactions_root().clone(),
305            ),
306            epoch_number: epoch_number.map(|e| U256::from(e)),
307            block_number,
308            gas_used,
309            gas_limit,
310            base_fee_per_gas,
311            timestamp: b.block_header.timestamp().into(),
312            difficulty: b.block_header.difficulty().clone().into(),
313            pow_quality: b
314                .block_header
315                .pow_hash
316                .map(|h| pow::pow_hash_to_quality(&h, &b.block_header.nonce())),
317            adaptive: b.block_header.adaptive(),
318            referee_hashes: b
319                .block_header
320                .referee_hashes()
321                .iter()
322                .map(|x| H256::from(*x))
323                .collect(),
324            nonce: b.block_header.nonce().into(),
325            transactions,
326            custom: b
327                .block_header
328                .custom()
329                .clone()
330                .into_iter()
331                .map(Into::into)
332                .collect(),
333            size: Some(b.size().into()),
334            pos_reference: b.block_header.pos_reference().clone(),
335        })
336    }
337}

349    fn get_goodput(&self) -> RpcResult<String> {
350        info!("RPC Request: get_goodput");
351        let mut all_block_set = HashSet::new();
352        for epoch_number in 1..self.consensus.best_epoch_number() {
353            for block_hash in self
354                .consensus
355                .get_block_hashes_by_epoch(epoch_number.into())
356                .map_err(|_| internal_error())?
357            {
358                all_block_set.insert(block_hash);
359            }
360        }
361        let mut set = HashSet::new();
362        let mut min = std::u64::MAX;
363        let mut max: u64 = 0;
364        for key in &all_block_set {
365            if let Some(block) =
366                self.data_man.block_by_hash(key, false /* update_cache */)
367            {
368                let timestamp = block.block_header.timestamp();
369                if timestamp < min && timestamp > 0 {
370                    min = timestamp;
371                }
372                if timestamp > max {
373                    max = timestamp;
374                }
375                for transaction in &block.transactions {
376                    set.insert(transaction.hash());
377                }
378            }
379        }
380        if max != min {
381            let lower_bound = min + ((max - min) as f64 * 0.3) as u64;
382            let upper_bound = min + ((max - min) as f64 * 0.8) as u64;
383            let mut ranged_set = HashSet::new();
384            for key in &all_block_set {
385                if let Some(block) = self
386                    .data_man
387                    .block_by_hash(key, false /* update_cache */)
388                {
389                    let timestamp = block.block_header.timestamp();
390                    if timestamp > lower_bound && timestamp < upper_bound {
391                        for transaction in &block.transactions {
392                            ranged_set.insert(transaction.hash());
393                        }
394                    }
395                }
396            }
397            if upper_bound != lower_bound {
398                Ok(format!(
399                    "full: {}, ranged: {}",
400                    set.len() as isize / (max - min) as isize,
401                    ranged_set.len() as isize
402                        / (upper_bound - lower_bound) as isize
403                ))
404            } else {
405                Ok(format!(
406                    "full: {}",
407                    set.len() as isize / (max - min) as isize
408                ))
409            }
410        } else {
411            Ok("-1".to_string())
412        }
413    }

682        &self, account_start_index: usize,
683    ) -> RpcResult<Bytes> {
684        info!(
685            "RPC Request: send_usable_genesis_accounts start from {:?}",

830    fn pos_get_chosen_proposal(&self) -> RpcResult<Option<PosBlock>> {
831        let maybe_block = self
832            .pos_handler
833            .get_chosen_proposal()
834            .map_err(|e| {
835                warn!("pos_get_chosen_proposal: err={:?}", e);
836                internal_error()
837            })?
838            .and_then(|b| {
839                let block_hash = b.id();
840                self.pos_handler
841                    .cached_db()
842                    .get_block(&block_hash)
843                    .ok()
844                    .map(|executed_block| {
845                        let executed_block = executed_block.lock();
846                        PosBlock {
847                            hash: hash_value_to_h256(b.id()),
848                            epoch: U64::from(b.epoch()),
849                            round: U64::from(b.round()),
850                            last_tx_number: executed_block
851                                .output()
852                                .version()
853                                .unwrap_or_default()
854                                .into(),
855                            miner: b.author().map(|a| H256::from(a.to_u8())),
856                            parent_hash: hash_value_to_h256(b.parent_id()),
857                            timestamp: U64::from(b.timestamp_usecs()),
858                            pivot_decision: executed_block
859                                .output()
860                                .pivot_block()
861                                .as_ref()
862                                .map(|d| Decision::from(d)),
863                            height: executed_block
864                                .output()
865                                .executed_trees()
866                                .pos_state()
867                                .current_view()
868                                .into(),
869                            signatures: vec![],
870                        }
871                    })
872            });
873        Ok(maybe_block)
874    }
