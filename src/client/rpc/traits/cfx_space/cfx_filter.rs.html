1use crate::rpc::types::{CfxFilterChanges, CfxRpcLogFilter, Log as RpcLog};

6/// Cfx filters rpc api (polling).
7#[rpc(server)]
8pub trait CfxFilter {
9    /// Returns id of new filter.
10    #[rpc(name = "cfx_newFilter")]
11    fn new_filter(&self, _: CfxRpcLogFilter) -> JsonRpcResult<H128>;
12
13    /// Returns id of new block filter.
14    #[rpc(name = "cfx_newBlockFilter")]
15    fn new_block_filter(&self) -> JsonRpcResult<H128>;
16
17    /// Returns id of new block filter.
18    #[rpc(name = "cfx_newPendingTransactionFilter")]
19    fn new_pending_transaction_filter(&self) -> JsonRpcResult<H128>;
20
21    /// Returns filter changes since last poll.
22    #[rpc(name = "cfx_getFilterChanges")]
23    fn filter_changes(&self, _: H128) -> JsonRpcResult<CfxFilterChanges>;
24
25    /// Returns all logs matching given filter (in a range 'from' - 'to').
26    #[rpc(name = "cfx_getFilterLogs")]
27    fn filter_logs(&self, _: H128) -> JsonRpcResult<Vec<RpcLog>>;
28
29    /// Uninstalls filter.
30    #[rpc(name = "cfx_uninstallFilter")]
31    fn uninstall_filter(&self, _: H128) -> JsonRpcResult<bool>;
32}
