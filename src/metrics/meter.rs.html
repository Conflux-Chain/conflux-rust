20// Meters count events to produce exponentially-weighted moving average rates
21// at one-, five-, and fifteen-minutes and a mean rate.
22pub trait Meter: Send + Sync {
23    fn count(&self) -> usize { 0 }
24    fn mark(&self, _n: usize) {}
25    fn rate1(&self) -> f64 { 0.0 }
26    fn rate5(&self) -> f64 { 0.0 }
27    fn rate15(&self) -> f64 { 0.0 }
28    fn rate_mean(&self) -> f64 { 0.0 }
29    fn snapshot(&self) -> Arc<dyn Meter> { Arc::new(MeterSnapshot::default()) }
30    fn stop(&self) {}
31}
32
33struct NoopMeter;
34impl Meter for NoopMeter {}
35
36pub fn register_meter(name: &str) -> Arc<dyn Meter> {
37    if !is_enabled() {
38        return Arc::new(NoopMeter);
39    }
40
41    let meter = Arc::new(StandardMeter::new(name.into()));
42    DEFAULT_REGISTRY
43        .write()
44        .register(name.into(), meter.clone());
45    ARBITER.meters.lock().insert(name.into(), meter.clone());
46
47    meter
48}
49
50pub fn register_meter_with_group(group: &str, name: &str) -> Arc<dyn Meter> {
51    if !is_enabled() {
52        return Arc::new(NoopMeter);
53    }
54
55    let mut full_meter_name = String::from(group);
56    full_meter_name.push('_');
57    full_meter_name.push_str(name);
58
59    let meter = Arc::new(StandardMeter::new(full_meter_name.clone()));
60    DEFAULT_GROUPING_REGISTRY.write().register(
61        group.into(),
62        name.into(),
63        meter.clone(),
64    );
65
66    let mut meters = ARBITER.meters.lock();
67    assert_eq!(meters.contains_key(&full_meter_name), false);
68    meters.insert(full_meter_name, meter.clone());
69
70    meter
71}

73#[derive(Default, Clone)]
74struct MeterSnapshot {
75    count: usize,
76    rates: [u64; 4], // m1, m5, m15 and mean
77}
78
79impl Meter for MeterSnapshot {
80    fn count(&self) -> usize { self.count }
81
82    fn rate1(&self) -> f64 { f64::from_bits(self.rates[0]) }
83
84    fn rate5(&self) -> f64 { f64::from_bits(self.rates[1]) }
85
86    fn rate15(&self) -> f64 { f64::from_bits(self.rates[2]) }
87
88    fn rate_mean(&self) -> f64 { f64::from_bits(self.rates[3]) }
89
90    fn snapshot(&self) -> Arc<dyn Meter> { Arc::new(self.clone()) }
91}

93pub struct StandardMeter {
94    name: String,
95    snapshot: RwLock<MeterSnapshot>,
96    ewmas: [EWMA; 3],
97    start_time: Instant,
98    stopped: AtomicBool,
99}
100
101impl StandardMeter {
102    pub fn new(name: String) -> Self {
103        StandardMeter {
104            name,
105            snapshot: RwLock::new(MeterSnapshot::default()),
106            ewmas: [EWMA::new(1.0), EWMA::new(5.0), EWMA::new(15.0)],
107            start_time: Instant::now(),
108            stopped: AtomicBool::new(false),
109        }
110    }
111
112    fn tick(&self) {
113        let mut snapshot = self.snapshot.write();
114
115        for i in 0..3 {
116            self.ewmas[i].tick();
117            snapshot.rates[i] = f64::to_bits(self.ewmas[i].rate());
118        }
119
120        let rate_mean_nano =
121            snapshot.count as f64 / self.start_time.elapsed().as_nanos() as f64;
122        snapshot.rates[3] = f64::to_bits(rate_mean_nano * 1e9);
123    }
124}
125
126impl Meter for StandardMeter {
127    fn count(&self) -> usize { self.snapshot.read().count }
128
129    fn mark(&self, n: usize) {
130        if self.stopped.load(ORDER) {
131            return;
132        }
133
134        let mut snapshot = self.snapshot.write();
135        snapshot.count += n;
136
137        self.ewmas[0].update(n);
138        self.ewmas[1].update(n);
139        self.ewmas[2].update(n);
140
141        let rate_mean_nano =
142            snapshot.count as f64 / self.start_time.elapsed().as_nanos() as f64;
143        snapshot.rates[3] = f64::to_bits(rate_mean_nano * 1e9);
144    }
145
146    fn rate1(&self) -> f64 { f64::from_bits(self.snapshot.read().rates[0]) }
147
148    fn rate5(&self) -> f64 { f64::from_bits(self.snapshot.read().rates[1]) }
149
150    fn rate15(&self) -> f64 { f64::from_bits(self.snapshot.read().rates[2]) }
151
152    fn rate_mean(&self) -> f64 { f64::from_bits(self.snapshot.read().rates[3]) }
153
154    fn snapshot(&self) -> Arc<dyn Meter> {
155        Arc::new(self.snapshot.read().clone())
156    }
157
158    fn stop(&self) {
159        if let Ok(false) =
160            self.stopped.compare_exchange(false, true, ORDER, ORDER)
161        {
162            ARBITER.meters.lock().remove(&self.name);
163        }
164    }
165}
166
167impl Metric for StandardMeter {
168    fn get_type(&self) -> &str { "Meter" }
169}
170
171impl Drop for StandardMeter {
172    fn drop(&mut self) { self.stop(); }
173}

179/// MeterArbiter ticks meters every 5s from a single thread.
180/// meters are references in a set for future stopping.
181struct MeterArbiter {
182    meters: Arc<Mutex<HashMap<String, Arc<StandardMeter>>>>,
183    timer: Timer,
184}
185
186unsafe impl Send for MeterArbiter {}
187unsafe impl Sync for MeterArbiter {}
188
189impl Default for MeterArbiter {
190    fn default() -> Self {
191        let arbiter = MeterArbiter {
192            meters: Arc::new(Mutex::new(HashMap::new())),
193            timer: Timer::new(),
194        };
195
196        let meters = arbiter.meters.clone();
197        arbiter
198            .timer
199            .schedule_repeating(Duration::seconds(5), move || {
200                for (_, meter) in meters.lock().iter() {
201                    meter.tick();
202                }
203            })
204            .ignore();
205
206        arbiter
207    }
208}

210/// A struct used to measure time in metrics.
211pub struct MeterTimer {
212    meter: &'static dyn Meter,
213    start: Instant,
214}
215
216impl MeterTimer {
217    /// Call this to measure the time to run to the end of the current scope.
218    /// It will add the time from the function called till the returned
219    /// instance is dropped to `meter`.
220    pub fn time_func(meter: &'static dyn Meter) -> Self {
221        Self {
222            meter,
223            start: Instant::now(),
224        }
225    }
226}
227
228impl Drop for MeterTimer {
229    fn drop(&mut self) {
230        self.meter
231            .mark((Instant::now() - self.start).as_nanos() as usize)
232    }
