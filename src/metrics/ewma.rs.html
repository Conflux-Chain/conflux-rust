5use crate::metrics::ORDER;
6use std::sync::atomic::{AtomicBool, AtomicU64, AtomicUsize};
7
8/// EWMAs continuously calculate an exponentially-weighted moving average based
9/// on an outside source of clock ticks.
10pub struct EWMA {
11    uncounted: AtomicUsize,
12    alpha: f64,
13    rate: AtomicU64,
14    init: AtomicBool,
15}
16
17impl EWMA {
18    /// Constructs a new EWMA for a n-minutes moving average.
19    pub fn new(n: f64) -> Self {
20        EWMA::new_with_alpha(1.0 - (-5.0 / 60.0 / n).exp())
21    }
22
23    fn new_with_alpha(alpha: f64) -> Self {
24        EWMA {
25            uncounted: AtomicUsize::new(0),
26            alpha,
27            rate: AtomicU64::new(0),
28            init: AtomicBool::new(false),
29        }
30    }
31
32    /// Rate returns the moving average rate of events per second.
33    pub fn rate(&self) -> f64 { f64::from_bits(self.rate.load(ORDER)) * 1e9 }
34
35    /// Update adds n uncounted events.
36    pub fn update(&self, n: usize) { self.uncounted.fetch_add(n, ORDER); }
37
38    /// Ticks the clock to update the moving average. It assumes it is called
39    /// every 5 seconds.
40    pub fn tick(&self) {
41        let count = self.uncounted.swap(0, ORDER) as f64;
42        let instant_rate = count / 5e9;
43
44        if let Err(true) = self.init.compare_exchange(false, true, ORDER, ORDER)
45        {
46            let mut current_rate = f64::from_bits(self.rate.load(ORDER));
47            current_rate += self.alpha * (instant_rate - current_rate);
48            self.rate.store(f64::to_bits(current_rate), ORDER);
49        } else {
50            self.rate.store(f64::to_bits(instant_rate), ORDER);
51        }
52    }
53}
