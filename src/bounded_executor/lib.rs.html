10//! A bounded tokio [`Handle`]. Only a bounded number of tasks can run
11//! concurrently when spawned through this executor, defined by the initial
12//! `capacity`.
13
14use futures::future::{Future, FutureExt};
15use std::sync::Arc;
16use tokio::{
17    runtime::Handle,
18    sync::{OwnedSemaphorePermit, Semaphore},
19    task::JoinHandle,
20};
21
22#[derive(Clone, Debug)]
23pub struct BoundedExecutor {
24    semaphore: Arc<Semaphore>,
25    executor: Handle,
26}
27
28impl BoundedExecutor {
29    /// Create a new `BoundedExecutor` from an existing tokio [`Handle`]
30    /// with a maximum concurrent task capacity of `capacity`.
31    pub fn new(capacity: usize, executor: Handle) -> Self {
32        let semaphore = Arc::new(Semaphore::new(capacity));
33        Self {
34            semaphore,
35            executor,
36        }
37    }
38
39    /// Spawn a [`Future`] on the `BoundedExecutor`. This function is async and
40    /// will block if the executor is at capacity until one of the other spawned
41    /// futures completes. This function returns a [`JoinHandle`] that the
42    /// caller can `.await` on for the results of the [`Future`].
43    pub async fn spawn<F>(&self, f: F) -> JoinHandle<F::Output>
44    where
45        F: Future + Send + 'static,
46        F::Output: Send + 'static,
47    {
48        let permit = self.semaphore.clone().acquire_owned().await.unwrap();
49        self.spawn_with_permit(f, permit)
50    }
51
52    /// Try to spawn a [`Future`] on the `BoundedExecutor`. If the
53    /// `BoundedExecutor` is at capacity, this will return an `Err(F)`,
54    /// passing back the future the caller attempted to spawn. Otherwise,
55    /// this will spawn the future on the executor and send back a
56    /// [`JoinHandle`] that the caller can `.await` on for the results of
57    /// the [`Future`].
58    pub fn try_spawn<F>(&self, f: F) -> Result<JoinHandle<F::Output>, F>
59    where
60        F: Future + Send + 'static,
61        F::Output: Send + 'static,
62    {
63        match self.semaphore.clone().try_acquire_owned().ok() {
64            Some(permit) => Ok(self.spawn_with_permit(f, permit)),
65            None => Err(f),
66        }
67    }
68
69    fn spawn_with_permit<F>(
70        &self, f: F, spawn_permit: OwnedSemaphorePermit,
71    ) -> JoinHandle<F::Output>
72    where
73        F: Future + Send + 'static,
74        F::Output: Send + 'static,
75    {
76        // Release the permit back to the semaphore when this task completes.
77        let f = f.map(move |ret| {
78            drop(spawn_permit);
79            ret
80        });
81        self.executor.spawn(f)
82    }
83}

85#[cfg(test)]
86mod test {
87    use super::*;
88    use futures::{channel::oneshot, executor::block_on, future::Future};
89    use std::{
90        sync::atomic::{AtomicU32, Ordering},
91        time::Duration,
92    };
93    use tokio::{runtime::Runtime, time::sleep};
94
95    #[test]
96    fn try_spawn() {
97        let rt = Runtime::new().unwrap();
98        let executor = rt.handle().clone();
99        let executor = BoundedExecutor::new(1, executor);
100
101        let (tx1, rx1) = oneshot::channel();
102        let (tx2, rx2) = oneshot::channel();
103
104        // executor has a free slot, spawn should succeed
105
106        let f1 = executor.try_spawn(rx1).unwrap();
107
108        // executor is full, try_spawn should return err and give back the task
109        // we attempted to spawn
110
111        let rx2 = executor.try_spawn(rx2).unwrap_err();
112
113        // complete f1 future, should open a free slot in executor
114
115        tx1.send(()).unwrap();
116        block_on(f1).unwrap().unwrap();
117
118        // should successfully spawn a new task now that the first is complete
119
120        let f2 = executor.try_spawn(rx2).unwrap();
121
122        // cleanup
123
124        tx2.send(()).unwrap();
125        block_on(f2).unwrap().unwrap();
126    }
127
128    fn yield_task() -> impl Future<Output = ()> {
129        sleep(Duration::from_millis(1)).map(|_| ())
130    }
131
132    // spawn NUM_TASKS futures on a BoundedExecutor, ensuring that no more than
133    // MAX_WORKERS ever enter the critical section.
134    #[test]
135    fn concurrent_bounded_executor() {
136        const MAX_WORKERS: u32 = 20;
137        const NUM_TASKS: u32 = 1000;
138        static WORKERS: AtomicU32 = AtomicU32::new(0);
139        static COMPLETED_TASKS: AtomicU32 = AtomicU32::new(0);
140
141        let rt = Runtime::new().unwrap();
142        let executor = rt.handle().clone();
143        let executor = BoundedExecutor::new(MAX_WORKERS as usize, executor);
144
145        for _ in 0..NUM_TASKS {
146            block_on(executor.spawn(async move {
147                // acquired permit, there should only ever be MAX_WORKERS in
148                // this critical section
149
150                let prev_workers = WORKERS.fetch_add(1, Ordering::SeqCst);
151                assert!(prev_workers < MAX_WORKERS);
152
153                // yield back to the tokio scheduler
154                yield_task().await;
155
156                let prev_workers = WORKERS.fetch_sub(1, Ordering::SeqCst);
157                assert!(prev_workers > 0 && prev_workers <= MAX_WORKERS);
158
159                COMPLETED_TASKS.fetch_add(1, Ordering::Relaxed);
160            }));
161        }
162
163        // spin until completed
164        loop {
165            let completed = COMPLETED_TASKS.load(Ordering::Relaxed);
166            if completed == NUM_TASKS {
167                break;
168            } else {
169                std::hint::spin_loop()
170            }
171        }
172    }
