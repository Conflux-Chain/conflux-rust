8use crate::Error;
9use consensus_types::common::{Author, Round};
10use diem_logger::Schema;
11use serde::Serialize;
12
13#[derive(Schema)]
14pub struct SafetyLogSchema<'a> {
15    name: LogEntry,
16    event: LogEvent,
17    round: Option<Round>,
18    preferred_round: Option<u64>,
19    last_voted_round: Option<u64>,
20    epoch: Option<u64>,
21    #[schema(display)]
22    error: Option<&'a Error>,
23    author: Option<Author>,
24}
25
26impl<'a> SafetyLogSchema<'a> {
27    pub fn new(name: LogEntry, event: LogEvent) -> Self {
28        Self {
29            name,
30            event,
31            round: None,
32            preferred_round: None,
33            last_voted_round: None,
34            epoch: None,
35            error: None,
36            author: None,
37        }
38    }
39}
40
41#[derive(Clone, Copy, Serialize)]
42#[serde(rename_all = "snake_case")]
43pub enum LogEntry {
44    ConsensusState,
45    ConstructAndSignVote,
46    Epoch,
47    Initialize,
48    KeyReconciliation,
49    LastVotedRound,
50    PreferredRound,
51    SignProposal,
52    SignTimeout,
53    State,
54}
55
56impl LogEntry {
57    pub fn as_str(&self) -> &'static str {
58        match self {
59            LogEntry::ConsensusState => "consensus_state",
60            LogEntry::ConstructAndSignVote => "construct_and_sign_vote",
61            LogEntry::Epoch => "epoch",
62            LogEntry::Initialize => "initialize",
63            LogEntry::LastVotedRound => "last_voted_round",
64            LogEntry::KeyReconciliation => "key_reconciliation",
65            LogEntry::PreferredRound => "preferred_round",
66            LogEntry::SignProposal => "sign_proposal",
67            LogEntry::SignTimeout => "sign_timeout",
68            LogEntry::State => "state",
69        }
70    }
71}
72
73#[derive(Clone, Copy, Serialize)]
74#[serde(rename_all = "snake_case")]
75pub enum LogEvent {
76    Error,
77    Request,
78    Success,
79    Update,
80}
