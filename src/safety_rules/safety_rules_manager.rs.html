26pub fn storage(config: &SafetyRulesConfig) -> PersistentSafetyStorage {
27    let backend = &config.backend;
28    let internal_storage: Storage =
29        backend.try_into().expect("Unable to initialize storage");
30    if let Err(error) = internal_storage.available() {
31        panic!("Storage is not available: {:?}", error);
32    }
33
34    if let Some(test_config) = &config.test {
35        let author = test_config.author;
36        let consensus_private_key = test_config
37            .consensus_key
38            .as_ref()
39            .expect("Missing consensus key in test config")
40            .private_key();
41        diem_debug!("safety_rules_manager: backend={:?}", config.backend,);
42
43        PersistentSafetyStorage::initialize(
44            internal_storage,
45            author,
46            consensus_private_key,
47            config.enable_cached_safety_data,
48        )
49    } else {
50        panic!("Remote consensus key storage not supported!")
51    }
52}
53
54enum SafetyRulesWrapper {
55    Local(Arc<RwLock<SafetyRules>>),
56    Process(ProcessService),
57    Serializer(Arc<RwLock<SerializerService>>),
58    Thread(ThreadService),
59}
60
61pub struct SafetyRulesManager {
62    internal_safety_rules: SafetyRulesWrapper,
63}
64
65impl SafetyRulesManager {
66    pub fn new(config: &SafetyRulesConfig) -> Self {
67        if let SafetyRulesService::Process(conf) = &config.service {
68            return Self::new_process(
69                conf.server_address(),
70                config.network_timeout_ms,
71            );
72        }
73
74        let storage = storage(config);
75        let verify_vote_proposal_signature =
76            config.verify_vote_proposal_signature;
77        let export_consensus_key = config.export_consensus_key;
78        let author = config.test.as_ref().map(|c| c.author).unwrap_or_default();
79        match config.service {
80            SafetyRulesService::Local => Self::new_local(
81                storage,
82                verify_vote_proposal_signature,
83                export_consensus_key,
84                config.vrf_private_key.as_ref().map(|key| key.private_key()),
85                author,
86            ),
87            SafetyRulesService::Serializer => Self::new_serializer(
88                storage,
89                verify_vote_proposal_signature,
90                export_consensus_key,
91                author,
92            ),
93            SafetyRulesService::Thread => Self::new_thread(
94                storage,
95                verify_vote_proposal_signature,
96                export_consensus_key,
97                config.network_timeout_ms,
98            ),
99            _ => {
100                panic!("Unimplemented SafetyRulesService: {:?}", config.service)
101            }
102        }
103    }
104
105    pub fn new_local(
106        storage: PersistentSafetyStorage, verify_vote_proposal_signature: bool,
107        export_consensus_key: bool,
108        vrf_private_key: Option<ConsensusVRFPrivateKey>,
109        author: AccountAddress,
110    ) -> Self {
111        let safety_rules = SafetyRules::new(
112            storage,
113            verify_vote_proposal_signature,
114            export_consensus_key,
115            vrf_private_key,
116            author,
117        );
118        Self {
119            internal_safety_rules: SafetyRulesWrapper::Local(Arc::new(
120                RwLock::new(safety_rules),
121            )),
122        }
123    }
124
125    pub fn new_process(server_addr: SocketAddr, timeout_ms: u64) -> Self {
126        let process_service = ProcessService::new(server_addr, timeout_ms);
127        Self {
128            internal_safety_rules: SafetyRulesWrapper::Process(process_service),
129        }
130    }
131
132    pub fn new_serializer(
133        storage: PersistentSafetyStorage, verify_vote_proposal_signature: bool,
134        export_consensus_key: bool, author: AccountAddress,
135    ) -> Self {
136        let safety_rules = SafetyRules::new(
137            storage,
138            verify_vote_proposal_signature,
139            export_consensus_key,
140            // TODO(lpl): Support this?
141            None,
142            author,
143        );
144        let serializer_service = SerializerService::new(safety_rules);
145        Self {
146            internal_safety_rules: SafetyRulesWrapper::Serializer(Arc::new(
147                RwLock::new(serializer_service),
148            )),
149        }
150    }
151
152    pub fn new_thread(
153        storage: PersistentSafetyStorage, verify_vote_proposal_signature: bool,
154        export_consensus_key: bool, timeout_ms: u64,
155    ) -> Self {
156        let thread = ThreadService::new(
157            storage,
158            verify_vote_proposal_signature,
159            export_consensus_key,
160            timeout_ms,
161        );
162        Self {
163            internal_safety_rules: SafetyRulesWrapper::Thread(thread),
164        }
165    }
166
167    pub fn client(&self) -> Box<dyn TSafetyRules + Send + Sync> {
168        match &self.internal_safety_rules {
169            SafetyRulesWrapper::Local(safety_rules) => {
170                Box::new(LocalClient::new(safety_rules.clone()))
171            }
172            SafetyRulesWrapper::Process(process) => Box::new(process.client()),
173            SafetyRulesWrapper::Serializer(serializer_service) => {
174                Box::new(SerializerClient::new(serializer_service.clone()))
175            }
176            SafetyRulesWrapper::Thread(thread) => Box::new(thread.client()),
177        }
178    }
179}
