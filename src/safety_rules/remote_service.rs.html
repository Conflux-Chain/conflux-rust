8use crate::{
9    persistent_safety_storage::PersistentSafetyStorage,
10    serializer::{
11        SafetyRulesInput, SerializerClient, SerializerService,
12        TSerializerClient,
13    },
14    Error, SafetyRules, TSafetyRules,
15};
16use diem_logger::warn as diem_warn;
17use diem_secure_net::{NetworkClient, NetworkServer};
18use diem_types::validator_config::ConsensusVRFPrivateKey;
19use std::net::SocketAddr;
20
21pub trait RemoteService {
22    fn client(&self) -> SerializerClient {
23        let network_client = NetworkClient::new(
24            "safety-rules",
25            self.server_address(),
26            self.network_timeout_ms(),
27        );
28        let service = Box::new(RemoteClient::new(network_client));
29        SerializerClient::new_client(service)
30    }
31
32    fn server_address(&self) -> SocketAddr;
33
34    /// Network Timeout in milliseconds.
35    fn network_timeout_ms(&self) -> u64;
36}
37
38pub fn execute(
39    storage: PersistentSafetyStorage, listen_addr: SocketAddr,
40    verify_vote_proposal_signature: bool, export_consensus_key: bool,
41    network_timeout_ms: u64, vrf_private_key: Option<ConsensusVRFPrivateKey>,
42) {
43    let mut safety_rules = SafetyRules::new(
44        storage,
45        verify_vote_proposal_signature,
46        export_consensus_key,
47        vrf_private_key,
48        Default::default(),
49    );
50    if let Err(e) = safety_rules.consensus_state() {
51        diem_warn!("Unable to print consensus state: {}", e);
52    }
53
54    let mut serializer_service = SerializerService::new(safety_rules);
55    let mut network_server =
56        NetworkServer::new("safety-rules", listen_addr, network_timeout_ms);
57
58    loop {
59        if let Err(e) =
60            process_one_message(&mut network_server, &mut serializer_service)
61        {
62            diem_warn!("Failed to process message: {}", e);
63        }
64    }
65}
66
67fn process_one_message(
68    network_server: &mut NetworkServer,
69    serializer_service: &mut SerializerService,
70) -> Result<(), Error> {
71    let request = network_server.read()?;
72    let response = serializer_service.handle_message(request)?;
73    network_server.write(&response)?;
74    Ok(())
75}
76
77struct RemoteClient {
78    network_client: NetworkClient,
79}
80
81impl RemoteClient {
82    pub fn new(network_client: NetworkClient) -> Self {
83        Self { network_client }
84    }
85
86    fn process_one_message(&mut self, input: &[u8]) -> Result<Vec<u8>, Error> {
87        self.network_client.write(&input)?;
88        self.network_client.read().map_err(|e| e.into())
89    }
90}
91
92impl TSerializerClient for RemoteClient {
93    fn request(&mut self, input: SafetyRulesInput) -> Result<Vec<u8>, Error> {
94        let input_message = bcs::to_bytes(&input)?;
95        loop {
96            match self.process_one_message(&input_message) {
97                Err(err) => diem_warn!(
98                    "Failed to communicate with SafetyRules service: {}",
99                    err
100                ),
101                Ok(value) => return Ok(value),
102            }
103        }
104    }
105}
