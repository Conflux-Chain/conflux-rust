80make_solidity_function! {
81    struct AddPrivilege(Vec<Address>, "addPrivilege(address[])");
82}
83impl_function_type!(AddPrivilege, "non_payable_write");
84
85impl UpfrontPaymentTrait for AddPrivilege {
86    fn upfront_gas_payment(
87        &self, input: &Vec<Address>, _: &ActionParams,
88        context: &InternalRefContext,
89    ) -> DbResult<U256> {
90        Ok(U256::from(context.spec.sstore_reset_gas) * input.len())
91    }
92}
93
94impl SimpleExecutionTrait for AddPrivilege {
95    fn execute_inner(
96        &self, addresses: Vec<Address>, params: &ActionParams,
97        context: &mut InternalRefContext,
98    ) -> vm::Result<()> {
99        if !context.is_contract_address(&params.sender)? {
100            return Err(vm::Error::InternalContract(
101                "normal account is not allowed to set commission_privilege"
102                    .into(),
103            ));
104        }
105        add_privilege(
106            params.sender,
107            addresses,
108            params,
109            context.state,
110            context.substate,
111        )
112    }
113}

226}
227impl_function_type!(IsWhitelisted, "query", gas: |spec: &Spec| spec.cold_sload_gas);
228
229impl SimpleExecutionTrait for IsWhitelisted {
230    fn execute_inner(
231        &self, (contract, user): (Address, Address), _: &ActionParams,
232        context: &mut InternalRefContext,
233    ) -> vm::Result<bool> {
234        if context.is_contract_address(&contract)? {
235            Ok(context.state.check_contract_whitelist(&contract, &user)?)
236        } else {
237            Ok(false)
238        }
239    }
240}

330make_solidity_function! {
331    struct AvailableStoragePoints(Address, "getAvailableStoragePoints(address)", U256);
332}
333impl_function_type!(AvailableStoragePoints, "query", gas: |spec: &Spec| spec.cold_sload_gas);
334
335impl SimpleExecutionTrait for AvailableStoragePoints {
336    fn execute_inner(
337        &self, contract: Address, _: &ActionParams,
338        context: &mut InternalRefContext,
339    ) -> vm::Result<U256> {
340        if context.is_contract_address(&contract)? {
341            Ok(context
342                .state
343                .available_storage_points_for_collateral(&contract)?
344                / *DRIPS_PER_STORAGE_COLLATERAL_UNIT)
345        } else {
346            Ok(U256::zero())
347        }
348    }
349}
