13impl OverlayAccount {
14    pub fn commit(
15        self, db: &mut StateDb, address: &AddressWithSpace,
16        mut debug_record: Option<&mut ComputeEpochDebugRecord>,
17    ) -> DbResult<()> {
18        // When committing an overlay account, the execution of an epoch has
19        // finished. In this case, all the checkpoints except the bottom one
20        // must be removed. (Each checkpoint is a mapping from addresses to
21        // overlay accounts.)
22
23        // Commit storage entries
24
25        assert!(self.storage_write_checkpoint.is_none());
26        assert_eq!(Arc::strong_count(&self.storage_write_cache), 1);
27        assert!(self.storage_write_cache.read().is_empty());
28        let write_cache = &mut self.storage_committed_cache.write();
29        for (k, mut v) in write_cache.drain() {
30            let address_key =
31                StorageKey::new_storage_key(&self.address.address, k.as_ref())
32                    .with_space(self.address.space);
33            let debug_record = debug_record.as_deref_mut();
34            if v.owner == Some(self.address.address) {
35                v.owner = None;
36            }
37            match v.value.is_zero() {
38                true => db.delete(address_key, debug_record)?,
39                false => {
40                    db.set::<StorageValue>(address_key, &v, debug_record)?
41                }
42            }
43        }
44
45        // Commit code
46
47        if let Some(code_info) = self.code.as_ref() {
48            let storage_key = StorageKey::new_code_key(
49                &self.address.address,
50                &self.code_hash,
51            )
52            .with_space(self.address.space);
53            db.set::<CodeInfo>(
54                storage_key,
55                code_info,
56                debug_record.as_deref_mut(),
57            )?;
58        }
59
60        // Commit deposit list
61
62        if let Some(deposit_list) = self.deposit_list.as_ref() {
63            self.address.assert_native();
64            let storage_key =
65                StorageKey::new_deposit_list_key(&self.address.address)
66                    .with_space(self.address.space);
67            db.set::<DepositList>(
68                storage_key,
69                deposit_list,
70                debug_record.as_deref_mut(),
71            )?;
72        }
73
74        // Commit votestake list
75
76        if let Some(vote_stake_list) = self.vote_stake_list.as_ref() {
77            self.address.assert_native();
78            let storage_key =
79                StorageKey::new_vote_list_key(&self.address.address)
80                    .with_space(self.address.space);
81            db.set::<VoteStakeList>(
82                storage_key,
83                vote_stake_list,
84                debug_record.as_deref_mut(),
85            )?;
86        }
87
88        // Commit storage layout
89
90        if let Some(layout) = self.storage_layout_change.clone() {
91            db.set_storage_layout(
92                &self.address,
93                layout,
94                debug_record.as_deref_mut(),
95            )?;
96        }
97
98        // Commit basic fields
99
100        db.set::<Account>(
101            StorageKey::new_account_key(&address.address)
102                .with_space(address.space),
103            &self.as_account(),
104            debug_record,
105        )?;
106
107        Ok(())
108    }
109
110    pub fn commit_cache(
111        &mut self, retain_transient_storage: bool, dirty: bool,
112    ) {
113        assert!(self.storage_write_checkpoint.is_none());
114        assert!(self.transient_storage_checkpoint.is_none());
115        assert_eq!(Arc::strong_count(&self.storage_write_cache), 1);
116
117        let mut storage_write_cache = self.storage_write_cache.write();
118        let mut storage_commit_cache = self.storage_committed_cache.write();
119
120        for (key, value) in storage_write_cache.drain() {
121            if let WriteCacheItem::Write(value) = value {
122                assert!(dirty);
123                storage_commit_cache.insert(key, value);
124            }
125        }
126
127        if !retain_transient_storage {
128            self.transient_storage_cache.write().clear();
129        }
130    }
131}
