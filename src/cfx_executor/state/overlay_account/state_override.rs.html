7impl AccountEntry {
8    pub fn from_loaded_with_override(
9        address: &AddressWithSpace, account: Option<Account>,
10        acc_overrides: &AccountOverride,
11    ) -> Self {
12        let acc = account.unwrap_or_else(|| Account::new_empty(address));
13        AccountEntry::Cached(
14            OverlayAccount::from_loaded_with_override(
15                address,
16                acc,
17                acc_overrides,
18            ),
19            true,
20        )
21    }
22}
23
24impl OverlayAccount {
25    pub fn from_loaded_with_override(
26        address: &AddressWithSpace, account: Account,
27        acc_overrides: &AccountOverride,
28    ) -> Self {
29        let mut acc = Self::from_loaded(address, account);
30
31        if let Some(balance) = acc_overrides.balance {
32            let curr_balance = *acc.balance();
33            if curr_balance > U256::zero() {
34                acc.sub_balance(&curr_balance);
35            }
36            acc.add_balance(&balance);
37        }
38
39        if let Some(nonce) = acc_overrides.nonce {
40            acc.set_nonce(&U256::from(nonce.as_u64()));
41        }
42
43        if let Some(code) = acc_overrides.code.as_ref() {
44            acc.override_code(code.clone(), address.address);
45        }
46
47        match &acc_overrides.state {
48            AccountStateOverrideMode::State(state_override) => {
49                acc.override_storage_read_cache(state_override, true);
50            }
51            AccountStateOverrideMode::Diff(diff) => {
52                acc.override_storage_read_cache(diff, false);
53            }
54            AccountStateOverrideMode::None => {}
55        }
56
57        if acc_overrides.move_precompile_to.is_some() {
58            // TODO: impl move precompile to logic
59        }
60
61        acc
62    }
63
64    fn override_storage_read_cache(
65        &mut self, account_storage: &HashMap<H256, H256>,
66        complete_override: bool,
67    ) {
68        assert!(self.storage_write_checkpoint.is_none());
69
70        self.storage_overrided = complete_override;
71
72        let read_cache = Arc::get_mut(&mut self.storage_read_cache)
73            .expect("override should happen when no checkpoint")
74            .get_mut();
75        for (key, value) in account_storage {
76            let key = key.as_bytes().to_vec();
77            let value = U256::from_big_endian(value.as_bytes());
78            let owner = if self.address.space == Space::Native {
79                Some(self.address.address)
80            } else {
81                None
82            };
83            let storage_value = StorageValue { owner, value };
84            read_cache.insert(key, storage_value);
85        }
86    }
