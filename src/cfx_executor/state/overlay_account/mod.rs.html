74/// The access and manipulation object during execution, which includes both
75/// database-stored information and in-execution data of an account. It is a
76/// basic unit of state caching map and the checkpoint layers (more
77/// specifically, its extented struct `AccountEntry`).
78///
79/// In Conflux consensus executor, after the execution of one epoch, the
80/// `OverlayAccount` in cache will be commit to the database.
81pub struct OverlayAccount {
82    /* ----------------------------------------
83    - Database-stored fields for all accounts -
84    ---------------------------------------- */
85    /// Address of the account
86    address: AddressWithSpace,
87    /// Balance (in Drip) of the account
88    balance: U256,
89    /// Nonce of the account,
90    nonce: U256,
91    /// Code hash of the account.
92    code_hash: H256,
93    /// Staking balance (in Drip) of the account
94    staking_balance: U256,
95    /// Collateral (in Drip) of the account
96    collateral_for_storage: U256,
97    /// Accumulated interest return (in Drip) of the account.
98    ///
99    /// Inactive after CIP-43.
100    accumulated_interest_return: U256,
101
102    /* ---------------------------------------------------
103    -  Database-stored fields for contract accounts only -
104    --------------------------------------------------- */
105    /// Administrator of the account (Only applicable for contract)
106    admin: Address,
107    /// Sponsor information of the account (Only applicable for contract)
108    sponsor_info: SponsorInfo,
109
110    /* ----------------------------------------------------------------
111    -  Lazily loaded database-stored fields, also called `ext_fields` -
112    ---------------------------------------------------------------- */
113    /// List of the deposit info of the account, sorted in increasing order of
114    /// `deposit_time`. (`None` indicates not loaded from db.)
115    ///
116    /// Cleared after CIP-97.
117    deposit_list: Option<DepositList>,
118    /// List of the vote info of the account. (`None` indicates not loaded from
119    /// db.)
120    ///
121    /// The `unlock_block_number` sorted in increasing order and the `amount`
122    /// is sorted in decreasing order. All the `unlock_block_number` and
123    /// `amount` is unique in the list.
124    vote_stake_list: Option<VoteStakeList>,
125    /// The code of the account.  (`None` indicates not loaded from db if
126    /// `code_hash` isn't `KECCAK_EMPTY`.)
127    code: Option<CodeInfo>,
128
129    /* -------------------
130    -  In-execution data -
131    ------------------- */
132    /// Storage layout change of the account
133    storage_layout_change: Option<StorageLayout>,
134
135    /// Read cache of database for the storage entries of this account for
136    /// recording unchanged values.
137    storage_read_cache: Arc<RwLock<HashMap<Vec<u8>, StorageValue>>>,
138
139    /// Committed storage after a transaction for the storage entries of this
140    /// account for recording unchanged values.
141    storage_committed_cache: Arc<RwLock<HashMap<Vec<u8>, StorageValue>>>,
142
143    /// Write cache for the storage entries of this account for recording
144    /// changed values.
145    storage_write_cache: Arc<RwLock<HashMap<Vec<u8>, WriteCacheItem>>>,
146    storage_write_checkpoint:
147        Option<RwLock<WriteCheckpointLayer<Vec<u8>, WriteCacheItem>>>,
148
149    /// Transient storage from CIP-142
150    transient_storage_cache: Arc<RwLock<HashMap<Vec<u8>, U256>>>,
151    transient_storage_checkpoint: Option<WriteCheckpointLayer<Vec<u8>, U256>>,
152
153    /* ---------------
154    -  Special flags -
155    --------------- */
156    /// Indicates whether it is a newly created contract since last commit.
157    is_newly_created_contract: bool,
158
159    /// Indicates whether all the storage entries and lazily loaded fields of
160    /// this account on the database should be regarded as deleted be and
161    /// cleared later. It will be set when such a contract has been killed
162    /// since last commit.
163    pending_db_clear: bool,
164
165    /// Indicates whether the storage cache entries of this account have been
166    /// overrided by the passed-in storage entries.
167    /// When this flag is set, the storage entries will only be read from the
168    /// cache
169    storage_overrided: bool,
170
171    /* ---------------
172    -  Helper fields -
173    --------------- */
174    /// For CIP-151, `None` indicates that this is either a non-contract, a
175    /// contract in the process of being created, or a contract not created
176    /// in the current epoch. A `Some(_)` value indicates that this is a
177    /// contract created by a specific transaction.
178    create_transaction_hash: Option<H256>,
179}
180
181impl OverlayAccount {
182    /// Inditcates if this account can execute bytecode
183    pub fn is_contract(&self) -> bool {
184        self.code_hash != KECCAK_EMPTY || self.is_newly_created_contract
185    }
186
187    /// Inditcates if this account has been killed and has not been re-created
188    /// (e.g. sending balance to killed address can recreate it) since last
189    /// commit.
190    pub fn removed_without_update(&self) -> bool {
191        self.pending_db_clear && self.as_account().is_default()
192    }
193
194    /// Inditcates if this account's storage entries and lazily loaded fields on
195    /// db should be cleared. Upon committing the overlay account, if this flag
196    /// is set, db clearing for this account will be triggerred.
197    pub fn pending_db_clear(&self) -> bool { self.pending_db_clear }
198
199    /// Inditcates if this account's storage entries and lazily loaded fields on
200    /// db are marked invalid (so an entry is empty if not in cache).
201    pub fn fresh_storage(&self) -> bool {
202        let builtin_address = self.address.space == Space::Native
203            && self.address.address.is_builtin_address();
204        (self.is_newly_created_contract && !builtin_address)
205            || self.pending_db_clear
206            || self.storage_overrided
207    }
208}

222    fn clone(&self) -> Self {
223        Self {
224            address: self.address.clone(),
225            balance: self.balance.clone(),
226            nonce: self.nonce.clone(),
227            code_hash: self.code_hash.clone(),
228            staking_balance: self.staking_balance.clone(),
229            collateral_for_storage: self.collateral_for_storage.clone(),
230            accumulated_interest_return: self
231                .accumulated_interest_return
232                .clone(),
233            admin: self.admin.clone(),
234            sponsor_info: self.sponsor_info.clone(),
235            deposit_list: self.deposit_list.clone(),
236            vote_stake_list: self.vote_stake_list.clone(),
237            code: self.code.clone(),
238            storage_layout_change: self.storage_layout_change.clone(),
239            storage_read_cache: self.storage_read_cache.clone(),
240            storage_committed_cache: self.storage_committed_cache.clone(),
241            storage_write_cache: self.storage_write_cache.clone(),
242            storage_write_checkpoint: self
243                .storage_write_checkpoint
244                .as_ref()
245                .map(|x| RwLock::new(x.read().clone())),
246            transient_storage_cache: self.transient_storage_cache.clone(),
247            transient_storage_checkpoint: self
248                .transient_storage_checkpoint
249                .clone(),
250            is_newly_created_contract: self.is_newly_created_contract.clone(),
251            pending_db_clear: self.pending_db_clear.clone(),
252            storage_overrided: self.storage_overrided.clone(),
253            create_transaction_hash: self.create_transaction_hash.clone(),
254        }
255    }

258#[cfg(test)]
259mod tests_another {
260    use super::*;
261    use crate::state::get_state_for_genesis_write;
262    use primitives::is_default::IsDefault;
263    use std::str::FromStr;
264
265    fn test_account_is_default(account: &mut OverlayAccount) {
266        let state = get_state_for_genesis_write();
267
268        assert!(account.as_account().is_default());
269
270        account.cache_ext_fields(true, true, &state.db).unwrap();
271        assert!(account.vote_stake_list().is_default());
272        assert!(account.deposit_list().is_default());
273    }
274
275    #[test]
276    fn new_overlay_account_is_default() {
277        let normal_addr =
278            Address::from_str("1000000000000000000000000000000000000000")
279                .unwrap()
280                .with_native_space();
281        let builtin_addr =
282            Address::from_str("0000000000000000000000000000000000000000")
283                .unwrap()
284                .with_native_space();
285
286        test_account_is_default(&mut OverlayAccount::new_basic(
287            &normal_addr,
288            U256::zero(),
289        ));
290        test_account_is_default(&mut OverlayAccount::new_basic(
291            &builtin_addr,
292            U256::zero(),
293        ));
294    }
