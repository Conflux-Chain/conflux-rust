15// calculate modexp: left-to-right binary exponentiation to keep multiplicands
16// lower
17pub fn modexp(mut base: BigUint, exp: Vec<u8>, modulus: BigUint) -> BigUint {
18    const BITS_PER_DIGIT: usize = 8;
19
20    // n^m % 0 || n^m % 1
21    if modulus <= BigUint::one() {
22        return BigUint::zero();
23    }
24
25    // normalize exponent
26    let mut exp = exp.into_iter().skip_while(|d| *d == 0).peekable();
27
28    // n^0 % m
29    if let None = exp.peek() {
30        return BigUint::one();
31    }
32
33    // 0^n % m, n > 0
34    if base.is_zero() {
35        return BigUint::zero();
36    }
37
38    base = base % &modulus;
39
40    // Fast path for base divisible by modulus.
41    if base.is_zero() {
42        return BigUint::zero();
43    }
44
45    // Left-to-right binary exponentiation (Handbook of Applied Cryptography -
46    // Algorithm 14.79). http://www.cacr.math.uwaterloo.ca/hac/about/chap14.pdf
47    let mut result = BigUint::one();
48
49    for digit in exp {
50        let mut mask = 1 << (BITS_PER_DIGIT - 1);
51
52        for _ in 0..BITS_PER_DIGIT {
53            result = &result * &result % &modulus;
54
55            if digit & mask > 0 {
56                result = result * &base % &modulus;
57            }
58
59            mask >>= 1;
60        }
61    }
62
63    result
64}

66impl Precompile for ModexpImpl {
67    fn execute(
68        &self, input: &[u8], output: &mut BytesRef,
69    ) -> Result<(), Error> {
70        let mut reader = input.chain(io::repeat(0));
71        let mut buf = [0; 32];
72
73        // read lengths as usize.
74        // ignoring the first 24 bytes might technically lead us to fall out of
75        // consensus, but so would running out of addressable memory!
76        let mut read_len = |reader: &mut io::Chain<&[u8], io::Repeat>| {
77            reader
78                .read_exact(&mut buf[..])
79                .expect("reading from zero-extended memory cannot fail; qed");
80            BigEndian::read_u64(&buf[24..]) as usize
81        };
82
83        let base_len = read_len(&mut reader);
84        let exp_len = read_len(&mut reader);
85        let mod_len = read_len(&mut reader);
86
87        // Gas formula allows arbitrary large exp_len when base and modulus are
88        // empty, so we need to handle empty base first.
89        let r = if base_len == 0 && mod_len == 0 {
90            BigUint::zero()
91        } else {
92            // read the numbers themselves.
93            let mut buf = vec![0; max(mod_len, max(base_len, exp_len))];
94            let mut read_num = |reader: &mut io::Chain<&[u8], io::Repeat>,
95                                len: usize| {
96                reader.read_exact(&mut buf[..len]).expect(
97                    "reading from zero-extended memory cannot fail; qed",
98                );
99                BigUint::from_bytes_be(&buf[..len])
100            };
101
102            let base = read_num(&mut reader, base_len);
103
104            let mut exp_buf = vec![0; exp_len];
105            reader
106                .read_exact(&mut exp_buf[..exp_len])
107                .expect("reading from zero-extended memory cannot fail; qed");
108
109            let modulus = read_num(&mut reader, mod_len);
110
111            modexp(base, exp_buf, modulus)
112        };
113
114        // write output to given memory, left padded and same length as the
115        // modulus.
116        let bytes = r.to_bytes_be();
117
118        // always true except in the case of zero-length modulus, which leads to
119        // output of length and value 1.
120        if bytes.len() <= mod_len {
121            let res_start = mod_len - bytes.len();
122            output.write(res_start, &bytes);
123        }
124
125        Ok(())
126    }
127}

129/// A special pricing model for modular exponentiation.
130#[derive(Debug)]
131pub(crate) enum ModexpPricer {
132    Byzantium { divisor: usize },
133    // CIP-645e: EIP-2565
134    Berlin { base: usize },
135}
136
137impl ModexpPricer {
138    pub(crate) fn new_byzantium(divisor: usize) -> ModexpPricer {
139        ModexpPricer::Byzantium { divisor }
140    }
141
142    pub(crate) fn new_berlin(base: usize) -> ModexpPricer {
143        ModexpPricer::Berlin { base }
144    }
145}
146
147impl Pricer for ModexpPricer {
148    fn cost(&self, input: &[u8]) -> U256 {
149        let mut reader = input.chain(io::repeat(0));
150        let mut buf = [0; 32];
151
152        // read lengths as U256 here for accurate gas calculation.
153        let mut read_len = || {
154            reader
155                .read_exact(&mut buf[..])
156                .expect("reading from zero-extended memory cannot fail; qed");
157            U256::from_big_endian(&buf[..])
158        };
159        let base_len = read_len();
160        let exp_len = read_len();
161        let mod_len = read_len();
162
163        if mod_len.is_zero() && base_len.is_zero() {
164            return match self {
165                Self::Byzantium { .. } => 0.into(),
166                Self::Berlin { base } => (*base).into(),
167            };
168        }
169
170        let max_len = U256::from(u32::max_value() / 2);
171
172        if base_len > max_len || mod_len > max_len || exp_len > max_len {
173            return U256::max_value();
174        }
175        let (base_len, exp_len, mod_len) =
176            (base_len.low_u64(), exp_len.low_u64(), mod_len.low_u64());
177
178        // read fist 32-byte word of the exponent.
179        let exp_low = if base_len + 96 >= input.len() as u64 {
180            U256::zero()
181        } else {
182            let mut buf = [0; 32];
183            let mut reader =
184                input[(96 + base_len as usize)..].chain(io::repeat(0));
185            let len = min(exp_len, 32) as usize;
186            reader
187                .read_exact(&mut buf[(32 - len)..])
188                .expect("reading from zero-extended memory cannot fail; qed");
189            U256::from_big_endian(&buf[..])
190        };
191        let iter_count = max(Self::adjusted_exp_len(exp_len, exp_low), 1);
192
193        match self {
194            ModexpPricer::Byzantium { divisor } => Self::byzantium_gas_calc(
195                base_len, mod_len, iter_count, *divisor,
196            ),
197            ModexpPricer::Berlin { base } => {
198                Self::berlin_gas_calc(base_len, mod_len, iter_count, *base)
199            }
200        }
201    }
202}
203
204impl ModexpPricer {
205    pub fn byzantium_gas_calc(
206        base_len: u64, mod_len: u64, iter_count: u64, divisor: usize,
207    ) -> U256 {
208        let m = max(mod_len, base_len);
209        let (gas, overflow) =
210            Self::mult_complexity(m).overflowing_mul(iter_count);
211        if overflow {
212            return U256::max_value();
213        }
214        (gas / divisor as u64).into()
215    }
216
217    fn adjusted_exp_len(len: u64, exp_low: U256) -> u64 {
218        let bit_index = if exp_low.is_zero() {
219            0
220        } else {
221            (255 - exp_low.leading_zeros()) as u64
222        };
223        if len <= 32 {
224            bit_index
225        } else {
226            8 * (len - 32) + bit_index
227        }
228    }
229
230    fn mult_complexity(x: u64) -> u64 {
231        match x {
232            x if x <= 64 => x * x,
233            x if x <= 1024 => (x * x) / 4 + 96 * x - 3072,
234            x => (x * x) / 16 + 480 * x - 199680,
235        }
236    }
237
238    pub fn berlin_gas_calc(
239        base_len: u64, mod_len: u64, iter_count: u64, base_gas: usize,
240    ) -> U256 {
241        fn calculate_multiplication_complexity(
242            base_len: u64, mod_len: u64,
243        ) -> U256 {
244            let max_len = max(base_len, mod_len);
245            let mut words = max_len / 8;
246            if max_len % 8 > 0 {
247                words += 1;
248            }
249            let words = U256::from(words);
250            words * words
251        }
252
253        let multiplication_complexity =
254            calculate_multiplication_complexity(base_len, mod_len);
255        let gas = (multiplication_complexity * U256::from(iter_count))
256            / U256::from(3);
257        let gas_u64 = if gas >= U256::from(u64::MAX) {
258            u64::MAX
259        } else {
260            gas.as_u64()
261        };
262        max(base_gas as u64, gas_u64).into()
263    }
