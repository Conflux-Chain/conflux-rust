5/// Execution error.
6#[derive(Debug)]
7pub struct Error(pub String);
8
9impl From<&'static str> for Error {
10    fn from(val: &'static str) -> Self { Error(val.into()) }
11}
12
13impl From<String> for Error {
14    fn from(val: String) -> Self { Error(val) }
15}
16
17impl Into<cfx_vm_types::Error> for Error {
18    fn into(self) -> cfx_vm_types::Error {
19        cfx_vm_types::Error::BuiltIn(self.0)
20    }
21}

23/// Native implementation of a built-in contract.
