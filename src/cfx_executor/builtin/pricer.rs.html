5/// A linear pricing model. This computes a price using a base cost and a cost
6/// per-word.

9    pub(crate) word: usize,
10}

18pub(crate) struct ConstPricer {
19    price: u64,
20}
21
22impl ConstPricer {

26impl Pricer for Linear {
27    fn cost(&self, input: &[u8]) -> U256 {
28        U256::from(self.base)
29            + U256::from(self.word) * U256::from((input.len() + 31) / 32)
30    }
31}
32
33impl Pricer for ConstPricer {
34    fn cost(&self, _input: &[u8]) -> U256 { U256::from(self.price) }
35}

37/// A alt_bn128_parinig pricing model. This computes a price using a base cost
38/// and a cost per pair.
39pub(crate) struct AltBn128PairingPricer {
40    base: usize,
41    pair: usize,
42}
43
44impl AltBn128PairingPricer {

46        AltBn128PairingPricer { base, pair }
47    }
48}
49
50impl Pricer for AltBn128PairingPricer {
51    fn cost(&self, input: &[u8]) -> U256 {
52        let cost = U256::from(self.base)
53            + U256::from(self.pair) * U256::from(input.len() / 192);
54        cost
55    }

58/// Pricing for Blake2 compression function: each call costs the same amount per
59/// round.
60pub(crate) struct Blake2FPricer {
61    /// Price per round of Blake2 compression function.
62    gas_per_round: u64,
63}
64
65impl Blake2FPricer {
66    pub fn new(gas_per_round: u64) -> Self { Self { gas_per_round } }
67}
68
69impl Pricer for Blake2FPricer {
70    fn cost(&self, input: &[u8]) -> U256 {
71        const FOUR: usize = std::mem::size_of::<u32>();
72        // Returning zero if the conversion fails is fine because `execute()`
73        // will check the length and bail with the appropriate error.
74        if input.len() < FOUR {
75            return U256::zero();
76        }
77        let (rounds_bytes, _) = input.split_at(FOUR);
78        let rounds =
79            u32::from_be_bytes(rounds_bytes.try_into().unwrap_or([0u8; 4]));
80        U256::from(self.gas_per_round * rounds as u64)
81    }
82}
