14/// Encodes a single finite field element into byte slice with padding.
15pub(super) fn fp_to_bytes(out: &mut [u8], input: *const blst_fp) {
16    if out.len() != PADDED_FP_LENGTH {
17        return;
18    }
19    let (padding, rest) = out.split_at_mut(PADDING_LENGTH);
20    padding.fill(0);
21    // SAFETY: Out length is checked previously, `input` is a blst value.
22    unsafe { blst_bendian_from_fp(rest.as_mut_ptr(), input) };
23}

25/// Removes zeros with which the precompile inputs are left padded to 64 bytes.
26pub(super) fn remove_padding(input: &[u8]) -> Result<&[u8; FP_LENGTH], String> {
27    if input.len() != PADDED_FP_LENGTH {
28        return Err(format!(
29            "Padded input should be {PADDED_FP_LENGTH} bytes, was {}",
30            input.len()
31        ));
32    }
33    let (padding, unpadded) = input.split_at(PADDING_LENGTH);
34    if !padding.iter().all(|&x| x == 0) {
35        return Err(format!(
36            "{PADDING_LENGTH} top bytes of input are not zero",
37        ));
38    }
39    Ok(unpadded.try_into().unwrap())
40}

42/// Extracts a scalar from a 32 byte slice representation, decoding the input as
43/// a big endian unsigned integer. If the input is not exactly 32 bytes long, an
44/// error is returned.
45///
46/// From [EIP-2537](https://eips.ethereum.org/EIPS/eip-2537):
47/// * A scalar for the multiplication operation is encoded as 32 bytes by
48///   performing BigEndian encoding of the corresponding (unsigned) integer.
49///
50/// We do not check that the scalar is a canonical Fr element, because the EIP
51/// specifies:
52/// * The corresponding integer is not required to be less than or equal than
53///   main subgroup order `q`.
54pub(super) fn extract_scalar_input(
55    input: &[u8],
56) -> Result<blst_scalar, String> {
57    if input.len() != SCALAR_LENGTH {
58        return Err(format!(
59            "Input should be {SCALAR_LENGTH} bytes, was {}",
60            input.len()
61        ));
62    }
63
64    let mut out = blst_scalar::default();
65    // SAFETY: `input` length is checked previously, out is a blst value.
66    unsafe {
67        // Note: We do not use `blst_scalar_fr_check` here because, from
68        // EIP-2537:
69        //
70        // * The corresponding integer is not required to be less than or equal
71        //   than main subgroup
72        // order `q`.
73        blst_scalar_from_bendian(&mut out, input.as_ptr())
74    };
75
76    Ok(out)
77}

79/// Checks if the input is a valid big-endian representation of a field element.
80fn is_valid_be(input: &[u8; 48]) -> bool {
81    for (i, modulo) in input.iter().zip(MODULUS_REPR.iter()) {
82        match i.cmp(modulo) {
83            Ordering::Greater => return false,
84            Ordering::Less => return true,
85            Ordering::Equal => continue,
86        }
87    }
88    // Return false if matching the modulus
89    false
90}

92/// Checks whether or not the input represents a canonical field element,
93/// returning the field element if successful.
94pub(super) fn fp_from_bendian(
95    input: &[u8; 48],
96) -> Result<blst_fp, &'static str> {
97    if !is_valid_be(input) {
98        return Err("non-canonical fp value");
99    }
100    let mut fp = blst_fp::default();
101    // SAFETY: `input` has fixed length, and `fp` is a blst value.
102    unsafe {
103        // This performs the check for canonical field elements
104        blst_fp_from_bendian(&mut fp, input.as_ptr());
105    }
106
107    Ok(fp)
108}
