20pub struct Bls12Wrapper<F>(F);
21
22impl<F: Send + Sync + Fn(&[u8]) -> Result<Vec<u8>, String>> Precompile
23    for Bls12Wrapper<F>
24{
25    fn execute(
26        &self, input: &[u8], output: &mut cfx_bytes::BytesRef,
27    ) -> Result<(), super::Error> {
28        let res = (self.0)(input)?;
29        output.write(0, &res[..]);
30        Ok(())
31    }
32}
33
34pub fn bls12_builtin_factory(name: &str) -> Box<dyn Precompile> {
35    match name {
36        "bls12_g1add" => Box::new(Bls12Wrapper(g1_add::g1_add)),
37        "bls12_g1msm" => Box::new(Bls12Wrapper(g1_msm::g1_msm)),
38        "bls12_g2add" => Box::new(Bls12Wrapper(g2_add::g2_add)),
39        "bls12_g2msm" => Box::new(Bls12Wrapper(g2_msm::g2_msm)),
40        "bls12_pairing_check" => Box::new(Bls12Wrapper(pairing::pairing)),
41        "bls12_map_fp_to_g1" => {
42            Box::new(Bls12Wrapper(map_fp_to_g1::map_fp_to_g1))
43        }
44        "bls12_map_fp2_to_g2" => {
45            Box::new(Bls12Wrapper(map_fp2_to_g2::map_fp2_to_g2))
46        }
47        _ => panic!("Unsupported BLS12 precompile function: {}", name),
48    }
49}

51macro_rules! bls12_precompile {
52    ($address:expr, $price_fn:path, $impl_fn:path) => {
53        (
54            $address,
55            Box::new(StaticPlan($price_fn())),
56            Box::new(Bls12Wrapper($impl_fn)),
57        )
58    };
59}
