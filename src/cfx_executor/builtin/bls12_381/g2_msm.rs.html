20/// Implements EIP-2537 G2MSM precompile.
21/// G2 multi-scalar-multiplication call expects `288*k` bytes as an input that
22/// is interpreted as byte concatenation of `k` slices each of them being a byte
23/// concatenation of encoding of G2 point (`256` bytes) and encoding of a scalar
24/// value (`32` bytes).
25/// Output is an encoding of multi-scalar-multiplication operation result -
26/// single G2 point (`256` bytes).
27/// See also: <https://eips.ethereum.org/EIPS/eip-2537#abi-for-g2-multiexponentiation>
28pub(super) fn g2_msm(input: &[u8]) -> Result<Vec<u8>, String> {
29    let input_len = input.len();
30    if input_len == 0 || input_len % G2_MSM_INPUT_LENGTH != 0 {
31        return Err(format!(
32            "G2MSM input length should be multiple of {}, was {}",
33            G2_MSM_INPUT_LENGTH, input_len
34        ));
35    }
36
37    let k = input_len / G2_MSM_INPUT_LENGTH;
38
39    let mut g2_points: Vec<blst_p2> = Vec::with_capacity(k);
40    let mut scalars: Vec<u8> = Vec::with_capacity(k * SCALAR_LENGTH);
41    for i in 0..k {
42        let slice = &input[i * G2_MSM_INPUT_LENGTH
43            ..i * G2_MSM_INPUT_LENGTH + G2_INPUT_ITEM_LENGTH];
44        // BLST batch API for p2_affines blows up when you pass it a point at
45        // infinity, so we must filter points at infinity (and their
46        // corresponding scalars) from the input.
47        if slice.iter().all(|i| *i == 0) {
48            continue;
49        }
50
51        // NB: Scalar multiplications, MSMs and pairings MUST perform a subgroup
52        // check.
53        //
54        // So we set the subgroup_check flag to `true`
55        let p0_aff = &extract_g2_input(slice, true)?;
56
57        let mut p0 = blst_p2::default();
58        // SAFETY: `p0` and `p0_aff` are blst values.
59        unsafe { blst_p2_from_affine(&mut p0, p0_aff) };
60
61        g2_points.push(p0);
62
63        scalars.extend_from_slice(
64            &extract_scalar_input(
65                &input[i * G2_MSM_INPUT_LENGTH + G2_INPUT_ITEM_LENGTH
66                    ..i * G2_MSM_INPUT_LENGTH
67                        + G2_INPUT_ITEM_LENGTH
68                        + SCALAR_LENGTH],
69            )?
70            .b,
71        );
72    }
73
74    // Return infinity point if all points are infinity
75    if g2_points.is_empty() {
76        return Ok([0; 256].into());
77    }
78
79    let points = p2_affines::from(&g2_points);
80    let multiexp = points.mult(&scalars, NBITS);
81
82    let mut multiexp_aff = blst_p2_affine::default();
83    // SAFETY: `multiexp_aff` and `multiexp` are blst values.
84    unsafe { blst_p2_to_affine(&mut multiexp_aff, &multiexp) };
85
86    let out = encode_g2_point(&multiexp_aff);
87    Ok(out)
88}
