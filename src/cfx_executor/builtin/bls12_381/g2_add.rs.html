16/// G2 addition call expects `512` bytes as an input that is interpreted as byte
17/// concatenation of two G2 points (`256` bytes each).
18///
19/// Output is an encoding of addition operation result - single G2 point (`256`
20/// bytes).
21/// See also <https://eips.ethereum.org/EIPS/eip-2537#abi-for-g2-addition>
22pub(super) fn g2_add(input: &[u8]) -> Result<Vec<u8>, String> {
23    if input.len() != G2_ADD_INPUT_LENGTH {
24        return Err(format!(
25            "G2ADD input should be {G2_ADD_INPUT_LENGTH} bytes, was {}",
26            input.len()
27        ));
28    }
29
30    // NB: There is no subgroup check for the G2 addition precompile.
31    //
32    // So we set the subgroup checks here to `false`
33    let a_aff = &extract_g2_input(&input[..G2_INPUT_ITEM_LENGTH], false)?;
34    let b_aff = &extract_g2_input(&input[G2_INPUT_ITEM_LENGTH..], false)?;
35
36    let mut b = blst_p2::default();
37    // SAFETY: `b` and `b_aff` are blst values.
38    unsafe { blst_p2_from_affine(&mut b, b_aff) };
39
40    let mut p = blst_p2::default();
41    // SAFETY: `p`, `b` and `a_aff` are blst values.
42    unsafe { blst_p2_add_or_double_affine(&mut p, &b, a_aff) };
43
44    let mut p_aff = blst_p2_affine::default();
45    // SAFETY: `p_aff` and `p` are blst values.
46    unsafe { blst_p2_to_affine(&mut p_aff, &p) };
47
48    let out = encode_g2_point(&p_aff);
49    Ok(out)
50}

52pub(super) fn g2_add_gas() -> impl Pricer {
53    ConstPricer::new(G2_ADD_BASE_GAS_FEE)
