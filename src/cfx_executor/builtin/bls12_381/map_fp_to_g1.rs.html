14/// Field-to-curve call expects 64 bytes as an input that is interpreted as an
15/// element of Fp. Output of this call is 128 bytes and is an encoded G1 point.
16/// See also: <https://eips.ethereum.org/EIPS/eip-2537#abi-for-mapping-fp-element-to-g1-point>
17pub(super) fn map_fp_to_g1(input: &[u8]) -> Result<Vec<u8>, String> {
18    if input.len() != PADDED_FP_LENGTH {
19        return Err(format!(
20            "MAP_FP_TO_G1 input should be {PADDED_FP_LENGTH} bytes, was {}",
21            input.len()
22        ));
23    }
24
25    let input_p0 = remove_padding(input)?;
26    let fp = fp_from_bendian(input_p0)?;
27
28    let mut p = blst_p1::default();
29    // SAFETY: `p` and `fp` are blst values.
30    // Third argument is unused if null.
31    unsafe { blst_map_to_g1(&mut p, &fp, core::ptr::null()) };
32
33    let mut p_aff = blst_p1_affine::default();
34    // SAFETY: `p_aff` and `p` are blst values.
35    unsafe { blst_p1_to_affine(&mut p_aff, &p) };
36
37    let out = encode_g1_point(&p_aff);
38    Ok(out)
39}
40
41pub(super) fn map_fp_to_g1_gas() -> impl Pricer {
42    ConstPricer::new(MAP_FP_TO_G1_BASE_GAS_FEE)
43}
