14// A maximum interval to update the mining block
15const BLOCK_FORCE_UPDATE_INTERVAL: Duration = Duration::from_secs(10);
16// Avoid busy loop: sleep for 30ms if the current loop iteration does nothing
17const BLOCKGEN_LOOP_SLEEP_DURATION: Duration = time::Duration::from_millis(30);
18
19pub(crate) struct MiningSession<'a> {
20    bg: &'a BlockGenerator,
21    miner: &'a dyn MineWorker,
22    solution_rx: SolutionReceiver,
23    state: MineState,
24}
25
26impl<'a> MiningSession<'a> {
27    pub fn new(
28        bg: &'a BlockGenerator, miner: &'a dyn MineWorker,
29        solution_rx: SolutionReceiver,
30    ) -> Self {
31        let state =
32            MineState::with_capacity(bg.pow_config.pow_problem_window_size);
33
34        MiningSession {
35            bg,
36            miner,
37            solution_rx,
38            state,
39        }
40    }
41
42    pub fn run(mut self) {
43        while self.bg.is_running() {
44            // TODO: #transations TBD
45            if !self.should_mine() {
46                thread::sleep(BLOCKGEN_LOOP_SLEEP_DURATION);
47                continue;
48            }
49
50            if self.is_mining_block_outdated() {
51                // Compute new mining task and send to workers
52                self.update_mining_task();
53            }
54            // Pull the mining solution
55            self.process_pending_solutions();
56        }
57    }
58
59    fn should_mine(&self) -> bool {
60        self.pow_config.test_mode || !self.sync.catch_up_mode()
61    }
62
63    /// Check if we need to mine on a new block
64    fn is_mining_block_outdated(&self) -> bool {
65        // TODO: update the logic here.
66
67        let Some(block) = self.state.mining_block() else {
68            return true;
69        };
70
71        // 1st Check: if the parent block changed
72        let best_block_hash = self.consensus.best_block_hash();
73        if best_block_hash != *block.block_header.parent_hash() {
74            return true;
75        }
76
77        // 2nd Check: if the last block is too old, we will generate a new
78        // block. Checking transaction updates and referees might be
79        // costly and the trade-off is unclear here. It is simple to
80        // just enforce a time here.
81
82        self.state.last_assemble_elapsed() > BLOCK_FORCE_UPDATE_INTERVAL
83    }
84
85    fn update_mining_task(&mut self) {
86        let next_mining_block = self.assembler.assemble_new_mining_block(None);
87        self.state.touch_last_assemble();
88
89        let problem = ProofOfWorkProblem::from_block_header(
90            &next_mining_block.block_header,
91        );
92
93        self.state.update_job(next_mining_block, problem);
94        self.send_problem(problem);
95    }
96
97    /// Pull mining solution from the workers, returns whether a valid solution
98    /// is found
99    fn process_pending_solutions(&mut self) {
100        while let Ok(solution) = self.solution_rx.try_recv() {
101            debug!("new solution: {:?}", solution);
102
103            if let Some(mut mined_block) =
104                self.state.validate_and_claim_solution(&solution, |p, s| {
105                    self.bg.pow.validate(p, s)
106                })
107            {
108                mined_block.block_header.set_nonce(solution.nonce);
109                mined_block.block_header.compute_hash();
110
111                self.on_mined_block(mined_block);
112                return;
113            } else {
114                warn!(
115                    "Received invalid solution from miner: nonce = {}!",
116                    &solution.nonce
117                );
118            }
119        }
120
121        // The solution channel is empty, or all pending solutions were invalid.
122        // We should ensure if the miner has current job.
123        self.ensure_miner_has_current_work();
124        // Avoid busy-waiting
125        thread::sleep(BLOCKGEN_LOOP_SLEEP_DURATION);
126    }
127
128    /// We will send out heartbeat because newcomers or
129    /// disconnected people may lose the previous message
130    fn ensure_miner_has_current_work(&mut self) {
131        if !self.pow_config.use_stratum() {
132            return;
133        }
134
135        if let Some(problem) = self.state.check_for_renotification() {
136            self.send_problem(problem);
137        }
138    }
139
140    fn send_problem(&mut self, problem: ProofOfWorkProblem) {
141        self.miner.receive_problem(problem);
142
143        trace!("send problem: {:?}", problem);
144        self.state.touch_last_notify();
145    }
146}
