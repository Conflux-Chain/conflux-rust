1use cfxcore::pow::{ProofOfWorkProblem, ProofOfWorkSolution};
2use primitives::Block;
3use std::{
4    collections::VecDeque,
5    time::{self, Instant},
6};
7use time::Duration;
8
9#[derive(Clone, Debug)]
10pub(crate) struct MineJob {
11    block: Block,
12    problem: ProofOfWorkProblem,
13}
14
15pub(crate) struct MineState {
16    job_history: VecDeque<MineJob>,
17    mining_job: Option<MineJob>,
18    last_notify_at: Instant,
19    last_assemble_at: Instant,
20    history_capacity: usize,
21}
22
23impl MineState {
24    /// Creates a new job cache with the specified history capacity.
25    pub fn with_capacity(capacity: usize) -> Self {
26        MineState {
27            job_history: VecDeque::with_capacity(capacity),
28            mining_job: None,
29            last_notify_at: Instant::now(),
30            last_assemble_at: Instant::now(),
31            history_capacity: capacity,
32        }
33    }
34
35    pub fn mining_block(&self) -> Option<&Block> {
36        self.mining_job.as_ref().map(|job| &job.block)
37    }
38
39    pub fn last_assemble_elapsed(&self) -> Duration {
40        self.last_assemble_at.elapsed()
41    }
42
43    pub fn touch_last_assemble(&mut self) {
44        self.last_assemble_at = Instant::now();
45    }
46
47    pub fn touch_last_notify(&mut self) {
48        self.last_notify_at = Instant::now();
49    }
50
51    /// Sets a new mining job as the current one and adds it to the history.
52    ///
53    /// This should be called whenever a new block template is created.
54    pub fn update_job(&mut self, block: Block, problem: ProofOfWorkProblem) {
55        if self.job_history.len() == self.history_capacity {
56            self.job_history.pop_front();
57        }
58        let new_job = MineJob { block, problem };
59        self.job_history.push_back(new_job.clone());
60        self.mining_job = Some(new_job);
61        self.last_assemble_at = Instant::now();
62    }
63
64    /// Validates a proof-of-work solution against the job history.
65    ///
66    /// If the solution is valid for any recent job, this method returns the
67    /// corresponding block and clears the current job.
68    pub fn validate_and_claim_solution(
69        &mut self, solution: &ProofOfWorkSolution,
70        validate: impl Fn(&ProofOfWorkProblem, &ProofOfWorkSolution) -> bool,
71    ) -> Option<Block> {
72        let result = self
73            .job_history
74            .iter()
75            .find(|job| validate(&job.problem, solution))
76            .map(|job| job.block.clone());
77
78        if result.is_some() {
79            // Clear current mining info after a solution is found
80            self.mining_job = None;
81        }
82        result
83    }
84
85    /// Checks if it's time to re-notify miners of the current proof-of-work
86    /// problem.
87    ///
88    /// This is a heartbeat mechanism to ensure that new or temporarily
89    /// disconnected miners receive the current job.
90    pub fn check_for_renotification(&self) -> Option<ProofOfWorkProblem> {
91        let Some(MineJob { problem, .. }) = self.mining_job.as_ref() else {
92            return None;
93        };
94        if self.last_notify_at.elapsed() > Duration::from_secs(60) {
95            Some(problem.clone())
96        } else {
97            None
98        }
99    }
100}
